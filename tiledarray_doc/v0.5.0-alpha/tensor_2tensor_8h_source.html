1246  const size_type n = pimpl_->range_.rank();

1248  const size_type* restrict const upper = pimpl_->range_.upbound_data();
1249  const size_type* restrict const stride = pimpl_->range_.stride_data();
1250 
1251  // Search for the largest lower bound and the smallest upper bound
1252  size_type lower_max = 0ul, upper_min =
1253  std::numeric_limits<size_type>::max();
1254  for(size_type i = 0ul; i < n; ++i) {
1255  const size_type lower_i = lower[i];
1256  const size_type upper_i = upper[i];
1257 
1258  lower_max = std::max(lower_max, lower_i);
1259  upper_min = std::min(upper_min, upper_i);
1260  }
1261 
1262  value_type result = 0;
1263 
1264  if(lower_max < upper_min) {
1265  // Compute the first and last ordinal index
1266  size_type first = 0ul, last = 0ul, trace_stride = 0ul;
1267  for(size_type i = 0ul; i < n; ++i) {
1268  const size_type lower_i = lower[i];
1269  const size_type stride_i = stride[i];
1270 
1271  first += (lower_max - lower_i) * stride_i;
1272  last += (upper_min - lower_i) * stride_i;
1273  trace_stride += stride_i;
1274  }
1275 
1276  // Compute the trace
1277  const value_type* restrict const data = pimpl_->data_;
1278  for(; first < last; first += trace_stride)
1279  result += data[first];
