78  void init_range_data(const Index& lower_bound, const Index& upper_bound) {
79  // Construct temp pointers
80  size_type* restrict const lower = data_;
81  size_type* restrict const upper = lower + rank_;
82  size_type* restrict const extent = upper + rank_;
83  size_type* restrict const stride = extent + rank_;
84  const auto* restrict const lower_data = detail::data(lower_bound);
85  const auto* restrict const upper_data = detail::data(upper_bound);
86 
87  // Set the volume seed
88  volume_ = 1ul;
89  offset_ = 0ul;
90 
91  // Compute range data
92  for(int i = int(rank_) - 1; i >= 0; --i) {
93  // Check input dimensions
94  TA_ASSERT(lower_data[i] >= 0ul);
95  TA_ASSERT(lower_data[i] < upper_data[i]);
96 
97  // Compute data for element i of lower, upper, and extent
98  const size_type lower_bound_i = lower_data[i];
99  const size_type upper_bound_i = upper_data[i];
100  const size_type extent_i = upper_bound_i - lower_bound_i;
101 
102  lower[i] = lower_bound_i;
103  upper[i] = upper_bound_i;
104  extent[i] = extent_i;
105  stride[i] = volume_;
106  offset_ += lower_bound_i * volume_;
107  volume_ *= extent_i;
108  }

158  const size_type* restrict const other_lower_bound,
159  const size_type* restrict const other_upper_bound)
160  {
161  // Create temporary pointers to this range data
162  size_type* restrict const lower = data_;
163  size_type* restrict const upper = lower + rank_;
164  size_type* restrict const extent = upper + rank_;
165  size_type* restrict const stride = extent + rank_;

167  // Copy the permuted lower, upper, and extent into this range.
168  for(unsigned int i = 0u; i < rank_; ++i) {
169  const size_type perm_i = perm[i];

172  const size_type other_lower_bound_i = other_lower_bound[i];
173  const size_type other_upper_bound_i = other_upper_bound[i];

178  upper[perm_i] = other_upper_bound_i;

186  const size_type lower_i = lower[i];
187  const size_type extent_i = extent[i];
188  stride[i] = volume_;
189  offset_ += lower_i * volume_;
