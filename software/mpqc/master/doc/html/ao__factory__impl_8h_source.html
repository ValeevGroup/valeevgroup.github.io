88 template <typename Tile, typename Policy>
89 typename AOFactory<Tile, Policy>::TArray AOFactory<Tile, Policy>::compute(
90  const Formula& formula) {
91  TA_USER_ASSERT(lcao::detail::if_all_ao(formula),
92  "AOFactory only accept AO index!\n");
93 
94  ExEnv::out0() << incindent;
95 
96  TArray result;
97 
98  auto& world = this->world();
99 
100  // find if in registry
101  auto iter = this->registry_.find(formula);
102 
103  if (iter != this->registry_.end()) {
104  result = iter->second;
105  if(this->verbose_){
106  ExEnv::out0() << indent;
107  ExEnv::out0() << "Retrieved AO Integral: "
108  << utility::to_string(formula.string());
109  double size = mpqc::detail::array_size(result);
110  ExEnv::out0() << " Size: " << size << " GB" << std::endl;
111  }
112  } else {
113  // find a permutation, currently, it won't store permutation in registry
114 
115  std::vector<Formula> permutes = permutations(formula);
116  typename FormulaRegistry<TArray>::iterator find_permute;
117 
118  for (auto& permute : permutes) {
119  find_permute = this->registry_.find(permute);
120  if (find_permute != this->registry_.end()) {
121  mpqc::time_point time0 = mpqc::now(world, this->accurate_time_);
122 
123  // permute the array
124  result(formula.to_ta_expression()) =
125  (find_permute->second)(permute.to_ta_expression());
126 
127  mpqc::time_point time1 = mpqc::now(world, this->accurate_time_);
128  double time = mpqc::duration_in_s(time0, time1);
129 
130  if(this->verbose_){
131  ExEnv::out0() << indent;
132  ExEnv::out0() << "Permuted AO Integral: "
133  << utility::to_string(formula.string()) << " From "
134  << utility::to_string(permute.string());
135  double size = mpqc::detail::array_size(result);
136  ExEnv::out0() << " Size: " << size << " GB"
137  << " Time: " << time << " s" << std::endl;
138  }
139 
140  // store current array and delete old one
141  this->registry_.insert(formula, result);
142  this->registry_.purge_formula(permute);
143  }
144  }
145 
146  // if formula not found
147  if (!result.is_initialized()) {
148  // compute formula
149  if (formula.rank() == 2) {
150  result = compute2(formula);
151  this->registry_.insert(formula, result);
152  } else if (formula.rank() == 3) {
153  result = compute3(formula);
154  this->registry_.insert(formula, result);
155  } else if (formula.rank() == 4) {
156  result = compute4(formula);
157  this->registry_.insert(formula, result);
158  }
159  }
160 
161  madness::print_meminfo(
162  world.rank(), "AOFactory: " + utility::to_string(formula.string()));
163  }
164  ExEnv::out0() << decindent;
165  return result;
166 }

168 template <typename Tile, typename Policy>
169 typename AOFactory<Tile, Policy>::TArray AOFactory<Tile, Policy>::compute2(
170  const Formula& formula) {
171  BasisVector bs_array;
172  double time = 0.0;
173  mpqc::time_point time0;
174  mpqc::time_point time1;
175  TArray result;
176  auto& world = this->world();
177 
178  // get the inverse square root instead
179  if (iterative_inv_sqrt_ && formula.has_option(Formula::Option::Inverse)) {
180  auto inv_sqrt_formula = formula;
181  inv_sqrt_formula.clear_option();
182  inv_sqrt_formula.add_option(Formula::Option::InverseSquareRoot);
183 
184  result = this->compute(inv_sqrt_formula);
185 
186  time0 = mpqc::now(world, this->accurate_time_);
187  result("p,q") = result("p,r") * result("r,q");
188 
189  if (formula.oper().type() == Operator::Type::cGTG ||
190  formula.oper().type() == Operator::Type::cGTGCoulomb) {
191  result("i,j") = -result("i,j");
192  }
193 
194  if(this->verbose_){
195  ExEnv::out0() << indent;
196  ExEnv::out0() << "Computed Inverse of Integral: "
197  << utility::to_string(formula.string());
198  double size = mpqc::detail::array_size(result);
199 
200  time1 = mpqc::now(world, this->accurate_time_);
201  time += mpqc::duration_in_s(time0, time1);
202  ExEnv::out0() << " Size: " << size << " GB"
203  << " Time: " << time << " s" << std::endl;
204  }
205  }
206  // continue with normal step
207  else {
208  // use one body engine
209  if (formula.oper().is_onebody()) {
210  // H = V + T
211  if (formula.oper().type() == Operator::Type::Core) {
212  auto v_formula = formula;
213  v_formula.set_operator_type(Operator::Type::Nuclear);
214 
215  auto t_formula = formula;
216  t_formula.set_operator_type(Operator::Type::Kinetic);
217 
218  auto v = this->compute(v_formula);
219  auto t = this->compute(t_formula);
220 
221  time0 = mpqc::now(world, this->accurate_time_);
222 
223  result("i,j") = v("i,j") + t("i,j");
224 
225  time1 = mpqc::now(world, this->accurate_time_);
226  time += mpqc::duration_in_s(time0, time1);
227  }
228  // one body integral S, V, T...
229  else {
230  time0 = mpqc::now(world, this->accurate_time_);
231 
232  std::shared_ptr<utility::TSPool<libint2::Engine>> engine_pool;
233  parse_one_body(formula, engine_pool, bs_array);
234  result = compute_integrals(world, engine_pool, bs_array);
235 
236  time1 = mpqc::now(world, this->accurate_time_);
237  time += mpqc::duration_in_s(time0, time1);
238  }
239 
240  if(this->verbose_){
241  ExEnv::out0() << indent;
242  ExEnv::out0() << "Computed One Body Integral: "
243  << utility::to_string(formula.string());
244  double size = mpqc::detail::array_size(result);
245  ExEnv::out0() << " Size: " << size << " GB"
246  << " Time: " << time << " s" << std::endl;
247  }
248  }
249  // use two body engine
250  else if (formula.oper().is_twobody()) {
251  time0 = mpqc::now(world, this->accurate_time_);
252 
253  // compute integral
254  std::shared_ptr<utility::TSPool<libint2::Engine>> engine_pool;
255  parse_two_body_two_center(formula, engine_pool, bs_array);
256  result = compute_integrals(world, engine_pool, bs_array);
257 
258  time1 = mpqc::now(world, this->accurate_time_);
259  time += mpqc::duration_in_s(time0, time1);
260 
261  if(this->verbose_){
262  ExEnv::out0() << indent;
263  ExEnv::out0() << "Computed Twobody Two Center Integral: "
264  << utility::to_string(formula.string());
265  double size = mpqc::detail::array_size(result);
266  ExEnv::out0() << " Size: " << size << " GB"
267  << " Time: " << time << " s" << std::endl;
268  }
269  }
270  // compute JK, requires orbital space registry
271  else if (formula.oper().is_jk()) {
272  // density fitting case
273 
274  // find the density
275  auto space_index = detail::get_jk_orbital_space(formula.oper());
276  auto& space = this->orbital_registry().retrieve(space_index);
277 
278  auto obs = space.ao_index().name();
279  if (formula.has_option(Formula::Option::DensityFitting)) {
280  auto three_center_formula = detail::get_jk_df_formula(formula, obs);
281 
282  auto left = compute_direct(three_center_formula[0]);
283  auto center = compute(three_center_formula[1]);
284  auto right = compute_direct(three_center_formula[2]);
285 
286  time0 = mpqc::now(world, this->accurate_time_);
287 
288  // J case
289  if (formula.oper().type() == Operator::Type::J) {
290  result("i,j") = space("k,a") * space("l,a") * right("Q,k,l") *
291  center("K,Q") * left("K,i,j");
292  }
293  // K case
294  else {
295  result("i,j") = (left("K,i,k") * space("k,a")) * center("K,Q") *
296  (right("Q,j,l") * space("l,a"));
297  }
298 
299  time1 = mpqc::now(world, this->accurate_time_);
300  time += mpqc::duration_in_s(time0, time1);
301  }
302  // four center case
303  else {
304  // find the density
305  auto space_index = detail::get_jk_orbital_space(formula.oper());
306  auto& space = this->orbital_registry().retrieve(space_index);
307  auto obs = space.ao_index().name();
308  // convert to ao formula
309  auto four_center_formula = detail::get_jk_formula(formula, obs);
310  auto four_center = this->compute(four_center_formula);
311 
312  time0 = mpqc::now(world, this->accurate_time_);
313 
314  if (formula.notation() == Formula::Notation::Chemical) {
315  if (formula.oper().type() == Operator::Type::J) {
316  result("rho,sigma") = four_center("rho,sigma,mu,nu") *
317  (space("mu,i") * space("nu,i"));
318  } else {
319  result("rho,sigma") = four_center("rho,mu,sigma,nu") *
320  (space("mu,i") * space("nu,i"));
321  }
322  } else {
323  if (formula.oper().type() == Operator::Type::K) {
324  result("rho,sigma") = four_center("rho,sigma,mu,nu") *
325  (space("mu,i") * space("nu,i"));
326  } else {
327  result("rho,sigma") = four_center("rho,mu,sigma,nu") *
328  (space("mu,i") * space("nu,i"));
329  }
330  }
331 
332  time1 = mpqc::now(world, this->accurate_time_);
333  time += mpqc::duration_in_s(time0, time1);
334  }
335 
336  if(this->verbose_){
337  ExEnv::out0() << indent;
338  ExEnv::out0() << "Computed Coulumb/Exchange Integral: "
339  << utility::to_string(formula.string());
340  double size = mpqc::detail::array_size(result);
341  ExEnv::out0() << " Size: " << size << " GB"
342  << " Time: " << time << " s" << std::endl;
343  }
344  }
345  // hJ = H + J
346  else if (formula.oper().type() == Operator::Type::hJ) {
347  auto h_formula = formula;
348  h_formula.set_operator(Operator(L"H"));
349 
350  auto j_formula = formula;
351  j_formula.set_operator_type(Operator::Type::J);
352 
353  auto h = this->compute(h_formula);
354  auto j = this->compute(j_formula);
355 
356  time0 = mpqc::now(world, this->accurate_time_);
357 
358  result("i,j") = h("i,j") + 2 * j("i,j");
359 
360  time1 = mpqc::now(world, this->accurate_time_);
361  time += mpqc::duration_in_s(time0, time1);
362 
363  if(this->verbose_){
364  ExEnv::out0() << indent;
365  ExEnv::out0() << "Computed Coulumb/Exchange Integral: "
366  << utility::to_string(formula.string());
367  double size = mpqc::detail::array_size(result);
368  ExEnv::out0() << " Size: " << size << " GB"
369  << " Time: " << time << " s" << std::endl;
370  }
371  }
372  // compute Fock, requires orbital space registry
373  else if (formula.oper().is_fock()) {
374  auto formulas = detail::get_fock_formula(formula);
375 
376  auto h = compute(formulas[0]);
377  auto j = compute(formulas[1]);
378  auto k = compute(formulas[2]);
379 
380  time0 = mpqc::now(world, this->accurate_time_);
381  // if closed shell
382  if (formula.oper().type() == Operator::Type::Fock) {
383  result("rho,sigma") =
384  h("rho,sigma") + 2 * j("rho,sigma") - k("rho,sigma");
385  }
386  // else if spin orbital
387  else {
388  result("rho,sigma") = h("rho,sigma") + j("rho,sigma") - k("rho,sigma");
389  }
390 
391  time1 = mpqc::now(world, this->accurate_time_);
392 
393  time += mpqc::duration_in_s(time0, time1);
394 
395  if(this->verbose_){
396 
397  ExEnv::out0() << indent;
398  ExEnv::out0() << "Computed Fock Integral: "
399  << utility::to_string(formula.string());
400  double size = mpqc::detail::array_size(result);
401  ExEnv::out0() << " Size: " << size << " GB"
402  << " Time: " << time << " s" << std::endl;
403  }
404  }
405  }
406 
407  // check for other options
408 
409  // compute inverse
410  if (!iterative_inv_sqrt_ && formula.has_option(Formula::Option::Inverse)) {
411  time0 = mpqc::now(world, this->accurate_time_);
412 
413  if (formula.oper().type() == Operator::Type::cGTG ||
414  formula.oper().type() == Operator::Type::cGTGCoulomb) {
415  result("i,j") = -result("i,j");
416  }
417 
418  auto tmp = math::eigen_inverse(result);
419  result("i,j") = tmp("i,j");
420 
421  if (formula.oper().type() == Operator::Type::cGTG ||
422  formula.oper().type() == Operator::Type::cGTGCoulomb) {
423  result("i,j") = -result("i,j");
424  }
425  time1 = mpqc::now(world, this->accurate_time_);
426  auto inv_time = mpqc::duration_in_s(time0, time1);
427  if(this->verbose_){
428  ExEnv::out0() << indent;
429  ExEnv::out0() << "Inverse Time: " << inv_time << " s" << std::endl;
430  }
431  }
432 
433  // inverse square root of integral
434  if (formula.has_option(Formula::Option::InverseSquareRoot)) {
435  time0 = mpqc::now(world, this->accurate_time_);
436  if (formula.oper().type() == Operator::Type::cGTG ||
437  formula.oper().type() == Operator::Type::cGTGCoulomb) {
438  result("i,j") = -result("i,j");
439  }
440 
441  if (iterative_inv_sqrt_) {
442  TArray tmp = math::inverse_sqrt(result);
443  result("i,j") = tmp("i,j");
444  } else {
445  auto result_eig = math::array_to_eigen(result);
446 
447  Eigen::SelfAdjointEigenSolver<decltype(result_eig)> es(result_eig);
448  decltype(result_eig) inv_eig = es.operatorInverseSqrt();
449 
450  auto tr_result = result.trange().data()[0];
451  result = math::eigen_to_array<Tile, Policy>(result.world(), inv_eig,
452  tr_result, tr_result);
453  }
454 
455  if (formula.oper().type() == Operator::Type::cGTG ||
456  formula.oper().type() == Operator::Type::cGTGCoulomb) {
457  result("i,j") = -result("i,j");
458  }
459  time1 = mpqc::now(world, this->accurate_time_);
460  auto inv_sqrt_time = mpqc::duration_in_s(time0, time1);
461  if(this->verbose_){
462  ExEnv::out0() << indent;
463  ExEnv::out0() << "Inverse Square Root Time: " << inv_sqrt_time << " s" << std::endl;
464  }
465  }
466 
467  return result;
468 }

548  mpqc::time_point time1;

551 
552  if (formula.has_option(Formula::Option::DensityFitting)) {
553  // convert formula to df formula

555 
556  // compute integral

558  auto right = compute(formula_strings[1]);
559 
560  time0 = mpqc::now(world, this->accurate_time_);
561 
562  std::string result_str = formula.notation() == Formula::Notation::Chemical
563  ? "i,j,k,l"

565

567  formula.oper().type() == Operator::Type::cGTGCoulomb)
568  {
569  result(result_str) = left("q,i,j") * (-right("q,k,l"));
570  } else {
571  result(result_str) = left("q,i,j") * right("q,k,l");
572  }
573 
574  time1 = mpqc::now(world, this->accurate_time_);
575  time += mpqc::duration_in_s(time0, time1);
576 
577  if(this->verbose_){

580  << utility::to_string(formula.string());
581  double size = mpqc::detail::array_size(result);
582  ExEnv::out0() << " Size: " << size << " GB"
583  << " Time: " << time << " s" << std::endl;
584  }
585 
586  } else {
587  time0 = mpqc::now(world, this->accurate_time_);
588 
589  BasisVector bs_array;
590  std::shared_ptr<Screener> p_screener =
591  std::make_shared<Screener>(Screener{});
592
