114 void Debugger::handle(int sig) {
115  if (sig >= NSIG) return;
116 #ifdef HAVE_SIGNAL
117  typedef void (*handler_type)(int);
118  signal(sig, (handler_type)handler);
119 #endif
120  signals[sig] = this;
121  mysigs_[sig] = 1;
122 }
123 
124 void Debugger::release(int sig) {
125  if (sig >= NSIG) return;
126  signal(sig, SIG_DFL);
127  signals[sig] = nullptr;
128  mysigs_[sig] = 0;
129 }

209 void Debugger::debug(const char *reason) {
210  ExEnv::outn() << prefix_ << "Debugger::debug: ";
211  if (reason)
212  ExEnv::outn() << reason;
213  else
214  ExEnv::outn() << "no reason given";
215  ExEnv::outn() << endl;
216 
217 #ifndef HAVE_SYSTEM
218  abort();
219 #else
220  if (!cmd_.empty()) {
221  int pid = getpid();
222  // contruct the command name
223  std::string cmd = cmd_;
224  std::string::size_type pos;
225  std::string pidvar("$(PID)");
226  while ((pos = cmd.find(pidvar)) != std::string::npos) {
227  std::string pidstr;
228  pidstr += std::to_string(pid);
229  cmd.replace(pos, pidvar.size(), pidstr);
230  }
231  std::string execvar("$(EXEC)");
232  while ((pos = cmd.find(execvar)) != std::string::npos) {
233  cmd.replace(pos, execvar.size(), exec_);
234  }
235  std::string prefixvar("$(PREFIX)");
236  while ((pos = cmd.find(prefixvar)) != std::string::npos) {
237  cmd.replace(pos, prefixvar.size(), prefix_);
238  }
239  // start the debugger
240  // before starting the debugger de-register signal handler for SIGTRAP to let the debugger take over
241  release(SIGTRAP);
242  ExEnv::outn() << prefix_ << "Debugger: starting \"" << cmd << "\"" << endl;
243  debugger_ready_ = 0;
244  const auto system_retvalue = system(cmd.c_str());
245  if (system_retvalue != 0) { // call to system() failed
246  ExEnv::outn() << prefix_
247  << "Failed debugger launch: system() did not succeed ..."
248  << endl;
249  } else { // call to system() succeeded
250  // wait until the debugger is ready
251  if (sleep_) {
252  ExEnv::outn() << prefix_ << "Sleeping " << sleep_
253  << " seconds to wait for debugger ..." << endl;
254  sleep(sleep_);
255  }
256  if (wait_for_debugger_) {
257  std::string make_ready_message;
258  if (cmd_ == "gdb_xterm") {
259  make_ready_message = " type 'set variable debugger_ready_=1' then type 'c' to continue running";
260  } else if (cmd_ == "lldb_xterm") {
261  make_ready_message = " type 'expr debugger_ready_=1' then type 'c' to continue running";
262  }
263 
264  ExEnv::outn() << prefix_
265  << ": Spinning until debugger_ready_ is set ..." << make_ready_message << endl;
266  while (!debugger_ready_)
267  ;
268  }
269  }
270  }
271 #endif
272 }

274 void Debugger::got_signal(int sig) {
275  const char *signame;
276  if (sig == SIGSEGV)
277  signame = "SIGSEGV";
278  else if (sig == SIGFPE)
279  signame = "SIGFPE";
280  else if (sig == SIGHUP)
281  signame = "SIGHUP";
282  else if (sig == SIGINT)
283  signame = "SIGINT";
284  else if (sig == SIGABRT)
285  signame = "SIGABRT";
286 #ifdef SIGBUS
287  else if (sig == SIGBUS)
288  signame = "SIGBUS";
289 #endif
290  else if (sig == SIGTRAP)
291  signame = "SIGTRAP";
292  else
293  signame = "UNKNOWN SIGNAL";
294 
295  if (traceback_) {
296  traceback(signame);
297  }
298  if (debug_) {
299  debug(signame);
300  }
301 
302  if (exit_on_signal_) {
303  ExEnv::outn() << prefix_ << "Debugger: exiting" << endl;
304  exit(1);
305  } else {
306  ExEnv::outn() << prefix_ << "Debugger: continuing" << endl;
307  }
308 
309  // handle(sig);
310 }

335 void Debugger::__traceback(const std::string &prefix, const char *reason) {
336  detail::Backtrace result(prefix);
337  const size_t nframes_to_skip = 2;
338 #if defined(HAVE_LIBUNWIND)
339  ExEnv::outn() << prefix << "Debugger::traceback(using libunwind):";
340 #elif defined(HAVE_BACKTRACE) // !HAVE_LIBUNWIND
341  ExEnv::outn() << prefix << "Debugger::traceback(using backtrace):";
342 #else // !HAVE_LIBUNWIND && !HAVE_BACKTRACE
343 #if defined(SIMPLE_STACK)
344  ExEnv::outn() << prefix << "Debugger::traceback:";
345 #else
346  ExEnv::outn() << prefix << "traceback not available for this arch" << endl;
347  return;
348 #endif // SIMPLE_STACK
349 #endif // HAVE_LIBUNWIND, HAVE_BACKTRACE
350 
351  if (reason)
352  ExEnv::outn() << reason;
353  else
354  ExEnv::outn() << "no reason given";
355  ExEnv::outn() << endl;
356 
357  if (result.empty())
358  ExEnv::outn() << prefix << "backtrace returned no state information"
359  << std::endl;
360  else
361  ExEnv::outn() << result.str(nframes_to_skip) << std::endl;
362 }
