90  template <typename Index>
91  BlkTsrExpr(const TsrExpr<A>& tsr, const Index& lower_bound,
92  const Index& upper_bound) :
93  Expr_(), array_(tsr.array()), vars_(tsr.vars()),
94  lower_bound_(std::begin(lower_bound), std::end(lower_bound)),
95  upper_bound_(std::begin(upper_bound), std::end(upper_bound))
96  {
97 #ifndef NDEBUG
98  const unsigned int rank = array_.trange().tiles().rank();
99  // Check the dimension of the lower block bound
100  if(TiledArray::detail::size(lower_bound) != rank) {
101  if(World::get_default().rank() == 0) {
102  TA_USER_ERROR_MESSAGE( \
103  "The size lower bound of the block is not equal to rank of " \
104  "the array: " \
105  << "\n array rank = " << array_.trange().tiles().rank() \
106  << "\n lower bound = " << lower_bound );
107 
108  TA_EXCEPTION("The size lower bound of the block is not equal to " \
109  "rank of the array.");
110  }
111  }
112 
113  // Check the dimension of the upper block bound
114  if(TiledArray::detail::size(upper_bound) != rank) {
115  if(World::get_default().rank() == 0) {
116  TA_USER_ERROR_MESSAGE( \
117  "The size upper bound of the block is not equal to rank of " \
118  "the array: " \
119  << "\n array rank = " << rank \
120  << "\n upper bound = " << upper_bound );
121 
122  TA_EXCEPTION("The size upper bound of the block is not equal to " \
123  "rank of the array.");
124  }
125  }
126 
127  const bool lower_bound_check =
128  std::equal(std::begin(lower_bound), std::end(lower_bound),
129  tsr.array().trange().tiles().lobound_data(),
130  [] (std::size_t l, std::size_t r) { return l >= r; });
131  const bool upper_bound_check =
132  std::equal(std::begin(upper_bound), std::end(upper_bound),
133  tsr.array().trange().tiles().upbound_data(),
134  [] (std::size_t l, std::size_t r) { return l <= r; });
135  if(! (lower_bound_check && upper_bound_check)) {
136  if(World::get_default().rank() == 0) {
137  TA_USER_ERROR_MESSAGE( \
138  "The block range is not a sub-block of the array range: " \
139  << "\n array range = " << array_.trange().tiles() \
140  << "\n block range = [ " << lower_bound << " , " << upper_bound << " )");
141  }
142 
143  TA_EXCEPTION("The block range is not a sub-block of the array range.");
144  }
145 
146  const bool lower_upper_bound_check =
147  std::equal(std::begin(lower_bound), std::end(lower_bound),
148  std::begin(upper_bound),
149  [] (std::size_t l, std::size_t r) { return l < r; });
150  if(! lower_upper_bound_check) {
151  if(World::get_default().rank() == 0) {
152  TA_USER_ERROR_MESSAGE( \
153  "The block lower bound is not less than the upper bound: " \
154  << "\n lower bound = " << lower_bound \
155  << "\n upper bound = " << upper_bound);
156  }
157 
158  TA_EXCEPTION("The block lower bound is not less than the upper bound.");
159  }
160 
161 #endif // NDEBUG
162  }
