159  auto &world = D.world();
160  auto j0 = mpqc::fenced_now(world);
161  ArrayType J;

173  ArrayType compute_K(ArrayType const &LMO_in, ArrayType const &D,
174  double target_precision) {
175  using ::mpqc::detail::array_storage;
176 
177  auto &world = M_.world();
178  auto Exch0 = mpqc::fenced_now(world);
179  ArrayType L, K; // Matrices
180  ArrayType Z, F; // Tensors
181 
182  // Deep copy LMO for chopping
183  ArrayType LMO;
184  LMO("mu, i") = LMO_in("mu, i");
185 
186  if (LMO_chop_threshold_ != 0.0) {
187  auto chop0 = mpqc::fenced_now(world);
188  TA::foreach_inplace(LMO, [&](TA::Tensor<double> &t) {
189  const auto norm = t.norm();
190  if (norm > LMO_chop_threshold_) {
191  return norm;
192  } else {
193  return 0.0;
194  }
195  });
196  auto chop1 = mpqc::fenced_now(world);
197  LMO_chop_times_.push_back(mpqc::duration_in_s(chop0, chop1));
198 
199  auto chop_sizes = array_storage(LMO);
200  LMO_chopped_sizes_.push_back(
201  std::array<double, 2>{{chop_sizes[0], chop_sizes[1]}});
202  }
203 
204  // Contract C with orbitals
205  auto z0 = mpqc::fenced_now(world);
206  Z("X, i, mu") = C_("X, mu, nu") * LMO("nu, i");
207  Z.truncate();
208  auto z1 = mpqc::fenced_now(world);
209  Z_times_.push_back(mpqc::duration_in_s(z0, z1));
210 
211  auto Z_sizes = array_storage(Z);
212  Z_sizes_.push_back(std::array<double, 2>{{Z_sizes[0], Z_sizes[1]}});
213 
214  // Get forced output shape
215  TA::SparseShape<float> forced_shape;
216  if (force_threshold_ > 0.0) {
217  auto cadf_df_k_shape = [&](TA::Tensor<float> const &input) {
218  auto &range = input.range();
219  auto extent = range.extent_data();
220 
221  TA::Tensor<float> t(range, 0.0);
222 
223  std::unordered_set<int64_t> sig_mu;
224  std::unordered_set<int64_t> sig_X;
225 
226  for (auto i = 0ul; i < extent[1]; ++i) {
227  sig_mu.clear();
228  sig_X.clear();
229 
230  // For every I we determine save a list of important X and mu
231  for (auto X = 0ul; X < extent[0]; ++X) {
232  for (auto mu = 0ul; mu < extent[2]; ++mu) {
233  const auto val = input(X, i, mu);
234 
235  if (val >= force_threshold_) {
236  sig_X.insert(X);
237  sig_mu.insert(mu);
238  }
239  }
240  }
241 
242  // Then for every X we mark all important mu. The output will have
243  // significant mu X pairs that the input did not.
244  for (auto const &X : sig_X) {
245  for (auto const &mu : sig_mu) {
246  t(X, i, mu) = std::numeric_limits<float>::max();
247  }
248  }
249  }
250 
251  return t;
252  };
253  auto shape_time0 = mpqc::fenced_now(world);
254  forced_shape = Z.shape().transform(cadf_df_k_shape);
255  auto shape_time1 = mpqc::fenced_now(world);
256  shape_times_.push_back(mpqc::duration_in_s(shape_time0, shape_time1));
257  }
258 
259  // Construct F
260  auto f0 = mpqc::fenced_now(world);
261  if (force_threshold_ == 0.0) {
262  F("X, i, mu") =
263  E_("X, mu, nu") * LMO("nu, i") - 0.5 * M_("X,Y") * Z("Y, i, mu");
264  } else {
265  F("X, i, mu") = (E_("X, mu, nu") * LMO("nu, i")).set_shape(forced_shape);
266  F.truncate();
267 
268  F("X, i, mu") +=
269  (-0.5 * M_("X, Y") * Z("Y, i, mu")).set_shape(forced_shape);
270  }
271  F.truncate();
272  auto f1 = mpqc::fenced_now(world);
273  F_times_.push_back(mpqc::duration_in_s(f0, f1));
274 
275  auto F_sizes = array_storage(F);
276  F_sizes_.push_back(std::array<double, 2>{{F_sizes[0], F_sizes[1]}});
277 
278  // Construct L
279  auto l0 = mpqc::fenced_now(world);
280  L("mu, nu") = Z("X, i, mu") * F("X, i, nu");
281  L.truncate();
282  auto l1 = mpqc::fenced_now(world);
283  L_times_.push_back(mpqc::duration_in_s(l0, l1));
284 
285  auto k0 = mpqc::fenced_now(world);
286  if (!secadf_) {
287  K("mu, nu") = L("mu, nu") + L("nu, mu");
288  } else {
289  if (K_off_diagonal_.empty()) {
290  K("mu, nu") = L("mu, nu") + L("nu, mu");
291  world.gop.fence();
292  auto off = [this](TA::Tensor<float> const &t) {
293  return this->off_diagonal_shape(t);
294  };
295  K_off_diagonal_ = K.shape().transform(off);
296  }
297  K("mu, nu") = (L("mu, nu") + L("nu, mu")).set_shape(K_off_diagonal_);
298  }
299  K.truncate();
300 
301  // seCADF correction
302  if (secadf_) {
303  auto se0 = mpqc::fenced_now(world);
304  auto K_se = exactK_->operator()(D, LMO, target_precision);
305 
306  if (K_diagonal_.empty()) {
307  auto diag = [this](TA::Tensor<float> const &t) {
308  return this->diagonal_shape(t);
309  };
310  K_diagonal_ = K_se.shape().transform(diag);
311  }
312  K_se("mu, nu") = K_se("mu, nu").set_shape(K_diagonal_);
313 
314  // Use - because exactK_ returns negative K
315  K("mu, nu") = K("mu, nu") - K_se("mu, nu");
316  auto se1 = mpqc::fenced_now(world);
317  se_times_.push_back(mpqc::duration_in_s(se0, se1));
318  }
319  auto k1 = mpqc::fenced_now(world);
320  K_times_.push_back(mpqc::duration_in_s(k0, k1));
321  Exch_times_.push_back(mpqc::duration_in_s(Exch0, k1));
322 
323  energy_K_ = D("i,j").dot(K("i,j"));
324 
325  return K;
326  }
