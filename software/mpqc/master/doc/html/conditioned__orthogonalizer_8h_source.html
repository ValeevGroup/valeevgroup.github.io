16 template <typename T>
17 std::tuple<RowMatrix<T>, RowMatrix<T>, size_t, double, double> gensqrtinv(
18  const RowMatrix<T>& S, bool symmetric = false,
19  double max_condition_number = 1e8) {
20  using Matrix = RowMatrix<T>;
21 
22  Eigen::SelfAdjointEigenSolver<Matrix> eig_solver(S);
23  auto U = eig_solver.eigenvectors();
24  auto s = eig_solver.eigenvalues();
25  auto s_max = s.maxCoeff();
26  auto condition_number = std::min(
27  s_max / std::max(s.minCoeff(), std::numeric_limits<double>::min()),
28  1.0 / std::numeric_limits<double>::epsilon());
29  auto threshold = s_max / max_condition_number;
30  long n = s.rows();
31  long n_cond = 0;
32  for (long i = n - 1; i >= 0; --i) {
33  if (s(i) >= threshold) {
34  ++n_cond;
35  } else
36  i = 0; // skip rest since eigenvalues are in ascending order
37  }
38 
39  auto sigma = s.bottomRows(n_cond);
40  auto result_condition_number = sigma.maxCoeff() / sigma.minCoeff();
41  auto sigma_sqrt = sigma.array().sqrt().matrix().asDiagonal();
42  auto sigma_invsqrt = sigma.array().sqrt().inverse().matrix().asDiagonal();
43 
44  // make canonical X/Xinv
45  auto U_cond = U.block(0, n - n_cond, n, n_cond);
46  Matrix X = U_cond * sigma_invsqrt;
47  Matrix Xinv = U_cond * sigma_sqrt;
48  // convert to symmetric, if needed
49  if (symmetric) {
50  X = X * U_cond.transpose();
51  Xinv = Xinv * U_cond.transpose();
52  }
53  return std::make_tuple(X, Xinv, size_t(n_cond), condition_number,
54  result_condition_number);
55 }

78  ExEnv::out0() << "\n\t Using Symmetric Orthogonalization. \n"
