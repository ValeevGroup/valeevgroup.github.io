87  template <typename Index>
88  void compute_range_data(const size_type n, const Index& start, const Index& finish) {
89  // Set the volume seed
90  volume_ = 1ul;
91 
92  // Compute range data
93  for(int i = n - 1; i >= 0; --i) {
94  TA_ASSERT(start[i] < finish[i]);
95  start_[i] = start[i];
96  finish_[i] = finish[i];
97  size_[i] = finish[i] - start[i];
98  weight_[i] = volume_;
99  volume_ *= size_[i];
100  }

205  start_(), finish_(), size_(), weight_(), volume_(0ul)
206  {

223  for(size_type i = 0ul; i < n; ++i) {
224  const size_type pi = perm[i];
225  start[pi] = other_start[i];

526  index idx(size_type index) const {
527  // Check that index is contained by range.
528  TA_ASSERT(includes(index));
529 
530  const unsigned int end = dim();
531 
532  // Construct result coordinate index object and allocate its memory.
533  Range_::index result(end, 0);
534 
535  // Get pointers to the data
536  size_type * restrict const result_data = & result.front();
537  size_type const * restrict const weight = weight_.data();
538  size_type const * restrict const start = start_.data();
539 
540  // Compute the coordinate index of o in range.
541  for(unsigned int i = 0u; i < end; ++i) {
542  const size_type weight_i = weight[i];
543  const size_type start_i = start[i];
544 
545  // Compute result index element i
546  const size_type result_i = (index / weight_i) + start_i;
547  index %= weight_i;
548 
549  // Store result
550  result_data[i] = result_i;
551  }

553  return result;
554  }
