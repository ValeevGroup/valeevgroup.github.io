29 /*
30  * Input is Z and some maps and output should be a std::pair of
31  * RowMatrixXds, the first being the Domained M and the second being the
32  * Domained Z where Z is (X_{IJ} | (I,J) A_{IJ}) and M is (X_{IJ} | Y_{IJ})
33  */
34 template <typename Tile, typename Policy>
35 auto domained_MZ(TA::DistArray<Tile, Policy> const& Z,
36  TA::DistArray<Tile, Policy> const& M, SparseMap const& IJX,
37  SparseMap const& IJA, int I, int J,
38  IdxMap<3, madness::Future<Tile>>& local_Ztile_cache,
39  int64_t& MZ_elements, bool serial) {
40  auto& world = Z.world();
41  const auto nprocs = world.size();
42 
43  auto const& Ztrange = Z.trange();
44 
45  // Get the TiledRange1 objects
46  auto const& X_tr1 = Ztrange.data()[0];
47  auto const& I_tr1 = Ztrange.data()[1];
48  auto const& A_tr1 = Ztrange.data()[2];
49 
50  const auto X_tiles = IJX.cols();
51  const auto A_tiles = IJA.cols();
52 
53  // Determine the number of elements in the combined I and J blocks
54  auto const& Irange = *(I_tr1.begin() + I);
55  const auto Isize = Irange.second - Irange.first;
56 
57  auto const& Jrange = *(I_tr1.begin() + J);
58  const auto Jsize = Jrange.second - Jrange.first;
59 
60  // Element offset of J or first element of J in our matrix
61  const auto Jstart = Isize;
62 
63  // Total size of IJ
64  const auto IJsize = (I == J) ? Isize : Isize + Jsize;
65 
66  // Figure out how many auxiliary elements in our outputs
67  auto naux_els = 0;
68  std::vector<std::pair<int, int>> aux_tile_to_first_el;
69  for (auto X = 0; X < X_tiles; ++X) {
70  if (IJX(X) > 0) {
71  // Get the element range for X
72  const auto size_pair = *(X_tr1.begin() + X);
73  aux_tile_to_first_el.emplace_back(std::make_pair(X, naux_els));
74  naux_els += size_pair.second - size_pair.first;
75  }
76  }
77 
78  auto npao_els = 0;
79  std::vector<std::pair<int, int>> pao_tile_to_first_el;
80  for (auto A = 0; A < A_tiles; ++A) {
81  if (IJA(A) > 0) {
82  // Get the element range for A
83  const auto size_pair = *(A_tr1.begin() + A);
84  pao_tile_to_first_el.emplace_back(std::make_pair(A, npao_els));
85  npao_els += size_pair.second - size_pair.first;
86  }
87  }
88 
89  RowMatrixXd Mldf(naux_els, naux_els);
90  // Mldf.setZero();
91  const auto msize = naux_els * naux_els;
92  std::fill(Mldf.data(), Mldf.data() + msize, 0.0);
93  RowMatrixXd Zldf(naux_els, IJsize * npao_els);
94  const auto Zsize = naux_els * IJsize * npao_els;
95  std::fill(Zldf.data(), Zldf.data() + Zsize, 0.0);
96  // Zldf.setZero();
97 
98  std::atomic<std::int64_t> counter(0);
99  std::atomic<std::int64_t> MZ_written(0);
100 
101  auto write_m_tile = [&Mldf, &counter, &MZ_written](
102  Tile const& t, int x_first_el, int y_first_el) {
103  auto const& range = t.range();
104  const auto ext = range.extent_data();
105  const auto x_ext = ext[0];
106  const auto y_ext = ext[1];
107 
108  auto tptr = t.data();
109 
110  for (auto x = 0; x < x_ext; ++x) {
111  double* M_ptr = &Mldf(x + x_first_el, y_first_el);
112  std::copy(tptr, tptr + y_ext, M_ptr);
113  tptr += y_ext;
114  }
115 
116  MZ_written += ext[0] * ext[1];
117  ++counter;
118  };
119 
120  auto write_Z_tile = [&Zldf, npao_els, &counter, &MZ_written](
121  Tile const& t, int x_first_el, int i_first_el,
122  int a_first_el) {
123  auto const& range = t.range();
124  const auto ext = range.extent_data();
125 
126  const auto x_ext = ext[0];
127  const auto i_ext = ext[1];
128  const auto a_ext = ext[2];
129 
130  auto tptr = t.data();
131  for (auto x = 0; x < x_ext; ++x) {
132  const auto Zx = x + x_first_el;
133 
134  for (auto i = 0; i < i_ext; ++i) {
135  const auto Zi = i + i_first_el;
136  const auto Zia_first = Zi * npao_els + a_first_el;
137 
138  auto Zptr = &Zldf(Zx, Zia_first);
139  std::copy(tptr, tptr + a_ext, Zptr);
140  tptr += a_ext;
141  }
142  }
143  MZ_written += x_ext * i_ext * a_ext;
144  ++counter;
145  };
146 
147  if (nprocs > 1) {
148  int64_t ntasks = 0;
149  for (auto const& aux_info : aux_tile_to_first_el) {
150  const auto X = aux_info.first;
151  const auto x_start_el = aux_info.second;
152 
153  // loop to write M tiles
154  for (auto const& aux_info_y : aux_tile_to_first_el) {
155  const auto Y = aux_info_y.first;
156  if (Y > X) break; // Only include lower T
157  const auto y_start_el = aux_info_y.second;
158 
159  if (M.is_zero({X, Y})) {
160  continue;
161  }
162  auto MXY = M.find({X, Y});
163 
164  world.taskq.add(write_m_tile, MXY, x_start_el, y_start_el);
165  ++ntasks;
166  }
167 
168  // loop to write Z tiles
169  for (auto const& pao_info : pao_tile_to_first_el) {
170  const auto A = pao_info.first;
171  const auto a_start_el = pao_info.second;
172 
173  // Do I first
174  if (!Z.is_zero({X, I, A})) {
175  const auto i_start_el = 0;
176 
177  if (!Z.is_local({X, I, A})) { // Don't insert local tiles
178  // Check if we have already cached a future to this tile locally
179  auto ZXIA_iter = local_Ztile_cache.find({{X, I, A}});
180  if (ZXIA_iter == local_Ztile_cache.end()) {
181  // The future to the tile
182  auto ZXIA = Z.find({X, I, A});
183 
184  // The INDEX of the TILE
185  std::array<int, 3> XIA{{X, I, A}};
186 
187  // unordered_map insert returns a pair<iter, bool> and we want the
188  // iter to the inserted element
189  ZXIA_iter =
190  local_Ztile_cache.insert(std::make_pair(XIA, ZXIA)).first;
191  }
192 
193  // unordered_map insert does not invalidate references to object
194  // held in the map only iterators, ZXIA_iter.second should be a
195  // future and since we only call inserts on the main thread the
196  // iterator can't be invalidated before we grab the future out of
197  // it.
198  world.taskq.add(write_Z_tile, ZXIA_iter->second, x_start_el,
199  i_start_el, a_start_el);
200  ++ntasks;
201  } else { // If local just use the Future from TA
202  world.taskq.add(write_Z_tile, Z.find({X, I, A}).get(), x_start_el,
203  i_start_el, a_start_el);
204  ++ntasks;
205  }
206  }
207 
208  if (I != J && !Z.is_zero({X, J, A})) {
209  const auto j_start_el = Jstart;
210 
211  if (!Z.is_local({X, J, A})) {
212  // Repeat the tile look up from above
213  auto ZXJA_iter = local_Ztile_cache.find({{X, J, A}});
214  if (ZXJA_iter == local_Ztile_cache.end()) {
215  auto ZXJA = Z.find({X, J, A});
216  std::array<int, 3> XJA{{X, J, A}};
217  ZXJA_iter =
218  local_Ztile_cache.insert(std::make_pair(XJA, ZXJA)).first;
219  }
220 
221  world.taskq.add(write_Z_tile, ZXJA_iter->second, x_start_el,
222  j_start_el, a_start_el);
223  ++ntasks;
224  } else { // Don't cache local tiles
225  world.taskq.add(write_Z_tile, Z.find({X, J, A}).get(), x_start_el,
226  j_start_el, a_start_el);
227  ++ntasks;
228  }
229  }
230  }
231  }
232 
233  world.await([&counter, ntasks] { return counter == ntasks; });
234  } // multi node
235  else { // Single node
236  auto inner_loop_task = [&](std::pair<int, int> const& aux_info) {
237  // for (auto const& aux_info : aux_tile_to_first_el) {
238  const auto X = aux_info.first;
239  const auto x_start_el = aux_info.second;
240 
241  // loop to write M tiles
242  for (auto const& aux_info_y : aux_tile_to_first_el) {
243  const auto Y = aux_info_y.first;
244  if (Y > X) break; // Only include lower T
245  const auto y_start_el = aux_info_y.second;
246 
247  if (M.is_zero({X, Y})) {
248  continue;
249  }
250 
251  write_m_tile(M.find({X, Y}), x_start_el, y_start_el);
252  }
253 
254  // loop to write Z tiles
255  for (auto const& pao_info : pao_tile_to_first_el) {
256  const auto A = pao_info.first;
257  const auto a_start_el = pao_info.second;
258 
259  // Do I first
260  if (!Z.is_zero({X, I, A})) {
261  const auto i_start_el = 0;
262  write_Z_tile(Z.find({X, I, A}), x_start_el, i_start_el, a_start_el);
263  }
264 
265  if (I != J && !Z.is_zero({X, J, A})) {
266  const auto j_start_el = Jstart;
267 
268  write_Z_tile(Z.find({X, J, A}), x_start_el, j_start_el, a_start_el);
269  }
270  }
271  };
272 
273  auto aux_start = aux_tile_to_first_el.begin();
274  auto aux_end = aux_tile_to_first_el.end();
275  if (!serial) {
276  tbb::parallel_for_each(aux_start, aux_end, inner_loop_task);
277  } else {
278  std::for_each(aux_start, aux_end, inner_loop_task);
279  }
280  }
281 
282  MZ_elements += MZ_written; // Add from our atomic
283  return std::make_pair<RowMatrixXd, RowMatrixXd>(std::move(Mldf),
284  std::move(Zldf));
285 }

287 template <typename Tile, typename Policy>
288 auto domained_FS(TA::DistArray<Tile, Policy> const& Fpao,
289  TA::DistArray<Tile, Policy> const& Spao,
290  SparseMap const& IJA) {
291  auto& world = Fpao.world();
292  const auto nprocs = world.size();
293  auto const& Ftrange = Fpao.trange();
294 
295  // Get the TiledRange 1 objects
296  auto const& A_tr1 = Ftrange.data()[1];
297  const auto A_tiles = IJA.cols();
298 
299  auto npao_els = 0;
300  std::vector<std::pair<int, int>> pao_tile_to_first_el;
301  for (auto A = 0; A < A_tiles; ++A) {
302  if (IJA(A) > 0) {
303  // Get the element range for A
304  const auto size_pair = *(A_tr1.begin() + A);
305  pao_tile_to_first_el.emplace_back(std::make_pair(A, npao_els));
306  npao_els += size_pair.second - size_pair.first;
307  }
308  }
309 
310  RowMatrixXd F(npao_els, npao_els);
311  std::fill(F.data(), F.data() + F.size(), 0.0);
312  RowMatrixXd S(npao_els, npao_els);
313  std::fill(S.data(), S.data() + S.size(), 0.0);
314 
315  std::atomic<std::int64_t> counter(0);
316  auto write_F_tile = [&F, &counter](Tile const& t, int a_first_el,
317  int b_first_el) {
318  auto const& range = t.range();
319  const auto ext = range.extent_data();
320  const auto a_ext = ext[0];
321  const auto b_ext = ext[1];
322 
323  auto tptr = t.data();
324 
325  for (auto a = 0; a < a_ext; ++a) {
326  double* F_ptr = &F(a + a_first_el, b_first_el);
327  std::copy(tptr, tptr + b_ext, F_ptr);
328  tptr += b_ext;
329  }
330  ++counter;
331  };
332 
333  auto write_S_tile = [&S, &counter](Tile const& t, int a_first_el,
334  int b_first_el) {
335  auto const& range = t.range();
336  const auto ext = range.extent_data();
337  const auto a_ext = ext[0];
338  const auto b_ext = ext[1];
339 
340  auto tptr = t.data();
341 
342  for (auto a = 0; a < a_ext; ++a) {
343  double* S_ptr = &S(a + a_first_el, b_first_el);
344  std::copy(tptr, tptr + b_ext, S_ptr);
345  tptr += b_ext;
346  }
347  ++counter;
348  };
349 
350  if (nprocs > 1) {
351  int64_t ntasks = 0;
352  for (auto const& pao_info : pao_tile_to_first_el) {
353  const auto A = pao_info.first;
354  const auto a_start_el = pao_info.second;
355 
356  for (auto const& pao_info_b : pao_tile_to_first_el) {
357  const auto B = pao_info_b.first;
358  const auto b_start_el = pao_info_b.second;
359 
360  if (!Fpao.is_zero({A, B})) {
361  auto F_fut = Fpao.find({A, B});
362  world.taskq.add(write_F_tile, F_fut.get(), a_start_el, b_start_el);
363  ++ntasks;
364  }
365 
366  if (B > A) { // We only need the lower triangle of S
367  continue;
368  }
369  if (!Spao.is_zero({A, B})) {
370  auto S_fut = Spao.find({A, B});
371  world.taskq.add(write_S_tile, S_fut.get(), a_start_el, b_start_el);
372  ++ntasks;
373  }
374  }
375  }
376 
377  world.await([&counter, ntasks] { return counter == ntasks; });
378  } else { // nprocs == 1
379  auto start = pao_tile_to_first_el.begin();
380  auto end = pao_tile_to_first_el.end();
381  tbb::parallel_for_each(
382  start, end, [&](std::pair<int, int> const& pao_info) {
383  const auto A = pao_info.first;
384  const auto a_start_el = pao_info.second;
385 
386  for (auto const& pao_info_b : pao_tile_to_first_el) {
387  const auto B = pao_info_b.first;
388  const auto b_start_el = pao_info_b.second;
389 
390  if (!Fpao.is_zero({A, B})) {
391  auto F_fut = Fpao.find({A, B});
392  write_F_tile(F_fut.get(), a_start_el, b_start_el);
393  }
394 
395  if (B > A) { // We only need the lower triangle of S
396  continue;
397  }
398 
399  if (!Spao.is_zero({A, B})) {
400  auto S_fut = Spao.find({A, B});
401  write_S_tile(S_fut.get(), a_start_el, b_start_el);
402  }
403  }
404  });
405  }
406 
407  return std::make_pair<RowMatrixXd, RowMatrixXd>(std::move(F), std::move(S));
408 }

410 template <typename Tile, typename Policy>
411 auto Spao_for_overlaps(TA::DistArray<Tile, Policy> const& Spao,
412  SparseMap const& IJA, SparseMap const& KJA) {
413  auto& world = Spao.world();
414  auto const& Strange = Spao.trange();
415 
416  // Get the TiledRange 1 objects
417  auto const& A_tr1 = Strange.data()[1];
418  const auto A_tiles = IJA.cols();
419 
420  auto npao_els_IJ = 0;
421  std::vector<std::pair<int, int>> pao_tile_to_first_el_IJ;
422  for (auto A = 0; A < A_tiles; ++A) {
423  if (IJA(A) > 0) {
424  // Get the element range for A
425  const auto size_pair = *(A_tr1.begin() + A);
426  pao_tile_to_first_el_IJ.emplace_back(std::make_pair(A, npao_els_IJ));
427  npao_els_IJ += size_pair.second - size_pair.first;
428  }
429  }
430 
431  auto npao_els_KJ = 0;
432  std::vector<std::pair<int, int>> pao_tile_to_first_el_KJ;
433  for (auto A = 0; A < A_tiles; ++A) {
434  if (KJA(A) > 0) {
435  // Get the element range for A
436  const auto size_pair = *(A_tr1.begin() + A);
437  pao_tile_to_first_el_KJ.emplace_back(std::make_pair(A, npao_els_KJ));
438  npao_els_KJ += size_pair.second - size_pair.first;
439  }
440  }
441 
442  RowMatrixXd S = RowMatrixXd::Zero(npao_els_IJ, npao_els_KJ);
443 
444  std::atomic<std::int64_t> counter(0);
445 
446  auto write_S_tile = [&S, &counter](Tile const& t, int a_first_el,
447  int b_first_el) {
448  auto const& range = t.range();
449  const auto ext = range.extent_data();
450  auto t_map = TA::eigen_map(t, ext[0], ext[1]);
451  auto Sb = S.block(a_first_el, b_first_el, ext[0], ext[1]);
452  Sb = t_map;
453  ++counter;
454  };
455 
456  int64_t ntasks = 0;
457  for (auto const& pao_info : pao_tile_to_first_el_IJ) {
458  const auto A = pao_info.first;
459  const auto a_start_el = pao_info.second;
460 
461  for (auto const& pao_info_b : pao_tile_to_first_el_KJ) {
462  const auto B = pao_info_b.first;
463  const auto b_start_el = pao_info_b.second;
464 
465  if (!Spao.is_zero({A, B})) {
466  auto S_fut = Spao.find({A, B});
467  world.taskq.add(write_S_tile, S_fut, a_start_el, b_start_el);
468  ++ntasks;
469  }
470  }
471  }
472 
473  if (ntasks != 0) {
474  world.await([&counter, ntasks] { return counter == ntasks; });
475  }
476 
477  return S;
478 }

480 inline SparseMap IJK_pao_unions(PnoPair const& IJ,
481  IdxMap<2, PnoPair> const& pairs, int I, int J,
482  int nI, RowMatrixXd const& Fij_norms,
483  double tcutFIJ) {
484  SparseMap union_paos = IJ.pao_domains;
485 
486  // Loop over K > 1 since theose are the only overlaps we will compute
487  for (auto K = I + 1; K < nI; ++K) {
488  if (Fij_norms(I, K) < tcutFIJ) continue;
489 
490  auto KJ_pair = pairs.find({{K, J}});
491  if (KJ_pair == pairs.end()) {
492  continue;
493  }
494 
495  union_paos += KJ_pair->second.pao_domains;
496  }
497 
498  for (auto r = 0; r < union_paos.rows(); ++r) {
499  for (auto c = 0; c < union_paos.cols(); ++c) {
500  const auto val = union_paos(r, c);
501  union_paos(r, c) = std::min(val, 1);
502  }
503  }
504 
505  return union_paos;
506 }

508 // Do the first multiplication of U_IJ^T * Spao so that we don't have to do
509 // redundant work
510 template <typename Tile, typename Policy>
511 std::pair<RowMatrixXd, std::vector<std::pair<int, int>>> pre_multiply_IJ(
512  TA::DistArray<Tile, Policy> const& Spao, SparseMap const& IJK_pao_unions,
513  PnoPair const* IJ_pair) {
514  RowMatrixXd const& U_IJ = IJ_pair->U;
515  SparseMap const& IJ_pao_domains = IJ_pair->pao_domains;
516 
517  const auto Stile_extent = Spao.trange().tiles_range().extent_data();
518 
519  const auto nL = Stile_extent[0];
520  const auto nN = Stile_extent[1];
521 
522  // Figure out domains of U_IJ that we need
523  auto L_tr1 = Spao.trange().data()[0];
524  std::vector<std::pair<int, int>> size_of_sig_L;
525  for (auto L = 0; L < nL; ++L) {
526  if (IJ_pao_domains(L) != 0) {
527  const auto Linfo = *(L_tr1.begin() + L);
528  const auto Lsize = Linfo.second - Linfo.first;
529  size_of_sig_L.emplace_back(std::make_pair(L, Lsize));
530  }
531  }
532 
533  // Figure out how many elements needed for the columns of our out matrix
534  auto N_tr1 = Spao.trange().data()[1];
535  auto ncols = 0;
536  std::vector<std::pair<int, int>> sig_N_start_col;
537  for (auto N = 0; N < nN; ++N) {
538  if (IJK_pao_unions(N) != 0) {
539  const auto Ninfo = *(N_tr1.begin() + N);
540  const auto Nsize = Ninfo.second - Ninfo.first;
541  sig_N_start_col.emplace_back(std::make_pair(N, ncols));
542  ncols += Nsize;
543  }
544  }
545 
546  RowMatrixXd out(U_IJ.cols(), ncols);
547  out.setZero();
548 
549  //
550  // Do Gemm out(M,N) = \sum_{L} U_{L,M} * S_{L,N}
551  // But the block of M is always just 1 since the PNO space is blocked by 1
552  //
553 
554  auto L_row_start = 0;
555  const auto npnos = U_IJ.cols();
556  // There would be a loop over PNO blocks but we only have 1
557  for (auto const& Lpair : size_of_sig_L) {
558  const auto L = Lpair.first;
559  const auto Lext = Lpair.second;
560 
561  // U_IJ for domain L
562  const auto L_block_of_U = U_IJ.block(L_row_start, 0, Lext, npnos);
563 
564  for (auto const& tile_start_pair : sig_N_start_col) {
565  const auto N = tile_start_pair.first;
566  const auto out_col_start = tile_start_pair.second;
567 
568  if (Spao.is_zero({L, N})) {
569  continue;
570  }
571 
572  // Safe since Spao replicated
573  auto const& tile = Spao.find({L, N}).get();
574  auto const& range = tile.range();
575  const auto ext = range.extent_data();
576  const auto tmap = TA::eigen_map(tile, ext[0], ext[1]);
577 
578  // Do gemm into the correct location in out
579  // The out block is S(pno, N) so we write to the Nth domain
580  auto out_block = out.block(0, out_col_start, npnos, ext[1]);
581 
582  // Must be += since multiple L sum into each out_block
583  out_block += L_block_of_U.transpose() * tmap;
584  }
585 
586  L_row_start += Lext;

592 RowMatrixXd finish_S(
593  std::pair<RowMatrixXd, std::vector<std::pair<int, int>>> const&
594  tmpS_and_N_starts,
595  PnoPair const& KJ_pno) {
596  RowMatrixXd const& tmpS = tmpS_and_N_starts.first;
597  auto N_col_starts = tmpS_and_N_starts.second;
598 
599  SparseMap const& KJ_pao_domains = KJ_pno.pao_domains;
600  RowMatrixXd const& U_KJ = KJ_pno.U;
601 
602  const auto nIJ_pnos = tmpS.rows();
603 
604  // First we need to figure out the number of columns in out output mat
605  // N_col_starts should be guareenteed to already have every N needed from
606  // KJ_pao_domains since it was built from a union of all K for this I
607  std::vector<std::pair<int, int>> N_start_size; // Don't need global N
608  const auto nM = N_col_starts.size();
609  for (auto M = 0; M < nM; ++M) {
610  auto const& N_and_start = N_col_starts[M];
611  const auto N = N_and_start.first;
612 
613  if (KJ_pao_domains(N) == 0) {
614  continue;
615  }
616 
617  const auto starting_col = N_and_start.second;
618 
619  // Figure out the (last col + 1) in N
620  auto last_col_plus1 = 0;
621  if (M < (nM - 1)) {
622  last_col_plus1 = N_col_starts[M + 1].second;
623  } else {
624  last_col_plus1 = tmpS.cols();
625  }
626 
627  // Compute the size of this N
628  const auto Nsize = last_col_plus1 - starting_col;
629 
630  // Save our N, our starting col and our size
631  N_start_size.emplace_back(std::make_pair(starting_col, Nsize));
632  }
633 
634  const auto nKJ_pnos = U_KJ.cols();
635  RowMatrixXd out(nIJ_pnos, nKJ_pnos);
636  out.setZero();
637 
638  // S_IJ_KJ(M, L) = \sum_{N} tmpS(M, N) * U_KJ(N, L)
639  // Only 1 loop here since M and L are single blocks
640  auto U_start_row = 0;
641  for (auto const& start_size : N_start_size) {
642  const auto S_col_start = start_size.first;
643  const auto domain_size = start_size.second;
644 
645  const auto U_KJ_block = U_KJ.block(U_start_row, 0, domain_size, nKJ_pnos);
646  const auto S_block = tmpS.block(0, S_col_start, nIJ_pnos, domain_size);
647 
648  out += S_block * U_KJ_block;
649 
650  // Jump to the next domain block of U_KJ
651  U_start_row += domain_size;
652  }
653 
654  return out;
655 }

657 template <typename Tile, typename Policy>
658 IdxMap<3, RowMatrixXd> pno_overlaps(IdxMap<2, PnoPair>& IJ_pairs,
659  TA::DistArray<Tile, Policy>& Spao,
660  TA::DistArray<Tile, Policy> const& Fij,
661  double tcutFIJ,
662  double flattened_pnos = false) {
663  // First get a world and ensure Spao is replicated
664  auto& world = Spao.world();
665  Spao.make_replicated();
666 
667  // Prepare for what we need to do if we previously flattened our pnos
668  RowMatrixXd Fij_norms_by_el;
669  const auto ni_els = Fij.trange().elements_range().extent_data()[0];
670  if (flattened_pnos) {
671  Fij_norms_by_el.resize(ni_els, ni_els);
672  Fij_norms_by_el.setZero();
673  }
674 
675  // Next get the FIJ block norms
676  // Assume that I_tr1 == J_tr1
677  TA::Tensor<float> Fij_norms_ta = Fij.shape().data().clone();
678  const auto i_extent = Fij_norms_ta.range().extent_data();
679  auto I_tr1 = Fij.trange().data()[0];
680 
681  const auto nI = i_extent[0];
682  const auto nJ = i_extent[1];
683  for (auto I = 0; I < nI; ++I) {
684  auto I_info = *(I_tr1.begin() + I);
685  const auto I_start = I_info.first;
686  const auto I_size = I_info.second - I_info.first;
687  for (auto J = 0; J < nJ; ++J) {
688  auto J_info = *(I_tr1.begin() + J);
689  const auto J_start = J_info.first;
690  const auto J_size = J_info.second - J_info.first;
691  Fij_norms_ta(I, J) *= I_size * J_size;
692 
693  if (flattened_pnos) { // Write the whole block to the constant
694  Fij_norms_by_el.block(I_start, J_start, I_size, J_size) =
695  RowMatrixXd::Constant(I_size, J_size, Fij_norms_ta(I, J));
696  }
697  }
698  }
699 
700  RowMatrixXd Fij_norms;
701  if (flattened_pnos) {
702  Fij_norms = Fij_norms_by_el;
703  } else {
704  Fij_norms =
705  TA::eigen_map(Fij_norms_ta, i_extent[0], i_extent[1]).cast<double>();
706  }
707 
708  // Declare our map
709  IdxMap<3, RowMatrixXd> S_IJ_KJs;
710 
711  madness::Mutex mut(0);
712 
713  // Now that we have our norms and Spao is replicated we can loop over pairs
714  // and build their overlaps very carefully. Note Spao must be replicated so
715  // that we can call find in our tasks
716  std::atomic<std::int64_t> counter(0);
717  int64_t ntasks = 0;
718  const auto nK = (flattened_pnos) ? ni_els : nI;
719  for (auto& IJ_pno_pair : IJ_pairs) {
720  auto compute_overlaps_task = [&mut, &S_IJ_KJs, &Fij_norms, &IJ_pairs,
721  &counter, &Spao, tcutFIJ,
722  nK](int I, int J, PnoPair* IJ_pno_ptr) {
723 
724  auto& IJ_pno = *IJ_pno_ptr;
725  auto IJK_union =
726  IJK_pao_unions(IJ_pno, IJ_pairs, I, J, nK, Fij_norms, tcutFIJ);
727  auto S_and_N_start_cols = pre_multiply_IJ(Spao, IJK_union, &IJ_pno);
728 
729  for (auto K = 0; K < nK; ++K) {
730  if (Fij_norms(I, K) < tcutFIJ) continue;
731 
732  const auto KJ_pno = IJ_pairs.find({{K, J}});
733  if (KJ_pno == IJ_pairs.end()) {
734  continue;
735  }
736 
737  IJ_pno.sig_ks.emplace_back(K);
738 
739  if (K > I) {
740  RowMatrixXd Spno = finish_S(S_and_N_start_cols, KJ_pno->second);
741  std::array<int, 3> IJK{{I, J, K}};
742  {
743  madness::ScopedMutex<madness::Mutex> smut(&mut);
744  S_IJ_KJs.insert({IJK, std::move(Spno)});
745  }
746  }
747  }
748  ++counter;
749  };
750 
751  const auto I = IJ_pno_pair.first[0];
752  const auto J = IJ_pno_pair.first[1];
753  auto* IJ_pno_ptr = &IJ_pno_pair.second;
754 
755  world.taskq.add(compute_overlaps_task, I, J, IJ_pno_ptr);
756  ++ntasks;
757  }
758 
759  world.await([&counter, ntasks] { return counter == ntasks; });
760 
761  return S_IJ_KJs;
762 }
