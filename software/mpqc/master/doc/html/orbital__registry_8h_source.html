73  const auto& uocc_space = this->retrieve("a");
74 
75  // can't always deduce user's occ and unocc blk sizes, just grab dimensions of the first tile in each
76  const auto& occ_tile0 = occ_space.trange().tile(0);
77  auto occ_blksize = occ_tile0.second - occ_tile0.first;
78  const auto& uocc_tile0 = uocc_space.trange().tile(0);
79  auto uocc_blksize = uocc_tile0.second - uocc_tile0.first;
80 
81  trange1_engine_ = std::make_shared<const TRange1Engine>(
82  ndocc, nobs, occ_blksize, uocc_blksize, n_frozen_core);
83  } catch (ProgrammingError&) {
84  throw ProgrammingError(
85  "OrbitalSpaceRegistry::trange1_engine failed -- likely some "

88  }
