195  template <typename T>
196  inline typename T::size_type inner_size_helper(const T& tensor) {
197  const auto* restrict const stride = tensor.range().stride_data();
198  const auto* restrict const size = tensor.range().extent_data();
199 
200  int i = int(tensor.range().rank()) - 1;
201  auto volume = size[i];
202 
203  for(--i; i >= 0; --i) {
204  const auto stride_i = stride[i];
205  const auto size_i = size[i];
206 
207  if(volume != stride_i)
208  break;
209  volume *= size_i;
210  }
211 
212  return volume;
213  }

237 
238  for(--i; i >= 0; --i) {
239  const auto stride1_i = stride1[i];
240  const auto stride2_i = stride2[i];

242  const auto size2_i = size2[i];
243 
244  if((volume1 != stride1_i) || (volume2 != stride2_i))
245  break;
246  volume1 *= size1_i;
247  volume2 *= size2_i;
