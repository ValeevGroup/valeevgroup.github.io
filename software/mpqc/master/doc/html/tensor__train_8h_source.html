13 RowMatrixXd one_over_delta(Eigen::VectorXd const &e_vals, int64_t occ) {
14  const auto size = e_vals.size();
15  const auto vir = size - occ;
16  Eigen::MatrixXd delta(occ * vir, occ * vir);
17  delta.setZero();
18 
19  for (auto i = 0; i < occ; ++i) {
20  const auto iv = -e_vals[i];
21 
22  for (auto a = 0; a < vir; ++a) {
23  const auto ia = i * vir + a;
24  const auto av = e_vals[a + occ];
25 
26  for (auto j = 0; j < occ; ++j) {
27  const auto jv = -e_vals[j];
28 
29  for (auto b = 0; b < vir; ++b) {
30  const auto jb = j * vir + b;
31 
32  const auto val = iv + av + jv + e_vals[b + occ];
33  delta(ia, jb) = 1.0 / val;
34  }
35  }
36  }
37  }
38 
39  return delta;
40 }
