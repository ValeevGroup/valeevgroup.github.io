75  template <typename Index>
76  void init_range_data(const Index& lower_bound, const Index& upper_bound) {
77  // Construct temp pointers
78  size_type* restrict const lower = data_;
79  size_type* restrict const upper = lower + rank_;
80  size_type* restrict const extent = upper + rank_;
81  size_type* restrict const stride = extent + rank_;
82  const auto* restrict const lower_data = detail::data(lower_bound);
83  const auto* restrict const upper_data = detail::data(upper_bound);
84 
85  // Set the volume seed
86  volume_ = 1ul;
87  offset_ = 0ul;
88 
89  // Compute range data
90  for(int i = int(rank_) - 1; i >= 0; --i) {
91  // Check input dimensions
92  TA_ASSERT(lower_data[i] >= 0ul);
93  TA_ASSERT(lower_data[i] < upper_data[i]);
94 
95  // Compute data for element i of lower, upper, and extent
96  const size_type lower_bound_i = lower_data[i];
97  const size_type upper_bound_i = upper_data[i];
98  const size_type extent_i = upper_bound_i - lower_bound_i;
99 
100  lower[i] = lower_bound_i;
101  upper[i] = upper_bound_i;
102  extent[i] = extent_i;
103  stride[i] = volume_;
104  offset_ += lower_bound_i * volume_;
105  volume_ *= extent_i;
106  }
107  }

155  void init_range_data(const Permutation& perm,
156  const size_type* restrict const other_lower_bound,
157  const size_type* restrict const other_upper_bound)
158  {
159  // Create temporary pointers to this range data
160  size_type* restrict const lower = data_;
161  size_type* restrict const upper = lower + rank_;
162  size_type* restrict const extent = upper + rank_;
163  size_type* restrict const stride = extent + rank_;
164 
165  // Copy the permuted lower, upper, and extent into this range.
166  for(unsigned int i = 0u; i < rank_; ++i) {
167  const size_type perm_i = perm[i];
168 
169  // Get the lower bound, upper bound, and extent from other for rank i.
170  const size_type other_lower_bound_i = other_lower_bound[i];
171  const size_type other_upper_bound_i = other_upper_bound[i];
172  const size_type other_extent_i = other_upper_bound_i - other_lower_bound_i;
173 
174  // Store the permuted lower bound, upper bound, and extent
175  lower[perm_i] = other_lower_bound_i;
176  upper[perm_i] = other_upper_bound_i;
177  extent[perm_i] = other_extent_i;
178  }
179 
180  // Recompute stride, offset, and volume
181  volume_ = 1ul;
182  offset_ = 0ul;
183  for(int i = int(rank_) - 1; i >= 0; --i) {
184  const size_type lower_i = lower[i];
185  const size_type extent_i = extent[i];
186  stride[i] = volume_;
187  offset_ += lower_i * volume_;
188  volume_ *= extent_i;
189  }
