67 template <typename T>
68 T norm(DecomposedTensor<T> const &t) {
69  if (t.empty()) {
70  return T(0.0);
71  }
72  // Norm is bounded by ||M|| <= ||S^M|| * ||T^M||
73  // if single tile then norm is exact
74  auto norm_bound = 1.0;
75  for (auto const &tensor : t.tensors()) {
76  norm_bound *= tensor.norm();
77  }
78  return norm_bound;
79 }
