168 template <>
169 struct TupleCanonicalizer<PetiteList::Symmetry::aa_aa> {
170  template <typename I>
171  static bool is_canonical(I idx0) {
172  return true;
173  }
174  template <typename I>
175  static bool is_canonical(I idx0, I idx1) {
176  return idx0 >= idx1;
177  }
178  template <typename I>
179  static bool is_canonical(I idx0, I idx1, I idx2) {
180  return is_canonical(idx0, idx1);
181  }
182  template <typename I>
183  static bool is_canonical(I idx0, I idx1, I idx2, I idx3) {
184  if (idx0 < idx1) return false;
185  if (idx2 < idx3) return false;
186  if (idx0 == idx2 && idx1 < idx3) return false;
187  return true;
188  }
189  template <typename I>
190  static int64_t multiplicity(I idx0) {
191  throw std::logic_error(
192  "underfined TupleCanonicalizer<aa_aa>::multiplicity()");
193  }
194  template <typename I>
195  static int64_t multiplicity(I idx0, I idx1) {
196  throw std::logic_error(
197  "underfined TupleCanonicalizer<aa_aa>::multiplicity(,)");
198  }
199  template <typename I>
200  static int64_t multiplicity(I idx0, I idx1, I idx2) {
201  throw std::logic_error(
202  "underfined TupleCanonicalizer<aa_aa>::multiplicity(,,)");
203  }
204  template <typename I>
205  static int64_t multiplicity(I idx0, I idx1, I idx2, I idx3) {
206  return ((idx0 == idx1) ? 1 : 2) * ((idx2 == idx3) ? 1 : 2) *
207  ((idx0 == idx2 && idx1 == idx3) ? 1 : 2);
208  }
