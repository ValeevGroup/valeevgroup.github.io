157  void perm_vars(const VariableList& target_vars) {
158  // Only permute if the arguments can be permuted
159  if((left_op_ == permute_to_no_trans) || (right_op_ == permute_to_no_trans)) {
160 
161  // Compute ranks
162  const unsigned int result_rank = target_vars.dim();
163  const unsigned int inner_rank = (left_.vars().dim() +
164  right_.vars().dim() - result_rank) >> 1;
165  const unsigned int left_outer_rank = left_.vars().dim() - inner_rank;
166 
167  // Check that the left- and right-hand outer variables are correctly
168  // partitioned in the target variable list.
169  bool target_partitioned = true;
170  for(unsigned int i = 0u; i < left_outer_rank; ++i)
171  target_partitioned = target_partitioned &&
172  (find(target_vars, left_vars_[i], 0u, left_outer_rank) < left_outer_rank);
173 
174  // If target is properly partitioned, then arguments can be permuted
175  // to fit the target.
176  if(target_partitioned) {
177 
178  if(left_op_ == permute_to_no_trans) {
179  // Copy left-hand target variables to left and result variable lists.
180  for(unsigned int i = 0u; i < left_outer_rank; ++i) {
181  const std::string& var = target_vars[i];
182  const_cast<std::string&>(left_vars_[i]) = var;
183  const_cast<std::string&>(vars_[i]) = var;
184  }
185 
186  // Permute the left argument with the new variable list.
187  left_.perm_vars(left_vars_);
188  } else {
189  // Copy left-hand outer variables to that of result.
190  for(unsigned int i = 0u; i < left_outer_rank; ++i)
191  const_cast<std::string&>(vars_[i]) = left_vars_[i];
192  }
193 
194  if(right_op_ == permute_to_no_trans) {
195  // Copy right-hand target variables to right and result variable lists.
196  for(unsigned int i = left_outer_rank, j = inner_rank; i < result_rank; ++i, ++j) {
197  const std::string& var = target_vars[i];
198  const_cast<std::string&>(right_vars_[j]) = var;
199  const_cast<std::string&>(vars_[i]) = var;
200  }
201 
202  // Permute the left argument with the new variable list.
203  right_.perm_vars(right_vars_);
204  } else {
205  // Copy right-hand outer variables to that of result.
206  for(unsigned int i = left_outer_rank, j = inner_rank; i < result_rank; ++i, ++j)
207  const_cast<std::string&>(vars_[i]) = right_vars_[j];
208  }
209  }
210  }

218  void init_vars() {
219  const unsigned int left_rank = left_.vars().dim();
220  const unsigned int right_rank = right_.vars().dim();
221 
222  // Get non-const references to the argument variable lists.
223  std::vector<std::string>& left_vars =
224  const_cast<std::vector<std::string>&>(left_vars_.data());
225  left_vars.reserve(left_rank);
226  std::vector<std::string>& right_vars =
227  const_cast<std::vector<std::string>&>(right_vars_.data());
228  right_vars.reserve(right_rank);
229  std::vector<std::string>& result_vars =
230  const_cast<std::vector<std::string>&>(vars_.data());
231  result_vars.reserve(std::max(left_rank, right_rank));
232 
233  // Extract left-most result and inner variables from the left-hand argument.
234  for(unsigned int i = 0ul; i < left_rank; ++i) {
235  const std::string& var = left_.vars()[i];
236  if(find(right_.vars(), var, 0u, right_rank) == right_rank) {
237  // Store outer left variable
238  left_vars.push_back(var);
239  result_vars.push_back(var);
240  } else {
241  // Store inner left variable
242  right_vars.push_back(var);
243  }
244  }
245 
246  // Compute the inner and outer dimension ranks.
247  const unsigned int inner_rank = right_vars.size();
248  const unsigned int left_outer_rank = left_vars.size();
249  const unsigned int right_outer_rank = right_rank - inner_rank;
250  const unsigned int result_rank = left_outer_rank + right_outer_rank;
251 
252  // Resize result variables if necessary.
253  result_vars.reserve(result_rank);
254 
255  // Check for an outer product
256  if(inner_rank == 0u) {
257 
258  // Extract the right most outer variables from right hand argument.
259  for(unsigned int i = 0ul; i < right_rank; ++i) {
260  const std::string& var = right_.vars()[i];
261  right_vars.push_back(var);
262  result_vars.push_back(var);
263  }
264  return; // Quick exit
265  }
266 
267  // Initialize flags that will be used to determine the type of permutation
268  // that will be applied to the arguments (i.e. no permutation, transpose,
269  // or arbitrary permutation).
270  bool inner_vars_ordered = true, left_is_no_trans = true, left_is_trans = true,
271  right_is_no_trans = true, right_is_trans = true;
272 
273  // If the inner variable lists of the arguments are not in the same
274  // order, one of them will need to be permuted. Here, we determine which
275  // argument, left or right, will be permuted if a permutation is
276  // required. The argument with the lowest rank is preferred since it is
277  // likely to have the smaller memory footprint, or the fewest leaves to
278  // minimize the number of permutations in the expression.
279  const bool perm_left = (left_rank < right_rank) || ((left_rank == right_rank)
280  && (left_type::leaves <= right_type::leaves));
281 
282  // Extract variables from the right-hand argument, collect information
283  // about the layout of the variable lists, and ensure the inner variable
284  // lists are in the same order.
285  for(unsigned int i = 0ul; i < right_rank; ++i) {
286  const std::string& var = right_.vars()[i];
287  const unsigned int j = find(left_.vars(), var, 0u, left_rank);
288  if(j == left_rank) {
289  // Store outer right variable
290  right_vars.push_back(var);
291  result_vars.push_back(var);
292  } else {
293  const unsigned int x = left_vars.size() - left_outer_rank;
294 
295  // Collect information about the relative position of variables
296  inner_vars_ordered = inner_vars_ordered && (right_vars[x] == var);
297  left_is_no_trans = left_is_no_trans && (j >= left_outer_rank);
298  left_is_trans = left_is_trans && (j < inner_rank);
299  right_is_no_trans = right_is_no_trans && (i < inner_rank);
300  right_is_trans = right_is_trans && (i >= right_outer_rank);
301 
302  // Store inner right variable
303  if(inner_vars_ordered) {
304  // Left and right inner variable list order is equal.
305  left_vars.push_back(var);
306  } else if(perm_left) {
307  // Permute left so we need to store inner variables according to
308  // the order of the right-hand argument.
309  left_vars.push_back(var);
310  right_vars[x] = var;
311  left_is_no_trans = left_is_trans = false;
312  } else {
313  // Permute right so we need to store inner variables according to
314  // the order of the left-hand argument.
315  left_vars.push_back(right_vars[x]);
316  right_is_no_trans = right_is_trans = false;
317  }
318  }
319  }
320 
321  // Here we set the type of permutation that will be applied to the
322  // argument tensors. If an argument is in matrix form, permutation of
323  // the tiles is disabled.
324  if(left_is_no_trans) {
325  left_op_ = no_trans;
326  left_.permute_tiles(false);
327  } else if(left_is_trans) {
328  left_op_ = trans;
329  left_.permute_tiles(false);
330  } else {
331  left_.perm_vars(left_vars_);
332  }
333  if(right_is_no_trans) {
334  right_op_ = no_trans;
335  right_.permute_tiles(false);
336  } else if(right_is_trans) {
337  right_op_ = trans;
338  right_.permute_tiles(false);
339  } else {
340  right_.perm_vars(right_vars_);
341  }
342 
343  }

350  void init_struct(const VariableList& target_vars) {
351  // Initialize children
352  left_.init_struct(left_vars_);
353  right_.init_struct(right_vars_);
354 
355  // Initialize the tile operation in this function because it is used to
356  // evaluate the tiled range and shape.
357 
358  const madness::cblas::CBLAS_TRANSPOSE left_op =
359  (left_op_ == trans ? madness::cblas::Trans : madness::cblas::NoTrans);
360  const madness::cblas::CBLAS_TRANSPOSE right_op =
361  (right_op_ == trans ? madness::cblas::Trans : madness::cblas::NoTrans);
362 
363 
364  if(target_vars != vars_) {
365  // Initialize permuted structure
366  perm_ = ExprEngine_::make_perm(target_vars);
367  op_ = op_type(left_op, right_op, factor_, vars_.dim(), left_vars_.dim(),
368  right_vars_.dim(), (permute_tiles_ ? perm_ : Permutation()));
369  trange_ = ContEngine_::make_trange(perm_);
370  shape_ = ContEngine_::make_shape(perm_);
371  } else {
372  // Initialize non-permuted structure
373  op_ = op_type(left_op, right_op, factor_, vars_.dim(), left_vars_.dim(),
374  right_vars_.dim());
375  trange_ = ContEngine_::make_trange();
376  shape_ = ContEngine_::make_shape();
377  }
378 
379  if(ExprEngine_::struct_override_ptr_ != nullptr){
380  shape_ = shape_.mask(ExprEngine_::struct_override_ptr_->shape);
381  }

390  void init_distribution(World* world, std::shared_ptr<pmap_interface> pmap) {
391  const unsigned int inner_rank = op_.gemm_helper().num_contract_ranks();
392  const unsigned int left_rank = op_.gemm_helper().left_rank();
393  const unsigned int right_rank = op_.gemm_helper().right_rank();
394  const unsigned int left_outer_rank = left_rank - inner_rank;
395 
396  // Get pointers to the argument sizes
397  const size_type* restrict const left_tiles_size =
398  left_.trange().tiles().extent_data();
399  const size_type* restrict const left_element_size =
400  left_.trange().elements().extent_data();
401  const size_type* restrict const right_tiles_size =
402  right_.trange().tiles().extent_data();
403  const size_type* restrict const right_element_size =
404  right_.trange().elements().extent_data();
405 
406  // Compute the fused sizes of the contraction
407  size_type M = 1ul, m = 1ul, N = 1ul, n = 1ul;
408  unsigned int i = 0u;
409  for(; i < left_outer_rank; ++i) {
410  M *= left_tiles_size[i];
411  m *= left_element_size[i];
412  }
413  for(; i < left_rank; ++i)
414  K_ *= left_tiles_size[i];
415  for(i = inner_rank; i < right_rank; ++i) {
416  N *= right_tiles_size[i];
417  n *= right_element_size[i];
418  }
419 
420  // Construct the process grid.
421  proc_grid_ = TiledArray::detail::ProcGrid(*world, M, N, m, n);
422 
423  // Initialize children
424  left_.init_distribution(world, proc_grid_.make_row_phase_pmap(K_));
425  right_.init_distribution(world, proc_grid_.make_col_phase_pmap(K_));
426 
427  // Initialize the process map in not already defined
428  if(! pmap)
429  pmap = proc_grid_.make_pmap();
430  ExprEngine_::init_distribution(world, pmap);
431  }

437  trange_type make_trange(const Permutation& perm = Permutation()) const {
438  // Compute iteration limits
439  const unsigned int left_rank = op_.gemm_helper().left_rank();
440  const unsigned int right_rank = op_.gemm_helper().right_rank();
441  const unsigned int inner_rank = op_.gemm_helper().num_contract_ranks();
442  const unsigned int left_outer_rank = left_rank - inner_rank;
443 
444  // Construct the trange input and compute the gemm sizes
445  typename trange_type::Ranges ranges(op_.gemm_helper().result_rank());
446  unsigned int i = 0ul;
447  for(unsigned int x = 0ul; x < left_outer_rank; ++x, ++i) {
448  const unsigned int pi = (perm ? perm[i] : i);
449  ranges[pi] = left_.trange().data()[x];
450  }
451  for(unsigned int x = inner_rank; x < right_rank; ++x, ++i) {
452  const unsigned int pi = (perm ? perm[i] : i);
453  ranges[pi] = right_.trange().data()[x];
454  }
455 
456 #ifndef NDEBUG
457 
458  // Get left and right tile extents.
459  const auto* restrict const left_extent =
460  left_.trange().tiles().extent_data();
461  const auto* restrict const right_extent =
462  right_.trange().tiles().extent_data();
463 
464  // Check that the contracted dimensions are coformal (equal).
465  for(unsigned int l = left_outer_rank, r = 0ul; l < left_rank; ++l, ++r) {
466  if(left_.trange().data()[l] != right_.trange().data()[r]) {
467  if(World::get_default().rank() == 0) {
468 
469  if(left_extent[l] == right_extent[r]) {
470  TA_USER_ERROR_MESSAGE( "The tiling of the contracted dimensions " \
471  "of the left- and right-hand arguments are not equal.");
472 
473  } else {
474  TA_USER_ERROR_MESSAGE( "The contracted dimensions of the left- " \
475  "and right-hand arguments are not coformal:" \
476  << "\n left = " << left_.trange() \
477  << "\n right = " << right_.trange() );
478 
479  TA_EXCEPTION("The contracted dimensions of the left- and " \
480  "right-hand expressions are not coformal.");
481  }
482  }
483 
484  TA_EXCEPTION("The contracted dimensions of the left- and "
485  "right-hand expressions are not coformal.");
486  }
487  }
488 #endif // NDEBUG
489 
490  return trange_type(ranges.begin(), ranges.end());
491  }
