153  static void valid_(RandIter first, RandIter last) {
154  // Verify at least 2 elements are present if the vector is not empty.

164  template <typename RandIter>
165  void init_tiles_(RandIter first, RandIter last, size_type start_tile_index) {
166 #ifndef NDEBUG
167  valid_(first, last);
168 #endif // NDEBUG
169  range_.first = start_tile_index;
170  range_.second = start_tile_index + last - first - 1;
171  element_range_.first = *first;
172  element_range_.second = *(last - 1);
173  for (; first != (last - 1); ++first)
174  tile_ranges_.push_back(range_type(*first, *(first + 1)));
175  }

178  void init_map_() {
179  // check for 0 size range.
180  if((element_range_.second - element_range_.first) == 0)
181  return;
182 
183  // initialize elem2tile map
184  elem2tile_.resize(element_range_.second - element_range_.first);
185  const size_type end = range_.second - range_.first;
186  for(size_type t = 0; t < end; ++t)
187  for(size_type e = tile_ranges_[t].first; e < tile_ranges_[t].second; ++e)
188  elem2tile_[e - element_range_.first] = t + range_.first;
189  }
