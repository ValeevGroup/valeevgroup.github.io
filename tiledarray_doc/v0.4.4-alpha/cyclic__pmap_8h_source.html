67  CyclicPmap(World& world, size_type rows, size_type cols,
68  size_type proc_rows, size_type proc_cols) :
69  Pmap(world, rows * cols), rows_(rows), cols_(cols),
70  proc_cols_(proc_cols), proc_rows_(proc_rows)
71  {
72  // Check that the size is non-zero
73  TA_ASSERT(rows_ >= 1ul);
74  TA_ASSERT(cols_ >= 1ul);
75 
76  // Check limits of process rows and columns
77  TA_ASSERT(proc_rows_ >= 1ul);
78  TA_ASSERT(proc_cols_ >= 1ul);
79  TA_ASSERT((proc_rows_ * proc_cols_) <= procs_);
80 
81  // Initialize local tile list
82  if(rank_ < (proc_rows_ * proc_cols_)) {
83  // Compute rank coordinates
84  const size_type rank_row = rank_ / proc_cols_;
85  const size_type rank_col = rank_ % proc_cols_;
86 
87  const size_type local_rows =
88  (rows_ / proc_rows_) + ((rows_ % proc_rows_) < rank_row ? 1ul : 0ul);
89  const size_type local_cols =
90  (cols_ / proc_cols_) + ((cols_ % proc_cols_) < rank_col ? 1ul : 0ul);
91 
92  // Allocate memory for the local tile list
93  local_.reserve(local_rows * local_cols);
94 
95  // Iterate over local tiles
96  for(size_type i = rank_row; i < rows_; i += proc_rows_) {
97  const size_type row_end = (i + 1) * cols_;
98  for(size_type tile = i * cols_ + rank_col; tile < row_end; tile += proc_cols_) {
99  TA_ASSERT(CyclicPmap::owner(tile) == rank_);
100  local_.push_back(tile);
101  }
102  }
103  }
104  }
