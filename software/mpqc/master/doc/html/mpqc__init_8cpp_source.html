161 std::tuple<std::shared_ptr<mpqc::KeyVal>,KeyVal::InputFormat>
162  make_keyval(madness::World &world, const std::string &filename) {
163  std::shared_ptr<mpqc::KeyVal> kv;
164  using InputFormat = KeyVal::InputFormat;
165  InputFormat input_format;
166  try {
167  kv = __make_keyval(world, filename, InputFormat::json);
168  input_format = InputFormat::json;
169  } catch (...) {
170  }
171  if (input_format == InputFormat::invalid) {
172  try {
173  kv = __make_keyval(world, filename, InputFormat::xml);
174  input_format = InputFormat::xml;
175  } catch (std::exception &e) {
176  }
177  }
178  if (input_format == InputFormat::invalid) {
179  try {
180  kv = __make_keyval(world, filename, InputFormat::info);
181  input_format = InputFormat::info;
182  } catch (...) {
183  std::cerr << "failed read_info" << std::endl;
184  }
185  }
186  if (input_format == InputFormat::invalid)
187  throw InputError(
188  "did not recognize input file format (recognized formats: JSON, XML, "
189  "INFO)",
190  __FILE__, __LINE__);
191 
192  return std::make_tuple(kv, input_format);
193 }

231 void MPQCInit::init_work_dir() {
232  char *mpqc_work_dir;
233  mpqc_work_dir = std::getenv("MPQC_WORK_DIR");
234 
235  // if not user defined, use current path
236  if (mpqc_work_dir == nullptr) {
237  path curr_path = current_path();
238  // set the work dir in FormIO
239  FormIO::set_default_work_dir(curr_path.string());
240  } else {
241  // check the correctness of path
242  path work_path(mpqc_work_dir);
243  bool path_exists = exists(work_path);
244  if (!path_exists) {
245  throw FileOperationFailed(
246  "Path ${MPQC_WORK_DIR} does not exists! Please set environment "
247  "variable MPQC_WORK_DIR to a valid path.\n",
248  __FILE__, __LINE__, mpqc_work_dir, FileOperationFailed::Exists);
249  }
250  bool is_dir = is_directory(work_path);
251  if (!is_dir) {
252  throw FileOperationFailed(
253  "Path ${MPQC_WORK_DIR} is not a directory! Please set environment "
254  "variable MPQC_WORK_DIR to an existing directory.\n",
255  __FILE__, __LINE__, mpqc_work_dir, FileOperationFailed::Chdir);
256  }
257  // set the work dir in FormIO
258  FormIO::set_default_work_dir(mpqc_work_dir);
259  }

262 void MPQCInit::set_basename(const std::string &input_filename,
263  const std::string &output_filename) {
264  // get the basename for output files:
265  // 1) if output filename is given, use it minus the suffix
266  // 2) if output filename is not given, use input's basename minus the suffix
267  const char *basename_source;
268  auto input_filename_copy_cstr = std::make_unique<char[]>(input_filename.size() + 1);
269  input_filename_copy_cstr[input_filename.size()] = '\0';
270  std::copy(cbegin(input_filename), cend(input_filename), input_filename_copy_cstr.get());
271  if (!output_filename.empty())
272  basename_source = output_filename.c_str();
273  else {
274  // get the basename(input). basename() in libgen.h is in POSIX standard
275  basename_source = basename(input_filename_copy_cstr.get());
276  }
277  // if basename_source does not contain '.' this will return a null pointer
278  const char *dot_position = ::strrchr(basename_source, '.');
279  const int nfilebase = (dot_position) ? (int)(dot_position - basename_source)
280  : strlen(basename_source);
281  std::string basename(basename_source, basename_source+nfilebase);
282  FormIO::set_default_basename(std::move(basename));
283 }

322 std::tuple<std::string, std::string> process_options(
323  const std::shared_ptr<GetLongOpt> &options) {
324  // set the working dir
325  if (*options->retrieve("W") != ".") {
326  std::string dir = *options->retrieve("W");
327  auto err = chdir(dir.c_str());
328  if (err)
329  throw FileOperationFailed("could not change directory", __FILE__,
330  __LINE__, dir.c_str(),
331  FileOperationFailed::Chdir);
332  }
333 
334  // redirect the output, if needed
335  auto output_opt = options->retrieve("o");
336  std::string output_filename = output_opt ? *output_opt : std::string();
337 
338  auto print_std_header = []() {
339  ExEnv::out0() << indent << "MPQC version " << MPQC_VERSION << std::endl
340  << indent << "compiled for " << TARGET_ARCH << std::endl
341  << FormIO::copyright << std::endl;
342  };
343 
344  if (options->retrieve("h")) {
345  print_std_header();
346  options->usage(ExEnv::out0());
347  std::exit(0);
348  }
349 
350  if (options->retrieve("v")) {
351  print_std_header();
352  std::exit(0);
353  }
354 
355  if (options->retrieve("w")) {
356  print_std_header();
357  ExEnv::out0() << FormIO::warranty << std::endl;
358  std::exit(0);
359  }
360 
361  if (options->retrieve("L")) {
362  print_std_header();
363  ExEnv::out0() << FormIO::license << std::endl;
364  std::exit(0);
365  }
366 
367  if (options->retrieve("k")) {
368  print_std_header();
369  ExEnv::out0() << std::endl
370  << indent
371  << "DescribedClass KeyVal-ctor registry:" << incindent
372  << std::endl;
373  for (const auto &entry : DescribedClass::keyval_ctor_registry()) {
374  ExEnv::out0() << indent << entry.first << std::endl;
375  }
376  std::exit(0);
377  }
378 
379  if (options->retrieve("t")) {
380  KeyVal::set_throw_if_deprecated_path(true);
381  }
382 
383  // get input file name ... can be given as the last option
384  auto input_opt = options->retrieve("i");
385  std::string input_filename;
386  if (input_opt) {
387  input_filename = *input_opt;
388  } else if (MPQCInit::instance().argc() - options->first_unprocessed_arg() ==
389  1) {
390  input_filename =
391  MPQCInit::instance().argv()[options->first_unprocessed_arg()];
392  } else {
393  options->usage();
394  throw std::invalid_argument("input filename not given");
395  }
396 
397  return std::make_tuple(input_filename, output_filename);
398 }
