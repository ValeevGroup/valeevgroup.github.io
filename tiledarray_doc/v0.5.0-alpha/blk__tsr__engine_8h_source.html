164  trange_type make_trange() const {
165  const unsigned int rank = array_.trange().tiles().rank();
166 
167  std::vector<TiledRange1> trange_data;
168  trange_data.reserve(rank);
169  std::vector<std::size_t> trange1_data;
170 
171  // Get temporary data pointers
172  const auto* restrict const trange = array_.trange().data().data();
173  const auto* restrict const lower = lower_bound_.data();
174  const auto* restrict const upper = upper_bound_.data();
175 
176  for(unsigned int d = 0u; d < rank; ++d) {
177  // Copy the tiling for the block
178  const auto lower_d = lower[d];
179  const auto upper_d = upper[d];
180 
181  // Copy and shift the tiling for the block
182  auto i = lower_d;
183  const auto base_d = trange[d].tile(i).first;
184  trange1_data.emplace_back(0ul);
185  for(; i < upper_d; ++i)
186  trange1_data.emplace_back(trange[d].tile(i).second - base_d);
187 
188  // Add the trange1 to the tiled range data
189  trange_data.emplace_back(trange1_data.begin(), trange1_data.end());
190  trange1_data.resize(0ul);
191  }
192 
193  return TiledRange(trange_data.begin(), trange_data.end());
194  }

200  trange_type make_trange(const Permutation& perm) const {
201  const unsigned int rank = array_.trange().tiles().rank();
202 
203  std::vector<TiledRange1> trange_data;
204  trange_data.reserve(rank);
205  std::vector<std::size_t> trange1_data;
206 
207  // Get temporary data pointers
208  const auto* restrict const trange = array_.trange().data().data();
209  const auto* restrict const lower = lower_bound_.data();
210  const auto* restrict const upper = upper_bound_.data();
211 
212  // Construct the inverse permutation
213  const Permutation inv_perm = -perm;
214  for(unsigned int d = 0u; d < rank; ++d) {
215  const auto inv_perm_d = inv_perm[d];
216 
217  // Copy the tiling for the block
218  const auto lower_i = lower[inv_perm_d];
219  const auto upper_i = upper[inv_perm_d];
220 
221  // Copy, shift, and permute the tiling of the block
222  auto i = lower_i;
223  const auto base_d = trange[inv_perm_d].tile(i).first;
224  trange1_data.emplace_back(0ul);
225  for(; i < upper_i; ++i)
226  trange1_data.emplace_back(trange[inv_perm_d].tile(i).second - base_d);
227 
228  // Add the trange1 to the tiled range data
229  trange_data.emplace_back(trange1_data.begin(), trange1_data.end());
230  trange1_data.resize(0ul);
231  }
232 
233  return TiledRange(trange_data.begin(), trange_data.end());
234  }
