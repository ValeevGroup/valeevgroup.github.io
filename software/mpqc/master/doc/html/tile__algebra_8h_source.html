28 bool inline Decompose_Matrix(RowMatrix<T> input, RowMatrix<T> &L,
29  RowMatrix<T> &R, double cut) {
30  Eigen::ColPivHouseholderQR<RowMatrix<T>> qr(input);
31  qr.setThreshold(cut);
32  auto rank = qr.rank();
33 
34  auto full_rank = std::min(input.cols(), input.rows());
35  if (rank >= double(full_rank) / 2.0) {
36  return true;
37  }
38 
39  R = RowMatrix<T>(qr.matrixR()
40  .topLeftCorner(rank, input.cols())
41  .template triangularView<Eigen::Upper>()) *
42  qr.colsPermutation().transpose();
43  L = RowMatrix<T>(qr.householderQ()).leftCols(rank);
44 
45  return false;
46 }

179 // Only works on the output of dqeqp3
180 int inline qr_rank(
181  Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic> const &M,
182  double thresh) {
183  const auto full_rank = std::min(M.cols(), M.rows());
184  auto out_rank = full_rank;
185 
186  auto squared_sum = 0.0;
187  for (auto i = (full_rank - 1); i >= 0; --i) { // rows of R
188 
189  for (auto j = (M.cols() - 1); j >= i; --j) { // cols of R
190  squared_sum += M(i, j) * M(i, j);
191  }
192 
193  if (std::sqrt(squared_sum) >= thresh) {
194  return out_rank;
195  }
196 
197  --out_rank; // Decriment rank and go to next row.
198  }
199 
200  return out_rank;
201 }

203 bool inline Decompose_Matrix(
204  Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic> input,
205  Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic> &L,
206  Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic> &R, double cut) {
207  assert(input.size() >= 0);
208  integer M = input.rows();
209  integer N = input.cols();
210  auto full_rank = std::min(M, N);
211  typedef Eigen::Matrix<integer, Eigen::Dynamic, 1> VectorXi;
212  VectorXi J = VectorXi::Zero(N);
213  double Tau[full_rank];
214  double work;
215  integer LWORK = -1; // Ask LAPACK how much space we need.
216  integer INFO;
217  integer LDA = M;
218 
219  dgeqp3_(&M, &N, input.data(), &LDA, J.data(), Tau, &work, &LWORK, &INFO);
220  LWORK = work;
221  auto W = std::make_unique<double[]>(LWORK);
222  dgeqp3_(&M, &N, input.data(), &LDA, J.data(), Tau, W.get(), &LWORK, &INFO);
223 
224  const double thresh = cut;
225  integer rank = qr_rank(input, thresh);
226 
227  if (rank > 0.5 * double(full_rank)) {
228  return true; // Input is full rank
229  }
230 
231  // LAPACK assumes 1 based indexing, but we need zero.
232  std::for_each(J.data(), J.data() + J.size(), [](integer &val) { --val; });
233  Eigen::PermutationWrapper<VectorXi> P(J);
234  R = Eigen::MatrixXd(input.topLeftCorner(rank, N)
235  .template triangularView<Eigen::Upper>()) *
236  P.transpose();
237 
238  // Form Q.
239  dorgqr_(&M, &rank, &rank, input.data(), &M, Tau, W.get(), &LWORK, &INFO);
240  L = input.leftCols(rank);
241 
242  return false; // Input is not full rank
243 }
