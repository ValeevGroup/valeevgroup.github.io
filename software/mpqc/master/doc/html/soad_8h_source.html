26 inline RowMatrixXd soad_density_eig_matrix(Molecule const &mol) {
27  auto nao = 0;
28  for (const auto &atom : mol.atoms()) {
29  const auto Z = atom.charge();
30  nao += libint2::sto3g_num_ao(Z);
31  }
32 
33  RowMatrixXd D(nao, nao);
34  D.setZero();
35 
36  size_t ao = 0;
37  for (const auto &atom : mol.atoms()) {
38  const auto Z = atom.charge();
39  const auto &occvec = libint2::sto3g_ao_occupation_vector(Z);
40  for (const auto &occ : occvec) {
41  D(ao, ao) = occ;
42  ++ao;
43  }
44  }
45 
46  return D * 0.5; // we use densities normalized to # of electrons/2
47 }

198  const auto min_bs_shells =

200 
201  // make engine pool

203  utility::make_array_of_refs(obs,min_bs));
204

207  auto const &shape_range = H.shape().data().range();
208

211  TA::SparseShape<float> F_shape(shape_norms, trange);
212 
213  TA::DistArray<Tile,Policy> F(world, trange, F_shape);
214 
215  // Loop over lower diagonal tiles
216  const auto F_extent = F.trange().tiles_range().extent();
217  for (auto i = 0ul; i < F_extent[0]; ++i) {
218  const auto i_ord = i * F_extent[1];
219 
220  for (auto j = 0ul; j < F_extent[1]; ++j) {
221  const auto ord = i_ord + j;
222 
223  if (!F.is_zero(ord) && F.is_local(ord)) {
224  auto const &obs_row = obs.cluster_shells()[i];
225  auto const &obs_col = obs.cluster_shells()[j];
226 
227  world.taskq.add(soad_task<decltype(engs),Tile,Policy>, engs, ord, &obs_row,

229  }
230  }
