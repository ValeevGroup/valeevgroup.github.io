21 // coeffs of intermediates linear in the geminal, i.e. V and C
22 constexpr double C_ijij = (1. / 2 + 1. / 4) / 2; // 3/8
23 constexpr double C_ijji = (1. / 2 - 1. / 4) / 2; // 1/8
24 // coeffs of intermediates quadratic in the geminal, i.e. X and B
25 constexpr double CC_ijij = C_ijij * C_ijij + C_ijji * C_ijji; // 10/64
26 constexpr double CC_ijji =
27  2 * C_ijij * C_ijji; // C_ijij * C_ijji + C_ijji * C_ijij;
28  // 6/64
29 // closed-shell-antisymmetrized versions of above
30 constexpr double C_ijij_bar = 2 * C_ijij - C_ijji; // 5/8
31 constexpr double C_ijji_bar = 2 * C_ijji - C_ijij; // -1/8
32 constexpr double CC_ijij_bar = 2 * CC_ijij - CC_ijji; // 14/64
33 constexpr double CC_ijji_bar = 2 * CC_ijji - CC_ijij; // 2/64

72 template <bool P1_eq_P2 = false>
73 TiledArray::SparseShape<float> make_ijij_ijji_shape(
74  const TiledArray::TiledRange &trange) {
75  // number of occ tiles
76  auto n_occ = trange.data()[0].tiles_range().second;
77 
78  TiledArray::Tensor<float> tile_norms(trange.tiles_range(), 0.0);
79 
80  float max = (float) n_occ*n_occ;
81 
82  // set sparse tile
83  for (auto i = 0; i < n_occ; i++) {
84  const auto j_start = P1_eq_P2 ? i : 0;
85  for (auto j = j_start; j < n_occ; j++) {
86  const auto ijij = ((i * n_occ + j) * n_occ + i) * n_occ + j;
87  const auto ijji = ((i * n_occ + j) * n_occ + j) * n_occ + i;
88  tile_norms[ijij] = max;
89  tile_norms[ijji] = max;
90  }
91  }
92 
93  TiledArray::SparseShape<float> shape(tile_norms, trange);
94  return shape;
95 }

97 template <typename Tile, typename Policy>
98 void convert_X_ijkl(TiledArray::Array<double, 4, Tile, Policy> &ijkl,
99  const Eigen::MatrixXd &F) {
100  auto convert = [&F](Tile &result_tile) {
101 
102  // compute index
103  const auto i0 = result_tile.range().lobound()[0];
104  const auto in = result_tile.range().upbound()[0];
105  const auto j0 = result_tile.range().lobound()[1];
106  const auto jn = result_tile.range().upbound()[1];
107  const auto k0 = result_tile.range().lobound()[2];
108  const auto kn = result_tile.range().upbound()[2];
109  const auto l0 = result_tile.range().lobound()[3];
110  const auto ln = result_tile.range().upbound()[3];
111 
112  auto tile_idx = 0;
113  typename Tile::value_type norm = 0.0;
114  for (auto i = i0; i < in; ++i) {
115  const auto f_ii = F(i, i);
116  for (auto j = j0; j < jn; ++j) {
117  const auto f_jj = F(j, j);
118  for (auto k = k0; k < kn; ++k) {
119  for (auto l = l0; l < ln; ++l, ++tile_idx) {
120  const auto old = result_tile[tile_idx];
121  const auto result_abij = old * (f_ii + f_jj);
122  norm += result_abij * result_abij;
123  result_tile[tile_idx] = result_abij;
124  }
125  }
126  }
127  }
128  return std::sqrt(norm);
129  };
130 
131  TiledArray::foreach_inplace(ijkl, convert);
132 }

134 template <typename Tile, typename Policy>
135 TiledArray::Array<double, 4, Tile, Policy> convert_C_ijab(
136  TiledArray::Array<double, 4, Tile, Policy> &ijab, const std::size_t n_occ,
137  const std::size_t n_frozen, const Eigen::VectorXd &ens) {
138  auto convert = [&ens, n_occ, n_frozen](Tile &result_tile,
139  const Tile &arg_tile) {
140 
141  result_tile = Tile(arg_tile.range());
142 
143  // compute index
144  const auto i0 = result_tile.range().lobound()[0];
145  const auto in = result_tile.range().upbound()[0];
146  const auto j0 = result_tile.range().lobound()[1];
147  const auto jn = result_tile.range().upbound()[1];
148  const auto a0 = result_tile.range().lobound()[2];
149  const auto an = result_tile.range().upbound()[2];
150  const auto b0 = result_tile.range().lobound()[3];
151  const auto bn = result_tile.range().upbound()[3];
152 
153  auto tile_idx = 0;
154  typename Tile::value_type norm = 0.0;
155  for (auto i = i0; i < in; ++i) {
156  auto en_i = ens[i + n_frozen];
157  for (auto j = j0; j < jn; ++j) {
158  auto en_j = ens[j + n_frozen];
159  for (auto a = a0; a < an; ++a) {
160  auto en_a = ens[a + n_occ];
161  for (auto b = b0; b < bn; ++b, ++tile_idx) {
162  auto en_b = ens[b + n_occ];
163  const auto old = arg_tile[tile_idx];
164  const auto divde = 1 / (en_i + en_j - en_a - en_b);
165  const auto result_abij = old * divde;
166  norm += result_abij * result_abij;
167  result_tile[tile_idx] = result_abij;
168  }
169  }
170  }
171  }
172  return std::sqrt(norm);
173  };
174 
175  return TiledArray::foreach (ijab, convert);
176 }

200 template <typename Tile>
201 struct F12PairEnergyReductor {
202  using result_type = RowMatrix<typename Tile::scalar_type>;
203  using argument_type = Tile;
204 
205  enum class PairSpin {
206  none, // closed-shell only
207  aa, // same-spin
208  bb, // same-spin
209  ab, // opposite-spin
210  s0, // S=0 (singlet pair)
211  s1 // S=1 (triplet pair)
212  };
213  double c_ijij;
214  double c_ijji;
215  int n1;
216  int n2;
217  PairSpin spincase;
218 
219  F12PairEnergyReductor() = default;
220  F12PairEnergyReductor(F12PairEnergyReductor const &) = default;
221  F12PairEnergyReductor(double cijij, double cijji, int _n1, int _n2 = -1,
222  PairSpin sc = PairSpin::none)
223  : c_ijij(cijij),
224  c_ijji(cijji),
225  n1(_n1),
226  n2(_n2 == -1 ? _n1 : _n2),
227  spincase(sc) {
228  TA_USER_ASSERT(spincase != PairSpin::s0 && spincase != PairSpin::s1,
229  "spin-adapted F12PairEnergyReductor not implemented yet");
230  TA_USER_ASSERT(spincase != PairSpin::aa && spincase != PairSpin::bb &&
231  spincase != PairSpin::ab,
232  "spin-orbital F12PairEnergyReductor not implemented yet");
233  if (spincase != PairSpin::ab)
234  TA_USER_ASSERT(n1 == n2,
235  "same-spin case but #s of orbitals for "
236  "electrons 1 and 2 differ");
237  }
238 
239  result_type operator()() const { return result_type::Zero(n1, n2); }
240 
241  result_type operator()(result_type const &t) const { return t; }
242 
243  void operator()(result_type &me, result_type const &other) const {
244  TA_USER_ASSERT(me.rows() == other.rows() && me.cols() == other.cols(),
245  "result vector size mismatch");
246  me += other;
247  }
248 
249  void operator()(result_type &me, argument_type const &tile) const {
250  auto const &range = tile.range();
251 
252  TA_ASSERT(range.rank() == 4);
253 
254  auto const st = range.lobound_data();
255  auto const fn = range.upbound_data();
256 
257  auto sti = st[0];
258  auto fni = fn[0];
259  auto stj = st[1];
260  auto fnj = fn[1];
261  auto stk = st[2];
262  auto fnk = fn[2];
263  auto stl = st[3];
264  auto fnl = fn[3];
265 
266  const auto *value_ptr = tile.data();
267  for (auto i = sti; i < fni; ++i) {
268  for (auto j = stj; j < fnj; ++j) {
269  const auto ii = std::min(i, j);
270  const auto jj = std::max(i, j);
271  for (auto k = stk; k < fnk; ++k) {
272  for (auto l = stl; l < fnl; ++l, ++value_ptr) {
273  // ijij
274  if (k == i && l == j) {
275  me(ii, jj) += c_ijij * (*value_ptr);
276  }
277  // ijji
278  if (l == i && k == j) {
279  me(ii, jj) += c_ijji * (*value_ptr);
280  }
281  }
282  }
283  }
284  }
285  }
286 };
