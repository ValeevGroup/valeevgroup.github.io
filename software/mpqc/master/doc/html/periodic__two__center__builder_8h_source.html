237  // if reducer does not exist, create entry and store F, else accumulate F to

272  void compute_task_ab(size_t RJ, std::array<size_t, 2> tile_idx) {
273  const auto tile0 = tile_idx[0];
274  const auto tileR = tile_idx[1];
275 
276  // get reference to basis sets
277  const auto &basis0 = basis0_;
278  const auto &basisR = basisR_;
279 
280  // shell clusters for this tile
281  const auto &cluster0 = basis0->cluster_shells()[tile0];
282  const auto &clusterR = basisR->cluster_shells()[tileR];
283 
284  // number of shells in each cluster
285  const auto nshells0 = cluster0.size();
286  const auto nshellsR = clusterR.size();
287 
288  // 1-d tile ranges
289  const auto &tr0 = result_trange_.dim(0);
290  const auto &tr1 = result_trange_.dim(1);
291  const auto ntilesR = tr1.tile_extent();
292  const auto &rng0 = tr0.tile(tile0);
293  const auto &rngR = tr1.tile(tileR);
294  const auto rngR_size = rngR.second - rngR.first;
295 
296  // 2-d tile ranges describing the contribution blocks produced by this
297  auto result_rng = TA::Range({rng0, rngR});
298  // initialize contribution to the result matrices
299  auto result_tile = Tile(std::move(result_rng), 0.0);
300  // grab ptrs to tile data to make addressing more efficient
301  auto *result_ptr = result_tile.data();
302 
303  using ::mpqc::lcao::gaussian::detail::compute_func_offset_list;
304 
305  {
306  // index of first shell in this cluster
307  const auto sh0_offset = basis0_shell_offset_map_[tile0];
308  const auto shR_offset = basisR_shell_offset_map_[tileR];
309 
310  // index of last shell in this cluster
311  const auto sh0_max = sh0_offset + nshells0;
312  const auto shR_max = shR_offset + nshellsR;
313 
314  // determine if this task contains significant shell-pairs
315  auto is_significant = false;
316  {
317  auto sh0_in_basis = sh0_offset;
318  for (; sh0_in_basis != sh0_max; ++sh0_in_basis) {
319  for (const auto shR_in_basis : sig_shellpair_list_[sh0_in_basis]) {
320  if (shR_in_basis >= shR_offset && shR_in_basis < shR_max) {
321  is_significant = true;
322  break;
323  }
324  }
325  if (is_significant) break;
326  }
327  }
328 
329  if (is_significant) {
330  auto engine = engines_[RJ]->local();
331  const auto engine_precision = target_precision_;
332  engine.set_precision(engine_precision);
333  const auto &computed_shell_sets = engine.results();
334 
335  // compute offset list of cluster1 and cluster3
336  auto offset_list = compute_func_offset_list(clusterR, rngR.first);
337 
338  // this is the index of the first basis functions for each shell *in
339  // this shell cluster*
340  auto cf0_offset = 0;
341  // this is the index of the first basis functions for each shell *in the
342  // basis set*
343  auto bf0_offset = rng0.first;
344 
345  size_t cfR_offset, bfR_offset;
346  // loop over all shell sets
347  for (auto sh0 = 0; sh0 != nshells0; ++sh0) {
348  const auto &shell0 = cluster0[sh0];
349  const auto nf0 = shell0.size();
350 
351  const auto sh0_in_basis = sh0 + sh0_offset;
352  for (const auto &shR_in_basis : sig_shellpair_list_[sh0_in_basis]) {
353  if (shR_in_basis < shR_offset || shR_in_basis >= shR_max) continue;
354 
355  const auto shR = shR_in_basis - shR_offset;
356  std::tie(cfR_offset, bfR_offset) = offset_list[shR];
357 
358  const auto &shellR = clusterR[shR];
359  const auto nfR = shellR.size();
360 
361  // TODO add screening for 2-center ints
362  engine.compute1(shell0, shellR);
363 
364  const auto *ints = computed_shell_sets[0];
365  if (ints != nullptr) {
366  for (auto f0 = 0, f0R = 0; f0 != nf0; ++f0) {
367  const auto cf0 = f0 + cf0_offset;
368  for (auto fR = 0; fR != nfR; ++fR, ++f0R) {
369  const auto cfR = fR + cfR_offset;
370  const auto cf0R = cf0 * rngR_size + cfR;
371 
372  const auto value = ints[f0R];
373  result_ptr[cf0R] += value;
374  }
375  }
376  }
377  }
378 
379  cf0_offset += nf0;
380  bf0_offset += nf0;
381  }
382  }
383  }
384 
385  // accumulate the local contributions
386  {
387  auto tile0R = tile0 * ntilesR + tileR;
388  PeriodicTwoCenterBuilder_::accumulate_local_task(result_tile, tile0R);
389  }
390  }
