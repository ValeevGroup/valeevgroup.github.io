88 template <typename Tile, typename Policy>
89 double evaluate_energy(lcao::LCAOFactoryBase<Tile, Policy>& lcao_factory,
90  gaussian::AOFactoryBase<Tile, Policy>& ao_factory,
91  std::shared_ptr<Eigen::VectorXd> orbital_energy,
92  int num_laplace_points) {
93  auto& world = lcao_factory.world();
94 
95  TA::DistArray<Tile, Policy> Cv =
96  lcao_factory.orbital_registry().retrieve(OrbitalIndex(L"a")).coefs();
97 
98  TA::DistArray<Tile, Policy> Ci =
99  lcao_factory.orbital_registry().retrieve(OrbitalIndex(L"i")).coefs();
100 
101  // Needed for Localization stuffs
102  auto basis = *ao_factory.basis_registry()->retrieve(OrbitalIndex(L"λ"));
103 
104  const auto bs_array = utility::make_array(basis, basis);
105 
106  auto multi_pool = gaussian::make_engine_pool(
107  libint2::Operator::emultipole1, utility::make_array_of_refs(basis));
108 
109  auto r_xyz =
110  gaussian::xyz_integrals<Tile, Policy>(world, multi_pool, bs_array);
111 
112  auto U = mpqc::scf::FosterBoysLocalization{}(Ci, r_xyz);
113 
114  TA::DistArray<Tile, Policy> Ci_loc;
115  Ci_loc("mu, i") = Ci("mu, k") * U("k, i");
116 
117  auto eri3 = ao_factory.compute(L"(Λ | G| μ ν)");
118  auto M = ao_factory.compute(L"(Λ| G| Κ)");
119 
120  auto Linv = array_ops::cholesky_inverse(M);
121 
122  TA::DistArray<Tile, Policy> g, t, Z, B;
123  Z("X, i, nu") = Linv("X,Y") * (eri3("Y, mu, nu") * Ci_loc("mu, i"));
124  Z.truncate();
125 
126  auto z_storage = ::mpqc::detail::array_storage(Z);
127  ExEnv::out0() << indent << indent << "Z Storage " << z_storage[1] << ", "
128  << 100.0 - 100.0 * double(z_storage[1]) / double(z_storage[0])
129  << "% sparsity"
130  << "\n";
131 
132  auto k0 = mpqc::fenced_now(world);
133  decltype(g) K;
134  g("i, nu, j, sig") = Z("X, i, nu") * Z("X, j, sig");
135  g.truncate();
136  K("i,a,j,b") = 2 * g("i,a,j,b") - g("i,b,j,a");
137  K.truncate();
138  auto k1 = mpqc::fenced_now(world);
139  auto k_time = mpqc::duration_in_s(k0, k1);
140  ExEnv::out0() << indent << indent << "Time for K: " << k_time << std::endl;
141  auto k_storage = ::mpqc::detail::array_storage(K);
142  ExEnv::out0() << indent << indent << "K Storage " << k_storage[1] << ", "
143  << 100.0 - 100.0 * double(k_storage[1]) / double(k_storage[0])
144  << "% sparsity"
145  << "\n";
146 
147  auto nocc =
148  lcao_factory.orbital_registry().retrieve(OrbitalIndex(L"m")).rank();
149 
150  auto nactive =
151  lcao_factory.orbital_registry().retrieve(OrbitalIndex(L"i")).rank();
152 
153  auto nfrozen = nocc - nactive;
154 
155  auto const& energies = *orbital_energy;
156  const auto gap2 = 2 * (energies[nocc] - energies[nocc - 1]);
157 
158  Eigen::VectorXd w, x;
159  math::gauss_legendre(num_laplace_points, w, x);
160 
161  TA::DistArray<Tile, Policy> Di, Dv;
162  auto Iocc = lcao_factory.compute(L"<i|I|j>");
163  auto Ivir = lcao_factory.compute(L"<a|I|b>");
164 
165  auto t0_outer = mpqc::fenced_now(world);
166  for (auto i = 0; i < num_laplace_points; ++i) {
167  auto t0 = mpqc::fenced_now(world);
168  TA::foreach_inplace(Iocc, [&](TA::TensorD& t) {
169  auto const& range = t.range();
170  auto const lo = range.lobound_data();
171  auto const up = range.upbound_data();
172  for (auto n = lo[0]; n != up[0]; ++n) {
173  auto idx = n + nfrozen; // Account for activeness of space
174  const auto energy = energies[idx];
175  t(n, n) = std::pow(x[i], -energy / (gap2)-0.25);
176  }
177 
178  return t.norm();
179  });
180 
181  Di("ip,i") = U("l,ip") * Iocc("l,k") * U("k,i");
182  Di.truncate();
183 
184  TA::foreach_inplace(Ivir, [&](TA::TensorD& t) {
185  auto const& range = t.range();
186  auto const lo = range.lobound_data();
187  auto const up = range.upbound_data();
188  for (auto n = lo[0]; n != up[0]; ++n) {
189  auto idx = n + nocc; // Account for activeness of space
190  const auto energy = energies[idx];
191  t(n, n) = std::pow(x[i], energy / (gap2)-0.25);
192  }
193 
194  return t.norm();
195  });
196  Dv("mu,nu") = Cv("mu,l") * Ivir("l, k") * Cv("nu,k");
197  Dv.truncate();
198 
199  B("X, ip, nu") = Z("X, ip, nup") * Dv("nup, nu");
200  B.truncate();
201  B("X, i, nu") = B("X, ip, nu") * Di("ip, i");
202  B.truncate();
203 
204  if (i == 0) {
205  t("i, nu, j, sig") = (-w[i] / gap2) * B("X, i, nu") * B("X, j, sig");
206  } else {
207  t("i, nu, j, sig") += (-w[i] / gap2) * B("X, i, nu") * B("X, j, sig");
208  }
209  t.truncate();
210  auto t1 = mpqc::fenced_now(world);
211  auto t_time_iter = mpqc::duration_in_s(t0, t1);
212  ExEnv::out0() << indent << indent << indent << "Time for T[" << i + 1
213  << "]: " << t_time_iter << ", ";
214  auto t_storage = ::mpqc::detail::array_storage(t);
215  ExEnv::out0() << "t Storage " << t_storage[1] << ", "
216  << 100.0 - 100.0 * double(t_storage[1]) / double(t_storage[0])
217  << "% sparsity"
218  << "\n";
219  }
220  auto t1_outer = mpqc::fenced_now(world);
221  auto t_time = mpqc::duration_in_s(t0_outer, t1_outer);
222  ExEnv::out0() << indent << "Total Time for T: " << t_time << std::endl;
223 
224 
225  auto e0 = mpqc::fenced_now(world);
226  double energy = K("i,nu,j,sig").dot(t("i,nu,j,sig"));
227  auto e1 = mpqc::fenced_now(world);
228  auto e_time = mpqc::duration_in_s(e0, e1);
229  ExEnv::out0() << indent << "MP2 Correlation Energy = " << energy << "\n";
230  ExEnv::out0() << indent << "Energy Time: " << e_time << "\n";
231 
232  world.gop.fence();
233  return energy;
234 }

237 template <typename Tile, typename Policy>
238 void RLaplaceMP2<Tile, Policy>::evaluate(Energy* result) {
239  auto target_precision = result->target_precision(0);
240 
241  if (!this->computed() || computed_precision_ > target_precision) {
242  // compute reference to higher precision than required of correlation energy
243  auto target_ref_precision = target_precision / 100.;
244  auto ref_energy = std::make_shared<Energy>(ref_wfn_, target_ref_precision);
245  ::mpqc::evaluate(*ref_energy, ref_wfn_);
246 
247  this->init_sdref(ref_wfn_, target_ref_precision);
248 
249  mp2_corr_energy_ = detail::evaluate_energy(
250  this->lcao_factory(), this->ao_factory(),
251  make_diagonal_fpq(this->lcao_factory(), this->ao_factory(), true),
252  num_laplace_points_);
253 
254  this->computed_ = true;
255  this->set_value(result, ref_energy->energy() + mp2_corr_energy_.get());
256  computed_precision_ = target_precision;
257  }
258 }
