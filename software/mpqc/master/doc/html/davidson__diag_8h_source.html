128  template <typename Operator>
129  EigenVector<element_type> solve(value_type& guess, const Operator& op,
130  const DavidsonDiagPred<D>* const pred,
131  double convergence, std::size_t max_iter) {
132  double norm_e = 1.0;
133  double norm_r = 1.0;
134  std::size_t iter = 0;
135  auto& world = TA::get_default_world();
136 
137  EigenVector<element_type> eig = EigenVector<element_type>::Zero(n_roots_);
138 
139  while (iter < max_iter &&
140  (norm_r > 10 * convergence || norm_e > convergence) &&
141  !guess.empty()) {
142  auto time0 = mpqc::fenced_now(world);
143 
144  // compute product of H with guess vector
145  value_type HC = op(guess);
146 
147  auto time1 = mpqc::fenced_now(world);
148  EigenVector<element_type> eig_new, norms;
149  std::tie(eig_new, norms) = extrapolate(HC, guess, pred);
150  auto time2 = mpqc::fenced_now(world);
151 
152  EigenVector<element_type> delta_e = (eig - eig_new);
153  delta_e = delta_e.cwiseAbs();
154  norm_e =
155  *std::max_element(delta_e.data(), delta_e.data() + delta_e.size());
156  norm_r = *std::max_element(norms.data(), norms.data() + norms.size());
157 
158  util::print_davidson_energy_iteration(iter, delta_e, norms, eig_new,
159  mpqc::duration_in_s(time0, time1),
160  mpqc::duration_in_s(time1, time2));
161 
162  eig = eig_new;
163  iter++;
164 
165  } // end of while loop
166 
167  if (iter == max_iter) {
168  throw MaxIterExceeded("Davidson Diagonalization Exceeded Max Iteration",
169  __FILE__, __LINE__, max_iter, "DavidsonDiag");
170  }
171 
172  return eig;
