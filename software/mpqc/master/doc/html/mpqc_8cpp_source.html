91 int try_main(int argc, char *argv[], madness::World &world) {
92  using namespace mpqc;
93 
94  // define default MPQC options
95  auto options = make_options();
96 
97  // initialize MPQC
98  initialize(argc, argv, world, options);
99 
100  // parse and process options
101  options->parse(argc, argv);
102  std::string input_filename, output_filename;
103  std::tie(input_filename, output_filename) = process_options(options);
104 
105  // redirect the output to output_file
106  std::ofstream output;
107  if (!output_filename.empty()) output.open(output_filename);
108  if (!output.good())
109  throw FileOperationFailed("failed to open output file", __FILE__, __LINE__,
110  output_filename.c_str(),
111  FileOperationFailed::OpenW);
112  auto cout_streambuf_reset = [](std::streambuf *p) { std::cout.rdbuf(p); };
113  std::unique_ptr<std::streambuf, decltype(cout_streambuf_reset)>
114  cout_buffer_holder(std::cout.rdbuf(), cout_streambuf_reset);
115  if (!output_filename.empty()) std::cout.rdbuf(output.rdbuf());
116 
117  MPQCInit::instance().set_basename(input_filename, output_filename);
118 
119  // make keyval
120  std::shared_ptr<KeyVal> kv =
121  MPQCInit::instance().make_keyval(world, input_filename);
122  // set default class keys assuming that we are doing LCAO
123  kv->set_default_class_key<::mpqc::WavefunctionWorld>("LCAOWfnWorld");
124 
125  // now set up the debugger
126  auto debugger = kv->class_ptr<Debugger>("debugger");
127  // use default-constructed debugger if -D given and debugger keyword is
128  // missing
129  if (!debugger) {
130  auto debugger_opt = options->retrieve("D");
131  if (debugger_opt) {
132  const auto debugger_json_str = *debugger_opt;
133  if (debugger_json_str
134  .empty()) { // no JSON spec for Debugger given, use default ctor
135  debugger = std::make_shared<Debugger>();
136  } else { // JSON spec for Debugger given
137  std::istringstream iss(debugger_json_str);
138  KeyVal kv;
139  kv.read_json(iss);
140  debugger = std::make_shared<Debugger>(kv);
141  }
142  }
143  }
144  if (debugger) {
145  Debugger::set_default_debugger(debugger);
146  debugger->set_exec(argv[0]);
147  debugger->set_prefix(world.rank());
148  if (options->retrieve("d"))
149  debugger->debug("Starting debugger because -d given on command line.");
150  }
151 
152  // redirect filenames in KeyVal to the directory given by -p cmdline option
153  auto prefix_opt = options->retrieve("p");
154  if (prefix_opt) { // set file prefix, if given
155  kv->assign("file_prefix", *prefix_opt);
156  }
157 
158  // configure the units system
159  auto units_opt = options->retrieve("u");
160  std::string units_str;
161  if (units_opt) {
162  units_str = *units_opt;
163  } else {
164  if (kv->exists("units")) {
165  units_str = kv->value<std::string>("units");
166  }
167  }
168  if (!units_str.empty()) UnitFactory::set_default(units_str);
169 
170  // announce ourselves to the world
171  announce(world);
172 
174  // print computation metadata
176 
177  // the input keyval for reference
178  ExEnv::out0() << indent << "Input KeyVal (format="
179  << to_string(MPQCInit::instance().input_format()) << "):\n";
180  switch (MPQCInit::instance().input_format()) {
181  case MPQCInit::InputFormat::json:
182  kv->write_json(ExEnv::out0());
183  break;
184  case MPQCInit::InputFormat::xml:
185  kv->write_xml(ExEnv::out0());
186  break;
187  case MPQCInit::InputFormat::info:
188  kv->write_info(ExEnv::out0());
189  break;
190  default:
191  throw ProgrammingError(
192  "unrecognized input file format returned by MPQCInit::input_format()",
193  __FILE__, __LINE__);
194  }
195  ExEnv::out0() << std::endl;
196 
197  // units
198  ExEnv::out0() << indent << "Using fundamental constants system "
199  << UnitFactory::get_default()->system() << std::endl;
200 
201  // run
202  // set the default World to the top-level keyword "world"
203  const auto kv_has_top_world = kv->exists("world");
204  if (kv_has_top_world) {
205  throw InputError("top-level keyword \"world\" is reserved for use by MPQC",
206  __FILE__, __LINE__, "world");
207  }
208  kv->assign("world", &world); // set "$:world" keyword to &world to define
209  // the default execution context for this input
210  TA::set_default_world(world); // must specify default world to avoid
211  // madness::World::get_default() getting called
212  MPQCTask task(world, kv);
213  task.run();
214  kv->erase("world");
215  if (prefix_opt) { // unset file prefix, if did previously
216  kv->erase("file_prefix");
217  }
218 
219  // print the final keyval
220  ExEnv::out0() << indent << "Output KeyVal (format="
221  << to_string(MPQCInit::instance().input_format()) << "):\n";
222  switch (MPQCInit::instance().input_format()) {
223  case MPQCInit::InputFormat::json:
224  kv->write_json(ExEnv::out0());
225  break;
226  case MPQCInit::InputFormat::xml:
227  kv->write_xml(ExEnv::out0());
228  break;
229  case MPQCInit::InputFormat::info:
230  kv->write_info(ExEnv::out0());
231  break;
232  default:
233  throw ProgrammingError(
234  "unrecognized input file format returned by MPQCInit::input_format()",
235  __FILE__, __LINE__);
236  }
237  ExEnv::out0() << std::endl;
238 
239  return 0;
240 }
