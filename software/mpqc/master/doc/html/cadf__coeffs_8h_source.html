475 template <typename Array, typename DirectArray>
476 TA::DistArray<TA::Tensor<double>, TA::SparsePolicy> cadf_by_atom_array(
477  Array const &M, DirectArray const &eri3, TA::TiledRange const &trange) {
478  auto &world = M.world();
479  auto Cshape = eri3.array().shape(); // cadf_shape(world, trange);
480 
481  // Use same pmap to ensure some locality
482  auto pmap = eri3.array().pmap();
483  TA::DistArray<TA::Tensor<double>, TA::SparsePolicy> C(
484  world, trange, std::move(Cshape), pmap);
485 
486  const auto natoms = trange.tiles_range().extent_data()[0];
487  auto const &eri3_tiles = trange.tiles_range();
488 
489  using ArrayType =
490  typename std::remove_reference<decltype(eri3.array())>::type;
491  using DirectTile = typename ArrayType::value_type;
492  using Tile = TA::Tensor<double>;
493 
494  for (auto i = 0ul; i < natoms; ++i) {
495  std::array<unsigned long, 3> idx_iii = {{i, i, i}};
496  std::array<unsigned long, 2> idx_ii = {{i, i}};
497 
498  if (!C.is_zero(idx_iii) && C.is_local(idx_iii)) {
499  auto eri3_iii = eri3.array().find(idx_iii);
500  auto M_ii = M.find(idx_ii);
501 
502  unsigned long ord_iii = eri3_tiles.ordinal(idx_iii);
503  world.taskq.add(create_ii_tile<DirectTile, Tile>, &C, eri3_iii, M_ii,
504  ord_iii);
505  }
506 
507  for (auto j = 0ul; j < natoms; ++j) {
508  if (j == i) {
509  continue;
510  }
511 
512  std::array<unsigned long, 3> idx_iij = {{i, i, j}};
513  std::array<unsigned long, 3> idx_jij = {{j, i, j}};
514 
515  if (!C.is_zero(idx_iij) && !C.is_zero(idx_jij)) {
516  if (C.is_local(idx_iij) || C.is_local(idx_jij)) {
517  auto ord_iij = eri3_tiles.ordinal(idx_iij);
518  auto ord_jij = eri3_tiles.ordinal(idx_jij);
519 
520  std::array<unsigned long, 2> M_idx_ii = {{i, i}};
521  std::array<unsigned long, 2> M_idx_ij = {{i, j}};
522  std::array<unsigned long, 2> M_idx_ji = {{j, i}};
523  std::array<unsigned long, 2> M_idx_jj = {{j, j}};
524 
525  auto eri3_iij = eri3.array().find(idx_iij);
526  auto eri3_jij = eri3.array().find(idx_jij);
527 
528  auto M_ii = M.find(M_idx_ii);
529  auto M_ij = M.find(M_idx_ij);
530  auto M_ji = M.find(M_idx_ji);
531  auto M_jj = M.find(M_idx_jj);
532 
533  world.taskq.add(create_ij_tile<DirectTile, Tile>, &C, eri3_iij,
534  eri3_jij, M_ii, M_ij, M_ji, M_jj, ord_iij, ord_jij);
535  }
536  }
537  }
538  }
539  world.gop.fence();
540  C.truncate();
541 
542  return C;
543 }

570  using ::mpqc::detail::direct_3D_idx;
571  using ::mpqc::detail::direct_ord_idx;
572  using ::mpqc::detail::is_in_lattice_range;

575  const auto uc0_ord = i / natoms_per_uc;
576  const Vector3i uc0_3D =
577  direct_3D_idx(uc0_ord, lattice_range0) + lattice_center0;
578  const auto uc0_ord_in_df =
579  direct_ord_idx(uc0_3D - lattice_center_df, lattice_range_df);
580  const auto i_in_df = uc0_ord_in_df * natoms_per_uc + i % natoms_per_uc;
581 
582  if (is_in_lattice_range(uc0_3D, lattice_range1, lattice_center1)) {
583  const auto uc0_ord_in_bs1 =
584  direct_ord_idx(uc0_3D - lattice_center1, lattice_range1);
585  const auto i_in_bs1 = uc0_ord_in_bs1 * natoms_per_uc + i % natoms_per_uc;
586  std::array<unsigned long, 3> idx_iii = {{i_in_df, i, i_in_bs1}};
587  std::array<unsigned long, 2> idx_ii = {{i_in_df, i_in_df}};
588

590  auto eri3_iii = eri3.array().find(idx_iii);
591  auto M_ii = M.find(idx_ii);
592 
593  unsigned long ord_iii = eri3_tiles.ordinal(idx_iii);
594  world.taskq.add(create_ii_tile<DirectTile, Tile>, &C, eri3_iii, M_ii,
595  ord_iii);
596  }
597  }
598

600  const auto uc1_ord = j / natoms_per_uc;
601  const Vector3i uc1_3D =
602  direct_3D_idx(uc1_ord, lattice_range1) + lattice_center1;
603  if (uc0_3D == uc1_3D && (j % natoms_per_uc) == (i % natoms_per_uc)) {
604  continue;
605  }
606 
607  const auto uc1_ord_in_df =
608  direct_ord_idx(uc1_3D - lattice_center_df, lattice_range_df);
609  const auto j_in_df = uc1_ord_in_df * natoms_per_uc + j % natoms_per_uc;
610  std::array<unsigned long, 3> idx_iij = {{i_in_df, i, j}};
611  std::array<unsigned long, 3> idx_jij = {{j_in_df, i, j}};
612 
613  if (!C.is_zero(idx_iij) && !C.is_zero(idx_jij)) {
614  if (C.is_local(idx_iij) || C.is_local(idx_jij)) {
615  auto ord_iij = eri3_tiles.ordinal(idx_iij);
616  auto ord_jij = eri3_tiles.ordinal(idx_jij);
617 
618  std::array<unsigned long, 2> M_idx_ii = {{i_in_df, i_in_df}};
619  std::array<unsigned long, 2> M_idx_ij = {{i_in_df, j_in_df}};
620  std::array<unsigned long, 2> M_idx_ji = {{j_in_df, i_in_df}};
621  std::array<unsigned long, 2> M_idx_jj = {{j_in_df, j_in_df}};
622 
623  auto eri3_iij = eri3.array().find(idx_iij);
624  auto eri3_jij = eri3.array().find(idx_jij);
625 
626  auto M_ii = M.find(M_idx_ii);
627  auto M_ij = M.find(M_idx_ij);
628  auto M_ji = M.find(M_idx_ji);
629  auto M_jj = M.find(M_idx_jj);
630 
631  world.taskq.add(create_ij_tile<DirectTile, Tile>, &C, eri3_iij,
632  eri3_jij, M_ii, M_ij, M_ji, M_jj, ord_iij, ord_jij);
633  }
634  }
635  }
636  }

638

640  world.gop.fence();
641

646 template <typename DirectTile, typename Tile>
647 void create_ii_tile(TA::DistArray<TA::Tensor<double>, TA::SparsePolicy> *C,
648  DirectTile eri3_iii, Tile M_tile, unsigned long ord) {
649  TA::Tensor<double> eri3_tile = eri3_iii.operator TA::Tensor<double>();
650  auto eri3_extent = eri3_tile.range().extent();
651  RowMatrixXd eri3_eig =
652  TA::eigen_map(eri3_tile, eri3_extent[0], eri3_extent[1] * eri3_extent[2]);
653 
654  auto M_extent = M_tile.range().extent();
655  RowMatrixXd M_eig = TA::eigen_map(M_tile, M_extent[0], M_extent[1]);
656 
657  // Use pivoted QR for stablility reasons.
658  RowMatrixXd out_eig = M_eig.colPivHouseholderQr().solve(eri3_eig);
659  TA::TensorD out_tile(eri3_tile.range(), 0.0);
660 
661  TA::eigen_map(out_tile, eri3_extent[0], eri3_extent[1] * eri3_extent[2]) =
662  out_eig;
663 
664  C->set(ord, out_tile);
665 }

667 template <typename DirectTile, typename Tile>
668 void create_ij_tile(TA::DistArray<TA::Tensor<double>, TA::SparsePolicy> *C,
669  DirectTile direct_eri3_iij, DirectTile direct_eri3_jij,
670  Tile M_tile_ii, Tile M_tile_ij, Tile M_tile_ji,
671  Tile M_tile_jj, unsigned long ord_iij,
672  unsigned long ord_jij) {
673  TA::Tensor<double> eri3_iij = direct_eri3_iij.operator TA::Tensor<double>();
674  TA::Tensor<double> eri3_jij = direct_eri3_jij.operator TA::Tensor<double>();
675 
676  auto eri3_extent_iij = eri3_iij.range().extent_data();
677  auto eri3_extent_jij = eri3_jij.range().extent_data();
678 
679  RowMatrixXd eri3_eig_iij = TA::eigen_map(
680  eri3_iij, eri3_extent_iij[0], eri3_extent_iij[1] * eri3_extent_iij[2]);
681 
682  RowMatrixXd eri3_eig_jij = TA::eigen_map(
683  eri3_jij, eri3_extent_jij[0], eri3_extent_jij[1] * eri3_extent_jij[2]);
684 
685  auto M_extentii = M_tile_ii.range().extent_data();
686  auto M_extentij = M_tile_ij.range().extent_data();
687  auto M_extentji = M_tile_ji.range().extent_data();
688  auto M_extentjj = M_tile_jj.range().extent_data();
689 
690  auto rows = M_extentii[0] + M_extentji[0];
691  auto cols = M_extentii[1] + M_extentij[1];
692 
693  RowMatrixXd M_combo(rows, cols);
694 
695  // Write Mii
696  M_combo.block(0, 0, M_extentii[0], M_extentii[1]) =
697  TA::eigen_map(M_tile_ii, M_extentii[0], M_extentii[1]);
698 
699  // Write Mij
700  M_combo.block(0, M_extentii[1], M_extentij[0], M_extentij[1]) =
701  TA::eigen_map(M_tile_ij, M_extentij[0], M_extentij[1]);
702 
703  // Write Mji
704  M_combo.block(M_extentii[0], 0, M_extentji[0], M_extentji[1]) =
705  TA::eigen_map(M_tile_ji, M_extentji[0], M_extentji[1]);
706 
707  // Write Mjj
708  M_combo.block(M_extentii[0], M_extentii[1], M_extentjj[0], M_extentjj[1]) =
709  TA::eigen_map(M_tile_jj, M_extentjj[0], M_extentjj[1]);
710 
711  // Write the two eri3 tiles into one larger fused tile
712  const auto combo_rows = eri3_eig_iij.rows() + eri3_eig_jij.rows();
713  const auto combo_cols = eri3_eig_iij.cols();
714 
715  RowMatrixXd eri_combo(combo_rows, combo_cols);
716 
717  eri_combo.block(0, 0, eri3_eig_iij.rows(), combo_cols) = eri3_eig_iij;
718  eri_combo.block(eri3_eig_iij.rows(), 0, eri3_eig_jij.rows(), combo_cols) =
719  eri3_eig_jij;
720 
721  // Solver for the fitting Coeffs using more robust solver than LLT WARNING
722  // this was done because it turned out that LLT was not sufficent for basis
723  // set exploration and specifically LLT failed for d-aug-cc-pV5Z basis on s66
724  // benzene
725  RowMatrixXd C_total = M_combo.colPivHouseholderQr().solve(eri_combo);
726 
727  if (C->is_local(ord_iij)) {
728  RowMatrixXd C_iij = C_total.topRows(eri3_eig_iij.rows());
729 
730  TA::Tensor<double> out_iij(eri3_iij.range());
731  TA::eigen_map(out_iij, eri3_extent_iij[0],
732  eri3_extent_iij[1] * eri3_extent_iij[2]) = C_iij;
733 
734  C->set(ord_iij, out_iij);
735  }
736  if (C->is_local(ord_jij)) {
737  RowMatrixXd C_jij = C_total.bottomRows(eri3_eig_jij.rows());
738 
739  TA::Tensor<double> out_jij(eri3_jij.range());
740  TA::eigen_map(out_jij, eri3_extent_jij[0],
741  eri3_extent_jij[1] * eri3_extent_jij[2]) = C_jij;
742 
743  C->set(ord_jij, out_jij);
744  }
745 }
