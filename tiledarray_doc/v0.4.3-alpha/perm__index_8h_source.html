46  PermIndex(const Range& range, const Permutation& perm) :
47  weights_(NULL), ndim_(perm.dim())
48  {
49  if(ndim_ > 0) {
50  // Check the input data
51  TA_ASSERT(range.dim() == perm.dim());
52 
53  // Construct the inverse permutation
54  const Permutation inv_perm_ = -perm;
55 
56  // Allocate memory for this object
57  weights_ = static_cast<std::size_t*>(malloc((ndim_ + ndim_) * sizeof(std::size_t)));
58  if(! weights_)
59  throw std::bad_alloc();
60 
61  // Construct restricted pointers to the input data
62  const Permutation::index_type* restrict const inv_perm = & inv_perm_.data().front();
63  const Range::size_type* restrict const range_size = range.size().data();
64  const Range::size_type* restrict const range_weight = range.weight().data();
65 
66  // Construct restricted pointers to the object data
67  std::size_t* restrict const input_weight = weights_;
68  std::size_t* restrict const output_weight = weights_ + ndim_;
69 
70  // Initialize input and output weights
71  std::size_t volume = 1;
72  for(int i = ndim_ - 1; i >= 0; --i) {
73  // Load input data for iteration i.
74  const Permutation::index_type inv_perm_i = inv_perm[i];
75  const Range::size_type weight = range_weight[i];
76  const Range::size_type size = range_size[inv_perm_i];
77 
78  // Store the input and output weights
79  output_weight[inv_perm_i] = volume;
80  volume *= size;
81  input_weight[i] = weight;
82  }
83  }
84  }
