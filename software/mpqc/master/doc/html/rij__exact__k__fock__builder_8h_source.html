26  RIJEXACTKBuilder(array_type const &M, Integral const &eri3,
27  std::shared_ptr<const Basis> bra_basis,
28  std::shared_ptr<const Basis> ket_basis,

31  K_builder_(std::make_shared<Direct4CBuilder>(

44  array_type operator()(array_type const &D, array_type const &C,
45  double target_precision =
46  std::numeric_limits<double>::epsilon()) override {
47  auto &world = D.world();
48  auto j0 = mpqc::fenced_now(world);
49  array_type J;
50  J("m, n") =
51  eri3_("Z, m, n") *
52  (L_inv_("X, Z") * (L_inv_("X, Y") * (eri3_("Y, r, s") * D("r, s"))));
53  // symmetrize to account for permutational symmetry
54  J("m, n") = 0.5 * (J("m, n") + J("n, m"));
55  auto j1 = mpqc::fenced_now(world);
56  time_J_ = mpqc::duration_in_s(j0, j1);
57 
58  // Make K
59  auto k0 = mpqc::fenced_now(world);
60  array_type minus_K = (*K_builder_)(D, C, target_precision);
61  auto k1 = mpqc::fenced_now(world);
62  time_K_ = mpqc::duration_in_s(k0, k1);
63 
64  // Make and return G
65  array_type G;
66  G("mu, nu") = 2 * J("mu, nu") + minus_K("mu, nu");
67 
68  energy_J_ = D("i,j").dot(J("i,j"));
69  energy_K_ = -D("i,j").dot(minus_K("i,j"));
70 
71  return G;
72  }
