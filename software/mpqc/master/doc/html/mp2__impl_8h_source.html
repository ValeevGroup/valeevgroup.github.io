12 template <typename Tile, typename Policy>
13 double compute_mp2(
14  lcao::LCAOFactoryBase<Tile, Policy>& lcao_factory,
15  const std::shared_ptr<const Eigen::VectorXd>& orbital_energy,
16  const std::shared_ptr<const ::mpqc::utility::TRange1Engine>& tr1_engine,
17  bool df) {
18  auto& world = lcao_factory.world();
19  TA::DistArray<Tile, Policy> g_ijab;
20  g_ijab = lcao_factory.compute(df ? L"<i j|G|a b>[df]" : L"<i j|G|a b>");
21  // compute mp2 energy
22  double energy_mp2 =
23  (g_ijab("i,j,a,b") * (2 * g_ijab("i,j,a,b") - g_ijab("i,j,b,a")))
24  .reduce(mbpt::detail::Mp2Energy<Tile>(orbital_energy,
25  tr1_engine->get_occ(),
26  tr1_engine->get_nfrozen()));
27 
28  utility::print_par(world, (df ? "RI-" : ""), "MP2 Energy: ", energy_mp2,
29  "\n");
30  return energy_mp2;
31 }

59 template <typename Tile, typename Policy>
60 void RMP2<Tile, Policy>::evaluate(Energy* result) {
61  auto target_precision = result->target_precision(0);
62  // compute only if never computed, or requested with higher precision than
63  // before
64  if (!this->computed() || computed_precision_ > target_precision) {
65  // compute reference to higher precision than required of correlation energy
66  auto target_ref_precision = target_precision / 100.;
67  auto ref_energy = std::make_shared<Energy>(ref_wfn_, target_ref_precision);
68  ::mpqc::evaluate(*ref_energy, ref_wfn_);
69 
70  this->init_sdref(ref_wfn_, target_ref_precision);
71 
72  // compute
73  mp2_corr_energy_ = compute();
74 
75  this->computed_ = true;
76  this->set_value(result, ref_energy->energy() + mp2_corr_energy_);
77  computed_precision_ = target_precision;
78  }
79 };
