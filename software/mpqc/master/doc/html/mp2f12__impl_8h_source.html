50 template <typename Tile>
51 void RMP2F12<Tile>::evaluate(Energy* result) {
52  auto target_precision = result->target_precision(0);
53  if (!this->computed() || computed_precision_ > target_precision) {
54  // compute reference to higher precision than required of correlation energy
55  auto target_ref_precision = target_precision / 100.;
56  auto ref_energy = std::make_shared<Energy>(ref_wfn_, target_ref_precision);
57  ::mpqc::evaluate(*ref_energy, ref_wfn_);
58 
59  auto& world = this->wfn_world()->world();
60  auto time1 = mpqc::fenced_now(world);
61 
62  // initialize
63  init(target_ref_precision);
64 
65  // compute
66  RowMatrix<double> mp2_eij, f12_eij;
67  std::tie(mp2_eij, f12_eij) = compute();
68 
69  auto nocc = this->trange1_engine()->get_active_occ();
70  ExEnv::out0() << printf(
71  " i0 i1 eij(mp2) eij(f12) eij(mp2-f12) \n"
72  "====== ====== =============== =============== ===============\n");
73  for (int i = 0; i != nocc; ++i)
74  for (int j = i; j != nocc; ++j)
75  ExEnv::out0() << printf("%4d %4d %15.12lf %15.12lf %15.12lf\n", i,
76  j, mp2_eij(i, j), f12_eij(i, j),
77  mp2_eij(i, j) + f12_eij(i, j));
78 
79  mp2_corr_energy_ = mp2_eij.sum();
80  mp2_f12_energy_ = f12_eij.sum();
81  utility::print_par(world, "E_MP2: ", mp2_corr_energy_, "\n");
82  utility::print_par(world, "E_F12: ", mp2_f12_energy_, "\n");
83 
84  if (cabs_singles_) {
85  singles_energy_ = compute_cabs_singles();
86  }
87 
88  utility::print_par(world, "E_S: ", singles_energy_, "\n");
89 
90  double energy = ref_energy->energy() + mp2_corr_energy_ + mp2_f12_energy_ +
91  singles_energy_;
92 
93  this->computed_ = true;
94  this->set_value(result, energy);
95 
96  auto time2 = mpqc::fenced_now(world);
97  auto time = mpqc::duration_in_s(time1, time2);
98  utility::print_par(world, "F12 Time in MP2F12: ", time, " S \n");
99  }

102 template <typename Tile>
103 std::tuple<RowMatrix<double>, RowMatrix<double>> RMP2F12<Tile>::compute() {
104  auto& lcao_factory = to_lcao_factory(this->lcao_factory());
105  auto& world = lcao_factory.world();
106 
107  utility::print_par(world, "\n Computing MP2F12 ", approximation_,
108  " Approximation \n");
109 
110  RowMatrix<double> Eij_MP2, Eij_F12;
111 
112  auto n_active_occ = this->trange1_engine()->get_active_occ();
113  auto n_occ = this->trange1_engine()->get_occ();
114  auto n_frozen = this->trange1_engine()->get_nfrozen();
115 
116  auto ijij_ijji_shape = ijij_ijji_shape_;
117 
118  // compute B term
119  {
120  TA::DistArray<Tile, TA::SparsePolicy> B_ijij_ijji = compute_B();
121  RowMatrix<double> Eij_b =
122  B_ijij_ijji("i1,j1,i2,j2")
123  .reduce(f12::F12PairEnergyReductor<Tile>(
124  f12::CC_ijij_bar, f12::CC_ijji_bar, n_active_occ));
125  utility::print_par(world, "E_B: ", Eij_b.sum(), "\n");
126  Eij_F12 = Eij_b;
127  }
128 
129  // compute X term
130  {
131  TA::DistArray<Tile, TA::SparsePolicy> X_ijij_ijji = compute_X();
132  lcao_factory.purge_operator(L"R2");
133 
134  RowMatrix<double> Eij_x =
135  X_ijij_ijji("i1,j1,i2,j2")
136  .reduce(f12::F12PairEnergyReductor<Tile>(
137  f12::CC_ijij_bar, f12::CC_ijji_bar, n_active_occ));
138  Eij_x *= -1.0;
139  utility::print_par(world, "E_X: ", Eij_x.sum(), "\n");
140  Eij_F12 += Eij_x;
141  }
142 
143  // compute V term
144  {
145  TA::DistArray<Tile, TA::SparsePolicy> V_ijij_ijji = compute_V();
146  // G integral in MO not needed, still need G integral in AO to compute F, K,
147  // hJ
148  lcao_factory.registry().purge_operator(L"G");
149  lcao_factory.purge_operator(L"GR");
150 
151  // contribution from V_ijij_ijji
152  // NB factor of 2 from the Hylleraas functional
153  RowMatrix<double> e_ij =
154  V_ijij_ijji("i1,j1,i2,j2")
155  .reduce(f12::F12PairEnergyReductor<Tile>(
156  2 * f12::C_ijij_bar, 2 * f12::C_ijji_bar, n_active_occ));
157  Eij_F12 += e_ij;
158  utility::print_par(world, "E_V: ", e_ij.sum(), "\n");
159  }
160 
161  TA::DistArray<Tile, TA::SparsePolicy> t2;
162  {
163  TA::DistArray<Tile, TA::SparsePolicy> g_abij;
164  std::tie(t2, g_abij) = compute_T();
165 
166  // compute MP2 energy and pair energies
167  TA::DistArray<Tile, TA::SparsePolicy> TG_ijij_ijji;
168  TG_ijij_ijji("i1,j1,i2,j2") =
169  (t2("a,b,i1,j1") * g_abij("a,b,i2,j2")).set_shape(ijij_ijji_shape);
170  Eij_MP2 =
171  TG_ijij_ijji("i1,j1,i2,j2")
172  .reduce(f12::F12PairEnergyReductor<Tile>(2, -1, n_active_occ));
173  }
174 
175  // compute C term
176  TA::DistArray<Tile, TA::SparsePolicy> C_ijab = compute_C();
177 
178  {
179  utility::print_par(world, "Compute CT \n");
180  TA::DistArray<Tile, TA::SparsePolicy> CT;
181  CT("i1,j1,i2,j2") =
182  (C_ijab("i1,j1,a,b") * t2("a,b,i2,j2")).set_shape(ijij_ijji_shape);
183 
184  // NB factor of 2 from the Hylleraas functional
185  RowMatrix<double> Eij_ct =
186  CT("i1,j1,i2,j2")
187  .reduce(f12::F12PairEnergyReductor<Tile>(
188  2 * f12::C_ijij_bar, 2 * f12::C_ijji_bar, n_active_occ));
189  utility::print_par(world, "E_CT: ", Eij_ct.sum(), "\n");
190  Eij_F12 += Eij_ct;
191  }
192 
193  {
194  utility::print_par(world, "Compute CC Term \n");
195  TA::DistArray<Tile, TA::SparsePolicy> CC;
196  auto C_bar_ijab =
197  f12::convert_C_ijab(C_ijab, n_occ, n_frozen, *(this->orbital_energy()));
198  CC("i1,j1,i2,j2") = (C_ijab("i1,j1,a,b") * C_bar_ijab("i2,j2,a,b"))
199  .set_shape(ijij_ijji_shape);
200 
201  RowMatrix<double> Eij_cc =
202  CC("i1,j1,i2,j2")
203  .reduce(f12::F12PairEnergyReductor<Tile>(
204  f12::CC_ijij_bar, f12::CC_ijji_bar, n_active_occ));
205  utility::print_par(world, "E_CC: ", Eij_cc.sum(), "\n");
206  Eij_F12 += Eij_cc;
207  }
208 
209  return std::make_tuple(Eij_MP2, Eij_F12);
210 }
