33  void init() {
34  const std::size_t rank = ranges_.size();
35 
36  // Indices used to store range start and finish.
37  std::vector<size_type> start;
38  std::vector<size_type> finish;
39  std::vector<size_type> start_element;
40  std::vector<size_type> finish_element;
41 
42  start.reserve(rank);
43  finish.reserve(rank);
44  start_element.reserve(rank);
45  finish_element.reserve(rank);
46 
47  // Find the start and finish of the over all tiles and element ranges.
48  for(unsigned int i = 0; i < rank; ++i) {
49  start.push_back(ranges_[i].tiles().first);
50  finish.push_back(ranges_[i].tiles().second);
51 
52  start_element.push_back(ranges_[i].elements().first);
53  finish_element.push_back(ranges_[i].elements().second);
54  }
55  range_type(start, finish).swap(range_);
56  tile_range_type(start_element, finish_element).swap(element_range_);
57  }

148  template <typename Index>
149  typename std::enable_if<! std::is_integral<Index>::value, tile_range_type>::type
150  make_tile_range(const Index& index) const {
151  const unsigned int rank = range_.rank();
152  TA_ASSERT(index.size() == rank);
153  TA_ASSERT(range_.includes(index));
154  typename tile_range_type::index lower;
155  typename tile_range_type::index upper;
156  lower.reserve(rank);
157  upper.reserve(rank);
158  for(unsigned int i = 0; i < rank; ++i) {
159  lower.push_back(data()[i].tile(index[i]).first);
160  upper.push_back(data()[i].tile(index[i]).second);
161  }
162 
163  return tile_range_type(lower, upper);
164  }
