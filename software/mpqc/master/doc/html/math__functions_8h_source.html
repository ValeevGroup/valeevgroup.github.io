45 inline RowMatrixXd llt_solve(RowMatrixXd& A, RowMatrixXd const& B) {
46  // Call lapack verson of LLT dpotrf_, we have to reverse stuff since lapack
47  // will think all of our matrices are Col Major
48  // RowMatrixXd A_copy = A;

50  const char uplo = 'U'; // Do lower, but need to use U because Row -> Col
51  integer n = A.rows();

53  integer lda = n;
54  integer info;
55 
56  dpotrf_(&uplo, &n, a, &lda, &info, sizeof(char));
57 
58  // Not sure how to avoid this

60  integer nrhs = B_col_major.cols();
61  real8* b = B_col_major.data();
62  integer ldb = n;
63  dpotrs_(&uplo, &n, &nrhs, a, &lda, b, &ldb, &info);
64 
65  RowMatrixXd X = B_col_major;
66  return X;

69 inline void cholesky_inverse(Eigen::MatrixXd& A) {
70  const char uplo = 'L'; // Do lower, but need to use U because Row -> Col
71  integer n = A.rows();
72  // real8* a = A.data();
73  real8* a = A.data();
74  integer lda = n;
75  integer info;
76  dpotrf_(&uplo, &n, a, &lda, &info, sizeof(char));
77 
78  // Cholesky L uses the diag
79  const char diag = 'N';
80 
81  dtrtri_(&uplo, &diag, &n, a, &lda, &info);
82 
83  // Zero the upper right corner of A
84  for (auto r = 0; r < n; ++r) {
85  for (auto c = r + 1; c < n; ++c) {
86  A(r, c) = 0.0;
87  }

116 inline void dgemm(char Atrans, char Btrans, RowMatrixXd const& A,
117  RowMatrixXd const& B, RowMatrixXd& C, real8 alpha = 1.0,
118  real8 beta = 1.0) {
119  // We are just going to let TA do this for u
120  madness::cblas::CBLAS_TRANSPOSE left_op =
121  (Atrans == 'N') ? madness::cblas::NoTrans : madness::cblas::Trans;
122 
123  madness::cblas::CBLAS_TRANSPOSE right_op =
124  (Btrans == 'N') ? madness::cblas::NoTrans : madness::cblas::Trans;
125 
126  TA::math::GemmHelper gh(left_op, right_op, 2, 2, 2);
127  TA::Range Arange({A.rows(), A.cols()});
128  TA::Range Brange({B.rows(), B.cols()});
129 
130  integer m, n, k;
131 
132  gh.compute_matrix_sizes(m, n, k, Arange, Brange);
133 
134  integer lda = (left_op == madness::cblas::NoTrans) ? k : m;
135  integer ldb = (right_op == madness::cblas::NoTrans) ? n : k;
136 
137  TA::math::gemm(left_op, right_op, m, n, k, alpha, A.data(), lda, B.data(),
138  ldb, beta, C.data(), n);
139 }

214 // Return the eigenvectors and eigen values of a symetric matrix
215 inline std::pair<RowMatrixXd, VectorD> SymmEVD(RowMatrixXd& A) {
216  // Eigen::MatrixXd A_copy = A;
217 
218  const char jobz = 'V'; // Compute evecs
219  const char uplo = 'U'; // Lower triangular
220  const char range = 'A'; // compute all evals
221  integer n = A.rows();
222  real8* a = A.data();
223  integer lda = n;
224 
225  // Not used
226  real8 vl = 0;
227  real8 vu = 0;
228 
229  // Not used
230  integer il = 0;
231  integer iu = 0;
232 
233  // Tolerence
234  // According to MKL documentation
235  // If abstol < n *eps*||T||, then n *eps*||T|| is used instead, where eps is
236  // the machine precision, and ||T|| is the 1-norm of the matrix T. The
237  // eigenvalues are computed to an accuracy of eps*||T|| irrespective of
238  // abstol.
239  real8 abstol = 1e-16;
240 
241  // Number of evals found, we don't need this
242  integer m = n;
243 
244  // Evals
245  Eigen::VectorXd evals(A.rows());
246  evals.setZero();
247  real8* w = evals.data();
248 
249  // Evecs
250  Eigen::MatrixXd evecs(A.rows(), A.cols());
251  evecs.setZero();
252  real8* z = evecs.data();
253 
254  integer ldz = n;
255 
256  /* isuppz Doc, I'm not sure how to use this
257  * Array, size at least 2 *max(1, m).
258  * The support of the eigenvectors in z, i.e., the indices indicating the
259  * nonzero elements in z. The i-th eigenvector is nonzero only in elements
260  * isuppz[2i - 2] through isuppz[2i - 1]. Referenced only if eigenvectors are
261  * needed (jobz = 'V') and all eigenvalues are needed, that is, range = 'A' or
262  * range = 'I' and il = 1 and iu = n. */
263  EigenVector<int> eval_map(2 * n);
264  eval_map.setZero();
265 
266  real8 work;
267  integer lwork = -1; // Ask LAPACK how much space we need.
268 
269  integer iwork;
270  integer ilwork = -1; // Ask LAPACK how much space we need.
271 
272  integer info;
273 
274  // On the first go round ask for how much space to allocate for work
275  dsyevr_(&jobz, &range, &uplo, &n, a, &lda, &vl, &vu, &il, &iu, &abstol, &m, w,
276  z, &ldz, eval_map.data(), &work, &lwork, &iwork, &ilwork, &info);
277 
278  // Save the space requirements
279  lwork = work;
280  ilwork = iwork;
281 
282  // Allocate work space
283  std::unique_ptr<real8[]> real_work{new real8[lwork]};
284  std::unique_ptr<integer[]> real_iwork{new integer[ilwork]};
285 
286  // On the first go round ask for how much space to allocate for work
287  dsyevr_(&jobz, &range, &uplo, &n, a, &lda, &vl, &vu, &il, &iu, &abstol, &m, w,
288  z, &ldz, eval_map.data(), real_work.get(), &lwork, real_iwork.get(),
289  &ilwork, &info);
290 
291  return std::make_pair(RowMatrixXd(evecs), evals);
292 }
