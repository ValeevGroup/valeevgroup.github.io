90 template <typename Tile, typename Policy>
91 double evaluate_energy_pao(lcao::LCAOFactoryBase<Tile, Policy>& lcao_factory,
92  gaussian::AOFactoryBase<Tile, Policy>& ao_factory,
93  std::shared_ptr<Eigen::VectorXd> orbital_energy,
94  int num_laplace_points) {
95  auto& world = lcao_factory.world();
96 
97  TA::DistArray<Tile, Policy> Cv =
98  lcao_factory.orbital_registry().retrieve(OrbitalIndex(L"a")).coefs();
99 
100  TA::DistArray<Tile, Policy> Ci =
101  lcao_factory.orbital_registry().retrieve(OrbitalIndex(L"i")).coefs();
102 
103  // Active Space Localization
104  auto basis = *ao_factory.basis_registry()->retrieve(OrbitalIndex(L"λ"));
105  const auto bs_array = utility::make_array(basis, basis);
106  auto multi_pool = gaussian::make_engine_pool(
107  libint2::Operator::emultipole1, utility::make_array_of_refs(basis));
108  auto r_xyz =
109  gaussian::xyz_integrals<Tile, Policy>(world, multi_pool, bs_array);
110 
111  auto U = mpqc::scf::FosterBoysLocalization{}(Ci, r_xyz);
112  TA::DistArray<Tile, Policy> Ci_loc;
113  Ci_loc("mu, i") = Ci("mu, k") * U("k, i");
114  Ci_loc.truncate();
115 
116  auto Fij = lcao_factory.compute(L"<i|F|j>[df]");
117  Fij("i,j") = U("k,i") * Fij("k,l") * U("l,j");
118 
119  auto eri3 = ao_factory.compute(L"(Λ | G| μ ν)");
120  auto M = ao_factory.compute(L"(Λ| G| Κ)");
121  auto Linv = array_ops::cholesky_inverse(M);
122 
123  auto CadfC = ao_factory.compute(L"(Λ | CADF| μ ν)");
124  decltype(CadfC) Cdf;
125  auto L = array_ops::cholesky(M);
126  Cdf("X, mu, nu") = L("Y, X") * CadfC("Y, mu, nu");
127 
128  //
129  // PAO space set up
130  //
131  auto Sao = ao_factory.compute(L"(μ | ν)");
132  TA::DistArray<Tile, Policy> Q, Spao, Qbar, Pao;
133 
134  // Q from Phys. Chem. Chem. Phys., 2008, 10, 3430–3439
135  Q("a,nu") = Cv("mu, a") * Sao("mu, nu");
136  Q.truncate();
137  Pao("mu, nu") = Cv("mu, a") * Q("a, nu");
138  Pao.truncate();
139  Spao("mu, nu") = Pao("x, mu") * Sao("x,y") * Pao("y,nu");
140  Spao.truncate();
141  ExEnv::out0() << "Calling Pseudoinverse\n";
142  auto Vpao = array_ops::pseudoinverse(Spao, true);
143  Qbar("mu, nu") = Sao("mu, k") * Vpao("k, nu");
144  Qbar.truncate();
145 
146  TA::DistArray<Tile, Policy> g, t, Z, Zc, B;
147  Z("Y, i, mu") = eri3("Y, mu, nu") * Ci_loc("nu, i");
148  Zc("Y, i, mu") = Cdf("Y, mu, nu") * Ci_loc("nu, i");
149  Z("X, i, mu_pao") = Z("X, i, mu") * Pao("mu, mu_pao");
150  Zc("Y, i, mu_pao") = Zc("Y, i, mu") * Pao("mu, mu_pao");
151  Z.truncate();
152  Zc.truncate();
153 
154  auto z_storage = ::mpqc::detail::array_storage(Z);
155  auto zc_storage = ::mpqc::detail::array_storage(Zc);
156  ExEnv::out0() << indent << indent << "Zc Storage " << zc_storage[1] << ", "
157  << 100.0 - 100.0 * double(zc_storage[1]) / double(zc_storage[0])
158  << "% sparsity"
159  << "\n";
160  ExEnv::out0() << indent << indent << "Z Storage " << z_storage[1] << ", "
161  << 100.0 - 100.0 * double(z_storage[1]) / double(z_storage[0])
162  << "% sparsity"
163  << "\n";
164 
165  if (z_storage[1] < 1.0) {
166  ::mpqc::detail::write_array_to_csv(Z, "Z_mat.csv");
167  ::mpqc::detail::write_array_to_csv(Zc, "Zc_mat.csv");
168  }
169 
170  auto nocc =
171  lcao_factory.orbital_registry().retrieve(OrbitalIndex(L"m")).rank();
172 
173  auto nactive =
174  lcao_factory.orbital_registry().retrieve(OrbitalIndex(L"i")).rank();
175 
176  auto nfrozen = nocc - nactive;
177 
178  auto const& energies = *orbital_energy;
179  const auto gap2 = 2 * (energies[nocc] - energies[nocc - 1]);
180 
181  Eigen::VectorXd w, x;
182  math::gauss_legendre(num_laplace_points, w, x);
183 
184  TA::DistArray<Tile, Policy> Di, Dv;
185  auto Iocc = lcao_factory.compute(L"<i|I|j>");
186  auto Ivir = lcao_factory.compute(L"<a|I|b>");
187 
188  auto Fao = ao_factory.compute(L"<μ|F|ν>[df]");
189  decltype(Fao) Fpao;
190  Fpao("a,b") = Pao("mu, a") * Fao("mu, nu") * Pao("nu, b");
191  Fpao.truncate();
192 
193  auto t0_outer = mpqc::fenced_now(world);
194  for (auto i = 0; i < num_laplace_points; ++i) {
195  auto t0 = mpqc::fenced_now(world);
196  TA::foreach_inplace(Iocc, [&](TA::TensorD& t) {
197  auto const& range = t.range();
198  auto const lo = range.lobound_data();
199  auto const up = range.upbound_data();
200  for (auto n = lo[0]; n != up[0]; ++n) {
201  auto idx = n + nfrozen; // Account for activeness of space
202  const auto energy = energies[idx];
203  t(n, n) = std::pow(x[i], -energy / (gap2)-0.25);
204  }
205 
206  return t.norm();
207  });
208 
209  Di("ip,i") = U("l,ip") * Iocc("l,k") * U("k,i");
210  Di.truncate();
211 
212  TA::foreach_inplace(Ivir, [&](TA::TensorD& t) {
213  auto const& range = t.range();
214  auto const lo = range.lobound_data();
215  auto const up = range.upbound_data();
216  for (auto n = lo[0]; n != up[0]; ++n) {
217  auto idx = n + nocc; // Account for activeness of space
218  const auto energy = energies[idx];
219  t(n, n) = std::pow(x[i], energy / (gap2)-0.25);
220  }
221 
222  return t.norm();
223  });
224  Dv("mu,nu") = Cv("mu,l") * Ivir("l, k") * Cv("nu,k");
225  Dv("mu_pao,nu_pao") =
226  Qbar("mu, mu_pao") * Dv("mu, nu") * Qbar("nu, nu_pao");
227  Dv.truncate();
228 
229  B("X, nu, ip") = Zc("X, ip, nup") * Dv("nup, nu");
230  B.truncate();
231  B("X, i, nu") = B("X, nu, ip") * Di("ip, i");
232  B.truncate();
233 
234  if (i == 0) {
235  t("i, nu, j, sig") = (-w[i] / gap2) * B("X, i, nu") * B("X, j, sig");
236  } else {
237  t("i, nu, j, sig") += (-w[i] / gap2) * B("X, i, nu") * B("X, j, sig");
238  }
239  t.truncate();
240  auto t1 = mpqc::fenced_now(world);
241  auto t_time_iter = mpqc::duration_in_s(t0, t1);
242  ExEnv::out0() << indent << indent << indent << "Time for T[" << i + 1
243  << "]: " << t_time_iter << ", ";
244  auto t_storage = ::mpqc::detail::array_storage(t);
245  ExEnv::out0() << "t Storage " << t_storage[1] << ", "
246  << 100.0 - 100.0 * double(t_storage[1]) / double(t_storage[0])
247  << "% sparsity"
248  << "\n";
249  }
250  auto t1_outer = mpqc::fenced_now(world);
251  auto t_time = mpqc::duration_in_s(t0_outer, t1_outer);
252  ExEnv::out0() << indent << "Total Time for T: " << t_time << std::endl;
253 
254  auto k0 = mpqc::fenced_now(world);
255  decltype(g) K;
256  g("i, nu, j, sig") = Z("X, i, nu") * Z("X, j, sig");
257  g.truncate();
258  K("i,a,j,b") = 2 * g("i,a,j,b") - g("i,b,j,a");
259  K.truncate();
260  auto k1 = mpqc::fenced_now(world);
261  auto k_time = mpqc::duration_in_s(k0, k1);
262  ExEnv::out0() << indent << indent << "Time for K: " << k_time << std::endl;
263  auto k_storage = ::mpqc::detail::array_storage(K);
264  ExEnv::out0() << indent << indent << "K Storage " << k_storage[1] << ", "
265  << 100.0 - 100.0 * double(k_storage[1]) / double(k_storage[0])
266  << "% sparsity"
267  << "\n";
268 
269  double energy = K("i,nu,j,sig").dot(t("i,nu,j,sig"));
270  world.gop.fence();
271  ExEnv::out0() << indent << indent
272  << "MP2 PAO Laplace Correlation Energy = " << energy << "\n";
273 
274  return energy;
275 }

278 template <typename Tile, typename Policy>
279 void RPAOLaplaceMP2<Tile, Policy>::evaluate(Energy* result) {
280  auto target_precision = result->target_precision(0);
281 
282  if (!this->computed() || computed_precision_ > target_precision) {
283  // compute reference to higher precision than required of correlation energy
284  auto target_ref_precision = target_precision / 100.;
285  auto ref_energy = std::make_shared<Energy>(ref_wfn_, target_ref_precision);
286  ::mpqc::evaluate(*ref_energy, ref_wfn_);
287 
288  this->init_sdref(ref_wfn_, target_ref_precision);
289 
290  mp2_corr_energy_ = detail::evaluate_energy_pao(
291  this->lcao_factory(), this->ao_factory(),
292  make_diagonal_fpq(this->lcao_factory(), this->ao_factory()),
293  num_laplace_points_);
294 
295  this->computed_ = true;
296  this->set_value(result, ref_energy->energy() + mp2_corr_energy_.get());
297  computed_precision_ = target_precision;
298  }
299 }
