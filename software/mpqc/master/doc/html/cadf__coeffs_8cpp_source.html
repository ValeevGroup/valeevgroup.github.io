12 gaussian::Basis by_center_basis(gaussian::Basis const &in) {
13  auto const shells = in.flattened_shells();
14  auto it = shells.begin();
15  auto end = shells.end();
16 
17  std::vector<gaussian::ShellVec> out;
18  while (it != end) {
19  auto center = it->O;
20  gaussian::ShellVec atom;
21  while (it != end && it->O == center) {
22  atom.push_back(*it);
23  ++it;
24  }
25 
26  out.emplace_back(std::move(atom));
27  }
28 
29  return gaussian::Basis(std::move(out));
30 }

33  auto rank = t.range().rank();

35 
36  std::ofstream out_file(file_name);
37

40  auto ord = 0ul;

42  for (auto b = 0ul; b < ext[2]; ++b, ++ord) {
43  out_file << t(X, a, b);
44  if (ord < (ext[1] * ext[2] - 1)) {
45  out_file << ", ";

47  }
48  }
49  out_file << "\n";
50  }
51  } else if (rank == 4) {

53  for (auto b = 0ul; b < ext[1]; ++b) {
54  auto ord = 0ul;

60  }

90 TA::SparseShape<float> cadf_shape(madness::World &world,
91  TA::TiledRange const &trange) {
92  auto &tiles = trange.tiles_range();
93  TA::Tensor<float> norms(tiles, 0.0);
94  auto natoms = tiles.extent_data()[0]; // trange is tiled by atom
95 
96  for (auto i = 0ul; i < natoms; ++i) {
97  for (auto j = 0ul; j < natoms; ++j) {
98  norms(i, i, j) = std::numeric_limits<float>::max();
99  norms(j, i, j) = std::numeric_limits<float>::max();
100  }
101  }
102 
103  return TA::SparseShape<float>(world, norms, trange);

106 TA::SparseShape<float> cadf_shape_cluster(
107  TA::DistArray<TA::Tensor<double>, TA::SparsePolicy> const &C_atom,
108  TA::TiledRange const &trange,
109  std::unordered_map<int64_t, std::vector<int64_t>> &c2a // cluster to atom
110 ) {
111  auto const &tiles = trange.tiles_range();
112  auto norms = TA::Tensor<float>(tiles, 0.0);
113 
114  auto const &trange_atom = C_atom.trange();
115  const auto volume_atom = trange_atom.tiles_range().volume();
116 
117  // Loop over all tiles of C_atom to figure which tiles of C_cluster they got
118  // to
119  for (auto ord_atom = 0ul; ord_atom < volume_atom; ++ord_atom) {
120  if (C_atom.is_zero(ord_atom)) {
121  continue;
122  }
123  TA::Range range_atom = trange_atom.make_tile_range(ord_atom);
124  auto lo_atom = range_atom.lobound_data();
125  auto ord_cluster = tiles.ordinal(trange.element_to_tile(lo_atom));
126 
127  norms[ord_cluster] = std::numeric_limits<float>::max();
128 
129  // Write the atom to the cluster
130  auto it = c2a.find(int64_t(ord_cluster));
131  if (it != c2a.end()) {
132  it->second.push_back(ord_atom);
133  } else {
134  c2a[int64_t(ord_cluster)] = std::vector<int64_t>{int64_t(ord_atom)};
135  }
136  }
137 
138  // No world needed every node computed entire shape
139  return TA::SparseShape<float>(norms, trange);
140 }

171 TA::DistArray<TA::Tensor<double>, TA::SparsePolicy> cadf_by_atom_coeffs(
172  madness::World &world, gaussian::Basis const &by_cluster_obs,
173  gaussian::Basis const &by_cluster_dfbs) {
174  auto obs = detail::by_center_basis(by_cluster_obs);
175  auto dfbs = detail::by_center_basis(by_cluster_dfbs);
176 
177  const auto dfbs_ref_array = utility::make_array_of_refs(dfbs, dfbs);
178  const auto dfbs_array = gaussian::BasisVector{{dfbs, dfbs}};
179  auto eng2 = make_engine_pool(libint2::Operator::coulomb, dfbs_ref_array,
180  libint2::BraKet::xs_xs);
181  auto M = gaussian::sparse_integrals(world, eng2, dfbs_array);
182 
183  auto screener = detail::cadf_by_atom_screener(world, obs, dfbs, 1e-12);
184  auto eng3 = make_engine_pool(libint2::Operator::coulomb,
185  utility::make_array_of_refs(dfbs, obs, obs),
186  libint2::BraKet::xs_xx);
187 
188  auto eri3_norms = [&](TA::Tensor<float> const &in) {
189  const auto thresh = screener->skip_threshold();
190 
191  auto const &range = in.range();
192  auto ext = range.extent_data();
193 
194  std::vector<std::pair<std::array<int, 3>, float>> norms;
195  norms.reserve(ext[1] * ext[2]);
196 
197  using idx_type = std::array<int, 3>;
198  auto val_max = std::numeric_limits<float>::max();
199  for (auto a = 0; a < int(ext[1]); ++a) {
200  for (auto b = 0; b < int(ext[2]); ++b) {
201  auto in_val = std::max(in(a, a, b), in(b, a, b));
202 
203  if (in_val >= thresh) {
204  norms.emplace_back(std::make_pair(idx_type{{a, a, b}}, val_max));
205  norms.emplace_back(std::make_pair(idx_type{{b, a, b}}, val_max));
206  }
207  }
208  }
209 
210  return norms;
211  };
212 
213  auto three_array = std::vector<gaussian::Basis>{dfbs, obs, obs};
214 
215  auto trange = gaussian::detail::create_trange(three_array);
216  auto pmap =
217  TA::SparsePolicy::default_pmap(world, trange.tiles_range().volume());
218 
219  bool replicate = true;
220  auto norms =
221  eri3_norms(screener->norm_estimate(world, three_array, *pmap, replicate));
222 
223  auto eri3 = direct_sparse_integrals(world, eng3, three_array, norms,
224  std::move(screener));
225 
226  return cadf_by_atom_array(M, eri3, detail::cadf_trange(obs, dfbs));
227 }

229 TA::DistArray<TA::Tensor<double>, TA::SparsePolicy> cadf_by_atom_coeffs(
230  const TA::DistArray<TA::Tensor<double>, TA::SparsePolicy> &M,
231  gaussian::Basis const &by_cluster_bs0,
232  gaussian::Basis const &by_cluster_bs1,

235  Vector3i const &lattice_range_df, Vector3i const &lattice_center0,

237  auto &world = M.world();
238 
239  auto bs0 = detail::by_center_basis(by_cluster_bs0);
240  auto bs1 = detail::by_center_basis(by_cluster_bs1);
241  auto dfbs = detail::by_center_basis(by_cluster_dfbs);
242 
243  auto screener = detail::cadf_by_atom_screener(world, bs0, bs1, dfbs, 1e-12);
244 
245  auto eng3 = make_engine_pool(libint2::Operator::coulomb,
246  utility::make_array_of_refs(dfbs, bs0, bs1),

252  auto eri3_norms = [&](TA::Tensor<float> const &in) {
253  const auto thresh = screener->skip_threshold();
254 
255  auto const &range = in.range();
256  auto ext = range.extent_data();
257 
258  std::vector<std::pair<std::array<int, 3>, float>> norms;
259  norms.reserve(ext[1] * ext[2]);
260 
261  using idx_type = std::array<int, 3>;
262  auto val_max = std::numeric_limits<float>::max();
263  for (auto a = 0; a < int(ext[1]); ++a) {
264  const auto uc0_ord = a / natoms_per_uc;
265  const Vector3i uc0_3D =
266  direct_3D_idx(uc0_ord, lattice_range0) + lattice_center0;
267  const auto uc0_ord_in_df =
268  direct_ord_idx(uc0_3D - lattice_center_df, lattice_range_df);
269  const auto a_in_df = uc0_ord_in_df * natoms_per_uc + a % natoms_per_uc;
270 
271  for (auto b = 0; b < int(ext[2]); ++b) {
272  const auto uc1_ord = b / natoms_per_uc;
273  const Vector3i uc1_3D =
274  direct_3D_idx(uc1_ord, lattice_range1) + lattice_center1;
275  const auto uc1_ord_in_df =
276  direct_ord_idx(uc1_3D - lattice_center_df, lattice_range_df);
277  const auto b_in_df = uc1_ord_in_df * natoms_per_uc + b % natoms_per_uc;
278 
279  auto in_val = std::max(in(a_in_df, a, b), in(b_in_df, a, b));
280 
281  if (in_val >= thresh) {
282  norms.emplace_back(
283  std::make_pair(idx_type{{int(a_in_df), a, b}}, val_max));
284  norms.emplace_back(
285  std::make_pair(idx_type{{int(b_in_df), a, b}}, val_max));
286  }
287  }
288  }
289 
290  return norms;
291  };
