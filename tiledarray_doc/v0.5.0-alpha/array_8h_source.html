80  static void lazy_deleter(const impl_type* const pimpl) {
81  if(pimpl) {
82  if(madness::initialized()) {
83  World& world = pimpl->get_world();
84  const madness::uniqueidT id = pimpl->id();
85  cleanup_counter_++;
86 
87  try {
88  world.gop.lazy_sync(id, [pimpl]() {
89  delete pimpl;
90  Array_::cleanup_counter_--;
91  });
92  }
93  catch(madness::MadnessException& e) {
94  fprintf(stderr, "!! ERROR TiledArray: madness::MadnessException thrown in Array::lazy_deleter().\n"
95  "%s\n"
96  "!! ERROR TiledArray: The exception has been absorbed.\n"
97  "!! ERROR TiledArray: rank=%i\n", e.what(), world.rank());
98 
99  cleanup_counter_--;
100  delete pimpl;
101  }
102  catch(std::exception& e) {
103  fprintf(stderr, "!! ERROR TiledArray: std::exception thrown in Array::lazy_deleter().\n"
104  "%s\n"
105  "!! ERROR TiledArray: The exception has been absorbed.\n"
106  "!! ERROR TiledArray: rank=%i\n", e.what(), world.rank());
107 
108  cleanup_counter_--;
109  delete pimpl;
110  }
111  catch(...) {
112  fprintf(stderr, "!! ERROR TiledArray: An unknown exception was thrown in Array::lazy_deleter().\n"
113  "!! ERROR TiledArray: The exception has been absorbed.\n"
114  "!! ERROR TiledArray: rank=%i\n", world.rank());
115 
116  cleanup_counter_--;
117  delete pimpl;
118  }
119  } else {
120  delete pimpl;
121  }
122  }
123  }

131  static std::shared_ptr<impl_type>
132  init(World& world, const trange_type& trange, const shape_type& shape,
133  std::shared_ptr<pmap_interface> pmap)
134  {
135  // User level validation of input
136 
137  // Validate the tiled range
138  TA_USER_ASSERT(trange.tiles().rank() == DIM,
139  "Array::Array() -- The dimension of tiled range is not equal to the array.");
140 
141  if(! pmap) {
142  // Construct a default process map
143  pmap = Policy::default_pmap(world, trange.tiles().volume());
144  } else {
145  // Validate the process map
146  TA_USER_ASSERT(pmap->size() == trange.tiles().volume(),
147  "Array::Array() -- The size of the process map is not equal to the number of tiles in the TiledRange object.");
148  TA_USER_ASSERT(pmap->rank() == typename pmap_interface::size_type(world.rank()),
149  "Array::Array() -- The rank of the process map is not equal to that of the world object.");
150  TA_USER_ASSERT(pmap->procs() == typename pmap_interface::size_type(world.size()),
151  "Array::Array() -- The number of processes in the process map is not equal to that of the world object.");
152  }
153 
154  // Validate the shape
155  TA_USER_ASSERT(! shape.empty(),
156  "Array::Array() -- The shape is not initialized.");
157  TA_USER_ASSERT(shape.validate(trange.tiles()),
158  "Array::Array() -- The range of the shape is not equal to the tiles range.");
159 
160  return std::shared_ptr<impl_type>(new impl_type(world, trange, shape, pmap), lazy_deleter);
161  }

535  void make_replicated() {
536  check_pimpl();
537  if((! pimpl_->pmap()->is_replicated()) && (get_world().size() > 1)) {
538  // Construct a replicated array
539  std::shared_ptr<pmap_interface> pmap(new detail::ReplicatedPmap(get_world(), size()));
540  Array_ result = Array_(get_world(), trange(), get_shape(), pmap);
541 
542  // Create the replicator object that will do an all-to-all broadcast of
543  // the local tile data.
544  std::shared_ptr<detail::Replicator<Array_> > replicator(
545  new detail::Replicator<Array_>(*this, result));
546 
547  // Put the replicator pointer in the deferred cleanup object so it will
548  // be deleted at the end of the next fence.
549  TA_ASSERT(replicator.unique()); // Required for deferred_cleanup
550  madness::detail::deferred_cleanup(get_world(), replicator);
