17 // reduce matrix 1/(ei + ej - ea - eb)
18 template <typename Tile, typename Policy>
19 void d_abij_inplace(TA::Array<double, 4, Tile, Policy> &abij,
20  const EigenVector<typename Tile::numeric_type> &ens,
21  std::size_t n_occ, std::size_t n_frozen,
22  typename Tile::numeric_type shift = 0.0) {
23  auto convert = [&ens, n_occ, n_frozen, shift](Tile &result_tile) {
24 
25  // compute index
26  const auto a0 = result_tile.range().lobound()[0];
27  const auto an = result_tile.range().upbound()[0];
28  const auto b0 = result_tile.range().lobound()[1];
29  const auto bn = result_tile.range().upbound()[1];
30  const auto i0 = result_tile.range().lobound()[2];
31  const auto in = result_tile.range().upbound()[2];
32  const auto j0 = result_tile.range().lobound()[3];
33  const auto jn = result_tile.range().upbound()[3];
34 
35  auto tile_idx = 0;
36  typename Tile::numeric_type norm = 0.0;
37  for (auto a = a0; a < an; ++a) {
38  const auto e_a = shift - ens[a + n_occ];
39  for (auto b = b0; b < bn; ++b) {
40  const auto e_ab = e_a - ens[b + n_occ];
41  for (auto i = i0; i < in; ++i) {
42  const auto e_abi = e_ab + ens[i + n_frozen];
43  for (auto j = j0; j < jn; ++j, ++tile_idx) {
44  const auto e_abij = e_abi + ens[j + n_frozen];
45  const auto old = result_tile[tile_idx];
46  const auto result_abij = old / (e_abij);
47  norm += result_abij * result_abij;
48  result_tile[tile_idx] = result_abij;
49  }
50  }
51  }
52  }
53  return std::sqrt(norm);
54  };
55 
56  TA::foreach_inplace(abij, convert);
57  abij.world().gop.fence();
58 }

165  Tile, typename std::enable_if<std::is_same<Policy, TA::SparsePolicy>::value,
166  TA::SparsePolicy>::type>
167 create_d_ai(madness::World &world, const TA::TiledRange &trange,
168  const EigenVector<typename Tile::numeric_type> ens,
169  std::size_t n_occ, std::size_t n_frozen) {
170  typedef typename TA::DistArray<Tile, Policy>::range_type range_type;
171 
172  auto make_tile = [&ens, n_occ, n_frozen](const range_type &range, std::size_t ord,
173  Tile *out_tile, TA::TensorF *norms) {
174 
175  auto result_tile = Tile(range);
176  const auto a0 = result_tile.range().lobound()[0];
177  const auto an = result_tile.range().upbound()[0];
178  const auto i0 = result_tile.range().lobound()[1];
179  const auto in = result_tile.range().upbound()[1];
180 
181  auto ai = 0;
182  const typename Tile::value_type identity = 1.0;
183  for (auto a = a0; a < an; ++a) {
184  const auto e_a = ens[a + n_occ];
185  for (auto i = i0; i < in; ++i, ++ai) {
186  const auto e_i = ens[i + n_frozen];
187  const auto e_ia = e_i - e_a;
188  const auto result_ai = identity / (e_ia);
189  result_tile[ai] = result_ai;
190  }
191  }
192  const auto tile_volume = result_tile.range().volume();
193  const auto tile_norm = result_tile.norm();
194  bool save_norm =
195  tile_norm >= tile_volume * TA::SparseShape<float>::threshold();
196  if (save_norm) {
197  *out_tile = result_tile;
198  (*norms)[ord] = tile_norm;
199  }
200  };
201 
202  const auto tvolume = trange.tiles_range().volume();
203  std::vector<Tile> tiles(tvolume);
204  TA::TensorF tile_norms(trange.tiles_range(), 0.0);
205  auto pmap = TA::SparsePolicy::default_pmap(world, tvolume);
206 
207  for (auto const ord : *pmap) {
208  world.taskq.add(make_tile, trange.make_tile_range(ord), ord, &tiles[ord],
209  &tile_norms);
210  }
211 
212  world.gop.fence();
213 
214  TA::SparseShape<float> shape(world, tile_norms, trange);
215  TA::DistArray<Tile, Policy> result(world, trange, shape, pmap);
216 
217  for (auto const ord : *pmap) {
218  if (result.is_local(ord) && !result.is_zero(ord)) {
219  auto &tile = tiles[ord];
220  assert(!tile.empty());
221  result.set(ord, tile);
222  }
223  }
224 
225  world.gop.fence();
226  result.truncate();
227  return result;
228 }
