106 Exception::Exception(const char *description,
107  const char *file,
108  int line,
109  const char *exception_type) MPQC__NOEXCEPT:
110  detail::Exception(description, file, line),
111  exception_type_(exception_type),
112  backtrace_("=mpqcbacktrace=: ")
113 {
114  try {
115  elaboration_ = std::make_unique<std::ostringstream>();
116  if (exception_type_) {
117  elaborate() << "exception: " << exception_type_
118  << std::endl;
119  }
120  if (detail::Exception::description()) {
121  elaborate() << "description: " << detail::Exception::description()
122  << std::endl;
123  }
124  if (detail::Exception::file()) {
125  elaborate() << "location: " << detail::Exception::file() << ":" << line
126  << std::endl;
127  }
128  const size_t nframes_to_skip = 1;
129  elaborate() << "backtrace:" << std::endl << backtrace_.str(nframes_to_skip);
130  }
131  catch (...) {
132  // info in the elaboration is incomplete
133  // only provide the basic description
134  elaboration_ = 0;
135  }
136 }

159 const char*
160 Exception::what() const MPQC__NOEXCEPT
161 {
162  try {
163  if (elaboration_) {
164  std::string elab(elaboration_->str());
165  elaboration_c_str_ = std::make_unique<char[]>(1+elab.size());
166  for (std::string::size_type i=0; i<elab.size(); i++) elaboration_c_str_[i] = elab[i];
167  elaboration_c_str_[elab.size()] = '\0';
168  return elaboration_c_str_.get();
169  }
170  }
171  catch (...) {
172  // Ignore the exception and return the next available string.
173  }
174  return Exception::what();
175 }
