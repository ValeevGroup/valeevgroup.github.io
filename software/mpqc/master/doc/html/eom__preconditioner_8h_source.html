28  EEPred() : eps_o_(), eps_v_() {}

33  virtual void operator()(
34  const EigenVector<element_type> &e,
35  std::vector<::mpqc::cc::TPack<Array>> &guess) const override {
36  std::size_t n_roots = e.size();
37  TA_ASSERT(n_roots == guess.size());
38  for (std::size_t i = 0; i < n_roots; i++) {
39  compute(e[i], guess[i]);
40  }

43  void compute(const element_type &e, ::mpqc::cc::TPack<Array> &guess) const {
44  const auto &eps_v = this->eps_v_;
45  const auto &eps_o = this->eps_o_;
46 
47  auto task1 = [&eps_v, &eps_o, e](Tile &result_tile) {
48  const auto &range = result_tile.range();
49  float norm = 0.0;
50  for (const auto &i : range) {
51  const auto result = result_tile[i] / (e + eps_o[i[1]] - eps_v[i[0]]);
52  result_tile[i] = result;
53  norm += result * result;
54  }
55  return std::sqrt(norm);
56  };
57 
58  auto task2 = [&eps_v, &eps_o, e](Tile &result_tile) {
59  const auto &range = result_tile.range();
60  float norm = 0.0;
61  for (const auto &i : range) {
62  const auto result = result_tile[i] / (e - eps_v[i[0]] - eps_v[i[1]] +
63  eps_o[i[2]] + eps_o[i[3]]);
64  result_tile[i] = result;
65  norm += result * result;
66  }
67  return std::sqrt(norm);
68  };
69 
70  TA::foreach_inplace(guess[0], task1);
71  TA::foreach_inplace(guess[1], task2);
72 
73  guess[0].world().gop.fence();
74  }
75 
76  private:
78  EigenVector<element_type> eps_o_;
80  EigenVector<element_type> eps_v_;
81 };

266  virtual typename Tile::numeric_type norm(
267  const ::mpqc::cc::TPack<Array> &t1t2) const override {
268  auto r1_reblock = detail::reblock_t1(t1t2[0], reblock_i_, reblock_a_);
269  auto r2_reblock = detail::reblock_t2(t1t2[1], reblock_i_, reblock_a_);

272  detail::R2SquaredNormReductionOp<Array> op2(pnos_);

274  return sqrt(r1_reblock("a,i").reduce(op1).get() +
275  r2_reblock("a,b,i,j").reduce(op2).get()) /
276  (size(r1_reblock) + size(r2_reblock));

285  void compute(const typename Tile::numeric_type &e,
286  ::mpqc::cc::TPack<Array> &guess) const {
287  // reblock
288  guess[0] = detail::reblock_t1(guess[0], reblock_i_, reblock_a_);
289  guess[1] = detail::reblock_t2(guess[1], reblock_i_, reblock_a_);
290 
291  // pno update
292  guess[0] =
293  detail::pno_jacobi_update_t1(guess[0], eps_o_, F_osv_diag_, osvs_, -e);
294  guess[1] =
295  detail::pno_jacobi_update_t2(guess[1], eps_o_, F_pno_diag_, pnos_, -e);
296 
297  // unblock
298  guess[0] = detail::unblock_t1(guess[0], reblock_i_, reblock_a_);
299  guess[1] = detail::unblock_t2(guess[1], reblock_i_, reblock_a_);
300 
301  // change the sign
302  guess[0]("a,i") = -guess[0]("a,i");
303  guess[1]("a,b,i,j") = -guess[1]("a,b,i,j");
304  }

308  std::size_t n_unocc = T2.trange().dim(0).extent();
309  std::size_t n_occ = T2.trange().dim(2).extent();
310  auto &world = T2.world();

313

315  for (std::size_t i = 0; i <= n_occ; ++i) {
316  occ_blocks.push_back(i);
317  }
318  const TA::TiledRange1 occ_col =
319  TA::TiledRange1(occ_blocks.begin(), occ_blocks.end());
320 
321  reblock_i_ = mpqc::math::create_diagonal_array_from_eigen<
322  Tile, TA::detail::policy_t<Array>>(world, occ_row, occ_col, 1.0);
323 
324  const TA::TiledRange1 uocc_row = T2.trange().dim(0);

327  const TA::TiledRange1 uocc_col =
328  TA::TiledRange1(uocc_blocks.begin(), uocc_blocks.end());
329 
330  reblock_a_ = mpqc::math::create_diagonal_array_from_eigen<
331  Tile, TA::detail::policy_t<Array>>(world, uocc_row, uocc_col, 1.0);
332  }

365  void transfer_pnos() {
366 
367  const auto nocc_act = size(nosvs_);
368  const auto me = this->get_world().rank();
369  for(auto i = 0; i < nocc_act; ++i) {
370  for (auto j = i + 1; j < nocc_act; ++j) {
371  const auto ij = i * nocc_act + j;
372  if (ij_pmap_->owner(ij) == me) {
373  auto ji_owner = ij_pmap_->owner(j * nocc_act + i);
374  WorldObject_::task(ji_owner, &PNOEEPred::copy_pnoij, i, j, pnos_[ij], npnos_[ij], F_pno_diag_[ij]);
375  }
376  }
377  }

379  this->get_world().gop.fence();
380  }
