17 class OrthTraceResettingPurifier {
18  public:

21  OrthTraceResettingPurifier(ArrayType const &sqrt_inv)
22  : sqrt_inv_(sqrt_inv), I(create_diagonal_matrix(sqrt_inv, 1.0)) {}
23 
24  double trace(ArrayType const &A) const { return A("i,j").trace(); }
25 
26  ArrayType operator()(ArrayType const &Fao, std::size_t occ) const {
27  // F = Z F_{ao} Z^{T}
28  ArrayType F;
29  F("i,j") = sqrt_inv_("i,k") * Fao("k,l") * sqrt_inv_("l,j");
30 
31  auto eig_pair = eval_guess(F);
32  auto emax = eig_pair[1];
33  auto emin = eig_pair[0];
34  auto scale = 1.0 / (emax - emin);
35 
36  // D_{o} = \frac{emax I - F}{emax - emin}
37  ArrayType D;
38  D("i,j") = scale * (emax * I("i,j") - F("i,j"));
39  auto tr = trace(D);
40 
41  occte = occ / 2;
42  auto iter = 1;
43  ArrayType D2;
44  auto error = std::abs(tr - occ);
45  while (error >= 1e-13 && iter <= 100) {
46  // Compute D2
47  D2("i,j") = D("i,k") * D("k,j");
48  if (tr > occ) {
49  D = D2;
50  } else {
51  D("i,j") = 2 * D("i,j") - D2("i,j");
52  }
53  D.truncate();
54  tr = trace(D);
55  error = std::abs(tr - occ);
56  ++iter;
57  }
58  if (iter >= 100) {
59  if (D.world().rank() == 0) {
60  std::cout << "Purification took " << iter
61  << " iterations with error " << error
62  << " this is likely unconverged." << std::endl;
63  }
64  }
65 
66  auto d_norm = D("i,j").norm().get();
67  if (d_norm != d_norm) { // check for nan
68  std::cout << "D norm was nan" << std::endl;
69  throw;
70  }
71 
72  // D_{ao} = Z^{T} D Z
73  D("i,j") = sqrt_inv_("i,k") * D("k,l") * sqrt_inv_("l,j");
74  // Symmetrize D
75  // D("i,j") = 0.5 * D("i,j") + D("j,i");
76  D.truncate(); // necessary since norms blow up otherwise
77  return D;
78  }

83  const ArrayType I;
