66  void init_range_data(const Index& start, const Index& finish) {
67  // Construct temp pointers
68  size_type* restrict const start_ptr = data_;
69  size_type* restrict const finish_ptr = start_ptr + rank_;
70  size_type* restrict const size_ptr = finish_ptr + rank_;
71  size_type* restrict const weight_ptr = size_ptr + rank_;
72 
73  // Set the volume seed
74  volume_ = 1ul;
75 
76  // Compute range data
77  for(int i = int(rank_) - 1; i >= 0; --i) {
78  // Check input dimensions
79  TA_ASSERT(start[i] >= 0ul);
80  TA_ASSERT(start[i] < finish[i]);
81

84  const size_type finish_i = finish[i];

86 
87  start_ptr[i] = start_i;
88  finish_ptr[i] = finish_i;
89  size_ptr[i] = size_i;
90  weight_ptr[i] = volume_;
91  volume_ *= size_i;
92  }

138  void init_range_data(const Permutation& perm,
139  const size_type* restrict const other_start,
140  const size_type* restrict const other_finish)
141  {
142  // Create temporary pointers to this range data
143  size_type* restrict const start_ptr = data_;
144  size_type* restrict const finish_ptr = start_ptr + rank_;
145  size_type* restrict const size_ptr = finish_ptr + rank_;
146  size_type* restrict const weight_ptr = size_ptr + rank_;
147 
148  // Copy the permuted start, finish, and size into this range.
149  for(unsigned int i = 0u; i < rank_; ++i) {
150  const size_type perm_i = perm[i];
151 
152  // Get the start, finish, and size from other for rank i.
153  const size_type other_start_i = other_start[i];
154  const size_type other_finish_i = other_finish[i];
155  const size_type other_size_i = other_finish_i - other_start_i;
156 
157  // Store the permuted start, finish and size
158  start_ptr[perm_i] = other_start_i;
159  finish_ptr[perm_i] = other_finish_i;
160  size_ptr[perm_i] = other_size_i;
161  }
162 
163  // Recompute weight and volume
164  volume_ = 1ul;
165  for(int i = int(rank_) - 1; i >= 0; --i) {
166  weight_ptr[i] = volume_;
167  volume_ *= size_ptr[i];
168  }
169  }

576  index idx(size_type index) const {
577  // Check that index is contained by range.
578  TA_ASSERT(includes(index));
579 
580  // Construct result coordinate index object and allocate its memory.
581  Range_::index result(rank_, 0);
582 
583  // Get pointers to the data
584  size_type * restrict const result_data = & result.front();
585  size_type const * restrict const weight_ptr = data_ + rank_ + rank_ + rank_;

587 
588  // Compute the coordinate index of o in range.
589  for(unsigned int i = 0u; i < rank_; ++i) {
590  const size_type weight_i = weight_ptr[i];
591  const size_type start_i = start_ptr[i];
