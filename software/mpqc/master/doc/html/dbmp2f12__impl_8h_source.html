23 template <typename Tile>
24 void RIDBRMP2F12<Tile>::evaluate(Energy* result) {
25  if (this->computed()) {
26  auto& world = this->wfn_world()->world();
27 
28  // compute MP2F12
29  RIRMP2F12<Tile>::evaluate(result);
30  double mp2f12_energy = this->get_value(result).derivs(0)[0];
31 
32  auto time0 = mpqc::fenced_now(world);
33 
34  double energy = 0.0;
35  if (mp2_method_ == "none") {
36  energy = mp2f12_energy;
37  }
38  else if(mp2_method_ == "redo") {
39  double new_mp2 = compute_new_mp2();
40  energy = new_mp2 + this->f12_energy() + this->cabs_singles_energy();
41  }
42  else if(mp2_method_ == "update"){
43  double new_mp2 = compute_new_mp2();
44  energy = mp2f12_energy - this->mp2_corr_energy()+ new_mp2;
45  }
46 
47  this->computed_ = true;
48  this->set_value(result, energy);
49 
50  auto time1 = mpqc::fenced_now(world);
51  auto time = mpqc::duration_in_s(time0, time1);
52  utility::print_par(world, "Total DBMP2F12 Time: ", time, " S \n");
53  }

76 template <typename Tile>
77 double RIDBRMP2F12<Tile>::compute_new_mp2() {
78  auto& world = this->wfn_world()->world();
79  auto time1 = mpqc::fenced_now(world);
80 
81  auto& lcao_factory = this->lcao_factory();
82 
83  double new_mp2 = 0.0;
85  if (mp2_method_ == "redo") {
86  auto fock = lcao_factory.ao_factory().compute(L"<Α|F|Α>[df]");
87 
88  // clear registry
89  this->obsolete();
90 
91  // change basis
92  auto vbs = lcao_factory.ao_factory().orbital_basis_registry().retrieve(
93  OrbitalIndex(L"Α"));
94  lcao_factory.ao_factory().orbital_basis_registry().remove(
95  OrbitalIndex(L"κ"));
96  lcao_factory.ao_factory().orbital_basis_registry().insert(
97  OrbitalIndex(L"κ"), vbs);
98  lcao_factory.ao_factory().orbital_basis_registry().remove(
99  OrbitalIndex(L"Α"));
100 
101  lcao::RIRMP2<Tile, TA::SparsePolicy> mp2(kv_);
102 
103  auto ref = mp2.refwfn();
104 
105  std::shared_ptr<RHF<Tile,TA::SparsePolicy>> rhf = std::dynamic_pointer_cast<RHF<Tile,TA::SparsePolicy>>(ref);
106  MPQC_ASSERT( rhf != nullptr);
107  if(rhf == nullptr){
108  throw std::runtime_error("\nCan not cast ref_wfn to RHF class! \n");
109  }
110 
111  rhf->set_fock(fock);
112 
113  //TODO fix this
114 // new_mp2 = mp2.compute();
115  new_mp2 = 0.0;
116  } else if (mp2_method_ == "update") {
117  std::size_t n = this->trange1_engine()->get_all();
118  std::size_t v = this->trange1_engine()->get_vir();
119  std::size_t o = this->trange1_engine()->get_occ();
120  std::size_t f = this->trange1_engine()->get_nfrozen();
121  // get fock matrix
122  RowMatrixXd fock;
123  {
124  auto f_ii = lcao_factory.compute(L"<m|F|m>[df]");
125  auto f_aa = lcao_factory.compute(L"<a|F|a>[df]");
126  auto f_ia = lcao_factory.compute(L"<m|F|a>[df]");
127 
128  RowMatrixXd f_ii_eigen = math::array_to_eigen(f_ii);
129  RowMatrixXd f_aa_eigen = math::array_to_eigen(f_aa);
130  RowMatrixXd f_ia_eigen = math::array_to_eigen(f_ia);
131 
132  // form the new fock matrix
133  fock = RowMatrixXd::Zero(n, n);
134  fock.block(0, 0, o, o) << f_ii_eigen;
135  fock.block(o, o, v, v) << f_aa_eigen;
136  fock.block(0, o, o, v) << f_ia_eigen;
137  fock.block(o, 0, v, o) << f_ia_eigen.transpose();
138  }
139 
140  // solve new orbitals
141  TA::DistArray<Tile,TA::SparsePolicy> C_occ_ta, C_corr_occ_ta, C_vir_ta;
142  Eigen::VectorXd ens_all;
143  {
144  // get the basis map
145  Eigen::RowVectorXi basis_map;
146  {
147  const auto vbs =
148  *this->wfn_world()->basis_registry()->retrieve(OrbitalIndex(L"Α"));
149  const auto obs =
150  *this->wfn_world()->basis_registry()->retrieve(OrbitalIndex(L"κ"));
151  basis_map = gaussian::sub_basis_map(vbs, obs);
152 
153 // std::cout << "Basis Map" << basis_map << std::endl;
154  }
155 
156  // update old C_occ to vbs
157  RowMatrixXd old_coeffs = RowMatrixXd::Zero(n, n);
158  {
159  auto old_occ =
160  lcao_factory.orbital_space().retrieve(OrbitalIndex(L"m")).coefs();
161  RowMatrixXd old_occ_eigen = math::array_to_eigen(old_occ);
162 
163  // convert occ to vbs
164  RowMatrixXd occ_eigen = RowMatrixXd::Zero(n, o);
165  for (std::size_t i = 0; i < n; i++) {
166  if (basis_map[i] != 0) {
167  occ_eigen.row(i) = old_occ_eigen.row(basis_map[i] - 1);
168  }
169  }
170 
171  auto old_vir =
172  lcao_factory.orbital_space().retrieve(OrbitalIndex(L"a")).coefs();
173  RowMatrixXd old_vir_eigen = math::array_to_eigen(old_vir);
174 
175  old_coeffs.block(0, 0, n, o) << occ_eigen;
176  old_coeffs.block(0, o, n, v) << old_vir_eigen;
177  }
178 
179  // eigen solve on node 0 + broadcast
180  RowMatrixXd C_all;
181  if (world.rank() == 0) {
182  Eigen::SelfAdjointEigenSolver<RowMatrixXd> es(fock);
183  ens_all = es.eigenvalues();
184  C_all = es.eigenvectors();
185  C_all = old_coeffs * C_all;
186  }
187  world.gop.broadcast_serializable(ens_all, 0);
188  world.gop.broadcast_serializable(C_all, 0);
189 
190  // update the coefficient
191 
192  RowMatrixXd C_occ = C_all.block(0, 0, n, o);
193  RowMatrixXd C_corr_occ = C_all.block(0, f, n, o - f);
194  RowMatrixXd C_vir = C_all.rightCols(n - o);
195 
196  auto tr_vbs = this->wfn_world()
197  ->basis_registry()
198  ->retrieve(OrbitalIndex(L"Α"))
199  ->create_trange1();
200  auto tr_corr_occ = this->trange1_engine()->get_active_occ_tr1();
201  auto tr_occ = this->trange1_engine()->get_occ_tr1();
202  auto tr_vir = this->trange1_engine()->get_vir_tr1();
203 
204  // convert to TA
205  C_occ_ta =
206  math::eigen_to_array<Tile,TA::SparsePolicy>(world, C_occ, tr_vbs, tr_occ);
207  C_corr_occ_ta = math::eigen_to_array<Tile,TA::SparsePolicy>(
208  world, C_corr_occ, tr_vbs, tr_corr_occ);
209  C_vir_ta =
210  math::eigen_to_array<Tile,TA::SparsePolicy>(world, C_vir, tr_vbs, tr_vir);
211  }
212 
213  lcao_factory.orbital_space().clear();
214  lcao_factory.registry().purge(world);
215  lcao_factory.ao_factory().registry().purge_operator(world,L"F");
216  lcao_factory.ao_factory().registry().purge_operator(world,L"J");
217  lcao_factory.ao_factory().registry().purge_operator(world,L"K");
218 
219  auto& orbital_registry = lcao_factory.orbital_space();
220 
221  // insert to registry
222  using OrbitalSpaceTArray =
223  OrbitalSpace<TA::DistArray<Tile, TA::SparsePolicy>>;
224  auto occ_space =
225  OrbitalSpaceTArray(OrbitalIndex(L"m"), OrbitalIndex(L"Α"), C_occ_ta);
226  orbital_registry.add(occ_space);
227 
228  auto corr_occ_space = OrbitalSpaceTArray(OrbitalIndex(L"i"),
229  OrbitalIndex(L"Α"), C_corr_occ_ta);
230  orbital_registry.add(corr_occ_space);
231 
232  auto vir_space =
233  OrbitalSpaceTArray(OrbitalIndex(L"a"), OrbitalIndex(L"Α"), C_vir_ta);
234  orbital_registry.add(vir_space);
235 
236  auto g_ijab = lcao_factory.compute(L"<i j|G|a b>[df]");
237 
238  new_mp2 =
239  (g_ijab("i,j,a,b") * (2 * g_ijab("i,j,a,b") - g_ijab("i,j,b,a")))
240  .reduce(mbpt::detail::Mp2Energy<Tile>(
241  std::make_shared<Eigen::VectorXd>(ens_all),
242 // this->orbital_energy_,
243  this->trange1_engine_->get_occ(), this->trange1_engine_->get_nfrozen()));
244 
245  utility::print_par(world, "New MP2: ", new_mp2, "\n");
246 
247 // double d_scf = ens_all.segment(0,o).sum() - this->orbital_energy_->segment(0,o).sum();
248 // utility::print_par(world, "SCF Correction: ", d_scf, "\n");
249 
250 // auto f_ia = lcao_factory.compute(L"<m|F|a>[df]");
251 // d_scf += 2*f_ia("i,a").reduce(mbpt::detail::ScfCorrection<Tile>(std::make_shared<Eigen::VectorXd>(ens_all), o));
252 
253 // utility::print_par(world, "SCF Correction: ", d_scf, "\n");
254 
255 // new_mp2 += d_scf;
256 
257  } else {
258  utility::print_par(
259  world, "\n Warning! Not valid method to update MP2 energy! Skip! \n");
260  }
261 
262  auto time2 = mpqc::fenced_now(world);
263  double time = mpqc::duration_in_s(time1, time2);
264  utility::print_par(world, "Total New MP2 Time: ", time, " S \n");
265 
266  return new_mp2;
267 }

309  auto F_ma = this->lcao_factory().compute(L"<m|F|a>[df]");
310  es -= 2 *

312  this->orbital_energy(), this->trange1_engine()->get_occ()));
