69  template <typename Index>
70  void init_range_data(const Index& lower_bound, const Index& upper_bound) {
71  // Construct temp pointers
72  size_type* restrict const lower = data_;
73  size_type* restrict const upper = lower + rank_;
74  size_type* restrict const extent = upper + rank_;
75  size_type* restrict const stride = extent + rank_;
76 
77  // Set the volume seed
78  volume_ = 1ul;
79  offset_ = 0ul;
80 
81  // Compute range data
82  for(int i = int(rank_) - 1; i >= 0; --i) {
83  // Check input dimensions
84  TA_ASSERT(lower_bound[i] >= 0ul);
85  TA_ASSERT(lower_bound[i] < upper_bound[i]);
86 
87  // Compute data for element i of lower, upper, and extent
88  const size_type lower_bound_i = lower_bound[i];
89  const size_type upper_bound_i = upper_bound[i];
90  const size_type extent_i = upper_bound_i - lower_bound_i;
91 
92  lower[i] = lower_bound_i;
93  upper[i] = upper_bound_i;
94  extent[i] = extent_i;
95  stride[i] = volume_;
96  offset_ += lower_bound_i * volume_;
97  volume_ *= extent_i;
98  }
99  }

146  void init_range_data(const Permutation& perm,
147  const size_type* restrict const other_lower_bound,
148  const size_type* restrict const other_upper_bound)
149  {
150  // Create temporary pointers to this range data
151  size_type* restrict const lower = data_;
152  size_type* restrict const upper = lower + rank_;
153  size_type* restrict const extent = upper + rank_;
154  size_type* restrict const stride = extent + rank_;
155 
156  // Copy the permuted lower, upper, and extent into this range.
157  for(unsigned int i = 0u; i < rank_; ++i) {
158  const size_type perm_i = perm[i];
159 
160  // Get the lower bound, upper bound, and extent from other for rank i.
161  const size_type other_lower_bound_i = other_lower_bound[i];
162  const size_type other_upper_bound_i = other_upper_bound[i];
163  const size_type other_extent_i = other_upper_bound_i - other_lower_bound_i;
164 
165  // Store the permuted lower bound, upper bound, and extent
166  lower[perm_i] = other_lower_bound_i;
167  upper[perm_i] = other_upper_bound_i;
168  extent[perm_i] = other_extent_i;
169  }
170 
171  // Recompute stride, offset, and volume
172  volume_ = 1ul;
173  offset_ = 0ul;
174  for(int i = int(rank_) - 1; i >= 0; --i) {
175  const size_type lower_i = lower[i];
176  stride[i] = volume_;
177  offset_ += lower_i * volume_;
178  volume_ *= extent[i];
179  }
180  }

610  Range_::index result(rank_, 0);
