125  template <typename EigMat>
126  EigMat operator()(EigMat &C, const EigMat &ao_x, const EigMat &ao_y,
127  const EigMat &ao_z, size_t ncols_of_C_to_skip = 0) const {
128  EigMat C_loc =
129  C.block(0, ncols_of_C_to_skip, C.rows(), C.cols() - ncols_of_C_to_skip);
130 
131  EigMat U_loc = EigMat::Identity(C_loc.cols(), C_loc.cols());
132  auto converged = fb_jacobi_sweeps(C_loc, U_loc, {ao_x, ao_y, ao_z},
133  jacobi_convergence_threshold_, jacobi_max_iter_);
134  if (!converged) {

138  throw AlgorithmException(oss.str().c_str(), __FILE__, __LINE__);
139  }

141  EigMat U = EigMat::Identity(C.cols(), C.cols());
142  U.block(ncols_of_C_to_skip, ncols_of_C_to_skip, U_loc.rows(),

176  template <typename EigMat>
177  EigMat operator()(EigMat &C, const EigMat &S,
178  size_t ncols_of_C_to_skip = 0) const {
179  EigMat C_loc =
180  C.block(0, ncols_of_C_to_skip, C.rows(), C.cols() - ncols_of_C_to_skip);
181 
182  Eigen::SelfAdjointEigenSolver<EigMat> eig_solver(S);
183  EigMat X = eig_solver.operatorInverseSqrt();
184  EigMat Psi_tr = ((X.inverse()) * C_loc).transpose();
185 
186  Eigen::ColPivHouseholderQR<RowMatrixXd> qr(Psi_tr);
187  EigMat Q = qr.householderQ().setLength(qr.nonzeroPivots());
188 
189  EigMat U = Eigen::MatrixXd::Identity(C.cols(), C.cols());
190  U.block(ncols_of_C_to_skip, ncols_of_C_to_skip, Q.rows(), Q.cols()) = Q;
191  return U;
192  }
