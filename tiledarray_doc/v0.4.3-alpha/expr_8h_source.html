150  A make_array(World& world, const std::shared_ptr<typename A::pmap_interface>& pmap,
151  const VariableList& target_vars) const
152  {
153  typedef madness::Range<typename engine_type::pmap_interface::const_iterator> range_type;
154 
155  // Construct the expression engine
156  engine_type engine(derived());
157  engine.init(world, pmap, target_vars);
158 
159  // Create the distributed evaluator from this expression
160  typename engine_type::dist_eval_type dist_eval = engine.make_dist_eval();
161  dist_eval.eval();
162 
163  // Create the result array
164  A result(dist_eval.get_world(), dist_eval.trange(),
165  dist_eval.shape(), dist_eval.pmap());
166 
167  // Move the data from disteval into the result array
168  int chuck_size = std::max<int>(1u,
169  dist_eval.pmap()->local_size() / (madness::ThreadPool::size() + 1));
170  world.taskq.for_each(range_type(dist_eval.pmap()->begin(), dist_eval.pmap()->end(), chuck_size),
171  EvalTiles<A, typename engine_type::dist_eval_type>(result, dist_eval)).get();
172 
173  // Wait for child expressions of dist_eval
174  dist_eval.wait();
175 
176  return result;
177  }

195  void eval_to(TsrExpr<A>& tsr) const {

197  // Get the target world.
198  World& world = (tsr.array().is_initialized() ?

246  template <typename Op>
247  Future<typename Op::result_type>
248  reduce(const Op& op, World& world = World::get_default()) const {
249  // Typedefs
250  typedef madness::TaggedKey<madness::uniqueidT, ExpressionReduceTag> key_type;
251  typedef TiledArray::math::UnaryReduceWrapper<typename engine_type::value_type,
252  Op> reduction_op_type;
253 
254  // Construct the expression engine
255  engine_type engine(derived());
256  engine.init(world, std::shared_ptr<typename engine_type::pmap_interface>(),
257  VariableList());
258 
259  // Create the distributed evaluator from this expression
260  typename engine_type::dist_eval_type dist_eval = engine.make_dist_eval();
261  dist_eval.eval();
262 
263  // Create a local reduction task
264  reduction_op_type wrapped_op(op);
265  TiledArray::detail::ReduceTask<reduction_op_type> reduce_task(world, wrapped_op);
266 
267  // Move the data from dist_eval into the local reduction task
268  typename engine_type::dist_eval_type::pmap_interface::const_iterator it =
269  dist_eval.pmap()->begin();
270  const typename engine_type::dist_eval_type::pmap_interface::const_iterator end =
271  dist_eval.pmap()->end();
272  for(; it != end; ++it)
273  if(! dist_eval.is_zero(*it))
274  reduce_task.add(dist_eval.get(*it));
275 
276  // All reduce the result of the expression
277  return world.gop.all_reduce(key_type(dist_eval.id()), reduce_task.submit(), op);
278  }

280  template <typename D, typename Op>
281  Future<typename Op::result_type>
282  reduce(const Expr<D>& right_expr, const Op& op,
283  World& world = World::get_default()) const
284  {
285  // Typedefs
286  typedef madness::TaggedKey<madness::uniqueidT, ExpressionReduceTag> key_type;
287  typedef TiledArray::math::BinaryReduceWrapper<typename engine_type::value_type,
288  typename D::engine_type::value_type, Op> reduction_op_type;
289 
290  // Evaluate this expression
291  engine_type left_engine(derived());
292  left_engine.init(world, std::shared_ptr<typename engine_type::pmap_interface>(),
293  VariableList());
294 
295  // Create the distributed evaluator for this expression
296  typename engine_type::dist_eval_type left_dist_eval =
297  left_engine.make_dist_eval();
298  left_dist_eval.eval();
299 
300  // Evaluate the right-hand expression
301  typename D::engine_type right_engine(right_expr.derived());
302  right_engine.init(world, left_engine.pmap(), left_engine.vars());
303 
304  // Create the distributed evaluator for the right-hand expression
305  typename D::engine_type::dist_eval_type right_dist_eval =
306  right_engine.make_dist_eval();
307  right_dist_eval.eval();
308 
309 #ifndef NDEBUG
310  if(left_dist_eval.trange() != right_dist_eval.trange()) {
311  if(World::get_default().rank() == 0) {
312  TA_USER_ERROR_MESSAGE( \
313  "The TiledRanges of the left- and right-hand arguments the binary reduction are not equal:" \
314  << "\n left = " << left_dist_eval.trange() \
315  << "\n right = " << right_dist_eval.trange() );
316  }
317 
318  TA_EXCEPTION("The TiledRange objects of a binary expression are not equal.");
319  }
320 #endif // NDEBUG
321 
322  // Create a local reduction task
323  reduction_op_type wrapped_op(op);
324  TiledArray::detail::ReducePairTask<reduction_op_type>
325  local_reduce_task(world, wrapped_op);
326 
327  // Move the data from dist_eval into the local reduction task
328  typename engine_type::dist_eval_type::pmap_interface::const_iterator it =
329  left_dist_eval.pmap()->begin();
330  const typename engine_type::dist_eval_type::pmap_interface::const_iterator end =
331  left_dist_eval.pmap()->end();
332  for(; it != end; ++it) {
333  const typename engine_type::size_type index = *it;
334  const bool left_not_zero = !left_dist_eval.is_zero(index);
335  const bool right_not_zero = !right_dist_eval.is_zero(index);
336 
337  if(left_not_zero && right_not_zero) {
338  local_reduce_task.add(left_dist_eval.get(index), right_dist_eval.get(index));
339  } else {
340  if(left_not_zero) left_dist_eval.get(index);
341  if(right_not_zero) right_dist_eval.get(index);
342  }
343  }
344 
345  return world.gop.all_reduce(key_type(left_dist_eval.id()),
346  local_reduce_task.submit(), op);
347  }
