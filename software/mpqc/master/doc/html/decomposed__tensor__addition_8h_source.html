12 DecomposedTensor<T> add_order2(DecomposedTensor<T> const &l,
13  DecomposedTensor<T> const &r, T factor) {
14  if (l.ndecomp() >= 2) {
15  if (r.ndecomp() >= 2) {
16  auto const &l_extent = l.tensor(0).range().extent();
17  auto const &r_extent = r.tensor(1).range().extent();
18  const auto l_rank = l.rank();
19  const auto r_rank = r.rank();
20  const auto out_rank = l_rank + r_rank;
21  TA::Range l_new_range{l_extent[0], out_rank};
22  TA::Range r_new_range{out_rank, r_extent[1]};
23 
24  TA::Tensor<T> L(l_new_range);
25  TA::Tensor<T> R(r_new_range);
26 
27  // copy into left
28  const auto l_vol = L.range().volume();
29  auto data = L.data();
30  auto ll_data = l.tensor(0).data();
31  auto rl_data = r.tensor(0).data();
32  // Access data in stride
33  for (auto i = 0ul; i < l_vol; i += out_rank) {
34  std::copy(ll_data, ll_data + l_rank, data + i);
35  ll_data += l_rank;
36  std::copy(rl_data, rl_data + r_rank, data + i + l_rank);
37  rl_data += r_rank;
38  }
39  L.scale_to(factor);
40 
41  // copy into R
42  data = R.data();
43  auto lr_data = l.tensor(1).data();
44  auto rr_data = r.tensor(1).data();
45  auto old_l_vol = l.tensor(1).range().volume();
46  auto old_r_vol = r.tensor(1).range().volume();
47  // R is copied by rows so we don't have to loop.
48  std::copy(lr_data, lr_data + old_l_vol, data);
49  std::copy(rr_data, rr_data + old_r_vol, data + old_l_vol);
50  return DecomposedTensor<T>(l.cut(), std::move(L), std::move(R));
51  } else {
52  auto left = combine(l);
53  return DecomposedTensor<T>(l.cut(), left.add_to(r.tensor(0), factor));
54  }
55  } else if (l.ndecomp() == 1) {
56  if (r.ndecomp() == 1) {
57  auto out = l.tensor(0).add(r.tensor(0), factor);
58  return DecomposedTensor<T>(l.cut(), std::move(out));
59  } else {
60  auto right = combine(r);
61  return DecomposedTensor<T>(l.cut(), l.tensor(0).add(right, factor));
62  }
63  }
64  assert(false);
65  return DecomposedTensor<T>(l.cut());
66 }

81 template <typename T>
82 DecomposedTensor<T> add(DecomposedTensor<T> const &l,
83  DecomposedTensor<T> const &r, const T factor) {
84  if (l.orders().back() == 2) {
85  return add_order2(l, r, factor);
86  }
87  if (l.ndecomp() >= 2) { // Low *
88  if (r.ndecomp() >= 2) { // Low Low
89  auto const &l_extent = l.tensor(0).range().extent();
90  auto const &r_extent = r.tensor(1).range().extent();
91  const auto l_rank = l.rank();
92  const auto r_rank = r.rank();
93  const auto out_rank = l_rank + r_rank;
94  TA::Range l_new_range{l_extent[0], out_rank};
95  TA::Range r_new_range{out_rank, r_extent[1], r_extent[2]};
96 
97  TA::Tensor<T> L(l_new_range);
98  TA::Tensor<T> R(r_new_range);
99 
100  // copy into left
101  const auto l_vol = L.range().volume();
102  auto data = L.data();
103  auto ll_data = l.tensor(0).data();
104  auto rl_clone = r.tensor(0).clone();
105  rl_clone.scale(factor);
106  auto rl_data = rl_clone.data();
107  // Access data in stride
108  for (auto i = 0ul; i < l_vol; i += out_rank) {
109  std::copy(ll_data, ll_data + l_rank, data + i);
110  ll_data += l_rank;
111  std::copy(rl_data, rl_data + r_rank, data + i + l_rank);
112  rl_data += r_rank;
113  }
114  // add the factor
115  L.scale_to(factor);
116 
117  // copy into R
118  data = R.data();
119  auto lr_data = l.tensor(1).data();
120  auto rr_data = r.tensor(1).data();
121  auto old_l_vol = l.tensor(1).range().volume();
122  auto old_r_vol = r.tensor(1).range().volume();
123  // R is copied by rows so we don't have to loop.
124  std::copy(lr_data, lr_data + old_l_vol, data);
125  std::copy(rr_data, rr_data + old_r_vol, data + old_l_vol);
126  return DecomposedTensor<T>(l.cut(), std::move(L), std::move(R));
127  } else { // Low Full
128  // Contraction is (X, r) * (r, mu nu) = (X, mu, nu)
129  TA::math::GemmHelper gh(NoT, NoT, 3, 2, 3);
130 
131  auto out = r.tensor(0).clone();
132 
133  out.gemm(l.tensor(0), l.tensor(1), 1.0, gh);
134  return DecomposedTensor<T>(l.cut(), std::move(out));
135  }
136  } else if (l.ndecomp() == 1) { // Full *
137  if (r.ndecomp() == 1) { // Full Full
138  auto out = l.tensor(0).add(r.tensor(0), factor);
139  return DecomposedTensor<T>(l.cut(), std::move(out));
140  } else { // Full Low
141  // Contraction is (X, r) * (r, mu nu) = (X, mu, nu)
142  TA::math::GemmHelper gh(NoT, NoT, 3, 2, 3);
143 
144  auto out = l.tensor(0).clone();
145 
146  out.gemm(r.tensor(0), r.tensor(1), 1.0, gh);
147  return DecomposedTensor<T>(l.cut(), std::move(out));
148  }
149  }
150  assert(false);
151  return DecomposedTensor<T>(l.cut());
152 }

173 DecomposedTensor<T> &add_to_order2(DecomposedTensor<T> &l,
174  DecomposedTensor<T> const &r) {
175  if (l.ndecomp() == 1) {
176  if (r.ndecomp() == 1) {
177  l.tensor(0).add_to(r.tensor(0));
178  } else {
179  constexpr auto NoT = madness::cblas::CBLAS_TRANSPOSE::NoTrans;
180  auto gh = TA::math::GemmHelper(NoT, NoT, 2, 2, 2);
181  l.tensor(0).gemm(r.tensor(0), r.tensor(1), 1.0, gh);
182  }
183 
184  } else {
185  if (r.ndecomp() == 1) {
186  constexpr auto NoT = madness::cblas::CBLAS_TRANSPOSE::NoTrans;
187  auto gh = TA::math::GemmHelper(NoT, NoT, 2, 2, 2);
188  auto temp = r.tensor(0).clone();
189  temp.gemm(l.tensor(0), l.tensor(1), 1.0, gh);
190  l = DecomposedTensor<T>(l.cut(), std::move(temp));
191  } else {
192  l = add(l, r);
193  auto const &l_left_extent = l.tensor(0).range().extent();
194  auto const &l_right_extent = l.tensor(1).range().extent();
195  auto out_dim = l.rank();
196  const auto full_rank = std::min(l_left_extent[0], l_right_extent[1]);
197 
198  if (out_dim >= full_rank / 6) {
199  recompress(l);
200  out_dim = l.rank();
201  }
202 
203  if (out_dim > full_rank / 2) {
204  l = DecomposedTensor<T>(l.cut(), combine(l));
205  }
206  }
207  }
