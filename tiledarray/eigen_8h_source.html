161  template <typename T, typename A, typename Derived>
162  inline void eigen_submatrix_to_tensor(const Eigen::MatrixBase<Derived>& matrix, Tensor<T, A>& tensor) {
163  typedef typename Tensor<T, A>::size_type size_type;
164  TA_ASSERT((tensor.range().dim() == 2u) || (tensor.range().dim() == 1u));
165 
166  if(tensor.range().dim() == 2u) {
167  TA_ASSERT(tensor.range().finish()[0] <= size_type(matrix.rows()));
168  TA_ASSERT(tensor.range().finish()[1] <= size_type(matrix.cols()));
169 
170  // Copy matrix
171  eigen_map(tensor, tensor.range().size()[0], tensor.range().size()[1]) =
172  matrix.block(tensor.range().start()[0], tensor.range().start()[1],
173  tensor.range().size()[0], tensor.range().size()[1]);
174  } else {
175  // Check that matrix is a vector.
176  TA_ASSERT((matrix.rows() == 1) || (matrix.cols() == 1));
177 
178  if(matrix.rows() == 1) {
179  TA_ASSERT(tensor.range().finish()[0] <= size_type(matrix.cols()));
180 
181  // Copy the row vector to tensor
182  eigen_map(tensor, 1, tensor.range().size()[0]) =
183  matrix.block(0, tensor.range().start()[0], 1, tensor.range().size()[0]);
184  } else {
185  TA_ASSERT(tensor.range().finish()[0] <= size_type(matrix.rows()));
186 
187  // Copy the column vector to tensor
188  eigen_map(tensor, tensor.range().size()[0], 1) =
189  matrix.block(tensor.range().start()[0], 0, tensor.range().size()[0], 1);
190  }
191  }
192  }

207  template <typename T, typename A, typename Derived>
208  inline void tensor_to_eigen_submatrix(const Tensor<T, A>& tensor, Eigen::MatrixBase<Derived>& matrix) {
209  typedef typename Tensor<T, A>::size_type size_type;
210  TA_ASSERT((tensor.range().dim() == 2u) || (tensor.range().dim() == 1u));
211 
212  if(tensor.range().dim() == 2) {
213  TA_ASSERT(tensor.range().finish()[0] <= size_type(matrix.rows()));
214  TA_ASSERT(tensor.range().finish()[1] <= size_type(matrix.cols()));
215 
216  // Copy tensor into matrix
217  matrix.block(tensor.range().start()[0], tensor.range().start()[1],
218  tensor.range().size()[0], tensor.range().size()[1]) =
219  eigen_map(tensor, tensor.range().size()[0], tensor.range().size()[1]);
220  } else {
221  TA_ASSERT((matrix.rows() == 1) || (matrix.cols() == 1));
222 
223  if(matrix.rows() == 1) {
224  TA_ASSERT(tensor.range().finish()[0] <= size_type(matrix.cols()));
225 
226  // Copy tensor into row vector
227  matrix.block(0, tensor.range().start()[0], 1, tensor.range().size()[0]) =
228  eigen_map(tensor, 1, tensor.range().size()[0]);
229  } else {
230  TA_ASSERT(tensor.range().finish()[0] <= size_type(matrix.rows()));
231 
232  // Copy tensor into column vector
233  matrix.block(tensor.range().start()[0], 0, tensor.range().size()[0], 1) =
234  eigen_map(tensor, tensor.range().size()[0], 1);
235  }
236  }

316  template <typename A, typename Derived>
317  A eigen_to_array(madness::World& world, const typename A::trange_type& trange,
318  const Eigen::MatrixBase<Derived>& matrix, bool replicated = false)
319  {
320  typedef typename A::size_type size_type;
321  // Check that trange matches the dimensions of other
322  if((matrix.cols() > 1) && (matrix.rows() > 1)) {
323  TA_USER_ASSERT(trange.tiles().dim() == 2,
324  "TiledArray::eigen_to_array(): The number of dimensions in trange is not equal to that of the Eigen matrix.");
325  TA_USER_ASSERT(trange.elements().size()[0] == size_type(matrix.rows()),
326  "TiledArray::eigen_to_array(): The number of rows in trange is not equal to the number of rows in the Eigen matrix.");
327  TA_USER_ASSERT(trange.elements().size()[1] == size_type(matrix.cols()),
328  "TiledArray::eigen_to_array(): The number of columns in trange is not equal to the number of columns in the Eigen matrix.");
329  } else {
330  TA_USER_ASSERT(trange.tiles().dim() == 1,
331  "TiledArray::eigen_to_array(): The number of dimensions in trange must match that of the Eigen matrix.");
332  TA_USER_ASSERT(trange.elements().size()[0] == size_type(matrix.size()),
333  "TiledArray::eigen_to_array(): The size of trange must be equal to the matrix size.");
334  }
335 
336  // Check that this is not a distributed computing environment
337  if(! replicated)
338  TA_USER_ASSERT(world.size() == 1,
339  "An array cannot be assigned with an Eigen::Matrix when the number of MPI processes is greater than 1.");
340 
341  // Create a new tensor
342  A array = (replicated && (world.size() > 1) ?
343  A(world, trange, std::static_pointer_cast<typename A::pmap_interface>(
344  std::shared_ptr<detail::ReplicatedPmap>(new detail::ReplicatedPmap(world, trange.tiles().volume())))) :
345  A(world, trange));
346 
347  // Spawn tasks to copy Eigen to an Array
348  madness::AtomicInt counter;
349  counter = 0;
350  std::size_t n = 0;
351  for(std::size_t i = 0; i < array.size(); ++i) {
352  world.taskq.add(& detail::counted_eigen_submatrix_to_tensor<A, Derived>,
353  &matrix, array, i, &counter);
354  ++n;
355  }
356 
357  // Wait until the write tasks are complete
358  detail::CounterProbe probe(counter, n);
359  array.get_world().await(probe);
360 
361  return array;
362  }

387  template <typename T, unsigned int DIM, typename Tile>
388  Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>
389  array_to_eigen(const Array<T, DIM, Tile>& array) {
390  // Check that the array will fit in a matrix or vector
391  TA_USER_ASSERT((DIM == 2u) || (DIM == 1u),
392  "TiledArray::array_to_eigen(): The array dimensions must be equal to 1 or 2.");
393 
394  // Check that this is not a distributed computing environment or that the
395  // array is replicated
396  if(! array.get_pmap()->is_replicated())
397  TA_USER_ASSERT(array.get_world().size() == 1,
398  "TiledArray::array_to_eigen(): Array cannot be assigned with an Eigen::Matrix when the number of MPI processes is greater than 1.");
399 
400  // Construct the Eigen matrix
401  Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>
402  matrix(array.trange().elements().size()[0],
403  (DIM == 2 ? array.trange().elements().size()[1] : 1));
404 
405  // Spawn tasks to copy array tiles to the Eigen matrix
406  madness::AtomicInt counter;
407  counter = 0;
408  std::size_t n = 0;
409  for(std::size_t i = 0; i < array.size(); ++i) {
410  if(! array.is_zero(i)) {
411  array.get_world().taskq.add(
412  & detail::counted_tensor_to_eigen_submatrix<Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>,
413  typename Array<T, DIM, Tile>::value_type>,
414  array.find(i), &matrix, &counter);
415  ++n;
416  }
417  }
418 
419  // Wait until the above tasks are complete. Tasks will be processed by this
420  // thread while waiting.
421  detail::CounterProbe probe(counter, n);
422  array.get_world().await(probe);
423 
424  return matrix;
425  }
