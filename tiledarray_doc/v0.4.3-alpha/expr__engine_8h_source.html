89  void init(World& world, std::shared_ptr<pmap_interface> pmap,
90  const VariableList& target_vars)
91  {
92  if(target_vars.dim()) {
93  derived().init_vars(target_vars);
94  derived().init_struct(target_vars);
95  } else {
96  derived().init_vars();
97  derived().init_struct(vars_);
98  }
99 
100  // Check for a valid process map.
101  if(pmap) {
102  // If process map is not valid, use the process map constructed by the
103  // expression engine.
104  if((typename pmap_interface::size_type(world.size()) != pmap->procs()) ||
105  (trange_.tiles().volume() != pmap->size()))
106  pmap.reset();
107  }
108 
109  derived().init_distribution(& world, pmap);
110  }

122  void init_struct(const VariableList& target_vars) {
123  if(target_vars != vars_) {
124  perm_ = derived().make_perm(target_vars);
125  trange_ = derived().make_trange(perm_);
126  shape_ = derived().make_shape(perm_);
127  } else {
128  trange_ = derived().make_trange();
129  shape_ = derived().make_shape();
130  }

140  void init_distribution(World* world,
