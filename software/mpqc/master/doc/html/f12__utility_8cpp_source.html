13 double basis_to_f12exponent(const std::string& basis_name) {

16  boost::to_lower(basis);
17 
18  double f12_gamma;
19 
20  if (basis == "cc-pvdz-f12") {

23  f12_gamma = 1.0;

45 double fngtg(const std::vector<double>& cc, const std::vector<double>& aa,
46  double x) {
47  double value = 0.0;
48  const double x2 = x * x;
49  const auto n = cc.size();
50 
51  for (auto i = 0ul; i < n; ++i) {
52  const auto cc_i = cc[i];
53  const auto aa_i = aa[i];
54  const double x_aa_i = - aa_i * x2;
55  double result;
56  if(x_aa_i < -708.4){
57  result = 0.0;
58  }
59  else{
60  TA_ASSERT( x_aa_i < 709.8 );
61  result = std::exp(x_aa_i);
62  }
63  value += cc_i * result;
64  }
65 
66  return value;
67 }

78 void LinearSolveDamped(int n, Eigen::MatrixXd& A, const double* b,
79  double lambda, double* x) {
80  Eigen::MatrixXd Acopy(A);
81  for (int m = 0; m < n; ++m) Acopy(m, m) *= (1 + lambda);
82  Eigen::VectorXd e(n);
83  for (int i = 0; i < n; i++) e(i) = b[i];
84  Eigen::VectorXd result = Acopy.colPivHouseholderQr().solve(e);
85  for (int i = 0; i < n; i++) x[i] = result(i);
86 }

105 std::vector<std::pair<double, double>> stg_ng_fit(std::size_t n, double zeta) {
106  const std::size_t nparams = 2 * n;
107  std::vector<double> cc(n, 1.0);
108  std::vector<double> aa(n);
109  for (auto i = 0ul; i < n; ++i) {
110  aa[i] = std::pow(3.0, (i + 2 - (n + 1) / 2.0));
111  }
112 
113  // first rescale cc for ff[x] to match the norm of f[x]
114  double ffnormfac = 0.0;
115  for (auto i = 0ul; i < n; ++i) {
116  for (auto j = 0ul; j < n; ++j) {
117  ffnormfac += cc[i] * cc[j] / std::sqrt(aa[i] + aa[j]);
118  }
119  }
120 
121  const double Nf = std::sqrt(2.0 * zeta) * zeta;
122  const double Nff =
123  std::sqrt(2.0) / (std::sqrt(ffnormfac) * std::sqrt(std::sqrt(M_PI)));
124  for (auto i = 0ul; i < n; ++i) cc[i] *= -Nff / Nf;
125 
126  const unsigned int npts = 1001;
127  const double xmin = 0.0;
128  const double xmax = 10.0;
129  double lambda0 = 1000; // damping factor is initially set to 1000, eventually
130  // should end up at 0
131  const double nu =
132  3.0; // increase/decrease the damping factor scale it by this
133  const double epsilon = 1e-15;
134  const unsigned int maxniter = 200;
135 
136  // grid points on which we will fit
137  std::vector<double> xi(npts);
138  for (auto i = 0u; i < npts; ++i)
139  xi[i] = xmin + (xmax - xmin) * i / (npts - 1);
140 
141  Eigen::VectorXd err(npts);
142  Eigen::MatrixXd J(npts, nparams);
143 
144  std::vector<double> delta(nparams);
145 
146  double errnormI;
147  double errnormIm1 = 1e3;
148  bool converged = false;
149  unsigned int iter = 0;
150  while (!converged && iter < maxniter) {
151  // std::cout << "Iteration " << ++iter << ": lambda = " << lambda0/nu <<
152  // std::endl;
153 
154  for (unsigned int i = 0; i < npts; ++i) {
155  const double x = xi[i];
156  err[i] = (detail::fstg(zeta, x) - detail::fngtg(cc, aa, x)) *
157  std::sqrt(detail::ww(x));
158  }
159  errnormI = err.norm() / std::sqrt((double)npts);
160 
161  // std::cout << "|err|=" << errnormI << std::endl;
162  converged = std::abs((errnormI - errnormIm1) / errnormIm1) <= epsilon;
163  if (converged) break;
164  errnormIm1 = errnormI;
165 
166  for (unsigned int i = 0; i < npts; ++i) {
167  const double x2 = xi[i] * xi[i];
168  const double sqrt_ww_x = std::sqrt(detail::ww(xi[i]));
169  for (auto j = 0u; j < n; ++j)
170  J(i, j) = (std::exp(-aa[j] * x2)) * sqrt_ww_x;
171  for (auto j = 0u; j < n; ++j)
172  J(i, n + j) = -sqrt_ww_x * x2 * cc[j] * std::exp(-aa[j] * x2);
173  }
174 
175  Eigen::MatrixXd A(nparams, nparams);
176 
177  A = J.transpose() * J;
178  // BLAS_Mat_x_Mat( A, J, J, true, false );
179 
180  Eigen::VectorXd b(nparams);
181 
182  b = J.transpose() * err;
183  // BLAS_Mat_x_Vec( b, J, err, true );
184 
185  // try decreasing damping first
186  // if not successful try increasing damping until it results in a decrease
187  // in the error
188  lambda0 /= nu;
189  for (int l = -1; l < 1000; ++l) {
190  detail::LinearSolveDamped(nparams, A, &(b[0]), lambda0, &(delta[0]));
191  std::vector<double> cc_0(cc);
192  for (auto i = 0u; i < n; ++i) cc_0[i] += delta[i];
193  std::vector<double> aa_0(aa);
194  for (auto i = 0u; i < n; ++i) aa_0[i] += delta[i + n];
195 
196  // if any of the exponents are negative the step is too large and need to
197  // increase damping
198  bool step_too_large = false;
199  for (auto i = 0u; i < n; ++i)
200  if (aa_0[i] < 0.0) {
201  step_too_large = true;
202  break;
203  }
204  if (!step_too_large) {
205  std::vector<double> err_0(npts);
206  for (auto i = 0u; i < npts; ++i) {
207  const double x = xi[i];
208  err_0[i] = (detail::fstg(zeta, x) - detail::fngtg(cc_0, aa_0, x)) *
209  std::sqrt(detail::ww(x));
210  }
211  const double errnorm_0 = detail::norm(err_0) / std::sqrt((double)npts);
212  if (errnorm_0 < errnormI) {
213  cc = cc_0;
214  aa = aa_0;
215  break;
216  } else // step lead to increase of the error -- try dampening a bit
217  // more
218  lambda0 *= nu;
219  } else // too large of a step
220  lambda0 *= nu;
221  } // done adjusting the damping factor
222 
223  } // end of iterative minimization
224 
225  // if reached max # of iterations throw if the error is too terrible
226  if (iter == maxniter && errnormI > 1e-10)
227  throw std::runtime_error(
228  "after the max # of iterations Levenberg-Marquardt failed to converge "
229  "to better than 1e-10");
230 
231  std::vector<std::pair<double, double>> result(n);
232  for (auto i = 0u; i < n; ++i) {
233  result[i].second = cc[i];
234  result[i].first = aa[i];
235  }
236  return result;
237 }
