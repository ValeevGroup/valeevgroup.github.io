144  template <typename Cluster, typename Clusterable>
145  void lloyds_algorithm(std::vector<Cluster> &clusters,
146  std::vector<Clusterable> const &cbls) {
147  // Save old centers
148  auto old_centers = centers(clusters);
149 
150  // Run an iteration to get new clusters
151  attach_clusterables(clusters, cbls);
152  auto new_centers = centers(clusters);
153 
154  auto iter = 0;
155  while (!kmeans_converged(old_centers, new_centers) &&
156  (max_iters_ > iter++)) {
157  old_centers = new_centers;
158  attach_clusterables(clusters, cbls);
159  new_centers = centers(clusters);
160  }
161  }

163  bool kmeans_converged(std::vector<Vector3d> const &old_centers,
164  std::vector<Vector3d> const &new_centers) {
165  auto compare_center_position = [](Vector3d const &oldc,
166  Vector3d const &newc) {
167  return 1e-7 < (oldc - newc).squaredNorm();
168  };
169 
170  return std::equal(old_centers.begin(), old_centers.end(),
171  new_centers.begin(), compare_center_position);
172  }

175  // k-means++: The Advantages of Careful Seeding,
176  // by: David Arthur and Sergei Vassilvitskii
177  template <typename Cluster, typename Clusterable>
178  std::vector<Cluster> initial_guess(std::vector<Clusterable> const &cbls,
179  int64_t nclusters) {
180  // initially each center has the same weight
181  std::vector<double> weights(cbls.size(), 1.0);
182  std::mt19937 engine(seed_);
183 
184  std::vector<Cluster> clusters(nclusters);
185 
186  auto end = clusters.end();
187  for (auto it = clusters.begin(); it != end; ++it) {
188  std::discrete_distribution<unsigned int> random_index(weights.begin(),
189  weights.end());
190 
191  // Determine a random center
192  auto idx = random_index(engine);
193  auto center_guess = center(cbls[idx]);
194 
195  // Set the current cluster to that center
196  set_center(*it, center_guess);
197 
198  // Update the weights based on the chosen cluster guesses
199  auto last_init_cluster = it;
200  std::advance(last_init_cluster, 1);
201  update_weights(clusters.begin(), last_init_cluster, cbls, weights);
202  }
203 
204  // Updates the clusters with current best guess
205  attach_clusterables(clusters, cbls);
206 
207  return clusters;
208  }

210  // Function to update the weights in the k-means++ initialization.
211  template <typename Iter, typename Clusterable>
212  void update_weights(Iter first, Iter last,
213  std::vector<Clusterable> const &cbls,
214  std::vector<double> &weights) {
215  const auto size = cbls.size();
216 
217  for (auto i = 0ul; i < size; ++i) {
218  const auto cbl_center = center(cbls[i]);
219 
220  const auto closest_cluster_center =
221  center(*closest_cluster(first, last, cbl_center));
222 
223  const auto cbl_to_cluster_dist2 =
224  (cbl_center - closest_cluster_center).squaredNorm();
225 
226  weights[i] = cbl_to_cluster_dist2;
227  }

242 template <typename Cluster>
243 double kmeans_objective(std::vector<Cluster> const &cs) {
244  double sum = 0.0;
245  // sum over clusters
246  for (auto const &cluster : cs) {
247  const auto cluster_center = center(cluster);
248 
249  // sum over squared distance of elem to cluster centers
250  for (auto const &elem : cluster) {
251  const auto elem_center = center(elem);
252  sum += (elem_center - cluster_center).squaredNorm();
253  }
254  }
255 
256  return sum;
257 }
