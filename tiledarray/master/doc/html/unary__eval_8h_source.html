121  virtual int internal_eval() {
122  // Convert pimpl to this object type so it can be used in tasks
123  std::shared_ptr<UnaryEvalImpl_> self =
124  std::enable_shared_from_this<UnaryEvalImpl_>::shared_from_this();
125 
126  // Evaluate argument
127  arg_.eval();
128 
129  // Counter for the number of tasks submitted by this object
130  size_type task_count = 0ul;
131 
132  // Make sure all local tiles are present.
133  const typename pmap_interface::const_iterator end = arg_.pmap()->end();
134  typename pmap_interface::const_iterator it = arg_.pmap()->begin();
135  for(; it != end; ++it) {
136  // Get argument tile index
137  const size_type index = *it;
138 
139  if(! arg_.is_zero(index)) {
140  // Get target tile index
141  const size_type target_index = DistEvalImpl_::perm_index_to_target(index);
142 
143  // Schedule tile evaluation task
144  TensorImpl_::world().taskq.add(self, & UnaryEvalImpl_::eval_tile,
145  target_index, arg_.get(index));
146 
147  ++task_count;
148  }
149  }
150 
151  // Wait for local tiles of argument to be evaluated
152  arg_.wait();
153 
154  return task_count;
155  }
