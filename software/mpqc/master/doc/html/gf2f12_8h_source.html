20 enum class Denominator {
21  pqrE, // ( e(p) + e(q) - e(r) - E)
22  rEpq // (-e(p) - e(q) + e(r) + E)
23 };

25 template <Denominator Denom, typename Tile, typename Policy>
26 TA::Array<double, 4, Tile, Policy> d_pqrE(
27  TA::Array<double, 4, Tile, Policy>& pqrs, const Eigen::VectorXd& evals_p,
28  const Eigen::VectorXd& evals_q, const Eigen::VectorXd& evals_r,
29  typename Tile::scalar_type E) {
30  auto convert = [evals_p, evals_q, evals_r, E](Tile& result_tile,
31  const Tile& arg_tile) {
32 
33  result_tile = Tile(arg_tile.range());
34 
35  // compute index
36  const auto p0 = result_tile.range().lobound()[0];
37  const auto p1 = result_tile.range().upbound()[0];
38  const auto q0 = result_tile.range().lobound()[1];
39  const auto q1 = result_tile.range().upbound()[1];
40  const auto r0 = result_tile.range().lobound()[2];
41  const auto r1 = result_tile.range().upbound()[2];
42  const auto s0 = result_tile.range().lobound()[3];
43  const auto s1 = result_tile.range().upbound()[3];
44 
45  auto tile_idx = 0;
46  typename Tile::value_type norm2 = 0.0;
47  for (auto p = p0; p != p1; ++p) {
48  const auto e_p = evals_p[p];
49  for (auto q = q0; q != q1; ++q) {
50  const auto e_q = evals_q[q];
51  for (auto r = r0; r < r1; ++r) {
52  const auto e_r = evals_r[r];
53  const auto denom =
54  1 / ((Denom == Denominator::pqrE) ? -e_r - E + e_p + e_q
55  : e_r + E - e_p - e_q);
56  for (auto s = s0; s < s1; ++s, ++tile_idx) {
57  const auto v_pqrs = arg_tile[tile_idx];
58  const auto vv_pqrs = v_pqrs * denom;
59  norm2 += vv_pqrs * vv_pqrs;
60  result_tile[tile_idx] = vv_pqrs;
61  }
62  }
63  }
64  }
65  return std::sqrt(norm2);
66  };
67 
68  return TA::foreach (pqrs, convert);
69 }

136  void evaluate(GFRealPole* pole) override {
137  auto target_precision = pole->target_precision(0);
138 
139  if (!this->computed()) {
140  // init orbitals
141  auto target_ref_precision = target_precision / 100.;
142  init(target_ref_precision);
143 
144  auto& world = this->lcao_factory().world();
145  auto time0 = mpqc::fenced_now(world);
146 
147  auto orbital = pole->target();
148  if (orbital < 0 &&
149  (abs(orbital) - 1 >= this->trange1_engine()->get_active_occ()))
150  throw std::runtime_error(
151  "GFRealPole::target is invalid (the number of holes exceeded)");
152  if (orbital > 0 && (orbital - 1 >= this->trange1_engine()->get_vir()))

154  "GFRealPole::target is invalid (the number of particles exceeded)");

156  std::string method_str = dyson_method_;
157  enum class Method { diag, nondiag };
158  Method method;
159  if (method_str.find("diagonal") == 0)
160  method = Method::diag;
161  else if (method_str == "nondiagonal")
162  method = Method::nondiag;
163  else
164  throw ProgrammingError("unknown method string", __FILE__, __LINE__);

166  ExEnv::out0() << "orbital = " << orbital << " method = " << method_str

170  if (method == Method::diag)
171  result = compute_diagonal(orbital, max_iter_);
172  else
173  result = compute_nondiagonal(orbital, max_iter_);
174 
175  this->computed_ = true;
176  this->set_value(pole, result);
177  auto time1 = mpqc::fenced_now(world);
178  auto time = mpqc::duration_in_s(time0, time1);
179 
180  ExEnv::out0() << "Total GF2F12 Time: " << time << " S\n";
181  }

246 template <typename Tile>
247 double GF2F12<Tile>::compute_diagonal(const int target_orbital,
248  const int max_niter) {
249  auto& lcao_factory = to_lcao_factory(this->lcao_factory());
250  auto& world = this->lcao_factory().world();
251 
252  auto nfzc = this->trange1_engine()->get_nfrozen();
253  auto nocc = this->trange1_engine()->get_active_occ();
254  auto nuocc = this->trange1_engine()->get_vir();
255  // map orbital_ (index relative to Fermi level) to orbital index in active
256  // orbital space
257  const auto orbital =
258  nfzc + nocc +
259  ((target_orbital < 0) ? target_orbital : target_orbital - 1);
260 
261  auto orbital_energy =
262  make_diagonal_fpq(this->lcao_factory(), this->ao_factory(), true); // use df by default
263 
264  auto SE = orbital_energy->operator()(orbital);
265 
266  Eigen::VectorXd occ_evals = orbital_energy->segment(nfzc, nocc + nfzc);
267  Eigen::VectorXd uocc_evals = orbital_energy->segment(nfzc + nocc, nuocc);
268 
269  // will use only the target orbital to transform ints
270  // create an OrbitalSpace here
271  { init_target_orbital_diagonal(target_orbital); }
272 
273  lcao_factory.keep_partial_transforms(true);
274 
275  TArray g_vvog = this->lcao_factory().compute(L"<a b|G|i x>[df]");
276  TArray g_oovg = this->lcao_factory().compute(L"<i j|G|a x>[df]");
277 
278  ExEnv::out0() << printf("Iter SE2(in) SE2(out) SE2(delta)\n");
279  ExEnv::out0() << printf("==== =========== =========== ===========\n");
280  size_t iter = 0;
281  decltype(SE) SE_diff;
282  do {
283  TArray Sigma_pph;
284  {
285  TArray dg_vvog = dyson::d_pqrE<dyson::Denominator::rEpq>(
286  g_vvog, uocc_evals, uocc_evals, occ_evals, SE);
287  Sigma_pph("x,y") = 0.5 * (4 * g_vvog("a,b,i,x") - 2 * g_vvog("b,a,i,x")) *
288  dg_vvog("a,b,i,y");
289  }
290  // std::cout << "Sigma_pph:\n" << Sigma_pph << std::endl;
291 
292  TArray Sigma_hhp;
293  {
294  TArray dg_oovg = dyson::d_pqrE<dyson::Denominator::rEpq>(
295  g_oovg, occ_evals, occ_evals, uocc_evals, SE);
296  Sigma_hhp("x,y") = 0.5 * (4 * g_oovg("i,j,a,x") - 2 * g_oovg("j,i,a,x")) *
297  dg_oovg("i,j,a,y");
298  }
299  // std::cout << "Sigma_hhp:\n" << Sigma_hhp << std::endl;
300 
301  RowMatrixXd Sigma;
302  {
303  TArray Sigma_ta;
304  Sigma_ta("x,y") = Sigma_pph("x,y") + Sigma_hhp("x,y");
305  Sigma = math::array_to_eigen(Sigma_ta);
306  }
307 
308  // std::cout << "Sigma = " << Sigma << std::endl;
309 
310  auto SE_updated = Sigma(0, 0) + orbital_energy->operator()(orbital);
311  SE_diff = SE_updated - SE;
312 
313  ExEnv::out0() << printf(" %3ld %10.4lf %10.4lf %10.4lf\n", iter, SE,
314  SE_updated, SE_diff);
315 
316  SE = SE_updated;
317  ++iter;
318 
319  } while ((fabs(SE_diff) > 1e-6) && (iter <= max_niter));
320 
321  this->lcao_factory().purge_formula(L"<a b|G|i x>[df]");
322  this->lcao_factory().purge_formula(L"<i j|G|a x>[df]");
323 
324  // for now the f12 contribution is energy-independent
325  TArray Sigma_pph_f12;
326  {
327  TArray V_ixix, V_ixxi;
328  // always do DF
329  std::tie(V_ixix, V_ixxi) = compute_V();
330 
331  Sigma_pph_f12("x,y") =
332  0.5 * (1.25 * V_ixix("i,x,j,y") - 0.25 * V_ixxi("i,x,y,j")) *
333  this->lcao_factory()(L"<i|I|j>");
334  }
335  // std::cout << "Sigma_pph_f12:\n" << Sigma_pph_f12 << std::endl;
336  Matrix Sigma_f12 = math::array_to_eigen(Sigma_pph_f12);
337 
338  // done with F12 ... remove all geminal ints and CABS indices
339  this->lcao_factory().purge_operator(L"R");
340  this->lcao_factory().purge_index(L"a'");
341  this->lcao_factory().purge_index(L"Ï");
342 
343  lcao_factory.keep_partial_transforms(false);
344 
345  if (world.rank() == 0) {
346  auto SE_F12 = SE + Sigma_f12(0, 0);
347  auto unit_factory = UnitFactory::get_default();
348  auto Hartree2eV = unit_factory->make_unit("eV").from_atomic_units();
349  std::string orblabel = std::string(target_orbital < 0 ? "IP" : "EA") +
350  std::to_string(abs(target_orbital));
351  ExEnv::out0() << printf("final GF2 %6s = %11.3lf eV (%10.4lf a.u.)\n",
352  orblabel.c_str(), SE * Hartree2eV, SE);
353  ExEnv::out0() << printf("final GF2-F12-V %6s = %11.3lf eV (%10.4lf a.u.)\n",
354  orblabel.c_str(), SE_F12 * Hartree2eV, SE_F12);
355  if (target_orbital > 0)
356  ExEnv::out0() << printf(
357  "WARNING: non-strongly-orthogonal F12 projector is used for the F12 "
358  "correction to EA!!! \n");
359  }
360 
361  return SE;
362 }
