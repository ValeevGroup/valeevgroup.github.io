138  constexpr bool compress_D = false;
139  auto dD = TA::to_new_tile_type(

142  darray_type dJ;

144

155  array_type compute_K(array_type const &C) {

160  C, TaToDecompTensor(clr_thresh_, compress_C));

170  F_df("X, i, mu") = (G_df_("X,mu, nu") * dC("nu,i")).set_shape(C_mo.shape());
171  F_df.truncate();
172  auto fdf1 = mpqc::fenced_now(world);
173  f_df_storages_.push_back(detail::array_storage(F_df));
174 
175  auto l0 = mpqc::fenced_now(world);
176  dL("mu, nu") = C_mo("X, i, mu") * F_df("X, i, nu");
177  dL.truncate();
178  auto l1 = mpqc::fenced_now(world);
179 
180  array_type K;
181  auto k0 = mpqc::fenced_now(world);
182  auto L = TA::to_new_tile_type(dL, DecompToTaTensor{});
183  K("mu, nu") = L("mu, nu") + L("nu, mu");
184  auto k1 = mpqc::fenced_now(world);
185 
186  c_mo_times_.push_back(mpqc::duration_in_s(cmo0, cmo1));
187  f_df_times_.push_back(mpqc::duration_in_s(fdf0, fdf1));
188  dl_times_.push_back(mpqc::duration_in_s(l0, l1));
189  dl_to_k_times_.push_back(mpqc::duration_in_s(k0, k1));
190 
191  return K;
