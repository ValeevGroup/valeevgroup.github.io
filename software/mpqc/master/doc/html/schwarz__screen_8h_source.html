84  template <typename E>
85  Qmatrix(madness::World &world, E const &eng, Basis const &bs,
86  norm_func_type norm_op = detail::inf_norm)
87  : is_aux_(true) {
88  // Get bs shells
89  auto shells = bs.flattened_shells();
90  const auto nshells = shells.size();
91 
92  // Compute Q_
93  Q_ = Eigen::VectorXd(nshells);
94 
95  auto Q_func = [=](Shell const *sh, double *Q_val) {
96  // Compute the shell set (a unit | a unit)
97  const auto &bufs = eng->local().compute(*sh, unit_shell, *sh, unit_shell);
98  const auto nsh = sh->size();
99  const auto bmap = Eigen::Map<const RowMatrixXd>(bufs[0], nsh, nsh);
100 
101  // Compute and write the sqrt(norm) of quartet (a unit | a unit)
102  *Q_val = norm_op(bmap);
103  };
104 
105  for (auto i = 0ul; i < nshells; ++i) {
106  auto *Q_val_ptr = &Q_(i);
107  world.taskq.add(Q_func, &shells[i], Q_val_ptr);
108  }
109  world.gop.fence();
110 
111  // Compute the function to shell map
112  f2s_map map;
113  auto func_ind = 0;
114  for (auto i = 0ul; i < nshells; ++i) {
115  map.emplace(func_ind, i);
116  func_ind += shells[i].size();
117  }
118  f2s_maps_[0] = std::move(map);
119 
120  // Compute the max elem in Q
121  max_elem_in_row_.resize(nshells);
122  for (auto i = 0ul; i < nshells; ++i) {
123  const auto max_elem = Q_(i);
124  max_elem_in_Q_ = std::max(max_elem_in_Q_, max_elem);
125  max_elem_in_row_[i] = max_elem;
126  }
127 
128  // Build Q_cluster_
129  auto const &cshells = bs.cluster_shells();
130  const auto nclusters = cshells.size();
131  Q_cluster_ = Eigen::VectorXd(nclusters);
132  Q_cluster_.setZero();
133 
134  // Because we are using Qbra * Qket >= (bra|ket)^2 just sum these values
135  // into
136  // Q_cluster_
137  auto first_shell_in_cluster = 0;
138  for (auto c = 0ul; c < nclusters; ++c) {
139  const auto nshells = cshells[c].size();
140  const auto last = first_shell_in_cluster + nshells;
141  for (std::size_t s = first_shell_in_cluster; s < last; ++s) {
142  Q_cluster_(c) += Q_(s);
143  }
144  first_shell_in_cluster += nshells;
145  }
146  }

166  template <typename E>
167  Qmatrix(TA::World &world, E const &eng, Basis const &bs0, Basis const &bs1,
168  norm_func_type norm_op = detail::inf_norm)
169  : is_aux_(false) {
170  // Get bs shells
171  const auto shells0 = bs0.flattened_shells();
172  const auto shells1 = bs1.flattened_shells();
173  const auto nshells0 = shells0.size();
174  const auto nshells1 = shells1.size();
175 
176  // Compute Q_
177  Q_.resize(nshells0, nshells1);
178 
179  auto Q_func = [=](Shell const *sh0, Shell const *sh1, double *Q_val) {
180  // Get shell sizes
181  const auto nsh0 = sh0->size();
182  const auto nsh1 = sh1->size();
183  const auto n2 = nsh0 * nsh1;
184 
185  // Capture local engine
186  auto &leng = eng->local();
187  leng.set_precision(0.);
188  auto const &result = leng.results();
189 
190  // Compute estimate
191  leng.compute(*sh0, *sh1, *sh0, *sh1);
192  const auto bmap = Eigen::Map<const RowMatrixXd>(result[0], n2, n2);
193  *Q_val = norm_op(bmap);
194 
195  // Reset precision to something better than 0.
196  leng.set_precision(std::numeric_limits<double>::epsilon());
197  };
198 
199  for (auto i = 0ul; i < nshells0; ++i) {
200  for (auto j = 0ul; j < nshells1; ++j) {
201  auto *Q_val_ptr = &Q_(i, j);
202  world.taskq.add(Q_func, &shells0[i], &shells1[j], Q_val_ptr);
203  }
204  }
205  world.gop.fence();
206 
207  // Compute the function to shell map for bs0
208  f2s_map map0;
209  auto func_ind = 0;
210  for (auto i = 0ul; i < nshells0; ++i) {
211  map0.emplace(func_ind, i);
212  func_ind += shells0[i].size();
213  }
214  f2s_maps_[0] = std::move(map0);
215 
216  // Compute the function to shell map for bs1
217  f2s_map map1;
218  func_ind = 0;
219  for (auto i = 0ul; i < nshells1; ++i) {
220  map1.emplace(func_ind, i);
221  func_ind += shells1[i].size();
222  }
223  f2s_maps_[1] = std::move(map1);
224 
225  // Compute the max elem in Q
226  max_elem_in_row_.resize(Q_.rows());
227  for (auto i = 0l; i < Q_.rows(); ++i) {
228  const auto max_elem = Q_.row(i).cwiseAbs().maxCoeff();
229  max_elem_in_Q_ = std::max(max_elem_in_Q_, max_elem);
230  max_elem_in_row_[i] = max_elem;
231  }
232 
233  // Make Q_cluster_
234  auto const &cshells0 = bs0.cluster_shells();
235  auto const &cshells1 = bs1.cluster_shells();
236 
237  auto const &nclusters0 = cshells0.size();
238  auto const &nclusters1 = cshells1.size();
239 
240  Q_cluster_ = RowMatrixXd(nclusters0, nclusters1);
241  Q_cluster_.setZero();
242 
243  auto first_shell_in_cluster0 = 0;
244  for (auto c0 = 0ul; c0 < nclusters0; ++c0) {
245  const auto nshells0 = cshells0[c0].size();
246  const auto last0 = first_shell_in_cluster0 + nshells0;
247 
248  auto first_shell_in_cluster1 = 0;
249  for (auto c1 = 0ul; c1 < nclusters1; ++c1) {
250  const auto nshells1 = cshells1[c1].size();
251  const auto last1 = first_shell_in_cluster1 + nshells1;
252 
253  auto val = 0.0;
254  for (std::size_t s0 = first_shell_in_cluster0; s0 < last0; ++s0) {
255  for (std::size_t s1 = first_shell_in_cluster1; s1 < last1; ++s1) {
256  val += Q_(s0, s1);
257  }
258  }
259  Q_cluster_(c0, c1) = val;
260 
261  first_shell_in_cluster1 += nshells1;
262  }
263  first_shell_in_cluster0 += nshells0;
264  }
265  }

308  /* skip_ takes a list of indices and returns true when the integral estimate
309  * is below the threshold and false when it is greater than or equal to the
310  * threshold. See function estimate for the implementation.
311  *
312  * \param idx is a series of indices from which to generate an integral
313  * estimate
314  *
315  * This is a one stop catch all for the overloads of the skip functions
316  */
317  template <typename... IDX>
318  bool skip_(IDX... idx) const {
319  auto est =
320  estimate(std::forward<IDX>(idx)...); // optional estimation for set idx
321  if (est) { // Check that we were able to estimate the integral set
322  return est < thresh2_; // If est below threshold then skip this set
323  } else { // We were unable to estimate this set for some reason
324  return false; // thus we should compute the integral
325  }
326  }
327 
328  /* skip_ takes norm of a density block and a list of indices.
329  * This returns true when the product of density norm and integral estimate
330  * is below the threshold and false when it is greater than or equal to the
331  * threshold. See function estimate for the implementation.
332  *
333  * \param D is infinity norm of the corresponding density block
334  * \param idx is a series of indices from which to generate an integral
335  * estimate
336  *
337  * This is a one stop catch all for the overloads of the skip functions
338  */
339  template <typename... IDX>
340  bool skip_(double D, IDX... idx) const {
341  auto est =
342  estimate(std::forward<IDX>(idx)...); // optional estimation for set idx
343  if (est && D != 0.0) { // Check that we were able to estimate the integral
344  // set AND density norm is large than zero
345  return (D * D * (*est)) < thresh2_; // If D^2 * est below threshold then skip this set
346  } else { // We were unable to estimate this set for some reason
347  return false; // thus we should compute the integral
348  }
349  }

430 template <typename E>
431 SchwarzScreen create_schwarz_screener(
432  TA::World &world, ShrPool<E> const &eng, std::vector<Basis> const &bs_array,
433  double thresh, decltype(detail::inf_norm) norm_func = detail::l2Norm) {
434  if (bs_array.size() == 3) { // One index must be auxiliary assume first
435  auto Q_bra =
436  std::make_shared<Qmatrix>(Qmatrix(world, eng, bs_array[0], norm_func));
437  auto Q_ket = std::make_shared<Qmatrix>(
438  Qmatrix(world, eng, bs_array[1], bs_array[2], norm_func));
439  return SchwarzScreen(std::move(Q_bra), std::move(Q_ket), thresh);
440  } else if (bs_array.size() == 4) { // Four center estimator
441  auto Q_bra = std::make_shared<Qmatrix>(
442  Qmatrix(world, eng, bs_array[0], bs_array[1], norm_func));
443  auto Q_ket = std::make_shared<Qmatrix>(
444  Qmatrix(world, eng, bs_array[2], bs_array[3], norm_func));
445  return SchwarzScreen(std::move(Q_bra), std::move(Q_ket), thresh);
446  } else {
447  throw InputError(
448  "The bs_array passed into create_schwarz_screener must be of length 3 "
449  "or 4",
450  __FILE__, __LINE__);
451  return SchwarzScreen();
452  }
