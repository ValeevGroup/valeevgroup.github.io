99  DistEvalImpl(World& world, const trange_type& trange,
100  const shape_type& shape, const std::shared_ptr<pmap_interface>& pmap,

174  void wait() const {
175  const int task_count = task_count_;
176  if(task_count > 0) {
177  try {
178  TensorImpl_::get_world().await([this,task_count] ()
179  { return this->set_counter_ == task_count; });
180  } catch(...) {
181  std::stringstream ss;
182  ss << "!! ERROR TiledArray: Aborting due to exception.\n"
183  << "!! ERROR TiledArray: rank=" << TensorImpl_::get_world().rank()
184  << " id=" << id_ << " " << set_counter_ << " of " << task_count << " tiles set\n";
185  std::cout << ss.str().c_str();
186  throw;
187  }
