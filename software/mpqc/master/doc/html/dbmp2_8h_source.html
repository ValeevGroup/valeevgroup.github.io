37  void operator()(result_type &me, argument_type const &tile) const {
38  auto const &range = tile.range();
39  auto const &vec = *vec_;
40  auto const st = range.lobound_data();
41  auto const fn = range.upbound_data();
42  auto tile_idx = 0;
43 
44  auto sti = st[0];
45  auto fni = fn[0];
46  auto sta = st[1];
47  auto fna = fn[1];
48 
49  for (auto i = sti; i < fni; ++i) {
50  const auto e_i = vec[i];
51  for (auto a = sta; a < fna; ++a, ++tile_idx) {
52  const auto e_ia = e_i - vec[a + n_occ_];
53  const auto data = tile.data()[tile_idx];
54  me += (data * data) / (e_ia);
55  }
56  }
57  }
58 };
