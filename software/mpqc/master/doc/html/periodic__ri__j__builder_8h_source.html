58  void init() {
59  print_detail_ = ao_factory_.print_detail();
60  force_hermiticity_ = ao_factory_.force_hermiticity();
61  auto &world = ao_factory_.world();
62 
63  mpqc::time_point t0, t1;
64 
65  // overlap matrix
66  M_ = ao_factory_.compute(L"<κ|λ>");
67 
68  // 1*N charge matrix of auxiliary basis within one unit cell
69  auto unit_basis = ::mpqc::lcao::gaussian::detail::create_unit_basis();
70  ao_factory_.basis_registry()->add(OrbitalIndex(L"U"), unit_basis);
71  auto charge_mat = ao_factory_.compute(L"< U | Κ >");
72  // charge vector of auxiliary basis within one unit cell
73  auto charge_vec =
74  ::mpqc::lcao::gaussian::detail::take_row_from_2D_array(charge_mat, 0);
75  double charge_2 = charge_vec("X").dot(charge_vec("X"));
76  q_ = std::sqrt(charge_2);
77  // normalized charge vector of auxiliary basis within one unit cell
78  n_("X") = (1.0 / q_) * charge_vec("X");
79 
80  // projection matrix that projects matrix X on to charge vector
81  P_para_("X, Y") = n_("X") * n_("Y");
82 
83  // projection matrix that projects matrix X orthogonal to charge vector
84  identity_ = ao_factory_.compute(L"< Κ | I | Λ >");
85  P_perp_("X, Y") = identity_("X, Y") - P_para_("X, Y");
86 
87  // 2-body 2-center integrals
88  V_ = ao_factory_.compute(L"( Κ | G | Λ )");
89 
90  // perpendicular part of 2-body 2-center integrals
91  t0 = mpqc::fenced_now(world);
92  V_perp_("X, Y") = P_perp_("X, Z") * V_("Z, W") * P_perp_("W, Y");
93  t1 = mpqc::fenced_now(world);
94  auto t_v_perp = mpqc::duration_in_s(t0, t1);
95 
96  // A inverse where A = V_perp + P_para
97  t0 = mpqc::fenced_now(world);
98  array_type A;
99  A("X, Y") = V_perp_("X, Y") + P_para_("X, Y");
100  t1 = mpqc::fenced_now(world);
101  auto t_a = mpqc::duration_in_s(t0, t1);
102 
103  t0 = mpqc::fenced_now(world);
104  inv_ = math::eigen_inverse(A);
105  t1 = mpqc::fenced_now(world);
106  auto t_a_inv = mpqc::duration_in_s(t0, t1);
107 
108  t0 = mpqc::fenced_now(world);
109  IP_("X, Y") = inv_("X, Z") * P_perp_("Z, Y");
110  t1 = mpqc::fenced_now(world);
111  auto t_im = mpqc::duration_in_s(t0, t1);
112 
113  t0 = mpqc::fenced_now(world);
114  three_center_builder_ = std::make_unique<PTC_Builder>(ao_factory_);
115  t1 = mpqc::fenced_now(world);

129  array_type compute_J(const array_type &D, double target_precision) {
130  auto &world = ao_factory_.world();
131 
132  mpqc::time_point t0, t1;
133  auto t0_j_builder = mpqc::fenced_now(world);
134 
135  // 3-center 2-electron direct integrals contracted with density matrix
136  t0 = mpqc::fenced_now(world);
137  G_ = three_center_builder_->template contract_with<1>(D, target_precision);
138  t1 = mpqc::fenced_now(world);
139  auto t_3c_d_contr = mpqc::duration_in_s(t0, t1);
140 
141  // Build [CD]_X = C_Xμν D_μν
142  double t_w_para, t_w;
143  {
144  // intermediate for C_para_Xμν D_μν
145  t0 = mpqc::fenced_now(world);
146  array_type interm;
147  double prefactor;
148  {
149  const auto R_max = ao_factory_.R_max();
150  const auto RD_max = ao_factory_.RD_max();
151  prefactor = ::mpqc::pbc::detail::dot_product(M_, D, R_max, RD_max);
152  }
153  interm("X") = (prefactor / q_) * n_("X");
154  t1 = mpqc::fenced_now(world);
155  t_w_para = mpqc::duration_in_s(t0, t1);
156 
157  // intermediate for C_Xμν D_μν
158  t0 = mpqc::fenced_now(world);
159  CD_ = interm;
160  CD_("X") += IP_("X, Y") * (G_("Y") - V_("Y, Z") * interm("Z"));
161  t1 = mpqc::fenced_now(world);
162  t_w = mpqc::duration_in_s(t0, t1);
163  }
164 
165  // Build DF Coulomb term J_μν
166  array_type J_unsymm, J, J_part1, J_part2;
167  array_type VCD; // just an intermediate
168  VCD("X") = V_("X, Y") * CD_("Y");
169 
170  // Build Part #1 of J_μν
171  double t_j1;
172  {
173  t0 = mpqc::fenced_now(world);
174  array_type interm;
175  interm("X") = 2.0 * CD_("X") - IP_("Y, X") * VCD("Y");
176 
177  J_part1 = three_center_builder_->template contract_with<2>(
178  interm, target_precision);
179  t1 = mpqc::fenced_now(world);
180  t_j1 = mpqc::duration_in_s(t0, t1);
181  }
182 
183  // Build Part #2 of J_μν
184  double t_j2;
185  {
186  t0 = mpqc::fenced_now(world);
187  array_type interm;
188  interm("X, Y") = IP_("X, Z") * V_("Z, Y") - identity_("X, Y");
189 
190  double prefactor = (interm("X, Z") * n_("Z")).dot(VCD("X"));
191  J_part2("mu, nu") = (prefactor / q_) * M_("mu, nu");
192  t1 = mpqc::fenced_now(world);
193  t_j2 = mpqc::duration_in_s(t0, t1);
194  }
195 
196  // Build J_μν
197  J_unsymm("mu, nu") = J_part1("mu, nu") + J_part2("mu, nu");
198  if (force_hermiticity_) {
199  // force hermiticity of Coulomb matrix
200  J = pbc::detail::symmetrize_matrix(J_unsymm);
201  } else {
202  // leave Coulomb matrix as non-hermitian (due to finite lattice range)
203  J = J_unsymm;
204  }
205 
206  auto t1_j_builder = mpqc::fenced_now(world);
207  auto t_tot = mpqc::duration_in_s(t0_j_builder, t1_j_builder);
208 
209  if (this->print_detail_) {
210  ExEnv::out0() << "\nRI-J time decomposition:\n"
211  << "\tSum_RJ (X|μν) D_μν: " << t_3c_d_contr << " s\n"
212  << "\tC_para_Xμν D_μν: " << t_w_para << " s\n"
213  << "\tC_Xμν D_μν: " << t_w << " s\n"
214  << "\tJ_part1: " << t_j1 << " s\n"
215  << "\tJ_part2: " << t_j2 << " s\n"
216  << "\nTotal J builder time: " << t_tot << " s" << std::endl;
217  }
218  return J;
219  }
