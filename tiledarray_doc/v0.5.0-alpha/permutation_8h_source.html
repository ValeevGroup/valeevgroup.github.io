56  inline void permute_array(const Perm& perm, const Arg& arg, Result& result) {
57  TA_ASSERT(perm.dim() == size(arg));
58  TA_ASSERT(perm.dim() == size(result));
59 
60  const unsigned int n = perm.dim();
61  for(unsigned int i = 0u; i < n; ++i) {
62  const typename Perm::index_type pi = perm[i];
63  result[pi] = arg[i];
64  }
65  }

248  Permutation pow(int n) const {
249  // Initialize the algorithm inputs
250  int power;
251  Permutation value;
252  if(n < 0) {
253  value = inv();
254  power = -n;
255  } else {
256  value = *this;
257  power = n;
258  }
259 
260  Permutation result = identity(p_.size());
261 
262  while(power) {
263  if(power & 1)
264  result = result.mult(value);
265  value = value.mult(value);
266  power >>= 1;
267  }
268 
269  return result;
270  }
