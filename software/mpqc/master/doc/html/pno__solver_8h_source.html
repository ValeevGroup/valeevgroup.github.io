108  double energy() {
109  ExEnv::out0() << "\nStarting the PNO-MP2 solver:\n";
110  ExEnv::out0() << indent << "Target energy precision: " << precision_
111  << "\n";
112  std::string time_iters = (time_iters_) ? "true" : "false";
113  ExEnv::out0() << indent << "Time individual iterations: " << time_iters
114  << "\n";
115 
116  // Accumulate the number of pairs total
117  auto npairs = ij_pairs_.size();
118  world_.gop.sum(&npairs, 1);
119 
120  auto num_paos = 0;
121  auto num_pnos = 0;
122  for (auto const& p : ij_pairs_) {
123  auto const& U = p.second.U;
124  num_paos += U.rows();
125  num_pnos += U.cols();
126  }
127 
128  world_.gop.sum(&num_paos, 1);
129  world_.gop.sum(&num_pnos, 1);
130 
131  auto avg_paos = double(num_paos) / double(npairs);
132  auto avg_pnos = double(num_pnos) / double(npairs);
133 
134  ExEnv::out0() << indent << "Number of IJ pairs: " << npairs << "\n";
135  ExEnv::out0() << indent << "Average number of PAOs per pairs: " << avg_paos
136  << "\n";
137  ExEnv::out0() << indent << "Average number of PNOs per pairs: " << avg_pnos
138  << "\n\n";
139 
140  double old_energy = 0.0;
141  double energy = old_energy;
142  double ediff = std::numeric_limits<double>::max();
143  double rnorm_over_pairs = std::numeric_limits<double>::max();
144  auto iter = 0;
145 
146  auto e_total0 = fenced_now(world_);
147  decltype(e_total0) e0, e1;
148  bool converged = false;
149  while (not converged) {
150  if (time_iters_) {
151  e0 = fenced_now(world_);

153  update_Rs();

155  energy = accumulate_pair_energies();

157  ediff = old_energy - energy;
158  old_energy = energy;
159  update_Ts();
160 
161  if (iter > 1) {
162  converged =
163  std::abs(ediff) < precision_ && rnorm_over_pairs < precision_;
164  }
165  if (!converged && iter > max_iters_) {
166  ExEnv::out0() << "Exceeded max iters, quiting now\n";
167  converged = true;
168  }
169  if (time_iters_) {
170  e1 = fenced_now(world_);
171  auto etime = duration_in_s(e0, e1);
172  ExEnv::out0() << "Iter(" << iter << ") took " << etime << "s\n";
173  } else {
174  ExEnv::out0() << "Iter(" << iter << ")\n";
175  }
176  ExEnv::out0() << indent << "Energy : " << energy << "\n";
177  ExEnv::out0() << indent << indent << "Energy change : " << ediff
178  << "\n";

181  << std::endl;
182  ++iter;
183  }
184  auto e_total1 = fenced_now(world_);
185  auto e_total_time = duration_in_s(e_total0, e_total1);
186  ExEnv::out0() << "Final PNO-MP2 energy: " << energy << "\n" << std::endl;

250  void update_ij_R(int i, int j, PnoPair* pd_ptr) {
251  // Make local copy of R so that we don't have to worry about concurrent
252  // writes
253  PnoPair& pd = *pd_ptr;
254  RowMatrixXd R_local = pd.K;
255 
256  auto const i_size = pd.i_size;
257  auto const i_start = pd.i_start;
258  auto const a_size = pd.a_size;
259 
260  auto const j_size = pd.j_size;
261  auto const b_size = pd.b_size;
262 
263  RowMatrixXd const& T = pd.T;
264  Eigen::VectorXd const& Fab = pd.epsil_ab;
265 
266  for (auto k = 0; k < i_size; ++k) {
267  for (auto a = 0; a < a_size; ++a) {
268  const auto ea = Fab[a];
269  const auto ka = k * a_size + a;
270 
271  for (auto l = 0; l < j_size; ++l) {
272  for (auto b = 0; b < b_size; ++b) {
273  const auto eb = Fab[b];
274  const auto lb = l * b_size + b;
275 
276  const auto scale = ea + eb;
277 
278  R_local(ka, lb) += scale * T(ka, lb);
279  }
280  }
281  }
282  }
283 
284  // Update the G matrix for this pair all J should be local
285  RowMatrixXd Gij = RowMatrixXd::Zero(R_local.rows(), R_local.cols());
286 
287  for (auto k : pd.sig_ks) {
288  std::array<int, 2> kj = {{k, j}};
289  auto const& pdk = ij_pairs_.at(kj);
290  RowMatrixXd const& Tkj = pdk.T;
291 
292  // Only store grab overlaps for K > I, otherwise swap I and K and
293  // transpose in equations
294  RowMatrixXd const* S_IJ_KJ_ptr;
295  bool ksmaller = k < i;
296  if (k != i) {
297  std::array<int, 3> ijk{{i, j, k}};
298  if (ksmaller) {
299  ijk = {{k, j, i}};
300  }
301  S_IJ_KJ_ptr = &overlaps_.at(ijk);
302  }
303 
304  auto const ki_size = pdk.i_size;
305  auto const ki_start = pdk.i_start;
306 
307  auto const ka_size = pdk.a_size;
308  auto const kb_size = pdk.b_size;
309 
310  for (auto h = 0; h < i_size; ++h) {
311  const auto hg = h + i_start;
312 
313  for (auto l = 0; l < j_size; ++l) {
314  auto Ghl = Gij.block(h * a_size, l * b_size, a_size, b_size);
315 
316  for (auto n = 0; n < ki_size; ++n) {
317  const auto ng = n + ki_start;
318  const auto Tnl =
319  Tkj.block(n * ka_size, l * kb_size, ka_size, kb_size);
320 
321  if (k == i) {
322  Ghl += Fij_(hg, ng) * Tnl;
323  } else if (ksmaller) {
324  // We need to transpose the overlaps since we really have S_kj_ij
325  RowMatrixXd const& S_kj_ij = *S_IJ_KJ_ptr;
326  Ghl += Fij_(hg, ng) * S_kj_ij.transpose() * Tnl * S_kj_ij;
327  } else {
328  // Do the normal thing since we have S_ij_kj
329  RowMatrixXd const& S_ij_kj = *S_IJ_KJ_ptr;
330  Ghl += Fij_(hg, ng) * S_ij_kj * Tnl * S_ij_kj.transpose();
331  }
332  }
333  }
334  }
335  }
336  R_local -= Gij;
337 
338  //
339  // We will assume that the Compute T function will Zero the R matrices
340  // it's the most reasonable thing to do I think
341  //
342 
343  /* THE FOLLOWING WORKS ONLY FOR SINGLE NODE JOBS */
344  // // After tested launch this in a task, that may require a bit of maths
345  // // to figure out which node to send it too.
346  // auto avoid_headache = [i, j, this](RowMatrixXd Gij) {
347  // this->update_ji_R(i, j, std::move(Gij));
348  // };
349  // world_.taskq.add(avoid_headache, std::move(Gij));
350 
351  /* UPDATE FOR REMOTE NODES */
352  // Sumbit task for updating pair JI, node I should own that tile.
353  const auto owner = i % world_size_;
354  WorldObject_::task(owner, &PnoMp2Solver::update_ji_R, i, j, Gij);
355 
356  madness::ScopedMutex<madness::Mutex> mut(&task_mutex_);
357  pd.R += R_local;
358  }

377  double pair_energy(PnoPair const* pd_ptr) {
378  auto& pd = *pd_ptr;
379  RowMatrixXd const& T = pd.T;
380  // RowMatrixXd Tt = 2 * T - T.transpose();
381 
382  RowMatrixXd const& R = pd.R;
383  RowMatrixXd const& K = pd.K;
384 
385  auto i_size = pd.i_size;
386  auto a_size = pd.a_size;
387 
388  auto j_size = pd.j_size;
389  auto b_size = pd.b_size;
390 
391  double energy = 0.0;
392  for (auto k = 0; k < i_size; ++k) {
393  for (auto l = 0; l < j_size; ++l) {
394  const auto block = T.block(k * a_size, l * b_size, a_size, b_size);
395  RowMatrixXd Tt = 2 * block - block.transpose();
396 
397  for (auto a = 0; a < a_size; ++a) {
398  const auto ka = k * a_size + a;
399 
400  for (auto b = 0; b < b_size; ++b) {
401  const auto lb = l * b_size + b;
402 
403  energy += (K(ka, lb) + R(ka, lb)) * Tt(a, b);
404  }
405  }
406  }
407  }
408 
409  return energy;
410  }

458  void update_T(int i, int j, PnoPair* pd_ptr) {
459  auto& pd = *pd_ptr;
460  RowMatrixXd& R = pd.R;
461  RowMatrixXd& T = pd.T;
462  Eigen::VectorXd const& Fab = pd.epsil_ab;
463 
464  auto const i_size = pd.i_size;
465  auto const i_start = pd.i_start;
466  auto const a_size = pd.a_size;
467 
468  auto const j_size = pd.j_size;
469  auto const j_start = pd.j_start;
470  auto const b_size = pd.b_size;
471 
472  for (auto k = 0; k < i_size; ++k) {
473  const auto kg = k + i_start;
474  const auto Fkk = Fij_(kg, kg);
475 
476  for (auto l = 0; l < j_size; ++l) {
477  const auto lg = l + j_start;
478  const auto Fll = Fij_(lg, lg);
479 
480  for (auto a = 0; a < a_size; ++a) {
481  const auto ea = Fab[a];
482  const auto ka = k * a_size + a;
483 
484  for (auto b = 0; b < b_size; ++b) {
485  const auto eb = Fab[b];
486  const auto lb = l * b_size + b;
487 
488  T(ka, lb) -= R(ka, lb) / (ea + eb - Fkk - Fll);
489  }
490  }
491  }
492  }
493 
494  // Set R to zero here so we can start the next iter fresh
495  R.setZero();
496  }

498  void update_Ts() {
499  for (auto& ij_pair : ij_pairs_) {
500  const auto idx = ij_pair.first;
501  const auto i = std::get<0>(idx);
502  const auto j = std::get<1>(idx);
503 
504  // Make Task
505  // update_T(i, j, &(ij_pair.second));
506  auto taskq_workaround = [i, j, this](PnoPair* pd_ptr) {
507  this->update_T(i, j, pd_ptr);
508  };
509 
510  world_.taskq.add(taskq_workaround, &(ij_pair.second));
511  }
512  // This will locally block on tasks, we have to locally block so that we
513  // don't start trying to write to R's before a T is updated, otherwise we
514  // have to make a future for each pair and submitting those tasks would be a
515  // bit tricky.
516  world_.taskq.fence();
517  }
