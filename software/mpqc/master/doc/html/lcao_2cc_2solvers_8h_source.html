688 TA::DistArray<Tile, Policy> form_T_from_K (
689  const TA::DistArray<Tile, Policy>& K,
690  const RowMatrix<typename Tile::numeric_type>& F_occ_act,
691  const RowMatrix<typename Tile::numeric_type>& F_uocc,
692  std::vector<RowMatrix<typename Tile::numeric_type>>& pnos,

699 
700  result_tile = Tile(arg_tile.range());
701 
702  // determine i and j indices
703  const int i = arg_tile.range().lobound()[2];
704  const int j = arg_tile.range().lobound()[3];
705 
706  Matrix pno_ij = pnos[i*nocc_act + j];

708  auto npno = pno_ij.cols();

710

712  Vector eps_v = F_pno.diagonal();

715

717  const auto e_j = eps_o[j];
718  const auto e_ij = -e_i - e_j;
719

721  for (int a = 0, tile_idx = 0; a != npno; ++a) {
722  const auto e_a = eps_v[a];
723 
724  for (int b = 0; b != npno; ++b, ++tile_idx) {
725  const auto e_b = eps_v[b];
726  const auto e_ab = e_a + e_b;
727

729  // const auto e_abij = e_a + e_b - e_i - e_j;
730  const auto K_abij = arg_tile[tile_idx];
731  const auto T_abij = -K_abij / e_abij;
732  const auto abs_result = std::abs(T_abij);
733  norm += abs_result * abs_result;
734  result_tile[tile_idx] = T_abij;
735 
736  } // b
737  } // a
738  return std::sqrt(norm);
739  }; // form_T

742  T2.world().gop.fence();

746 template <typename Tile, typename Policy>
747 TA::DistArray<Tile, Policy> construct_density(
748  const TA::DistArray<Tile, Policy>& t2) {
749  using Matrix = RowMatrix<typename Tile::numeric_type>;
751  // lambda function to transform T to D; implement using a for_each
752  std::size_t nuocc = t2.trange().dim(0).extent();
753  auto form_D = [nuocc](Tile& result_tile, const Tile& arg_tile) {
754 
755  result_tile = Tile(arg_tile.range());
756 
757  // Get values of i,j and compute delta_ij
758  const int i = arg_tile.range().lobound()[2];
759  const int j = arg_tile.range().lobound()[3];
760 
761  double delta_ij = (i == j) ? 1.0 : 0.0;
762 
763  // Form T_ij matrix from arg_tile
764  Matrix T_ij = TA::eigen_map(arg_tile, nuocc, nuocc);
765 
766  // Form T_ji matrix from T_ij
767  Matrix T_ji = T_ij.transpose();
768 
769  // Form D_ij from T_ij and T_ji
770  Matrix D_ij =
771  (1.0 / (1.0 + delta_ij)) * (4.0 * T_ji * T_ij - 2.0 * T_ij * T_ij +
772  4.0 * T_ij * T_ji - 2.0 * T_ji * T_ji);
773 
774  // Transform D_ij into a tile
775  auto norm = 0.0;
776  std::size_t tile_idx = 0;
777  for (std::size_t a = 0; a < nuocc; ++a) {
778  for (std::size_t b = 0; b < nuocc; ++b, ++tile_idx) {
779  const auto elem = D_ij(a, b);
780  norm += elem * elem;
781  result_tile[tile_idx] = elem;
782  }
783  }
784  return std::sqrt(norm);
785  }; // form_D
786 
787  auto D = TA::foreach (t2, form_D);
788  t2.world().gop.fence();
789 
790  return D;

1721  template <typename Factory, typename OFactory> auto compute_mp1_t2(const T& K, Factory&& fac, OFactory&& ofac) {
1722 
1723  auto& world = fac.world();
1724 
1725  auto nocc = ofac.retrieve("m").rank();
1726  auto nocc_act = ofac.retrieve("i").rank();
1727  auto nvir = ofac.retrieve("a").rank();
1728  auto nfzc = nocc - nocc_act;
1729 
1730  auto F = fac.compute(formula(L"(p|F|q)"));
1731  Eigen::VectorXd eps_p = math::array_to_eigen(F).diagonal();
1732  // replicated diagonal elements of Fo
1733  auto eps_o = eps_p.segment(nfzc, nocc_act);
1734  // replicated diagonal elements of Fv
1735  auto eps_v = eps_p.tail(nvir);
1736 
1737  // Fij
1738  auto Fo = fac.compute(formula(L"(i|F|j)"));
1739  // Fab
1740  auto Fv = fac.compute(formula(L"(a|F|b)"));
1741 
1742  // zero out amplitudes
1743  auto T2 = T(world, K.trange(), K.shape());
1744  T2.fill(0.0);
1745 
1746  // lambda function will be used to do a Jacobi update of the residual
1747  auto jacobi_update = [eps_o, eps_v](TA::TensorD& result_tile) {
1748 
1749  const auto& range = result_tile.range();
1750  double norm = 0.0;
1751  for (const auto& i : range) {
1752  const auto result_abij = result_tile[i] / (eps_o[i[2]] + eps_o[i[3]]
1753  - eps_v[i[0]] - eps_v[i[1]]);
1754  result_tile[i] = result_abij;
1755  norm += result_abij * result_abij;
1756  }
1757  return std::sqrt(norm);
1758  };
1759 
1760  // solve the MP1 equations
1761  auto converged = false;
1762  auto iter = 0;
1763  auto energy = +1.0;
1764  const auto target_precision = 1e-6;
1765  ExEnv::out0() << "MP1 T2 Solver\n";
1766  while (not converged) {
1767  T R;
1768  R("a,b,i,j") = K("a,b,i,j") + Fv("a,c") * T2("c,b,i,j") +
1769  Fv("b,c") * T2("a,c,i,j") - Fo("i,k") * T2("a,b,k,j") -
1770  Fo("j,k") * T2("a,b,i,k");
1771 
1772  // estimate the MP2 energy ... the Hylleraas formula is quadratic in error
1773  double updated_energy =
1774  (K("a,b,i,j") + R("a,b,i,j")).dot(2 * T2("a,b,i,j") - T2("b,a,i,j"));
1775  ExEnv::out0() << indent << "Iteration: " << iter
1776  << " Energy: " << updated_energy << std::endl;
1777 
1778  const auto computed_precision = std::abs(updated_energy - energy);
1779  energy = updated_energy;
1780 
1781  // update the amplitudes, if needed
1782  converged = computed_precision <= target_precision;
1783  if (not converged) {
1784  // R^{ij}_{ab} -> R^{ij}_{ab} / (F^i_i + F^j_j - F^a_a - F^b_b)
1785  TA::foreach_inplace(R, jacobi_update);
1786  // need a fence here since foreach_inplace mutates the contents of R
1787  // as a side effect.
1788  // N.B. most TiledArray ops will not need a fence (except to control
1789  // the resource use)
1790  world.gop.fence();
1791  T2("a,b,i,j") += R("a,b,i,j");
1792  ++iter;
1793  }
1794  }
1795 
1796  return T2;
1797  }
