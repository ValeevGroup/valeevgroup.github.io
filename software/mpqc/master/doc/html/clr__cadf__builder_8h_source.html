258  ArrayType compute_J(ArrayType const &D) {
259  auto &world = D.world();
260  auto j0 = mpqc::fenced_now(world);
261  ArrayType J;
262  J("mu, nu") =
263  E_("X, mu, nu") *
264  (Mchol_inv_("Z, X") *
265  (Mchol_inv_("Z, Y") * (E_("Y, rho, sigma") * D("rho, sigma"))));
266  auto j1 = mpqc::fenced_now(world);
267  j_times_.push_back(mpqc::duration_in_s(j0, j1));
268 
269  return J;
270  }

272  array_type compute_K(ArrayType const &C_in) {
273  auto &world = M_.world();
274  ArrayType K; // Matrices
275  DArrayType dC, dL; // Matrices
276  DArrayType C_mo, F_df; // Tensors
277 
278  // Deep copy C for chopping
279  ArrayType C;
280  C("mu, i") = C_in("mu, i");
281  // Capture C sizes
282  lcao_sizes_.push_back(detail::array_storage(C));
283 
284  if (lcao_chop_threshold_ != 0.0) {
285  auto chop0 = mpqc::fenced_now(world);
286  TA::foreach_inplace(C, [&](TA::Tensor<double> &t) {
287  const auto norm = t.norm();
288  if (norm > lcao_chop_threshold_) {
289  return norm;
290  } else {
291  return 0.0;
292  }
293  });
294  auto chop1 = mpqc::fenced_now(world);
295  lcao_chop_times_.push_back(mpqc::duration_in_s(chop0, chop1));
296 
297  lcao_chopped_sizes_.push_back(detail::array_storage(C));
298  }
299 
300  bool compress_C = false;
301  dC = TA::to_new_tile_type(
302  C, TaToDecompTensor(clr_threshold_, compress_C));
303 
304  // Contract C_df with orbitals
305  auto c_mo0 = mpqc::fenced_now(world);
306  C_mo("X, i, mu") = C_df_("X, mu, nu") * dC("nu, i");
307  C_mo.truncate();
308  auto c_mo1 = mpqc::fenced_now(world);
309  c_mo_times_.push_back(mpqc::duration_in_s(c_mo0, c_mo1));
310  c_mo_sizes_.push_back(detail::array_storage(C_mo));
311 
312  // Get forced output shape
313  TA::SparseShape<float> forced_shape;
314  if (use_forced_shape_) {
315  auto cadf_df_k_shape = [&](TA::Tensor<float> const &input) {
316  auto &range = input.range();
317  auto extent = range.extent_data();
318 
319  TA::Tensor<float> t(range, 0.0);
320 
321  std::unordered_set<int64_t> sig_mu;
322  std::unordered_set<int64_t> sig_X;
323 
324  for (auto i = 0ul; i < extent[1]; ++i) {
325  sig_mu.clear();
326  sig_X.clear();
327 
328  // For every I we determine save a list of important X and mu
329  for (auto X = 0ul; X < extent[0]; ++X) {
330  for (auto mu = 0ul; mu < extent[2]; ++mu) {
331  const auto val = input(X, i, mu);
332 
333  if (val >= force_threshold_) {
334  sig_X.insert(X);
335  sig_mu.insert(mu);
336  }
337  }
338  }
339 
340  // Then for every X we mark all important mu. The output will have
341  // significant mu X pairs that the input did not.
342  for (auto const &X : sig_X) {
343  for (auto const &mu : sig_mu) {
344  t(X, i, mu) = std::numeric_limits<float>::max();
345  }
346  }
347  }
348 
349  return t;
350  };
351 
352  auto shape_time0 = mpqc::fenced_now(world);
353  forced_shape = C_mo.shape().transform(cadf_df_k_shape);
354  auto shape_time1 = mpqc::fenced_now(world);
355  shape_times_.push_back(mpqc::duration_in_s(shape_time0, shape_time1));
356  }
357 
358  // Construct F_df
359  auto f_df0 = mpqc::fenced_now(world);
360  if (!use_forced_shape_) {
361  auto emo0 = mpqc::fenced_now(world);
362  DArrayType E_mo;
363  E_mo("X,i,mu") = dE_("X, mu, nu") * dC("nu,i");
364  E_mo.truncate();
365  auto emo1 = mpqc::fenced_now(world);
366  e_mo_times_.push_back(mpqc::duration_in_s(emo0, emo1));
367  e_mo_sizes_.push_back(detail::array_storage(E_mo));
368 
369  auto rdf0 = mpqc::fenced_now(world);
370  DArrayType R_df;
371  R_df("X, i, mu") = M_("X, Y") * C_mo("Y, i, mu");
372  auto rdf1 = mpqc::fenced_now(world);
373  r_df_times_.push_back(mpqc::duration_in_s(rdf0, rdf1));
374  F_df("X, i, mu") = E_mo("X, i, mu") - 0.5 * R_df("X, i, mu");
375  } else {
376  auto emo0 = mpqc::fenced_now(world);
377  DArrayType E_mo;
378  E_mo("X,i,mu") = (dE_("X, mu, nu") * dC("nu,i")).set_shape(forced_shape);
379  E_mo.truncate();
380  auto emo1 = mpqc::fenced_now(world);
381  e_mo_times_.push_back(mpqc::duration_in_s(emo0, emo1));
382  e_mo_sizes_.push_back(detail::array_storage(E_mo));
383 
384  auto rdf0 = mpqc::fenced_now(world);
385  DArrayType R_df;
386  R_df("X, i, mu") =
387  (M_("X, Y") * C_mo("Y, i, mu")).set_shape(forced_shape);
388  auto rdf1 = mpqc::fenced_now(world);
389  r_df_times_.push_back(mpqc::duration_in_s(rdf0, rdf1));
390 
391  F_df("X, i, mu") = E_mo("X, i, mu") - 0.5 * R_df("X, i, mu");
392  }
393  F_df.truncate();
394  auto f_df1 = mpqc::fenced_now(world);
395  f_df_times_.push_back(mpqc::duration_in_s(f_df0, f_df1));
396  f_df_sizes_.push_back(detail::array_storage(F_df));
397 
398  // Construct L
399  auto l0 = mpqc::fenced_now(world);
400  dL("mu, nu") = C_mo("X, i, mu") * F_df("X, i, nu");
401  dL.truncate();
402  auto L = TA::to_new_tile_type(dL, DecompToTaTensor{});
403  auto l1 = mpqc::fenced_now(world);
404  l_times_.push_back(mpqc::duration_in_s(l0, l1));
405 
406  auto k0 = mpqc::fenced_now(world);
407  l_times_.push_back(mpqc::duration_in_s(l0, l1));
408  K("mu, nu") = L("mu, nu") + L("nu, mu");
409  K.truncate();
410  auto k1 = mpqc::fenced_now(world);
411  k_times_.push_back(mpqc::duration_in_s(k0, k1));
412 
413  return K;
414  }
