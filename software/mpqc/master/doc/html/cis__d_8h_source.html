33 
34  if (df) {

37

39  result("a,b,i,j") += tmp("a,b,i,j") + tmp("b,a,j,i");
40 
41  } else {
42  auto g_iabc = lcao_factory.compute(L"<i a|G|b c>");
43 
44  TA::DistArray<Tile, Policy> tmp;
45  tmp("a,b,i,j") = g_iabc("i,c,a,b") * cis_ampl("j,c");
46  result("a,b,i,j") += tmp("a,b,i,j") + tmp("b,a,j,i");
47  }

50  auto f_ij = lcao_factory.compute(L"<i|F|j>" + postfix);

58  ens << eps_o, eps_v;
59

61 
62  return result;
