131 template <typename Tile, typename Policy>
132 double dot_product(TA::DistArray<Tile, Policy> const &L,
133  TA::DistArray<Tile, Policy> const &R,
134  Vector3i const &L_range, Vector3i const &R_range) {
135  using ::mpqc::detail::direct_ord_idx;
136  auto L_size = 1 + direct_ord_idx(L_range, L_range);
137  auto R_size = 1 + direct_ord_idx(R_range, R_range);
138  auto L_tiles_range = L.trange().tiles_range();
139  auto R_tiles_range = R.trange().tiles_range();
140  assert(L_tiles_range.extent(1) == L_tiles_range.extent(0) * L_size);
141  assert(R_tiles_range.extent(1) == R_tiles_range.extent(0) * R_size);
142 
143  double result;
144  if (L_range == R_range) {
145  result = L("m, n").dot(R("m, n"));
146  } else if (L_range(0) <= R_range(0) && L_range(1) <= R_range(1) &&
147  L_range(2) <= R_range(2)) {
148  auto reduced_R = reduced_size_array(R, R_range, L_range);
149  result = L("m, n").dot(reduced_R("m, n"));
150  } else if (L_range(0) >= R_range(0) && L_range(1) >= R_range(1) &&
151  L_range(2) >= R_range(2)) {
152  auto reduced_L = reduced_size_array(L, L_range, R_range);
153  result = reduced_L("m, n").dot(R("m, n"));
154  } else {
155  auto x = std::min(L_range(0), R_range(0));
156  auto y = std::min(L_range(1), R_range(1));
157  auto z = std::min(L_range(2), R_range(2));
158  Vector3i min_range(x, y, z);
159  auto reduced_L = reduced_size_array(L, L_range, min_range);
160  auto reduced_R = reduced_size_array(R, R_range, min_range);
161  result = reduced_L("m, n").dot(reduced_R("m, n"));
162  }
163  return result;
164 }

295 template <typename Tile, typename Policy>
296 Vector3i truncate_lattice_range(const TA::DistArray<Tile, Policy> &D,
297  const Vector3i &RD_max,
298  const double threshold) {
299  ExEnv::out0() << "\tUser specified lattice range = " << RD_max.transpose()
300  << std::endl;
301 
302  using ::mpqc::detail::direct_3D_idx;
303  using ::mpqc::detail::direct_ord_idx;
304 
305  const auto &Dnorms = D.shape().data();
306  std::vector<Vector3i> sig_density_uc_list;
307  const auto RD_size = 1 + direct_ord_idx(RD_max, RD_max);
308 
309  const auto ntiles = D.trange().tiles_range().extent(0);
310 
311  // determine significant unit cells
312  for (auto RD_ord = 0ul; RD_ord != RD_size; ++RD_ord) {
313  const auto RD_3D = direct_3D_idx(RD_ord, RD_max);
314  auto is_significant = false;
315  for (auto mu = 0ul; mu != ntiles; ++mu) {
316  for (auto nu = 0ul; nu != ntiles; ++nu) {
317  const auto nu_in_D = nu + RD_ord * ntiles;
318  std::array<size_t, 2> idx{{mu, nu_in_D}};
319  if (Dnorms(idx) >= threshold) {
320  is_significant = true;
321  sig_density_uc_list.emplace_back(RD_3D);
322  break;
323  }
324  }
325  if (is_significant) {
326  break;
327  }
328  }
329  }
330 
331  // renew lattice range based on the list of significant unit cells
332  auto x = 0;
333  auto y = 0;
334  auto z = 0;
335  for (const auto &RD_3D : sig_density_uc_list) {
336  x = std::max(x, RD_3D(0));
337  y = std::max(y, RD_3D(1));
338  z = std::max(z, RD_3D(2));
339  }
340  Vector3i new_RD_max = {x, y, z};
341  ExEnv::out0() << "\tUpdated lattice range = " << new_RD_max.transpose()
342  << std::endl;
343 
344  return new_RD_max;
345 }

356 template <typename Tile, typename Policy>
357 TA::DistArray<Tile, Policy> symmetrize_matrix(
358  const TA::DistArray<Tile, Policy> &M_unsymm) {
359  const auto &elements_range = M_unsymm.trange().elements_range();
360  const auto ext0 = elements_range.extent(0);
361  const auto ext1 = elements_range.extent(1);
362 
363  // compute total number of unit cells
364  assert(ext1 % ext0 == 0);
365  const auto ncells = ext1 / ext0;
366 
367  // determine the index of the reference cell
368  assert(ncells > 0 && ncells % 2 == 1);
369  const auto R_ref = (ncells - 1) / 2;
370 
371  // initialize an empty matrix
372  auto M_unsymm_eig = math::array_to_eigen(M_unsymm);
373  RowMatrixXd M_symm_eig(ext0, ext1);
374 
375  // fill matrix elements for each R
376  for (auto Rp_ord = R_ref; Rp_ord != ncells; ++Rp_ord) {
377  const auto distance = Rp_ord - R_ref;
378  const auto Rm_ord = R_ref - distance;
379  const auto Mp_unsymm = M_unsymm_eig.block(0, ext0 * Rp_ord, ext0, ext0);
380  const auto Mm_unsymm = M_unsymm_eig.block(0, ext0 * Rm_ord, ext0, ext0);
381  M_symm_eig.block(0, ext0 * Rp_ord, ext0, ext0) =
382  0.5 * (Mp_unsymm + Mm_unsymm.transpose());
383  if (distance != 0) {
384  M_symm_eig.block(0, ext0 * Rm_ord, ext0, ext0) =
385  0.5 * (Mm_unsymm + Mp_unsymm.transpose());
386  }
387  }
388 
389  // convert to an TA Array
390  auto &world = M_unsymm.world();
391  const auto &tr0 = M_unsymm.trange().dim(0);
392  const auto &tr1 = M_unsymm.trange().dim(1);
393  auto M_symm =
394  math::eigen_to_array<Tile, Policy>(world, M_symm_eig, tr0, tr1);
395  M_symm.truncate();
396  world.gop.fence();
397 
398  return M_symm;
399 }
