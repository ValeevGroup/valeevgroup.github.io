184 int GetLongOpt::setcell(Cell &c, const char *valtoken, const char *nexttoken,
185  const std::string &name) {
186  switch (c.type) {
187  case GetLongOpt::NoValue:
188  if (*valtoken == '=') {
189  throw std::runtime_error("unsolicited value for program option " +
190  std::string(1, optmarker) + c.option);
191  }
192  c.value = std::make_unique<std::string>("");
193  return 0;
194  case GetLongOpt::OptionalValue:
195  if (*valtoken == '=') {
196  c.value = std::make_unique<std::string>(++valtoken);
197  } else if (nexttoken != 0 && nexttoken[0] != optmarker) {
198  c.value = std::make_unique<std::string>(nexttoken);
199  return 1;
200  }
201  else

203  return 0;
204  case GetLongOpt::MandatoryValue:
205  if (*valtoken == '=') {

207  return 0;
