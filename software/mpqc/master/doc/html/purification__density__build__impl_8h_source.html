43 template<typename Tile, typename Policy>
44 typename PurificationDensityBuilder<Tile, Policy>::array_type
45 PurificationDensityBuilder<Tile, Policy>::purify(
46  typename PurificationDensityBuilder<Tile, Policy>::array_type const &F) {
47  auto &world = F.world();
48 
49  typename PurificationDensityBuilder<Tile, Policy>::array_type Fp, D, D2;
50  Fp("i,j") = M_inv_("i,k") * F("k,l") * M_inv_("j,l");
51 
52  auto eig_pair = math::eval_guess(Fp);
53  auto emax = eig_pair[1];
54  auto emin = eig_pair[0];
55  auto scale = 1.0 / (emax - emin);
56 
57  D("i,j") = scale * (emax * I_("i,j") - Fp("i,j"));
58 
59  auto trace = D("i,j").trace().get();
60 
61  auto iter = 0;
62  auto error = std::abs(trace - occ_);
63  while (error >= 1e-13 && iter <= 100) {
64  // Compute D2
65  D2("i,j") = D("i,k") * D("k,j");
66  if (trace > occ_) {
67  D = D2;
68  } else {
69  D("i,j") = 2 * D("i,j") - D2("i,j");
70  }
71  D.truncate();
72  trace = D("i,j").trace().get();
73  error = std::abs(trace - occ_);
74  ++iter;
75  }
76 
77  D("i,j") = M_inv_("i,k") * D("k,l") * M_inv_("l,j");
78  D.truncate();
79  world.gop.fence();
80 
81  return D;
82 }
