69  template <typename Op>
70  static vector_type
71  recursive_outer_product(const vector_type* const size_vectors,
72  const unsigned int dim, const Op& op)
73  {
74  vector_type result;
75 
76  if(dim == 1u) {
77  // Construct a modified copy of size_vector[0]
78  result = op(*size_vectors);
79  } else {
80  // Compute split the range and compute the outer products
81  const unsigned int middle = (dim >> 1u) + (dim & 1u);
82  const vector_type left = recursive_outer_product(size_vectors, middle, op);
83  const vector_type right = recursive_outer_product(size_vectors + middle, dim - middle, op);
84 
85  // Compute the outer product of left and right
86 
87  result = vector_type(left.size() * right.size());
88  result.outer_fill(left, right,
89  [] (const value_type left, const value_type right) { return left * right; });
90  }
91 
92  return result;
93  }

101  void normalize() {
102  const value_type threshold = threshold_;
103  const unsigned int dim = tile_norms_.range().dim();
104  const vector_type* restrict const size_vectors = size_vectors_.get();
105  size_type zero_tile_count = 0ul;
106 
107  if(dim == 1u) {
108  auto normalize_op = [threshold, &zero_tile_count] (value_type& norm, const value_type size) {
109  TA_ASSERT(norm >= value_type(0));
110  norm /= size;
111  if(norm < threshold) {
112  norm = value_type(0);
113  ++zero_tile_count;
114  }
115  };
116 
117  // This is the easy case where the data is a vector and can be
118  // normalized directly.
119  math::inplace_vector_op(normalize_op, size_vectors[0].size(),
120  tile_norms_.data(), size_vectors[0].data());
121 
122  } else {
123  // Here the normalization constants are computed and multiplied by the
124  // norm data using a recursive, outer-product algorithm. This is done to
125  // minimize temporary memory requirements, memory bandwidth, and work.
126 
127  auto inv_vec_op = [] (const vector_type& size_vector) {
128  return vector_type(size_vector,
129  [] (const value_type size) { return value_type(1) / size; });
130  };
131 
132  // Compute the left and right outer products
133  const unsigned int middle = (dim >> 1u) + (dim & 1u);
134  const vector_type left = recursive_outer_product(size_vectors, middle, inv_vec_op);
135  const vector_type right = recursive_outer_product(size_vectors + middle, dim - middle, inv_vec_op);
136 
137  auto normalize_op = [threshold, &zero_tile_count] (value_type& norm,
138  const value_type x, const value_type y)
139  {
140  TA_ASSERT(norm >= value_type(0));
141  norm *= x * y;
142  if(norm < threshold) {
143  norm = value_type(0);
144  ++zero_tile_count;
145  }
146  };
147 
148  math::outer(left.size(), right.size(), left.data(), right.data(),
149  tile_norms_.data(), normalize_op);
150  }
151 
152  zero_tile_count_ = zero_tile_count;
153  }

500  SparseShape_ add(value_type value) const {
501  TA_ASSERT(! tile_norms_.empty());
502  const value_type threshold = threshold_;
503  size_type zero_tile_count = 0ul;
504 
505  Tensor<T> result_tile_norms(tile_norms_.range());
506 
507  value = std::abs(value);
508  const unsigned int dim = tile_norms_.range().dim();
509  const vector_type* restrict const size_vectors = size_vectors_.get();
510 
511  if(dim == 1u) {
512  auto add_const_op = [threshold, &zero_tile_count, value] (value_type norm,
513  const value_type size)
514  {
515  norm += value / std::sqrt(size);
516  if(norm < threshold) {
517  norm = 0;
518  ++zero_tile_count;
519  }
520  return norm;
521  };
522 
523  // This is the easy case where the data is a vector and can be
524  // normalized directly.
525  math::vector_op(add_const_op, size_vectors[0].size(), result_tile_norms.data(),
526  tile_norms_.data(), size_vectors[0].data());
527 
528  } else {
529  // Here the normalization constants are computed and multiplied by the
530  // norm data using a recursive, outer algorithm. This is done to
531  // minimize temporary memory requirements, memory bandwidth, and work.
532 
533  auto inv_sqrt_vec_op = [] (const vector_type size_vector) {
534  return vector_type(size_vector,
535  [] (const value_type size) { return value_type(1) / std::sqrt(size); });
536  };
537 
538  // Compute the left and right outer products
539  const unsigned int middle = (dim >> 1u) + (dim & 1u);
540  const vector_type left = recursive_outer_product(size_vectors, middle, inv_sqrt_vec_op);
541  const vector_type right = recursive_outer_product(size_vectors + middle, dim - middle, inv_sqrt_vec_op);
542 
543  math::outer_fill(left.size(), right.size(), left.data(), right.data(),
544  tile_norms_.data(), result_tile_norms.data(),
545  [threshold, &zero_tile_count, value] (value_type& norm,
546  const value_type x, const value_type y)
547  {
548  norm += value * x * y;
549  if(norm < threshold) {
550  norm = value_type(0);
551  ++zero_tile_count;
552  }
553  });
554  }
555 
556  return SparseShape_(result_tile_norms, size_vectors_, zero_tile_count);
557  }

593  static size_type scale_by_size(Tensor<T>& tile_norms,
594  const vector_type* restrict const size_vectors)
595  {
596  const unsigned int dim = tile_norms.range().dim();
597  const value_type threshold = threshold_;
598  size_type zero_tile_count = 0ul;
599 
600  if(dim == 1u) {
601  // This is the easy case where the data is a vector and can be
602  // normalized directly.
603  math::inplace_vector_op(
604  [threshold, &zero_tile_count] (value_type& norm, const value_type size) {
605  norm *= size;
606  if(norm < threshold) {
607  norm = value_type(0);
608  ++zero_tile_count;
609  }
610  },
611  size_vectors[0].size(), tile_norms.data(), size_vectors[0].data());
612  } else {
613  // Here the normalization constants are computed and multiplied by the
614  // norm data using a recursive, outer algorithm. This is done to
615  // minimize temporary memory requirements, memory bandwidth, and work.
616 
617  auto noop = [](const vector_type& size_vector) -> const vector_type& {
618  return size_vector;
619  };
620 
621  // Compute the left and right outer products
622  const unsigned int middle = (dim >> 1u) + (dim & 1u);
623  const vector_type left = recursive_outer_product(size_vectors, middle, noop);
624  const vector_type right = recursive_outer_product(size_vectors + middle, dim - middle, noop);
625 
626  math::outer(left.size(), right.size(), left.data(), right.data(), tile_norms.data(),
627  [threshold, &zero_tile_count] (value_type& norm, const value_type x,
628  const value_type y)
629  {
630  norm *= x * y;
631  if(norm < threshold) {
632  norm = value_type(0);
633  ++zero_tile_count;
634  }
635  });
636  }
637 
638  return zero_tile_count;

695  SparseShape_ gemm(const SparseShape_& other, value_type factor,
696  const math::GemmHelper& gemm_helper) const
697  {
698  TA_ASSERT(! tile_norms_.empty());
699 
700  factor = std::abs(factor);
701  const value_type threshold = threshold_;
702  size_type zero_tile_count = 0ul;
703  integer M = 0, N = 0, K = 0;
704  gemm_helper.compute_matrix_sizes(M, N, K, tile_norms_.range(), other.tile_norms_.range());
705 
706  // Allocate memory for the contracted size vectors
707  std::shared_ptr<vector_type> result_size_vectors(new vector_type[gemm_helper.result_rank()],
708  madness::detail::CheckedArrayDeleter<vector_type>());
709 
710  // Initialize the result size vectors
711  unsigned int x = 0ul;
712  for(unsigned int i = gemm_helper.left_outer_begin(); i < gemm_helper.left_outer_end(); ++i, ++x)
713  result_size_vectors.get()[x] = size_vectors_.get()[i];
714  for(unsigned int i = gemm_helper.right_outer_begin(); i < gemm_helper.right_outer_end(); ++i, ++x)
715  result_size_vectors.get()[x] = other.size_vectors_.get()[i];
716 
717  // Compute the number of inner ranks
718  const unsigned int k_rank = gemm_helper.left_inner_end() - gemm_helper.left_inner_begin();
719 
720  // Construct the result norm tensor
721  Tensor<value_type> result_norms(gemm_helper.make_result_range<typename Tensor<T>::range_type>(
722  tile_norms_.range(), other.tile_norms_.range()), 0);
723 
724  if(k_rank > 0u) {
725 
726  // Compute size vector
727  const vector_type k_sizes =
728  recursive_outer_product(size_vectors_.get() + gemm_helper.left_inner_begin(),
729  k_rank, [] (const vector_type& size_vector) -> const vector_type&
730  { return size_vector; });
731 
732  // TODO: Make this faster. It can be done without using temporaries
733  // for the arguments, but requires a custom matrix multiply.
734 
735  Tensor<value_type> left(tile_norms_.range());
736  const size_type mk = M * K;
737  auto left_op = [] (const value_type left, const value_type right)
738  { return left * right; };
739  for(size_type i = 0ul; i < mk; i += K)
740  math::vector_op(left_op, K, left.data() + i,
741  tile_norms_.data() + i, k_sizes.data());
742 
743  Tensor<value_type> right(other.tile_norms_.range());
744  for(integer i = 0ul, k = 0; k < K; i += N, ++k) {
745  const value_type factor = k_sizes[k];
746  auto right_op = [=] (const value_type arg) { return arg * factor; };
747  math::vector_op(right_op, N, right.data() + i, other.tile_norms_.data() + i);
748  }
749 
750  result_norms = left.gemm(right, factor, gemm_helper);
751 
752  // Hard zero tiles that are below the zero threshold.
753  result_norms.inplace_unary(
