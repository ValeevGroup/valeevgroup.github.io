42  void compute_cc2() {
43  auto n_occ = tre_->get_active_occ();
44 
45  TArray2 f_ai;
46  f_ai("a,i") = fock_("a,i");
47 
48  auto g_abij = intermediate_->get_abij();
49 
50  TArray2 d1(f_ai.world(), f_ai.trange(), f_ai.shape(), f_ai.pmap());
51  // store d1 to local
52  d_ai(d1, ens_, n_occ);
53 
54  TArray4 d2(g_abij.world(), g_abij.trange(), g_abij.shape(), g_abij.pmap());
55  // store d2 distributed
56  d_abij_inplace(d2, ens_, n_occ);
57 
58  TArray2 t1;
59  TArray4 t2;
60  t1("a,i") = f_ai("a,i") * d1("a,i");
61  t2("a,b,i,j") = g_abij("a,b,i,j") * d2("a,b,i,j");
62 
63  TArray4 tau;
64  tau("a,b,i,j") = t2("a,b,i,j") + t1("a,i") * t1("b,j");
65 
66  double E0 = 0.0;
67  double E1 =
68  2.0 * TA::dot(f_ai("a,i"), t1("a,i")) +
69  TA::dot(2.0 * g_abij("a,b,i,j") - g_abij("b,a,i,j"), tau("a,b,i,j"));
70  double dE = std::abs(E1 - E0);
71 
72  // get all two electron integrals
73  TArray4 g_ijkl = intermediate_->get_ijkl();
74  TArray4 g_abcd = intermediate_->get_abcd();
75  TArray4 g_iajb = intermediate_->get_iajb();
76  TArray4 g_iabc = intermediate_->get_iabc();
77  TArray4 g_aibc = intermediate_->get_aibc();
78  TArray4 g_ijak = intermediate_->get_ijak();
79  TArray4 g_ijka = intermediate_->get_ijka();
80 
81  intermediate_->clean();
82 
83  // optimize t1 and t2
84  std::size_t iter = 0ul;
85  while (dE >= 1.0e-12) {
86  // intermediates for t1
87  // external index i and a
88  TArray2 h_ac, h_ki, h_ck;
89  {
90  h_ac("a,c") =
91  -(2.0 * g_abij("c,d,k,l") - g_abij("c,d,l,k")) * tau("a,d,k,l");
92 
93  h_ki("k,i") =
94  (2.0 * g_abij("c,d,k,l") - g_abij("d,c,k,l")) * tau("c,d,i,l");
95 
96  h_ck("c,k") = f_ai("c,k") +
97  (2.0 * g_abij("c,d,k,l") - g_abij("d,c,k,l")) * t1("d,l");
98  }
99 
100  // update t1
101  TArray2 r1;
102  {
103  t1("a,i") =
104  d1("a,i") *
105  (
106  //
107  f_ai("a,i") - 2.0 * f_ai("c,k") * t1("a,k") * t1("c,i")
108  //
109  + t1("c,i") * h_ac("a,c") - t1("a,k") * h_ki("k,i")
110  //
111  +
112  h_ck("c,k") * (2.0 * t2("c,a,k,i") - t2("c,a,i,k") +
113  t1("c,i") * t1("a,k"))
114  //
115  + (2.0 * g_abij("c,a,k,i") - g_iajb("k,a,i,c")) * t1("c,k")
116  //
117  + (2.0 * g_iacd("k,a,c,d") - g_iacd("k,a,d,c")) * tau("c,d,k,i")
118  //
119  -
120  (2.0 * g_klai("k,l,c,i") - g_klai("l,k,c,i")) * tau("c,a,k,l"));
121  }
122 
123  // intermediates for t2
124  // external index i j a b
125 
126  TArray4 a_klij, b_abcd;
127  TArray2 g_ki, g_ac;
128  {
129  a_klij("k,l,i,j") = g_ijkl("k,l,i,j") + g_klia("k,l,i,c") * t1("c,j") +
130  g_klai("k,l,c,j") * t1("c,i") +
131  g_abij("c,d,k,l") * t1("c,i") * t1("d,j");
132 
133  b_abcd("a,b,c,d") = g_abcd("a,b,c,d") - g_aicd("a,k,c,d") * t1("b,k") -
134  g_iacd("k,b,c,d") * t1("a,k");
135 
136  // g_ki("k,i") = h_ki("k,i") + f_ai("c,k")*t1("c,i") +
137  // (2.0*g_klia("k,l,i,a")-g_klia("l,k,i,a"))*t1("a,l");
138  // g_ac("a,c") = h_ac("a,c") - f_ai("c,k")*t1("a,k") +
139  // (2.0*g_aicd("a,k,c,d")-g_aicd("a,k,d,c"))*t1("d,k");
140  }
141 
142  TArray4 r2;
143  t2("a,b,i,j") =
144  d2("a,b,i,j") *
145  (
146  //
147  g_abij("a,b,i,j")
148  //
149  + a_klij("k,l,i,j") * t1("a,k") * t1("b,l")
150  //
151  + b_abcd("a,b,c,d") * t1("c,i") * t1("d,j")
152  //
153  //+ (fab("a,c") * t2("c,b,i,j") + fab("b,c") * t2("a,c,i,j"))
154  // * (1.0 - Iab("a,c"))
155  //+ (fij("k,i") * t2("a,b,k,j") + fij("k,j") * t2("a,c,i,k"))
156  // * (1 - Iij("k,i"))
157  //
158  // + (g_ac("a,c")*t2("c,b,i,j") -
159  // g_ki("k,i")*t2("a,b,k,j"))
160  // + (g_ac("b,c")*t2("c,a,j,i") -
161  // g_ki("k,j")*t2("b,a,k,i"))
162 
163  +
164  (g_iacd("i,c,a,b") - g_iajb("k,b,i,c") * t1("a,k")) * t1("c,j") +
165  (g_iacd("j,c,b,a") - g_iajb("k,a,j,c") * t1("b,k")) * t1("c,i")
166  //
167  -
168  (g_klai("i,j,a,k") + g_abij("a,c,i,k") * t1("c,j")) * t1("b,k") -
169  (g_klai("j,i,b,k") + g_abij("b,c,j,k") * t1("c,i")) * t1("a,k"));
170 
171  // t1("a,i") = r1("a,i");
172  // t2("a,b,i,j") = r2("a,b,i,j");
173  tau("a,b,i,j") = t2("a,b,i,j") + t1("a,i") * t1("b,j");
174 
175  // recompute energy
176  E0 = E1;
177  E1 = 2.0 * TA::dot(f_ai("a,i"), t1("a,i")) +
178  TA::dot((2.0 * g_abij("a,b,i,j") - g_abij("b,a,i,j")),
179  tau("a,b,i,j"));
180  dE = std::abs(E0 - E1);
181  iter += 1ul;
182  ExEnv::out0() << iter << " " << dE << std::endl;
183  }
184  ExEnv::out0() << E1 << std::endl;
185  }
