77 double BasisPairInfo::prim_pair_extent(const double exp0, const double exp1,
78  const double rab) {
79  const auto exp_sum = exp0 + exp1;
80  const auto exp_mult = exp0 * exp1;
81 
82  // compute overlap integral between two primitives, neglecting angular parts
83  const auto prefactor = std::pow(4.0 * exp_mult / (exp_sum * exp_sum), 0.75);
84  const auto exponent = std::exp(-1.0 * (exp_mult / exp_sum) * rab * rab);
85  const auto S = prefactor * exponent;
86 
87  const auto extent2 =
88  (std::log(S) - std::log(thresh_) + 0.5 * std::log(exp_sum)) / exp_sum;
89  const auto extent = (extent2 < 0) ? small_extent_ : std::sqrt(extent2);
90 
91  return extent;
92 }
