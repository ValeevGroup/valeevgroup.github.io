134 template <typename Policy, bool is_real = true>
135 TA::DistArray<TA::TensorD, Policy> tensorZ_to_tensorD(
136  const TA::DistArray<TA::TensorZ, Policy> &complex_array) {
137  TA::DistArray<TA::TensorD, Policy> result_array;
138 
139  auto take_part_from_tile = [=](TA::TensorD &result_tile,
140  const TA::TensorZ &arg_tile) {
141  const auto &range = arg_tile.range();
142  const auto volume = range.volume();
143  result_tile = TA::TensorD(range);
144 
145  float norm = 0.0;
146  for (auto ord = 0; ord < volume; ++ord) {
147  const auto z = arg_tile[ord];
148  auto result_el = (is_real) ? z.real() : z.imag();
149  norm += result_el * result_el;
150  result_tile[ord] = result_el;
151  }
152 
153  return std::sqrt(norm);
154  };
155 
156  result_array =
157  TA::foreach<TA::TensorD, TA::TensorZ>(complex_array, take_part_from_tile);
158 
159  complex_array.world().gop.fence();
160 
161  return result_array;
162 }

164 template <typename Tile, typename Policy>
165 TA::DistArray<Tile, Policy> compute_shellblock_norm(
166  const Basis &bs0, const Basis &bs1, const TA::DistArray<Tile, Policy> &D) {
167  auto &world = D.world();
168  // make trange1
169  auto make_shblk_trange1 = [](const Basis &bs) {
170  const auto &shells_Vec = bs.cluster_shells();
171  auto blocking = std::vector<int64_t>{0};
172  for (const auto &shells : shells_Vec) {
173  const auto nshell = shells.size();
174  auto next = blocking.back() + nshell;
175  blocking.emplace_back(next);
176  }
177  return TA::TiledRange1(blocking.begin(), blocking.end());
178  };
179 
180  const auto tr0 = make_shblk_trange1(bs0);
181  const auto tr1 = make_shblk_trange1(bs1);
182 
183  auto eig_D = ::mpqc::math::array_to_eigen(D);
184  // compute shell block norms
185  const auto shells0 = bs0.flattened_shells();
186  const auto shells1 = bs1.flattened_shells();
187  const auto nshell0 = shells0.size();
188  const auto nshell1 = shells1.size();
189  RowMatrixXd norm_D(nshell0, nshell1);
190  for (auto sh0 = 0ul, sh0_first = 0ul; sh0 != nshell0; ++sh0) {
191  const auto sh0_size = shells0[sh0].size();
192  for (auto sh1 = 0ul, sh1_first = 0ul; sh1 != nshell1; ++sh1) {
193  const auto sh1_size = shells1[sh1].size();
194 
195  norm_D(sh0, sh1) = eig_D.block(sh0_first, sh1_first, sh0_size, sh1_size)
196  .template lpNorm<Eigen::Infinity>();
197 
198  sh1_first += sh1_size;
199  }
200 
201  sh0_first += sh0_size;
202  }
203 
204  return math::eigen_to_array<Tile, Policy>(world, norm_D, tr0, tr1);
205 }

230 template <typename Array>
231 Formula lcao_to_ao(const Formula &formula,
232  const OrbitalSpaceRegistry<Array> &orbital_registry) {
233  std::vector<OrbitalIndex> ao_left_index, ao_right_index;
234 
235  int increment = 0;
236  auto left_index = formula.bra_indices();
237  for (const auto &index : left_index) {
238  // find the correspoding ao index
239  if (index.is_lcao()) {
240  auto ao_index = orbital_registry.retrieve(index).ao_index().name();
241  ao_index = ao_index + std::to_wstring(increment);
242  ao_left_index.push_back(ao_index);
243  increment++;
244  }
245  // if already ao, do nothing
246  else {
247  ao_left_index.push_back(index);
248  }
249  }
250 
251  auto right_index = formula.ket_indices();
252  for (const auto &index : right_index) {
253  // find the correspoding ao index
254  if (index.is_lcao()) {
255  auto ao_index = orbital_registry.retrieve(index).ao_index().name();
256  ao_index = ao_index + std::to_wstring(increment);
257  ao_right_index.push_back(ao_index);
258  increment++;
259  }
260  // if already ao, do nothing
261  else {
262  ao_right_index.push_back(index);
263  }
264  }
265 
266  // set formula with ao index
267  auto ao_formula = formula;
268  ao_formula.set_bra_indices(ao_left_index);
269  ao_formula.set_ket_indices(ao_right_index);
270 
271  return ao_formula;
272 }
