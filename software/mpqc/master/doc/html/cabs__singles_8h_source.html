74 template <typename Tile>
75 typename CABSSingles<Tile>::real_t CABSSingles<Tile>::compute(
76  bool df, bool d_approach, bool couple_virtual) {
77  TArray F_AB, F_MN;
78  if (df) {
79  if (d_approach) {
80  F_AB = lcao_factory_.compute(L"<A'|hJ|B'>[df]");
81 
82  } else {
83  F_AB = lcao_factory_.compute(L"<A'|F|B'>[df]");
84  }
85  F_MN = lcao_factory_.compute(L"<m|F|n>[df]");
86 
87  } else {
88  if (d_approach) {
89  F_AB = lcao_factory_.compute(L"<A'|hJ|B'>");
90 
91  } else {
92  F_AB = lcao_factory_.compute(L"<A'|F|B'>");
93  }
94  F_MN = lcao_factory_.compute(L"<m|F|n>");
95  }
96 
97  TArray F_MA;
99  if (couple_virtual) {
100  if (df) {
101  F_MA = lcao_factory_.compute(L"<m|F|A'>[df]");
102  } else {
103  F_MA = lcao_factory_.compute(L"<m|F|A'>");
104  }
105  }
107  else {
108  TArray F_Ma;
109  if (df) {
110  F_Ma = lcao_factory_.compute(L"<m|F|a'>[df]");
111  } else {
112  F_Ma = lcao_factory_.compute(L"<m|F|a'>");
113  }
114  RowMatrixXd F_Ma_eigen = math::array_to_eigen(F_Ma);
115  auto n_occ = F_Ma_eigen.rows();
116  auto n_cabs = F_Ma_eigen.cols();
117  auto n_allvir = F_AB.trange().elements_range().extent()[0];
118  auto n_vir = n_allvir - n_cabs;
119 
120  RowMatrixXd F_MA_eigen = RowMatrixXd::Zero(n_occ, n_allvir);
121  F_MA_eigen.block(0, n_vir, n_occ, n_cabs) << F_Ma_eigen;
122 
123  auto tr_m = F_Ma.trange().data()[0];
124  auto tr_A = F_AB.trange().data()[0];
125 
126  F_MA =
127  math::eigen_to_array<Tile,TA::SparsePolicy>(F_Ma.world(), F_MA_eigen, tr_m, tr_A);
128  }
129  // std::cout << F_MA << std::endl;
130 
131  TArray t;
132  t("i,A") = -F_MA("i,A");
133  t.truncate();
134 
135  // compute preconditioner
136  TArray P_MA = compute_preconditioner(F_MA.trange(), F_AB, F_MN);
137  // std::cout << P_MA << std::endl;
138 
139  CABSSingleEquation cabs_singles(F_AB, F_MN);
140 
141  double converge = 1e-12;
142 
143  TA::ConjugateGradientSolver<TArray, CABSSingleEquation> cg_solver;
144  auto resnorm = cg_solver(cabs_singles, F_MA, t, P_MA, converge);
145 
146  real_t E_S(0.0);
147  if (resnorm < converge) {
148  E_S = 2 * TA::dot(t("i,A"), F_MA("i,A"));
149  } else {
150  utility::print_par(t.world(),
151  "\n Warning! CABSSingles Not Converged!!! \n");
152  }
153  return E_S;
154 }

156 template <typename Tile>
157 TA::DistArray<Tile, TA::SparsePolicy> CABSSingles<Tile>::compute_preconditioner(
158  const TA::TiledRange& trange,
159  const TA::DistArray<Tile, TA::SparsePolicy>& F_AB,
160  const TA::DistArray<Tile, TA::SparsePolicy>& F_MN) {
161  auto& world = F_AB.world();
162 
163  Eigen::MatrixXd F_AB_eigen = math::array_to_eigen(F_AB);
164  Eigen::MatrixXd F_MN_eigen = math::array_to_eigen(F_MN);
165 
166  using range_type = typename TArray::range_type;
167 
168  auto make_tile = [&F_AB_eigen, &F_MN_eigen](
169  int64_t ord, range_type range, Tile* out_tile, TA::TensorF* norms) {
170  auto result_tile = Tile(range);
171  // compute index
172  const auto i0 = result_tile.range().lobound()[0];
173  const auto in = result_tile.range().upbound()[0];
174  const auto a0 = result_tile.range().lobound()[1];
175  const auto an = result_tile.range().upbound()[1];
176 
177  auto ia = 0;
178  typename Tile::value_type tmp = 1.0;
179  for (auto i = i0; i < in; ++i) {
180  const auto fi = F_MN_eigen(i, i);
181  for (auto a = a0; a < an; ++a, ++ia) {
182  const auto fa = F_AB_eigen(a, a);
183  const auto result_ai = tmp / (fi - fa);
184  result_tile[ia] = result_ai;
185  }
186  }
187 
188  const auto tile_volume = result_tile.range().volume();
189  const auto tile_norm = result_tile.norm();
190  bool save_norm =
191  tile_norm >= tile_volume * TA::SparseShape<float>::threshold();
192  if (save_norm) {
193  *out_tile = result_tile;
194  (*norms)[ord] = tile_norm;
195  }
196  };
197 
198  const auto tvolume = trange.tiles_range().volume();
199  std::vector<Tile> tiles(tvolume);
200  TA::TensorF tile_norms(trange.tiles_range(), 0.0);
201 
202  auto pmap = TA::SparsePolicy::default_pmap(world, tvolume);
203  for (auto const ord : *pmap) {
204  world.taskq.add(make_tile, ord, trange.make_tile_range(ord), &tiles[ord],
205  &tile_norms);
206  }
207 
208  world.gop.fence();
209 
210  TA::SparseShape<float> shape(world, tile_norms, trange);
211  TArray P_MA(world, trange, shape, pmap);
212 
213  for (auto const ord : *pmap) {
214  if (P_MA.is_local(ord) && !P_MA.is_zero(ord)) {
215  auto& tile = tiles[ord];
216  assert(!tile.empty());
217  P_MA.set(ord, tile);
218  }
219  }
220  world.gop.fence();
221 
222  P_MA.truncate();
223 
224  return P_MA;
225 }
