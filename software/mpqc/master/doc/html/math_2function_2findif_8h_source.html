72  virtual LinearCombinationOfDisplacements generate_displacements(
73  DerivIdx idx) const {
76  LinearCombinationOfDisplacements result;
77  switch (Order) {
78  case 1: {
79  const auto one_over_delta_ = 1 / delta_;
80  result.emplace_back(
81  std::make_pair(Displacement{{delta_}}, one_over_delta_));

84  } break;
85  default:

87  }

126  // for each derivative generate displacements

139  // compute
