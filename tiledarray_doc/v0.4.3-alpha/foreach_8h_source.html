162  template <typename T, unsigned int DIM, typename Tile, typename Op>
163  inline Array<T, DIM, Tile, SparsePolicy>
164  foreach(const Array<T, DIM, Tile, SparsePolicy> arg, Op&& op) {
165  typedef Array<T, DIM, Tile, SparsePolicy> array_type;
166  typedef typename array_type::value_type value_type;
167  typedef typename array_type::size_type size_type;
168  typedef typename array_type::shape_type shape_type;
169  typedef Future<value_type> future_type;
170  typedef std::pair<size_type, future_type> datum_type;
171 
172  // Create a vector to hold local tiles
173  std::vector<datum_type> tiles;
174  tiles.reserve(arg.get_pmap()->size());
175 
176  // Collect updated shape data.
177  TiledArray::Tensor<typename shape_type::value_type,
178  Eigen::aligned_allocator<typename shape_type::value_type> >
179  tile_norms(arg.trange().tiles(), 0);
180 
181  // Construct the new tile norms and
182  madness::AtomicInt counter; counter = 0;
183  int task_count = 0;
184  auto task = [&](const size_type index, const value_type& arg_tile) {
185  value_type result_tile;
186  tile_norms[index] = op(result_tile, arg_tile);
187  ++counter;
188  return result_tile;
189  };
190 
191  World& world = arg.get_world();
192 
193  // Get local tile index iterator
194  typename array_type::pmap_interface::const_iterator
195  it = arg.get_pmap()->begin(),
196  end = arg.get_pmap()->end();
197  for(; it != end; ++it) {
198  const size_type index = *it;
199  if(arg.is_zero(index))
200  continue;
201  future_type arg_tile = arg.find(index);
202  future_type result_tile = world.taskq.add(task, index, arg_tile);
203  ++task_count;
204  tiles.push_back(datum_type(index, result_tile));
205  }
206 
207  // Wait for tile norm data to be collected.
208  if(task_count > 0)
209  world.await([&counter,task_count] () {return counter == task_count; });
210 
211  // Construct the new array
212  array_type result(world, arg.trange(),
213  shape_type(world, tile_norms, arg.trange()), arg.get_pmap());
214  for(typename std::vector<datum_type>::const_iterator it = tiles.begin(); it != tiles.end(); ++it) {
215  const size_type index = it->first;
216  if(! result.is_zero(index))
217  result.set(it->first, it->second);
218  }
219 
220  return result;
221  }
