384 template <typename Tile, typename Policy>
385 std::pair<typename zRHF<Tile, Policy>::array_type,
386  typename zRHF<Tile, Policy>::array_type_z>
387 zRHF<Tile, Policy>::compute_density() {
388  auto& ao_factory = this->ao_factory();
389  auto& world = ao_factory.world();
390 
391  using ::mpqc::detail::extend_trange1;
392 
393  auto tr0 = Fk_.trange().data()[0];
394  auto tr1_real = extend_trange1(tr0, RD_size_);
395  auto tr1_recip = extend_trange1(tr0, k_size_);
396 
397  const auto ext0 = tr0.extent();
398  RowMatrixXd result_real_eig(ext0, tr1_real.extent());
399  MatrixZ result_recip_eig(ext0, tr1_recip.extent());
400  result_real_eig.setZero();
401  result_recip_eig.setZero();
402 
403  MatrixzVec F_recip_vec, D_recip_vec;
404  MatrixdVec D_real_vec;
405  F_recip_vec.resize(k_size_);
406  D_recip_vec.resize(k_size_);
407  D_real_vec.resize(RD_size_);
408 
409  auto fock_eig = math::array_to_eigen(Fk_);
410 
411  // parallel impl for F_k diagonalization and D_k build
412  auto compute_recip_density =
413  [this](MatrixZ* F_ptr, MatrixZ* X_ptr, MatrixZ* C_old_ptr, MatrixZ* C_ptr,
414  VectorD* eps_ptr, MatrixZ* D_ptr, bool is_gamma_point,
415  bool do_level_shift) {
416  // get references to matrix pointers
417  const auto& F = *F_ptr;
418  const auto& X = *X_ptr;
419  auto& C = *C_ptr;
420  auto& eps = *eps_ptr;
421  auto& D = *D_ptr;
422 
423  // symmetrize Fock
424  MatrixZ F_symm = 0.5 * (F.transpose().conjugate() + F);
425  if (is_gamma_point) {
426  F_symm = this->reverse_phase_factor(F_symm);
427  }
428 
429  // diagonalize Fock
430  if (do_level_shift) {
431  // transform Fock from AO to CO basis
432  const auto& C_old = *C_old_ptr;
433  MatrixZ FCO = C_old.transpose().conjugate() * F_symm * C_old;
434  // add energy level shift to diagonal elements of unoccupied orbitals
435  auto nobs = FCO.cols();
436  for (auto a = docc_; a != nobs; ++a) {
437  FCO(a, a) += level_shift_;
438  }
439  // diagonalize Fock in CO basis
440  Eigen::SelfAdjointEigenSolver<MatrixZ> comp_eig_solver_fco(FCO);
441  VectorD eps_temp = comp_eig_solver_fco.eigenvalues();
442  MatrixZ C_temp = comp_eig_solver_fco.eigenvectors();
443  // when k=0 (gamma point), reverse phase factor of complex
444  // eigenvectors
445  if (is_gamma_point) {
446  C_temp = this->reverse_phase_factor(C_temp);
447  }
448  // transform eigenvectors back to CO coefficients
449  C = C_old * C_temp;
450  // remove energy level shift from eigenvalues
451  for (auto p = 0; p != nobs; ++p) {
452  eps(p) = (p < docc_) ? eps_temp(p) : eps_temp(p) - level_shift_;
453  }
454  } else {
455  // Orthogonalize Fock matrix: F' = Xt * F * X
456  MatrixZ Ft = X.transpose().conjugate() * F_symm * X;
457  // Diagonalize F'
458  Eigen::SelfAdjointEigenSolver<MatrixZ> comp_eig_solver(Ft);
459  eps = comp_eig_solver.eigenvalues();
460  MatrixZ Ctemp = comp_eig_solver.eigenvectors();
461  // When k=0 (gamma point), reverse phase factor of complex
462  // eigenvectors
463  if (is_gamma_point) {
464  Ctemp = this->reverse_phase_factor(Ctemp);
465  }
466  // transform eigenvectors back to CO coefficients
467  C = X * Ctemp;
468  }
469 
470  // compute_density
471  MatrixZ C_occ = C_ptr->leftCols(docc_);
472  D = C_occ.conjugate() * C_occ.transpose();
473  };
474 
475  bool do_level_shift = (level_shift_ > 0.0 && iter_ > 0);
476  for (int64_t k = 0; k != k_size_; ++k) {
477  bool is_gamma_point = (k_size_ > 1 && k == ((k_size_ - 1) / 2));
478  MatrixZ* C_old = do_level_shift ? &C_[k] : nullptr;
479  F_recip_vec[k] = fock_eig.block(0, k * ext0, ext0, ext0);
480 
481  world.taskq.add(compute_recip_density, &(F_recip_vec[k]), &(X_[k]), C_old,
482  &(C_[k]), &(eps_[k]), &(D_recip_vec[k]), is_gamma_point,
483  do_level_shift);
484  }
485  world.gop.fence();
486 
487  // collect all D_k's to a big rectangular matrix
488  for (int64_t k = 0; k != k_size_; ++k) {
489  result_recip_eig.block(0, k * ext0, ext0, ext0) = D_recip_vec[k];
490  }
491 
492  // parallel impl for D_r
493  const auto denom_inv = 1.0 / double(nk_(0) * nk_(1) * nk_(2));
494  auto transform_recip2real = [ext0, denom_inv, this](MatrixZ* D_recip_ptr,
495  RowMatrixXd* D_real_ptr,
496  int64_t R) {
497  using ::mpqc::detail::k_vector;
498  using ::mpqc::detail::direct_vector;
499 
500  // get references to all matrix pointers
501  const auto& D_recip = *D_recip_ptr;
502  auto& D_real = *D_real_ptr;
503 
504  auto vec_R = direct_vector(R, RD_max_, dcell_);
505  D_real.setZero(ext0, ext0);
506 
507  for (int64_t k = 0; k != k_size_; ++k) {
508  auto vec_k = k_vector(k, nk_, dcell_);
509  auto exponent = std::exp(I * vec_k.dot(vec_R)) * denom_inv;
510  D_real += (exponent * D_recip.block(0, k * ext0, ext0, ext0)).real();
511  }
512  };
513 
514  for (int64_t Rd = 0; Rd != RD_size_; ++Rd) {
515  world.taskq.add(transform_recip2real, &result_recip_eig, &(D_real_vec[Rd]),
516  Rd);
517  }
518  world.gop.fence();
519 
520  // collect all D_r's to a big rectangular matrix
521  for (int64_t Rd = 0; Rd != RD_size_; ++Rd) {
522  result_real_eig.block(0, Rd * ext0, ext0, ext0) = D_real_vec[Rd];
523  }
524 
525  // convert rectangular matrices to TA::DistArray
526  auto result_real = math::eigen_to_array<Tile, Policy>(
527  world, result_real_eig, tr0, tr1_real);
528  auto result_recip = math::eigen_to_array<TA::TensorZ, Policy>(
529  world, result_recip_eig, tr0, tr1_recip);
530 
531  return std::make_pair(result_real, result_recip);
532 }

534 template <typename Tile, typename Policy>
535 typename zRHF<Tile, Policy>::array_type_z
536 zRHF<Tile, Policy>::transform_real2recip(const array_type& matrix,
537  const Vector3i& real_lattice_range,
538  const Vector3i& recip_lattice_range) {
539  // Make sure range values are all positive
540  MPQC_ASSERT((real_lattice_range.array() >= 0).all() &&
541  (recip_lattice_range.array() > 0).all());
542 
543  using ::mpqc::detail::direct_ord_idx;
544  using ::mpqc::detail::direct_vector;
545  using ::mpqc::detail::extend_trange1;
546  using ::mpqc::detail::k_ord_idx;
547  using ::mpqc::detail::k_vector;
548 
549  const auto real_lattice_size =
550  1 + direct_ord_idx(real_lattice_range, real_lattice_range);
551  const Vector3i k_end_3D_idx = (recip_lattice_range.array() - 1).matrix();
552  const auto recip_lattice_size =
553  1 + k_ord_idx(k_end_3D_idx, recip_lattice_range);
554  const auto tiles_range = matrix.trange().tiles_range();
555  MPQC_ASSERT(tiles_range.extent(1) % tiles_range.extent(0) == 0);
556  MPQC_ASSERT(uint64_t(real_lattice_size) ==
557  tiles_range.extent(1) / tiles_range.extent(0));
558 
559  array_type_z result;
560  auto tr0 = matrix.trange().data()[0];
561  auto tr1 = extend_trange1(tr0, recip_lattice_size);
562  auto& world = matrix.world();
563 
564  // Perform real->reciprocal transformation with Eigen
565  // TODO: perform it with TA
566  auto matrix_eig = math::array_to_eigen(matrix);
567  MatrixZ result_eig(tr0.extent(), tr1.extent());
568  result_eig.setZero();
569 
570  auto threshold = std::numeric_limits<double>::epsilon();
571  for (auto R = 0; R < real_lattice_size; ++R) {
572  auto bmat =
573  matrix_eig.block(0, R * tr0.extent(), tr0.extent(), tr0.extent());
574  if (bmat.norm() < bmat.size() * threshold)
575  continue;
576  else {
577  auto vec_R = direct_vector(R, real_lattice_range, dcell_);
578  for (auto k = 0; k < recip_lattice_size; ++k) {
579  auto vec_k = k_vector(k, recip_lattice_range, dcell_);
580  auto exponent = std::exp(I * vec_k.dot(vec_R));
581  result_eig.block(0, k * tr0.extent(), tr0.extent(), tr0.extent()) +=
582  bmat * exponent;
583  }
584  }
585  }
586 
587  result = math::eigen_to_array<TA::TensorZ, TA::SparsePolicy>(
588  world, result_eig, tr0, tr1);
589 
590  return result;
591 }

605 template <typename Tile, typename Policy>
606 MatrixZ zRHF<Tile, Policy>::reverse_phase_factor(const MatrixZ& mat0) {
607  MatrixZ result(mat0);
608 
609  for (auto row = 0; row < mat0.rows(); ++row) {
610  for (auto col = 0; col < mat0.cols(); ++col) {
611  std::complex<double> comp0 = mat0(row, col);
612 
613  double norm = std::abs(comp0);
614  if (norm == 0.0) {
615  result(row, col) = comp0;
616  } else {
617  double real = comp0.real();
618  double imag = comp0.imag();
619 
620  double phi = std::atan(imag / real);
621 
622  std::complex<double> comp1 = comp0 * std::exp(-1.0 * I * phi);
623 
624  result(row, col) = comp1;
625  }
626  }
627  }
628 
629  return result;
630 }
