199 template <typename String, typename>
200 OrbitalIndex::OrbitalIndex(String &&symbol) {
201  name_ = utility::to_wstring(symbol);
202 
203  if (name_.find_first_of(L'_') == std::wstring::npos) {
204  init(name_);
205  spin_ = Spin::None;
206  } else {
207  auto left = name_.find_first_of(L'_');
208 
209  TA_ASSERT(left != std::wstring::npos);
210  TA_ASSERT(name_.size() - left == 2);
211  std::wstring sub_letter = name_.substr(0, left);
212 
213  init(sub_letter);
214 
215  wchar_t spin = name_[left + 1];
216 
217  if (spin == L'α') {
218  spin_ = Spin::Alpha;
219  } else if (spin == L'β') {
220  spin_ = Spin::Beta;
221  } else {
222  throw std::runtime_error("Wrong Spin Label");
223  }
224  }
225 }
