55  throw std::runtime_error(utility::to_string(oper) + " Empty Operation! \n");
56  }
57 
58  boost::trim(oper);
59 
60  auto iter = std::find_if(
61  begin(oper_to_string), end(oper_to_string),
62  [=](const std::pair<Operator::Type, std::wstring> item) {
63  return item.second == oper;
64  });
65 
66  if (iter == oper_to_string.end())
67  throw std::runtime_error(utility::to_string(oper) +
68  " Invalid Operation! \n");

70  type_ = iter->first;
