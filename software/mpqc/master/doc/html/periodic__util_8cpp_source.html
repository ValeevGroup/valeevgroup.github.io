55 Vector3d k_vector(int64_t ord_idx, Vector3i const &nk, Vector3d const &dcell) {
56  Vector3d result;
57  auto x = ord_idx / nk(2) / nk(1);
58  auto y = (ord_idx / nk(2)) % nk(1);
59  auto z = ord_idx % nk(2);
60  result(0) = (dcell(0) == 0.0)
61  ? 0.0
62  : (-1.0 + (2.0 * (x + 1) - 1.0) / nk(0)) * (M_PI / dcell(0));
63  result(1) = (dcell(1) == 0.0)
64  ? 0.0
65  : (-1.0 + (2.0 * (y + 1) - 1.0) / nk(1)) * (M_PI / dcell(1));
66  result(2) = (dcell(2) == 0.0)
67  ? 0.0
68  : (-1.0 + (2.0 * (z + 1) - 1.0) / nk(2)) * (M_PI / dcell(2));
