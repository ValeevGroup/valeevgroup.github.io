103  OrbitalSpace(const OrbitalIndex& idx, const OrbitalIndex& ao_idx,
104  const Array& tarray, std::shared_ptr<const Group> group,
105  std::vector<Group::ordinal_type> irrep_indices =
106  std::vector<Group::ordinal_type>())
107  : index_(make_base_index(idx)),
108  ao_index_(make_base_index(ao_idx)),
109  coefs_(tarray),
110  group_(group),
111  irrep_indices_(std::move(irrep_indices)) {
112  // expect a matrix of coefficients
113  if (coefs_.elements_range().rank() != 2)
114  throw ProgrammingError(
115  "OrbitalSpace ctor expects a matrix of coefficients", __FILE__,
116  __LINE__);
117 
118  const auto nirreps = group_->order();
119  const auto ntiles_col = coefs_.trange().tiles_range().extent_data()[1];
120 
121  // if not given irrep indices, assume the blocks are evenly divisible among
122  // irreps
123  if (irrep_indices_.empty()) {
124  if (ntiles_col % nirreps != 0)
125  throw ProgrammingError("OrbitalSpace: cannot deduce irrep indices",
126  __FILE__, __LINE__);
127 
128  irrep_indices_.resize(ntiles_col);
129  auto ntiles_per_irrep = ntiles_col / nirreps;
130  for (auto irrep = 0, tile = 0; irrep != nirreps; ++irrep)
131  for (auto t = 0; t != ntiles_per_irrep; ++t, ++tile)
132  irrep_indices_[tile] = irrep;
133  } else if (irrep_indices_.size() != ntiles_col)
134  throw ProgrammingError("OrbitalSpace: # of irrep indices != # of tiles",
135  __FILE__, __LINE__);
136  }
