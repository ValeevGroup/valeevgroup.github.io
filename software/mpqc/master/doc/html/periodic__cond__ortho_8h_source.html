29  // if symmetric, X = U.s_sqrtinv.Ut
30  // if canonical, X = U.s_sqrtinv
31  // where s and U are eigenvalues and eigenvectors of S
32  Eigen::SelfAdjointEigenSolver<MatrixZ> comp_eig_solver(S_eig);

38  auto s_min = s.minCoeff();
39

41  std::min(s_max / std::max(s_min, std::numeric_limits<double>::min()),
42  1.0 / std::numeric_limits<double>::epsilon());
43 
44  auto threshold = s_max / max_condition_num;
45

48  for (int64_t i = s_rows - 1; i >= 0; --i) {

50  ++s_cond;
51  } else
52  i = 0;
53  }
54

56  auto result_condition_num = sigma.maxCoeff() / sigma.minCoeff();

59  auto sigma_invsqrt = sigma_invsqrt_real.cast<std::complex<double>>();
60

63  X = U_cond * sigma_invsqrt;
64  // make symmetric X

67  }

70  if (nbf_omitted < 0) throw "Error: dropping negative number of functions!";

74  << " at k = " << k;
75  }

79  auto should_be_I = XtS * X;
80
