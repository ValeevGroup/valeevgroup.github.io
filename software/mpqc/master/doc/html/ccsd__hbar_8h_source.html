251  gaussian::AOFactoryBase<Tile, Policy>& ao_factory,
252  const Array& t1, const Array& t2, const Array& tau,
253  const Array& FIA, const Array& W_AbCd, bool df) {

255  auto& world = lcao_factory.world();

257  auto time0 = mpqc::now(world, accurate_time);

259 
260  auto g_ijab = lcao_factory.compute(L"<i j|G|a b>" + postfix);
261  auto g_iajb = lcao_factory.compute(L"<i a|G|j b>" + postfix);
262  auto g_iabc = lcao_factory.compute(L"<i a|G|b c>" + postfix);
263  auto g_ijka = lcao_factory.compute(L"<i j|G|k a>" + postfix);
264 
265  Array t22;
266  t22("a,b,i,j") = 2.0 * t2("a,b,i,j") - t2("a,b,j,i");
267 
268  Array W_AbCi;
269 
270  W_AbCi("a,b,c,i") =
271 
272  g_iabc("i,a,b,c")
273 
274  - FIA("k,c") * t2("a,b,k,i")
275 
276  + g_ijka("l,k,i,c") * tau("a,b,k,l")
277 
278  - g_iabc("k,b,c,d") * t2("a,d,k,i") + g_iabc("k,a,d,c") * t22("b,d,i,k") -
279  g_iabc("k,a,c,d") * t2("b,d,i,k")

283  t1("b,k") * (-g_iajb("k,a,i,c") + t2("a,d,l,i") * g_ijab("l,k,c,d"));
284 
285  // if W_AbCd term is computed and stored
286  if (W_AbCd.is_initialized()) {
287  W_AbCi("a,b,c,i") += t1("d,i") * W_AbCd("a,b,c,d");

292 
293  Array Xt;

295

299

303  t1("d,i") * g_ijab("k,l,c,d") * tau("a,b,k,l");

312  }
313  }
314 
315  auto time1 = mpqc::now(world, accurate_time);
316  auto time = mpqc::duration_in_s(time0, time1);

318  ExEnv::out0() << indent
319  << "\nTime to Initialize W_AbCi Intermediates: " << time
320  << " S\n";

364 template <typename Tile, typename Policy,
365  typename Array = TA::DistArray<Tile, Policy>>
366 std::tuple<Array, Array, Array> compute_cs_ccsd_F(
367  LCAOFactoryBase<Tile, Policy>& lcao_factory,
368  gaussian::AOFactoryBase<Tile, Policy>& ao_factory, const Array& t1,
369  const Array& tau, bool df) {
370  bool verbose = lcao_factory.verbose();
371  auto& world = lcao_factory.world();
372  bool accurate_time = lcao_factory.accurate_time();
373  auto time0 = mpqc::now(world, accurate_time);
374  std::wstring postfix = df ? L"[df]" : L"";
375 
376  auto g_ijab_bar = lcao_factory.compute(L"<i j|G|a b>" + postfix);
377  g_ijab_bar("i,j,a,b") = 2.0 * g_ijab_bar("i,j,a,b") - g_ijab_bar("j,i,a,b");
378 
379  auto f_ia = lcao_factory.compute(L"<i|F|a>" + postfix);
380 
381  // compute FIA
382  Array FIA;
383  { FIA("i,a") = f_ia("i,a") + g_ijab_bar("i,j,a,b") * t1("b,j"); }
384 
385  Array FIJ;
386  {
387  FIJ = lcao_factory.compute(L"<i|F|j>" + postfix);
388  auto g_ijka = lcao_factory.compute(L"<i j|G|k a>" + postfix);
389  FIJ("i,j") += t1("c,l") * (2.0 * g_ijka("i,l,j,c") - g_ijka("l,i,j,c")) +
390  tau("c,d,j,l") * g_ijab_bar("i,l,c,d") -
391  f_ia("i,a") * t1("a,j");
392  }
393 
394  Array FAB;
395  {
396  FAB = lcao_factory.compute(L"<a|F|b>" + postfix);
397  FAB("a,b") -=
398  t1("a,m") * f_ia("m,b") + tau("a,d,k,l") * g_ijab_bar("k,l,b,d");
399 
400  if (df) {
401  // refactorize with density fitting
402  auto Xia = lcao_factory.compute(L"( Λ |G|i a)[inv_sqr]");
403  auto Xab = lcao_factory.compute(L"( Λ |G|a b)[inv_sqr]");
404 
405  FAB("a,b") += 2.0 * t1("d,k") * Xia("X,k,d") * Xab("X,a,b") -
406  t1("d,k") * Xab("X,a,d") * Xia("X,k,b");
407 
408  } else {
409  auto g_iabc = lcao_factory.compute(L"<i a|G|b c>" + postfix);
410  FAB("a,b") += t1("d,k") * (2.0 * g_iabc("k,a,d,b") - g_iabc("k,a,b,d"));
411  }
412  }
413  auto time1 = mpqc::now(world, accurate_time);
414  auto time = mpqc::duration_in_s(time0, time1);
415  if (verbose) {
416  ExEnv::out0() << indent << "\nTime to Initialize F Intermediates: " << time
417  << " S\n";
418  }
419  return std::make_tuple(FAB, FIA, FIJ);
420 };

422 template <typename Tile, typename Policy,
423  typename Array = TA::DistArray<Tile, Policy>>
424 cc::Intermediates<Array> compute_eom_intermediates(
425  LCAOFactoryBase<Tile, Policy>& lcao_factory,
426  gaussian::AOFactoryBase<Tile, Policy>& ao_factory, const Array& t2,
427  const Array& t1, bool df, std::string option) {
428  cc::Intermediates<Array> imds;
429 
430  std::wstring postfix = df ? L"[df]" : L"";
431 
432  auto f_ia = lcao_factory.compute(L"<i|F|a>" + postfix);
433  auto f_ij = lcao_factory.compute(L"<i|F|j>" + postfix);
434  auto f_ab = lcao_factory.compute(L"<a|F|b>" + postfix);
435 
436  auto g_ijkl = lcao_factory.compute(L"<i j|G|k l>" + postfix);
437  auto g_ijab = lcao_factory.compute(L"<i j|G|a b>" + postfix);
438  auto g_iajb = lcao_factory.compute(L"<i a|G|j b>" + postfix);
439  auto g_iabc = lcao_factory.compute(L"<i a|G|b c>" + postfix);
440  auto g_ijka = lcao_factory.compute(L"<i j|G|k a>" + postfix);
441 
442  Array tau;
443  tau("a,b,i,j") = t2("a,b,i,j") + (t1("a,i") * t1("b,j"));
444 
445  Array g_ijab_bar;
446  g_ijab_bar("i,j,a,b") = 2.0 * g_ijab("i,j,a,b") - g_ijab("j,i,a,b");
447 
448  // one body
449  {
450  Array g_ijka_bar;
451  g_ijka_bar("i,j,k,a") = 2.0 * g_ijka("i,j,k,a") - g_ijka("j,i,k,a");
452  //
453  imds.FIJ =
454  compute_cs_ccsd_F_OO(f_ij, f_ia, g_ijab_bar, g_ijka_bar, tau, t1);
455  //
456  imds.FIA = compute_cs_ccsd_F_ov(f_ia, g_ijab_bar, t1);
457 
458  Array g_iabc_bar;
459  g_iabc_bar("k,a,d,b") = 2.0 * g_iabc("k,a,d,b") - g_iabc("k,a,b,d");
460  //
461  imds.FAB =
462  compute_cs_ccsd_F_VV(f_ab, f_ia, g_ijab_bar, g_iabc_bar, tau, t1);
463 
464  // std::tie(imds.FAB, imds.FIA, imds.FIJ) =
465  // compute_cs_ccsd_F(lcao_factory,ao_factory,t1,tau,df);
466  }
467 
468  // two body
469  {
470  imds.Wijab = g_ijab;
471 
472  imds.Wijka = compute_cs_ccsd_W_ooov(g_ijka, g_ijab, t1);
473 
474  if (option == "ip") {
475  imds.Wiajb =
