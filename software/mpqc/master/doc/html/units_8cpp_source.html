28 void Unit::parse() {
29  to_atomic_units_ = 1.0;
30 
31  // physical constants used for atomic unit conversion factors
32  const double a0 = constants_->bohr_radius(); // m
33  const double e = constants_->elementary_charge(); // C
34  const double me = constants_->electron_mass(); // kg
35  const double e0 = constants_->electric_constant(); // F/m
36  const double NA = constants_->Avogadro_constant(); // mol-1
37  const double h = constants_->Planck_constant(); // J s
38  const double hbar = h / (2 * M_PI); // J s
39  const double c = constants_->speed_of_light(); // m/s
40 
41  // derived au conversion factors
42  const double Ea = e * e / ((4.0 * M_PI * e0) * a0); // J
43 
44  // other conversions
45  const double amu = constants_->atomic_mass_unit(); // kg
46  const double cal = constants_->thermochemical_calorie(); // J
47 
48  int invert = 0;
49  const char *rest = strrep_.c_str();
50 
51  while (rest) {
52  const char *end = ::strpbrk(rest, " */");
53  int nchar;
54  if (end) {
55  nchar = end - rest;
56  } else {
57  nchar = strlen(rest);
58  }
59  std::unique_ptr<char[]> unitstring = std::make_unique<char[]>(nchar + 1);
60  memcpy(unitstring.get(), rest, nchar);
61  unitstring[nchar] = '\0';
62 
63  double factor = 1.0;
64  if (eq(unitstring, "bohr") || eq(unitstring, "bohrs")) {
65  } else if (eq(unitstring, "hartree") || eq(unitstring, "hartrees") ||
66  eq(unitstring, "Hartree") || eq(unitstring, "Hartrees")) {
67  } else if (eq(unitstring, "ev") || eq(unitstring, "eV")) {
68  factor = 1.0 / constants_->Hartree_to_electron_volt();
69  } else if (eq(unitstring, "energy_wavenumber[cm]")) {
70  factor = (100 * h * c) / Ea; // the energy of photon with wave length of 1 cm
71  } else if (eq(unitstring, "debye")) {
72  factor = 1.0 / constants_->atomic_unit_to_debye();
73  } else if (eq(unitstring, "radian") || eq(unitstring, "radians")) {
74  } else if (eq(unitstring, "mol") || eq(unitstring, "mole")) {
75  factor = NA;
76  } else if (eq(unitstring, "kcal")) {
77  factor = 1000.0 * cal / Ea;
78  } else if (eq(unitstring, "kcal_per_mol")) {
79  factor = 1000.0 * cal / (Ea * NA);
80  } else if (eq(unitstring, "N") || eq(unitstring, "newton")) {
81  factor = a0 / Ea;
82  } else if (eq(unitstring, "dyne")) {
83  factor = 1.0e-5 * a0 / Ea;
84  } else if (eq(unitstring, "m") || eq(unitstring, "meter") ||
85  eq(unitstring, "meters")) {
86  factor = 1.0 / a0;
87  } else if (eq(unitstring, "cm") || eq(unitstring, "centimeter")) {
88  factor = 1.0e-2 / a0;
89  } else if (eq(unitstring, "angstrom") || eq(unitstring, "angstroms") ||
90  eq(unitstring, "aangstrom") || eq(unitstring, "aangstroms")) {
91  factor = 1.0e-10 / a0;
92  } else if (eq(unitstring, "amu")) {
93  factor = amu / me;
94  } else if (eq(unitstring, "degree") || eq(unitstring, "degrees")) {
95  factor = M_PI / 180.0;
96  } else if (eq(unitstring, "second") || eq(unitstring, "seconds") ||
97  eq(unitstring, "s")) {
98  factor = Ea / hbar; // atomic unit of time = how long it takes electron
99  // to travel 1 radian in bohr orbit
100  } else {
101  throw InputError("unknown Unit string", __FILE__, __LINE__, "unitstring",
102  unitstring.get());
103  }
104 
105  unitstring.reset();
106  if (invert)
107  factor = 1.0 / factor;
108  to_atomic_units_ *= factor;
109  rest = ::strpbrk(rest, " */");
110  while (rest && (*rest == ' ' || *rest == '*' || *rest == '/')) {
111  if (*rest == '/')
112  invert = !invert;
113  rest++;
114  }
115  }
