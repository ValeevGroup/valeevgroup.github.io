71  template <typename Op>
72  static vector_type
73  recursive_outer_product(const vector_type* const size_vectors,
74  const unsigned int dim, const Op& op)
75  {
76  vector_type result;
77 
78  if(dim == 1u) {
79  // Construct a modified copy of size_vector[0]
80  result = op(*size_vectors);
81  } else {
82  // Compute split the range and compute the outer products
83  const unsigned int middle = (dim >> 1u) + (dim & 1u);
84  const vector_type left = recursive_outer_product(size_vectors, middle, op);
85  const vector_type right = recursive_outer_product(size_vectors + middle, dim - middle, op);
86 
87  // Compute the outer product of left and right
88 
89  result = vector_type(left.size() * right.size());
90  result.outer_fill(left, right,
91  [] (const value_type left, const value_type right) { return left * right; });
92  }
93 
94  return result;
95  }

103  void normalize() {
104  const value_type threshold = threshold_;
105  const unsigned int dim = tile_norms_.range().rank();
106  const vector_type* restrict const size_vectors = size_vectors_.get();
107  madness::AtomicInt zero_tile_count;
108  zero_tile_count = 0;
109 
110  if(dim == 1u) {
111  auto normalize_op = [threshold, &zero_tile_count] (value_type& norm, const value_type size) {
112  TA_ASSERT(norm >= value_type(0));
113  norm /= size;
114  if(norm < threshold) {
115  norm = value_type(0);
116  ++zero_tile_count;
117  }
118  };
119 
120  // This is the easy case where the data is a vector and can be
121  // normalized directly.
122  math::inplace_vector_op(normalize_op, size_vectors[0].size(), tile_norms_.data(),
123  size_vectors[0].data());
124 
125  } else {
126  // Here the normalization constants are computed and multiplied by the
127  // norm data using a recursive, outer-product algorithm. This is done to
128  // minimize temporary memory requirements, memory bandwidth, and work.
129 
130  auto inv_vec_op = [] (const vector_type& size_vector) {
131  return vector_type(size_vector,
132  [] (const value_type size) { return value_type(1) / size; });
133  };
134 
135  // Compute the left and right outer products
136  const unsigned int middle = (dim >> 1u) + (dim & 1u);
137  const vector_type left = recursive_outer_product(size_vectors, middle, inv_vec_op);
138  const vector_type right = recursive_outer_product(size_vectors + middle, dim - middle, inv_vec_op);
139 
140  auto normalize_op = [threshold, &zero_tile_count] (value_type& norm,
141  const value_type x, const value_type y)
142  {
143  TA_ASSERT(norm >= value_type(0));
144  norm *= x * y;
145  if(norm < threshold) {
146  norm = value_type(0);
147  ++zero_tile_count;
148  }
149  };
150 
151  math::outer(left.size(), right.size(), left.data(), right.data(),
152  tile_norms_.data(), normalize_op);
153  }
154 
155  zero_tile_count_ = zero_tile_count;
156  }

368  template <typename Index>
369  SparseShape update_block(const Index& lower_bound, const Index& upper_bound,
370  const SparseShape& other)
371  {
372  Tensor<value_type> result_tile_norms = tile_norms_.clone();
373 
374  auto result_tile_norms_blk = result_tile_norms.block(lower_bound, upper_bound);
375  const value_type threshold = threshold_;
376  madness::AtomicInt zero_tile_count;
377  zero_tile_count = zero_tile_count_;
378  result_tile_norms_blk.inplace_binary(other.tile_norms_,
379  [threshold,&zero_tile_count] (value_type& l, const value_type r) {
380  // Update the zero tile count for the result
381  if((l < threshold) && (r >= threshold))
382  ++zero_tile_count;
383  else if((l >= threshold) && (r < threshold))
384  --zero_tile_count;
385 
386  // Update the tile norm value
387  l = r;
388  });
389 
390  return SparseShape_(result_tile_norms, size_vectors_, zero_tile_count);
391  }

400  template <typename Index>
401  std::shared_ptr<vector_type>
402  block_range(const Index& lower_bound, const Index& upper_bound) const {
403  TA_ASSERT(detail::size(lower_bound) == tile_norms_.range().rank());
404  TA_ASSERT(detail::size(upper_bound) == tile_norms_.range().rank());
405 
406  // Get the number dimensions of the the shape
407  const auto rank = detail::size(lower_bound);
408  const auto* restrict const lower = detail::data(lower_bound);
409  const auto* restrict const upper = detail::data(upper_bound);
410 
411  std::shared_ptr<vector_type> size_vectors(new vector_type[rank],
412  std::default_delete<vector_type[]>());
413 
414  for(auto i = 0ul; i < rank; ++i) {
415  // Get the new range size
416  const auto lower_i = lower[i];
417  const auto upper_i = upper[i];
418  const auto extent_i = upper_i - lower_i;
419 
420  // Check that the input indices are in range
421  TA_ASSERT(lower_i < upper_i);
422  TA_ASSERT(upper_i <= tile_norms_.range().upbound_data()[i]);
423 
424  // Compute the trange data for the result shape
425  size_vectors.get()[i] = vector_type(extent_i,
426  size_vectors_.get()[i].data() + lower_i,
427  [=] (const size_type j) { return j - lower_i; });
428  }
429 
430  return size_vectors;
431  }

439  template <typename Index>
440  SparseShape block(const Index& lower_bound, const Index& upper_bound) const {
441  std::shared_ptr<vector_type> size_vectors =
442  block_range(lower_bound, upper_bound);
443 
444  // Copy the data from arg to result
445  const value_type threshold = threshold_;
446  madness::AtomicInt zero_tile_count;
447  zero_tile_count = 0;
448  auto copy_op = [threshold,&zero_tile_count] (value_type& restrict result,
449  const value_type arg)
450  {
451  result = arg;
452  if(arg < threshold)
453  ++zero_tile_count;
454  };
455 
456 
457  // Construct the result norms tensor
458  TensorConstView<value_type> block_view =
459  tile_norms_.block(lower_bound, upper_bound);
460  Tensor<value_type> result_norms((Range(block_view.range().extent())));

462

724  SparseShape_ add(value_type value) const {
725  TA_ASSERT(! tile_norms_.empty());
726  const value_type threshold = threshold_;
727  madness::AtomicInt zero_tile_count;
728  zero_tile_count = 0;
729 
730  Tensor<T> result_tile_norms(tile_norms_.range());
731 
732  value = std::abs(value);
733  const unsigned int dim = tile_norms_.range().rank();
734  const vector_type* restrict const size_vectors = size_vectors_.get();
735 
736  if(dim == 1u) {
737  auto add_const_op = [threshold, &zero_tile_count, value] (value_type norm,
738  const value_type size)
739  {
740  norm += value / std::sqrt(size);
741  if(norm < threshold) {
742  norm = 0;
743  ++zero_tile_count;
744  }
745  return norm;
746  };
747 
748  // This is the easy case where the data is a vector and can be
749  // normalized directly.
750  math::vector_op(add_const_op, size_vectors[0].size(), result_tile_norms.data(),
751  tile_norms_.data(), size_vectors[0].data());
752 
753  } else {
754  // Here the normalization constants are computed and multiplied by the
755  // norm data using a recursive, outer algorithm. This is done to
756  // minimize temporary memory requirements, memory bandwidth, and work.
757 
758  auto inv_sqrt_vec_op = [] (const vector_type size_vector) {
759  return vector_type(size_vector,
760  [] (const value_type size) { return value_type(1) / std::sqrt(size); });
761  };
762 
763  // Compute the left and right outer products
764  const unsigned int middle = (dim >> 1u) + (dim & 1u);
765  const vector_type left = recursive_outer_product(size_vectors, middle, inv_sqrt_vec_op);
766  const vector_type right = recursive_outer_product(size_vectors + middle, dim - middle, inv_sqrt_vec_op);
767 
768  math::outer_fill(left.size(), right.size(), left.data(), right.data(),
769  tile_norms_.data(), result_tile_norms.data(),

827  // This is the easy case where the data is a vector and can be
828  // normalized directly.
829  math::inplace_vector_op(
830  [threshold, &zero_tile_count] (value_type& norm, const value_type size) {
831  norm *= size;
832  if(norm < threshold) {
833  norm = value_type(0);
834  ++zero_tile_count;
835  }
836  },
837  size_vectors[0].size(), tile_norms.data(), size_vectors[0].data());
838  } else {
839  // Here the normalization constants are computed and multiplied by the
840  // norm data using a recursive, outer algorithm. This is done to
841  // minimize temporary memory requirements, memory bandwidth, and work.
842 
843  auto noop = [](const vector_type& size_vector) -> const vector_type& {
844  return size_vector;
845  };
846 
847  // Compute the left and right outer products
848  const unsigned int middle = (dim >> 1u) + (dim & 1u);
849  const vector_type left = recursive_outer_product(size_vectors, middle, noop);
850  const vector_type right = recursive_outer_product(size_vectors + middle, dim - middle, noop);
851 
852  math::outer(left.size(), right.size(), left.data(), right.data(), tile_norms.data(),
853  [threshold, &zero_tile_count] (value_type& norm, const value_type x,
854  const value_type y)
855  {
856  norm *= x * y;
857  if(norm < threshold) {
858  norm = value_type(0);
859  ++zero_tile_count;
860  }
861  });
862  }
863 
864  return zero_tile_count;
865  }

921  SparseShape_ gemm(const SparseShape_& other, value_type factor,
922  const math::GemmHelper& gemm_helper) const
923  {
924  TA_ASSERT(! tile_norms_.empty());
925 
926  factor = std::abs(factor);
927  const value_type threshold = threshold_;
928  madness::AtomicInt zero_tile_count;
929  zero_tile_count = 0;
930  integer M = 0, N = 0, K = 0;
931  gemm_helper.compute_matrix_sizes(M, N, K, tile_norms_.range(), other.tile_norms_.range());
932 
933  // Allocate memory for the contracted size vectors
934  std::shared_ptr<vector_type> result_size_vectors(new vector_type[gemm_helper.result_rank()],
935  std::default_delete<vector_type[]>());
936 
937  // Initialize the result size vectors
938  unsigned int x = 0ul;
939  for(unsigned int i = gemm_helper.left_outer_begin(); i < gemm_helper.left_outer_end(); ++i, ++x)
940  result_size_vectors.get()[x] = size_vectors_.get()[i];
941  for(unsigned int i = gemm_helper.right_outer_begin(); i < gemm_helper.right_outer_end(); ++i, ++x)
942  result_size_vectors.get()[x] = other.size_vectors_.get()[i];
943 
944  // Compute the number of inner ranks
945  const unsigned int k_rank = gemm_helper.left_inner_end() - gemm_helper.left_inner_begin();
946 
947  // Construct the result norm tensor
948  Tensor<value_type> result_norms(gemm_helper.make_result_range<typename Tensor<T>::range_type>(
949  tile_norms_.range(), other.tile_norms_.range()), 0);
950 
951  if(k_rank > 0u) {
952 
953  // Compute size vector
954  const vector_type k_sizes =
955  recursive_outer_product(size_vectors_.get() + gemm_helper.left_inner_begin(),
956  k_rank, [] (const vector_type& size_vector) -> const vector_type&
957  { return size_vector; });
958 
959  // TODO: Make this faster. It can be done without using temporaries
960  // for the arguments, but requires a custom matrix multiply.
961 
962  Tensor<value_type> left(tile_norms_.range());
963  const size_type mk = M * K;
964  auto left_op = [] (const value_type left, const value_type right)
965  { return left * right; };
966  for(size_type i = 0ul; i < mk; i += K)
967  math::vector_op(left_op, K, left.data() + i,
968  tile_norms_.data() + i, k_sizes.data());
969 
970  Tensor<value_type> right(other.tile_norms_.range());
971  for(integer i = 0ul, k = 0; k < K; i += N, ++k) {
972  const value_type factor = k_sizes[k];
973  auto right_op = [=] (const value_type arg) { return arg * factor; };
974  math::vector_op(right_op, N, right.data() + i, other.tile_norms_.data() + i);
975  }
976 
977  result_norms = left.gemm(right, factor, gemm_helper);
978 
979  // Hard zero tiles that are below the zero threshold.
980  result_norms.inplace_unary(
981  [threshold, &zero_tile_count] (value_type& value) {
982  if(value < threshold) {
983  value = value_type(0);
984  ++zero_tile_count;
985  }
986  });
987 
988  } else {
989 
990  // This is an outer product, so the inputs can be used directly
991  math::outer_fill(M, N, tile_norms_.data(), other.tile_norms_.data(), result_norms.data(),
992  [threshold, &zero_tile_count, factor] (const value_type left,
993  const value_type right)
994  {
995  value_type norm = left * right * factor;
996  if(norm < threshold) {
997  norm = value_type(0);
998  ++zero_tile_count;
999  }
1000  return norm;
1001  });
1002  }
1003 
1004  return SparseShape_(result_norms, result_size_vectors, zero_tile_count);
1005  }

1008  const math::GemmHelper& gemm_helper, const Permutation& perm) const
1009  {
1010  return gemm(other, factor, gemm_helper).perm(perm);
