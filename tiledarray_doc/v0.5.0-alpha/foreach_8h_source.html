217  template <typename T, unsigned int DIM, typename Tile, typename Op>
218  inline Array<T, DIM, Tile, SparsePolicy>
219  foreach(const Array<T, DIM, Tile, SparsePolicy> arg, Op&& op) {
220  typedef Array<T, DIM, Tile, SparsePolicy> array_type;
221  typedef typename array_type::value_type value_type;
222  typedef typename array_type::size_type size_type;
223  typedef typename array_type::shape_type shape_type;
224  typedef Future<value_type> future_type;
225  typedef std::pair<size_type, future_type> datum_type;
226 
227  // Create a vector to hold local tiles
228  std::vector<datum_type> tiles;
229  tiles.reserve(arg.get_pmap()->size());
230 
231  // Construct a tensor to hold updated tile norms for the result shape.
232  TiledArray::Tensor<typename shape_type::value_type,
233  Eigen::aligned_allocator<typename shape_type::value_type> >
234  tile_norms(arg.trange().tiles(), 0);
235 
236  // Construct the task function used to construct the result tiles.
237  madness::AtomicInt counter; counter = 0;
238  int task_count = 0;
239  auto task = [&](const size_type index, const value_type& arg_tile) -> value_type {
240  value_type result_tile;
241  tile_norms[index] = op(result_tile, arg_tile);
242  ++counter;
243  return result_tile;
244  };
245 
246  World& world = arg.get_world();
247 
248  // Get local tile index iterator
249  typename array_type::pmap_interface::const_iterator
250  it = arg.get_pmap()->begin(),
251  end = arg.get_pmap()->end();
252  for(; it != end; ++it) {
253  const size_type index = *it;
254  if(arg.is_zero(index))
255  continue;
256  future_type arg_tile = arg.find(index);
257  future_type result_tile = world.taskq.add(task, index, arg_tile);
258  ++task_count;
259  tiles.push_back(datum_type(index, result_tile));
260  }
261 
262  // Wait for tile norm data to be collected.
263  if(task_count > 0)
264  world.await([&counter,task_count] () -> bool { return counter == task_count; });
265 
266  // Construct the new array
267  array_type result(world, arg.trange(),
268  shape_type(world, tile_norms, arg.trange()), arg.get_pmap());
269  for(typename std::vector<datum_type>::const_iterator it = tiles.begin(); it != tiles.end(); ++it) {
270  const size_type index = it->first;
271  if(! result.is_zero(index))
272  result.set(it->first, it->second);
273  }
274 
275  return result;
276  }
