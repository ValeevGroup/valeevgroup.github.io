92  void init(World& world, std::shared_ptr<pmap_interface> pmap,
93  const VariableList& target_vars)
94  {
95  if(target_vars.dim()) {
96  derived().init_vars(target_vars);
97  derived().init_struct(target_vars);
98  } else {
99  derived().init_vars();
100  derived().init_struct(vars_);
101  }
102 
103  // Check for a valid process map.
104  if(pmap) {
105  // If process map is not valid, use the process map constructed by the
106  // expression engine.
107  if((typename pmap_interface::size_type(world.size()) != pmap->procs()) ||
108  (trange_.tiles().volume() != pmap->size()))
109  pmap.reset();
110  }
111 
112  derived().init_distribution(& world, pmap);
113  }

125  void init_struct(const VariableList& target_vars) {
126  if(target_vars != vars_) {
127  perm_ = derived().make_perm(target_vars);
128  trange_ = derived().make_trange(perm_);
129  shape_ = derived().make_shape(perm_);
130  } else {
131  trange_ = derived().make_trange();
132  shape_ = derived().make_shape();
133  }
134 
135  if(struct_override_ptr_ != nullptr){
136  shape_ = shape_.mask(struct_override_ptr_->shape);
137  }
138  }
