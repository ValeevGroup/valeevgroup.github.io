20 // Functor for sorting centers based on the distance from a point.
21 class sort_by_distance_from_point {
22  public:
23  sort_by_distance_from_point(Vector3d const &point) : point_(point) {}
24 
25  bool operator()(ABCbl const &a, ABCbl const &b) const {
26  Vector3d a_diff = center_of_mass(a) - point_;
27  Vector3d b_diff = center_of_mass(b) - point_;
28 
29  const auto a_dist2 = a_diff.squaredNorm();
30  const auto b_dist2 = b_diff.squaredNorm();
31 
32  if (a_dist2 != b_dist2) { // Not same distance
33  return a_dist2 < b_dist2;
34  }
35 
36  // a and b were the same distance from point
37  // so we need to check each dim separately
38  // check in x then y then z order
39  if (a_diff[0] != b_diff[0]) {
40  return a_diff[0] < b_diff[0];
41  } else if (a_diff[1] != b_diff[1]) {
42  return a_diff[1] < b_diff[1];
43  } else {
44  return a_diff[2] < b_diff[2];
45  }
46  }
47 
48  private:
49  Vector3d point_;
50 };

134  auto unit_factory = UnitFactory::get_default();
135  auto bohr_to_angstrom =
136  unit_factory->make_unit("angstrom").from_atomic_units();
137  auto libint_atoms = libint2::read_dotxyz(file, bohr_to_angstrom);
138 
139  using ABCbl = AtomBasedClusterable;
140  std::vector<ABCbl> atoms;

144  if (!most_abundant_mass)

176 double Molecule::nuclear_repulsion_energy() const {
177  auto const &atoms = this->atoms();
178 
179  double energy = 0.0;
180  for (auto i = 0ul; i < atoms.size(); ++i) {
181  for (auto j = i + 1; j < atoms.size(); ++j) {
182  const auto r = (atoms[i].center() - atoms[j].center()).norm();
183  energy += (atoms[i].charge() * atoms[j].charge()) / r;
184  }
