25 template <typename T, typename AT>
26 std::array<TiledArray::Tensor<T, AT>, 2> eigen_estimator(
27  std::array<TiledArray::Tensor<T, AT>, 2> &result,
28  TiledArray::Tensor<T, AT> const &tile) {
29  typedef typename TiledArray::Tensor<T, AT>::size_type size_type;
30 
31  if (result[0].empty()) {
32  // Construct result tensors
33  const std::array<size_type, 1> range_start = {{tile.range().lobound()[0]}};
34  const std::array<size_type, 1> range_finish = {{tile.range().upbound()[0]}};
35  TiledArray::Range range(range_start, range_finish);
36  result[0] = TiledArray::Tensor<T, AT>{range, 0.0};
37  result[1] = TiledArray::Tensor<T, AT>{range, 0.0};
38  }
39 
40  // Sum the rows of tile into result
41  auto reduce_op = [](T & MADNESS_RESTRICT result, const T arg) {
42  result += std::abs(arg);
43  };
44 
45  TiledArray::math::row_reduce(tile.range().extent()[0],
46  tile.range().extent()[1], tile.data(),
47  result[0].data(), reduce_op);
48 
49  TA::Range range = tile.range();
50  auto const start = range.lobound();
51  auto const finish = range.upbound();
52  auto const weight_ptr = tile.range().stride_data();
53  auto const dims = tile.range().rank();
54  std::vector<unsigned int> weight(weight_ptr, weight_ptr + dims);
55 
56  size_type const start_max = *std::max_element(start.begin(), start.end());
57  size_type const finish_min = *std::min_element(finish.begin(), finish.end());
58 
59  const size_type n = tile.range().rank();
60  if (start_max < finish_min) {
61  // Compute the first and last ordinal index
62  size_type tile_first = 0ul, tile_last = 0ul, tile_stride = 0ul;
63  for (size_type i = 0ul; i < n; ++i) {
64  const size_type start_i = start[i];
65  const size_type weight_i = weight[i];
66 
67  tile_first += (start_max - start_i) * weight_i;
68  tile_last += (finish_min - start_i) * weight_i;
69  tile_stride += weight_i;
70  }
71  size_type result_first = tile_first / weight[0];
72 
73  // Compute the trace
74  const T *MADNESS_RESTRICT const tile_data = tile.data();
75  T *MADNESS_RESTRICT const result_data = result[1].data();
76  for (; tile_first < tile_last; tile_first += tile_stride, ++result_first) {
77  result_data[result_first] = tile_data[tile_first];
78  }
79  }
80 
81  return result;
82 }

86  auto const &array_extent = S.trange().tiles_range().extent();
87  std::vector<Eigen::VectorXd> row_norms(array_extent[0]);
88 
89  for (auto it = S.begin(); it != S.end(); ++it) {
90  auto const &tile = it->get();
91  auto const extent = tile.range().extent();
92  auto matrix_map = TiledArray::eigen_map(tile, extent[0], extent[1]);
93 
94  Eigen::VectorXd tile_row_sums(extent[0]);
95 
96  for (auto i = 0l; i < matrix_map.rows(); ++i) {
97  auto row_sum = 0.0;
98  for (auto j = 0l; j < matrix_map.cols(); ++j) {
99  row_sum += std::abs(matrix_map(i, j));
100  }
101  tile_row_sums[i] = row_sum;
102  }
103 
104  auto tile_row = it.index()[0];
105  if (row_norms[tile_row].size() == 0) {
106  row_norms[tile_row] = tile_row_sums;
107  } else {
108  row_norms[it.index()[0]] += tile_row_sums;
109  }
110  }
111 
112  double max = 0.0;
113  for (auto const &row_vector : row_norms) {
114  max = std::max(max, row_vector.maxCoeff());
115  }
116  return max;

259 template <typename Array>
260 std::array<typename Array::value_type::numeric_type, 2> eval_guess(
261  Array const &A) {
262  A.world().gop.fence();
263 
264  using global_accumlator = pair_accumulator<Array>;
265  std::vector<TiledArray::detail::ReduceTask<global_accumlator>> tasks;
266  tasks.reserve(A.trange().tiles_range().extent()[0]);
267  for (auto i = 0ul; i < A.trange().tiles_range().extent()[0]; ++i) {
268  auto row_ranges = A.trange().data()[0].tile(i);
269  std::array<typename Array::size_type, 1> start = {{row_ranges.first}};
270  std::array<typename Array::size_type, 1> finish = {{row_ranges.second}};
271  TiledArray::Range range(start, finish);
272  tasks.emplace_back(A.world(), global_accumlator(range));
273  }
274 
275  auto end = A.end();
276  for (auto it = A.begin(); it != end; ++it) {
277  tasks[it.index()[0]].add(A.find(it.ordinal()));
278  }
279 
280  TiledArray::detail::ReduceTask<pair_smasher> local_reduce(A.world(),
281  pair_smasher{});
282 
283  auto counter = 0;
284  for (auto i = 0ul; i < A.trange().tiles_range().extent()[0]; ++i) {
285  auto pair = tasks[i].submit();
286  auto row_ranges = A.trange().data()[0].tile(i);
287  std::array<typename Array::size_type, 1> start = {{row_ranges.first}};
288  std::array<typename Array::size_type, 1> finish = {{row_ranges.second}};
289  TiledArray::Range range(start, finish);
290  madness::DistributedID key(A.id(), counter++);
291  local_reduce.add(
292  A.world().gop.all_reduce(key, pair, global_accumlator{range}));
293  }
294 
295  return local_reduce.submit();

319 template <typename Array>
320 void third_order_update(Array const &S, Array &Z) {
321  auto &world = Z.world();
322 
323  // Calculate the lambda parameter, since we are currently only trying to
324  // invert Positive definite matrices, we will cap the smallest eigen value
325  // guess at 0.
326 
327  if (S.world().rank() == 0) {
328  std::cout << "Starting inverse sqrt\n";
329  }
330  auto evg0 = mpqc::fenced_now(world);
331  auto spectral_range = eval_guess(S);
332  auto evg1 = mpqc::fenced_now(world);
333  auto eval_time = mpqc::duration_in_s(evg0, evg1);
334  if (S.world().rank() == 0) {
335  std::cout << "\tEigenvalue estimation time = " << eval_time << " s\n";
336  }
337 
338  const auto max_eval = spectral_range[1];
339  const auto min_eval = std::max(0.0, spectral_range[0]);
340  auto S_scale = 2.0 / (max_eval + min_eval);
341 
342  Array Y = S;
343  Array T;
344  Array X;
345  auto Tscale = 1.0 / 8.0;
346 
347  auto ident = create_diagonal_matrix(Z, 1.0);
348  Array approx_zero;
349  auto iter = 0;
350  auto norm_diff = std::numeric_limits<double>::max();
351  while (norm_diff > 1.0e-13 && iter < 50) {
352  // Xn = \lambda*Yn*Z
353  X("i,j") = S_scale * Y("i,k") * Z("k,j");
354  X.truncate();
355 
356  // Third order update
357  T("i,j") = -10 * X("i,j") + 3 * X("i,k") * X("k,j");
358  add_to_diag(T, 15);
359  T("i,j") = Tscale * T("i,j");
360  T.truncate();
361 
362  // Updating Z and Y
363  Z("i,j") = Z("i,k") * T("k,j"); // Zn+1 = Zn*Tn
364  Y("i,j") = T("i,k") * Y("k,j"); // Yn+1 = Tn*Yn
365  Z.truncate();
366  Y.truncate();
367 
368  approx_zero("i,j") = X("i,j") - T("i,j");
369  const auto current_norm = approx_zero("i,j").norm().get();
370 
371  if (S.world().rank() == 0) {
372  std::cout << "\tCurrent difference norm = " << current_norm << "\n";
373  }
374  if (current_norm >= norm_diff) { // Once norm is increasing exit!
375  if (S.world().rank() == 0) {
376  std::cout << "Error in sqrt inverse increased. Exiting\n";
377  }
378  Z("i,j") = std::sqrt(S_scale) * Z("i,j");
379  return;
380  }
381  norm_diff = current_norm;
382  ++iter;
383  }
384 
385  Z("i,j") = std::sqrt(S_scale) * Z("i,j");
386 }
