48  bool skip(int64_t a, int64_t c, int64_t d) override {
49  auto local_Qcd = Qcd(c, d);
50  auto local_Qa = Qab(a);
51  auto QaQcd = local_Qcd * local_Qa;
52 
53  auto ash = Qab_f2s(a);
54  auto csh = Qcd_f2s(c);
55  auto dsh = Qcd_f2s(d);
56 
57  if (QaQcd >= skip_threshold()) {
58  double Rnorm = (shell_info_.a_center(ash)
59  - shell_info_.cd_center(csh, dsh)).norm();
60 
61  if (Rnorm > shell_info_.a_extent(ash)
62  + shell_info_.cd_extent(csh, dsh)) {
63  QaQcd = local_Qcd
64  * shell_info_.scaling_factor(ash, csh, dsh, Rnorm);
65  }
66  }
67 
68  return (QaQcd < skip_threshold()) ? true : false;
