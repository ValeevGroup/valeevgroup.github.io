76 template <typename Tile, typename Policy>
77 TA::DistArray<Tile, Policy> EOM_CCSD<Tile, Policy>::compute_HSS_HSD_C(
78  const TArray& Cai, const TArray& Cabij,
79  const cc::Intermediates<TArray>& imds) {
80  TArray HSS_HSD_C;
81 
82  {
83  // HSS * C part
84  HSS_HSD_C("a,i") = // Fac C^c_i
85  imds.FAB("a,c") * Cai("c,i")
86  // - Fki C^a_k
87  - imds.FIJ("k,i") * Cai("a,k")
88  // + Wakic C^c_k
89  + (2.0 * imds.Wiabj("k,a,c,i") + imds.Wiajb("k,c,i,a")) * Cai("c,k");
90  }
91 
92  // HSD * C part
93  {
94  TArray C;
95  C("a,c,i,k") = 2.0 * Cabij("a,c,i,k") - Cabij("c,a,i,k");
96  HSS_HSD_C("a,i") += // Fkc C^ac_ik
97  imds.FIA("k,c") * C("a,c,i,k")
98  // - 1/2 Wklic C^ac_kl
99  - imds.Wijka("k,l,i,c") * C("a,c,k,l");
100 
101  if (this->df_) {
102  auto t1 = this->t1();
103  // + 1/2 Wakcd C^cd_ik
104 
105  HSS_HSD_C("a,i") += imds.Xia("K,k,d") * C("c,d,i,k") *
106  (imds.Xab("K,a,c") - t1("a,l") * imds.Xia("K,l,c"));
107 
108  } else {
109  // + 1/2 Wakcd C^cd_ik
110  HSS_HSD_C("a,i") += imds.Waibc("a,k,c,d") * C("c,d,i,k");
111  }
112  }
113 
114  return HSS_HSD_C;
115 }

118 template <typename Tile, typename Policy>
119 TA::DistArray<Tile, Policy> EOM_CCSD<Tile, Policy>::compute_HDS_HDD_C(
120  const TArray& Cai, const TArray& Cabij,
121  const cc::Intermediates<TArray>& imds) {
122  TArray HDS_HDD_C;
123  auto t2 = this->t2();
124  auto t1 = this->t1();
125 
126  // HDS * C part
127  {
128  HDS_HDD_C("a,b,i,j") =
129  // P(ij) Wabcj C^c_i
130  // WAbCj C^C_I + WbAcI
131  imds.Wabci("a,b,c,j") * Cai("c,i")
132  // P(ab) Wkaij C^b_k C^c_j
133  // - WkAjI C^b_k - WKbIj C^A_K
134  - imds.Wiajk("k,b,i,j") * Cai("a,k")
135 
136  // - P(ij) Wlkjc C^c_k t^ab_il
137  // - WlKjC C^C_K T^Ab_Il - Wlkjc C^c_k T^Ad_Ij
138  // + WLKIC C^C_K T^Ab_jL + WLkIc C^c_k T^Ab_jL
139  - (2.0 * imds.Wijka("l,k,j,c") - imds.Wijka("k,l,j,c")) * Cai("c,k") *
140  t2("a,b,i,l");
141 
142  if (this->df_) {
143  TArray tmp;
144  tmp("K,b,d") = imds.Xab("K,b,d") - t1("b,l") * imds.Xia("K,l,d");
145 
146  HDS_HDD_C("a,b,i,j") +=
147  (2.0 * imds.Xia("K,k,c") * Cai("c,k") * tmp("K,b,d") -
148  tmp("K,b,c") * Cai("c,k") * imds.Xia("K,k,d")) *
149  t2("a,d,i,j");
150 
151  } else {
152  // + P(ab) Wbkdc C^c_k T^ad_ij
153  // + WbKdC C^C_K T^Ad_Ij + Wbkdc C^c_k T^Ad_Ij
154  // - WAKDC C^C_K T^bD_Ij - WAkDc C^c_k T^bD_Ij
155  HDS_HDD_C("a,b,i,j") +=
156  (2.0 * imds.Waibc("b,k,d,c") - imds.Waibc("b,k,c,d")) * Cai("c,k") *
157  t2("a,d,i,j");
158  }
159  // HDS_HDD_C("a,b,i,j") += HDS_HDD_C("b,a,j,i");
160  }
161 
162  // HDD * C part
163  {
164  TArray C, GC_ab, GC_ij;
165  C("a,c,i,k") = 2.0 * Cabij("a,c,i,k") - Cabij("c,a,i,k");
166  GC_ab("a,b") = imds.Wijab("k,l,a,c") * C("b,c,k,l");
167  GC_ij("i,j") = imds.Wijab("i,k,c,d") * C("c,d,j,k");
168 
169  TArray tmp;
170  HDS_HDD_C("a,b,i,j") +=
171 
172  // P(ab) Fbc C^ac_ij
173  // Fbc C^ac_ij + Fac C^cb_ij
174  imds.FAB("a,c") * Cabij("c,b,i,j")
175 
176  // - P(ij) Fkj C^ab_ik
177  // - Fkj C^ab_ik - Fki C^ab_kj
178  - imds.FIJ("k,j") * Cabij("a,b,i,k")
179 
180  // + P(ab) P(ij) Wbkjc C^ac_ik
181  // + Wbkjc C^ac_ik - Wbkic C^ac_jk
182  // - Wakjc C^bc_ik + Wakic C^bc_jk
183  + imds.Wiabj("k,b,c,j") * C("a,c,i,k") +
184  imds.Wiajb("k,c,j,b") * Cabij("a,c,i,k") +
185  imds.Wiajb("k,c,i,b") * Cabij("a,c,k,j")
186 
187  // - 1/2 P(ab) g^lk_dc C^ca_kl t^db_ij
188  // - 1/2 g^kl_dc C^ac_kl t^db_ij
189  // - 1/2 g^kl_cd C^cb_kl t^ad_ij
190  - GC_ab("d,a") * t2("d,b,i,j")
191 
192  // + 1/2 P(ij) Wlkdc C^dc_ik t^ab_jl
193  // - 1/2 Wlkcd C^cd_ik t^ab_lj
194  // - 1/2 Wlkcd C^cd_jk t^ab_il
195  - GC_ij("l,j") * t2("a,b,i,l");
196 
197  HDS_HDD_C("a,b,i,j") += HDS_HDD_C("b,a,j,i")
198  // + 1/2 Wabcd C^cd_ij
199  // + WAbCd_("a,b,c,d") * Cabij("c,d,i,j")
200  // + 1/2 Wklij C^ab_kl
201  + imds.Wijkl("k,l,i,j") * Cabij("a,b,k,l");
202 
203  if (imds.Wabcd.is_initialized()) {
204  // + 1/2 Wabcd C^cd_ij
205  // + WAbCd_("a,b,c,d") * Cabij("c,d,i,j")
206  HDS_HDD_C("a,b,i,j") += imds.Wabcd("a,b,c,d") * Cabij("c,d,i,j");
207  } else {
208  TArray tau;
209  tau("a,b,i,j") = t2("a,b,i,j") + t1("a,i") * t1("b,j");
210 
211  // integral direct term
212  auto direct_integral = this->get_direct_ao_integral();
213 
214  auto Ca =
215  this->lcao_factory().orbital_registry().retrieve(OrbitalIndex(L"a"));
216  auto Ci =
217  this->lcao_factory().orbital_registry().retrieve(OrbitalIndex(L"i"));
218 
219  TArray U;
220  U("p,r,i,j") =
221  Cabij("c,d,i,j") * Ca("q,c") * Ca("s,d") * direct_integral("p,q,r,s");
222  // U("p,r,i,j") = 0.5 * (U("p,r,i,j") + U("r,p,j,i"));
223  HDS_HDD_C("a,b,i,j") +=
224  U("p,r,i,j") * Ca("p,a") * Ca("r,b") -
225  U("r,p,i,j") * Ci("p,k") * Ca("r,a") * t1("b,k") -
226  U("p,r,i,j") * Ci("p,k") * Ca("r,b") * t1("a,k") +
227  U("p,r,i,j") * Ci("p,k") * Ci("r,l") * tau("a,b,k,l");
228  }
229  }
230 
231  return HDS_HDD_C;
232 }
