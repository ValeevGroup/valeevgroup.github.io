16 // Compute the min eval guess for a row of tiles.
17 template <typename It>
18 double min_eval_guess(It first, It second) {
19  auto nrows = first->get().range().extent()[0];
20  std::vector<double> row_min_evals(nrows, 0);
21 
22  for (; first != second; ++first) {
23  auto tensor = first->get();
24  std::vector<std::size_t> extent = tensor.range().extent();
25  auto matrix = TiledArray::eigen_map(tensor, extent[0], extent[1]);
26 
27  auto idx = first.index();
28 
29  if (idx[0] == idx[1]) {
30  for (auto i = 0l; i < matrix.rows(); ++i) {
31  for (auto j = 0l; j < matrix.cols(); ++j) {
32  row_min_evals[i] += (i == j) ? matrix(i, i) : -std::abs(matrix(i, j));
33  }
34  }
35  } else {
36  for (auto i = 0l; i < matrix.rows(); ++i) {
37  for (auto j = 0l; j < matrix.cols(); ++j) {
38  row_min_evals[i] -= std::abs(matrix(i, j));
39  }
40  }
41  }
42  }
43 
44  return *std::min_element(row_min_evals.begin(), row_min_evals.end());
45 }

47 template <typename Array>
48 std::pair<double, double> symmetric_min_max_evals(Array const &S) {
49  auto const &array_extent = S.trange().tiles_range().extent();
50  std::vector<Eigen::VectorXd> row_norms(array_extent[0]);
51 
52  for (auto it = S.begin(); it != S.end(); ++it) {
53  auto const &tile = it->get();
54  auto const extent = tile.range().extent();
55  auto matrix_map = TiledArray::eigen_map(tile, extent[0], extent[1]);
56 
57  Eigen::VectorXd tile_row_sums(extent[0]);
58 
59  for (auto i = 0l; i < matrix_map.rows(); ++i) {
60  auto row_sum = 0.0;
61  for (auto j = 0l; j < matrix_map.cols(); ++j) {
62  row_sum += std::abs(matrix_map(i, j));
63  }
64  tile_row_sums[i] = row_sum;
65  }
66 
67  auto tile_row = it.index()[0];
68  if (row_norms[tile_row].size() == 0) {
69  row_norms[tile_row] = tile_row_sums;
70  } else {
71  row_norms[it.index()[0]] += tile_row_sums;
72  }
73  }
74 
75  double max = 0.0;
76  double min = std::numeric_limits<double>::max();
77  for (auto const &row_vector : row_norms) {
78  max = std::max(max, row_vector.maxCoeff());
79  min = std::min(min, row_vector.minCoeff());
80  }
81  return std::make_pair(min, max);
82 }

93 template <typename Array>
94 Array invert(Array const &S) {
95  Array X = init_X(S);
96  Array product;
97  product("i,j") = X("i,k") * S("k,j");
98 
99  auto iter = 0;
100  double trace_ideal = S.trange().tiles_range().extent()[0];
101  double trace_real = 0.0;
102  while (iter < 1000 && std::abs(trace_real - trace_ideal) >= 1e-10) {
103  X("i,j") = 2 * X("i,j") - X("i,k") * S("k,l") * X("l,j");
104  product("i,j") = X("i,k") * S("k,j");
105  trace_real = product("i,j").trace();
106  ++iter;
107  }
108  X.world().gop.fence();
109  return X;
110 }

112 template <typename Array>
113 double min_eval_est(Array const &H, Array const &S) {
114  Array HSinv;
115  HSinv("i,j") = invert(S)("i,k") * H("k,j");
116 
117  auto min = std::numeric_limits<double>::max();
118  auto extent = HSinv.trange().tiles_range().extent();
119 
120  // Check every row of tiles for its minimum eval guess
121  for (auto tile_row_it = HSinv.begin(); tile_row_it != HSinv.end();) {
122  auto next_tile_row = tile_row_it;
123  std::advance(next_tile_row, extent[1]);
124 
125  min = std::min(min, min_eval_guess(tile_row_it, next_tile_row));
126 
127  tile_row_it = next_tile_row; // advance iterator
128  }
129 
130  return min;
131 }
