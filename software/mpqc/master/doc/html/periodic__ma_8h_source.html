516  bool is_uc_in_CFF_condition1(const Vector3i &uc_ket,
517  const Vector3i &uc_bra = {0, 0, 0}) {
518  Vector3d vec_bra = uc_bra.cast<double>().cwiseProduct(dcell_);
519  Vector3d vec_ket = uc_ket.cast<double>().cwiseProduct(dcell_);
520 
521  const auto npairs = ref_pairs_->npairs();
522 
523  using CenterExtentPair = std::vector<std::pair<Vector3d, double>>;
524  CenterExtentPair uc0, uc1;
525  for (auto p = 0ul; p != npairs; ++p) {
526  const auto center = ref_pairs_->center(p);
527  const auto extent = ref_pairs_->extent(p);
528  uc0.emplace_back(std::make_pair(center + vec_bra, extent));
529  uc1.emplace_back(std::make_pair(center + vec_ket, extent));
530  }
531 
532  auto condition1 = true;
533  for (auto p0 = 0ul; p0 != npairs; ++p0) {
534  const auto &center0 = uc0[p0].first;
535  const auto &extent0 = uc0[p0].second;
536  for (auto p1 = 0ul; p1 != npairs; ++p1) {
537  const auto &center1 = uc1[p1].first;
538  const auto &extent1 = uc1[p1].second;
539 
540  const auto rab2 = (center1 - center0).squaredNorm();
541  const auto ex2 = (extent0 + extent1) * (extent0 + extent1);
542  if (rab2 < ex2) {
543  condition1 = false;
544  break;
545  }
546  }
547  if (!condition1) break;
548  }
549 
550  return condition1;
551  }

663  template <unsigned int lmax>
664  MultipoleMoment<double, lmax> build_interaction_kernel(const Vector3d &c12) {
665  const auto c12_norm = c12.norm();
666  MPQC_ASSERT(c12_norm > 0.0);
667 
668  using namespace boost::math;
669  MultipoleMoment<double, lmax> result;
670 
671  const auto cos_theta = c12(2) / c12_norm;
672  const auto c12_xy_norm = std::sqrt(c12(0) * c12(0) + c12(1) * c12(1));
673 
674  // When Rx^2 + Ry^2 = 0, φ cannot computed. We use the fact that in this
675  // case cosθ = 1 or -1, and then associated Legendre polynomials
676  // P_{l, m}(cosθ) = 0 if m != 0
677  // P_{l, m}(cosθ) = (cosθ)^l if m == 0
678  if (c12_xy_norm < std::numeric_limits<double>::epsilon() *
679  10.0) { // in case of c12_xy_norm ~ epsilon, let's
680  // multiply epsilon by 10
681  for (auto l = 0, ord_idx = 0; l <= lmax; ++l) {
682  auto cos_theta_l = ((cos_theta > 0.0) || (l % 2 == 0)) ? 1.0 : -1.0;
683  auto result_l_0 =
684  factorial<double>(l) / std::pow(c12_norm, l + 1) * cos_theta_l;
685  for (auto m = -l; m <= l; ++m, ++ord_idx) {
686  result[ord_idx] = (m == 0) ? result_l_0 : 0.0;
687  }
688  }
689  return result;
690  } else {
691  // build a map that returns cos(m * phi) for each m using recurrence
692  // relation
693  auto cos_m_phi_map = cos_m_phi_map_recursive<lmax>(c12);
694  // build a map that returns sin(m * phi) for each m using recurrence
695  // relation
696  auto sin_m_phi_map = sin_m_phi_map_recursive<lmax>(c12);
697  // build a map that returns associated Legendre polynomial P_{l, m}(cosθ)
698  // for each ordinal index of (l, m) pair, using recurrence relation
699  auto legendre_map = associated_legendre_map_recursive<lmax>(cos_theta);
700 
701  // fill the result
702  for (auto l = 0, ord_idx = 0; l <= lmax; ++l) {
703  const auto inv_denom = 1.0 / std::pow(c12_norm, l + 1);
704  for (auto m = -l; m <= l; ++m, ++ord_idx) {
705  const auto num = factorial<double>(l - m);
706  const auto phi_part = (m >= 0) ? cos_m_phi_map[m] : sin_m_phi_map[m];
707  result[ord_idx] = num * inv_denom * legendre_map[ord_idx] * phi_part;
708  }
709  }
710 
711  return result;
712  }

737  MultipoleMoment<double, lmax> build_multipole_expansion(const Vector3d &r) {
738  const auto r_norm = r.norm();
739 
740  MultipoleMoment<double, lmax> result;
741 
742  if (r_norm < std::numeric_limits<double>::epsilon()) {
743  result.fill(0.0);
744  return result;
745  } else {
746  using namespace boost::math;
747 
748  const auto cos_theta = r(2) / r_norm;
749  const auto r_xy_norm = std::sqrt(r(0) * r(0) + r(1) * r(1));
750 
751  // When Rx^2 + Ry^2 = 0, φ cannot computed. We use the fact that in this
752  // case cosθ = 1 or -1, and then associated Legendre polynomials
753  // P_{l, m}(cosθ) = 0 if m != 0
754  // P_{l, m}(cosθ) = (cosθ)^l if m == 0
755  if (r_xy_norm < std::numeric_limits<double>::epsilon() *
756  10.0) { // in case of r_xy_norm ~ epsilon, let's
757  // multiply epsilon by 10
758  for (auto l = 0, ord_idx = 0; l <= lmax; ++l) {
759  auto cos_theta_l = ((cos_theta > 0.0) || (l % 2 == 0)) ? 1.0 : -1.0;
760  auto result_l_0 =
761  std::pow(r_norm, l) / factorial<double>(l) * cos_theta_l;
762  for (auto m = -l; m <= l; ++m, ++ord_idx) {
763  result[ord_idx] = (m == 0) ? result_l_0 : 0.0;
764  }
765  }
766  return result;
767  } else {
768  // build a map that returns cos(m * phi) for each m using recurrence
769  // relation
770  auto cos_m_phi_map = cos_m_phi_map_recursive<lmax>(r);
771  // build a map that returns sin(m * phi) for each m using recurrence
772  // relation
773  auto sin_m_phi_map = sin_m_phi_map_recursive<lmax>(r);
774  // build a map that returns associated Legendre polynomial P_{l,
775  // m}(cosθ) for each ordinal index of (l, m) pair, using recurrence
776  // relation
777  auto legendre_map = associated_legendre_map_recursive<lmax>(cos_theta);
778 
779  // fill the result
780  for (auto l = 0, ord_idx = 0; l <= lmax; ++l) {
781  const auto num = std::pow(r_norm, l);
782  for (auto m = -l; m <= l; ++m, ++ord_idx) {
783  const auto inv_denom = 1.0 / factorial<double>(l + m);
784  const auto phi_part =
785  (m >= 0) ? cos_m_phi_map[m] : sin_m_phi_map[m];
786  result[ord_idx] =
787  num * inv_denom * legendre_map[ord_idx] * phi_part;
788  }
789  }
790 
791  return result;
792  }
793  }
794  }

803  template <unsigned int lmax>
804  std::unordered_map<int, double> sin_m_phi_map_recursive(const Vector3d &r) {
805  const auto r_xy_norm = std::sqrt(r(0) * r(0) + r(1) * r(1));
806  MPQC_ASSERT(r_xy_norm >= std::numeric_limits<double>::epsilon());
807 
808  const auto inv_x2py2 = 1.0 / r_xy_norm;
809  const auto cos_phi = r(0) * inv_x2py2;
810  const auto sin_phi = r(1) * inv_x2py2;
811 
812  std::unordered_map<int, double> sin_m_phi_map;
813  sin_m_phi_map.reserve(2 * lmax + 1);
814 
815  sin_m_phi_map[0] = 0.0;
816  if (lmax >= 1) {
817  sin_m_phi_map[1] = sin_phi;
818  sin_m_phi_map[-1] = -sin_phi;
819  }
820  if (lmax >= 2) {
821  for (auto m = 2; m <= lmax; ++m) {
822  sin_m_phi_map[m] =
823  2.0 * sin_m_phi_map[m - 1] * cos_phi - sin_m_phi_map[m - 2];
824  sin_m_phi_map[-m] = -sin_m_phi_map[m];
825  }
826  }
827 
828  return sin_m_phi_map;

838  template <unsigned int lmax>
839  std::unordered_map<int, double> cos_m_phi_map_recursive(const Vector3d &r) {
840  const auto r_xy_norm = std::sqrt(r(0) * r(0) + r(1) * r(1));
841  MPQC_ASSERT(r_xy_norm >= std::numeric_limits<double>::epsilon());
842 
843  const auto cos_phi = r(0) / r_xy_norm;
844 
845  std::unordered_map<int, double> cos_m_phi_map;
846  cos_m_phi_map.reserve(2 * lmax + 1);
847 
848  cos_m_phi_map[0] = 1.0;

857  cos_m_phi_map[-m] = cos_m_phi_map[m];

871  template <unsigned int lmax>
872  std::unordered_map<int, double> associated_legendre_map_recursive(
873  double cos_theta) {
874  using namespace boost::math;
875 
876  double sin_theta = std::sqrt(1.0 - cos_theta * cos_theta);
877 
878  std::unordered_map<int, double> legendre_map;
879  legendre_map.reserve((lmax + 1) * (lmax + 1));
880 
881  legendre_map[0] = 1.0; // (l = 0, m = 0)
882  if (lmax >= 1) {
883  // compute P_{l, m} for all l >= 1 and m = 0
884  legendre_map[2] = cos_theta; // (l = 1, m = 0)
885  if (lmax >= 2) {
886  for (auto l = 2; l <= lmax; ++l) {
887  const auto ord_l_0 = l * l + l; // (l, 0)
888  const auto ord_lm1_0 = l * l - l; // (l - 1, 0)
889  const auto ord_lm2_0 = l * l - 3 * l + 2; // (l - 2, 0)
890  legendre_map[ord_l_0] =
891  legendre_next(l - 1, 0, cos_theta, legendre_map[ord_lm1_0],
892  legendre_map[ord_lm2_0]);
893  }
894  }
895 
896  // compute P_{l, m} for all l >= 1 and m != 0
897  for (auto m = 1; m <= lmax; ++m) {
898  const auto sign = (m % 2 == 0) ? 1 : -1;
899  const auto m2 = m * m;
900 
901  const auto ord_m_m = m2 + m + m; // (l = m, m)
902  const auto ord_mm1_mm1 = m2 - 1; // (l = m - 1, m - 1)
903  legendre_map[ord_m_m] =
904  -1.0 * (2.0 * m - 1.0) * sin_theta * legendre_map[ord_mm1_mm1];
905  legendre_map[m2] = sign * legendre_map[ord_m_m] /
906  factorial<double>(m + m); // (l = m, -m)
907 
908  if (lmax >= 2 && lmax >= m + 1) {
909  const auto ord_mp1_m = m2 + 4 * m + 2; // (l = m + 1, m)
910  legendre_map[ord_mp1_m] =
911  (2.0 * m + 1.0) * cos_theta * legendre_map[ord_m_m];
912  legendre_map[ord_mp1_m - 2 * m] =
913  sign * legendre_map[ord_mp1_m] /
914  factorial<double>(2 * m + 1); // (l = m + 1, -m)
915  }
916 
917  if (lmax >= 3 && lmax >= m + 2) {
918  for (auto l = m + 2; l <= lmax; ++l) {
919  const auto l2 = l * l;
920  const auto ord_l_m = l2 + l + m; // (l, m)
921  const auto ord_lm1_m = l2 - l + m; // (l - 1, m)
922  const auto ord_lm2_m = l2 - 3 * l + 2 + m; // (l - 2, m)
923  legendre_map[ord_l_m] =
924  legendre_next(l - 1, m, cos_theta, legendre_map[ord_lm1_m],
925  legendre_map[ord_lm2_m]);
926  legendre_map[ord_l_m - 2 * m] =
927  sign * factorial<double>(l - m) * legendre_map[ord_l_m] /
928  factorial<double>(l + m); // (l, -m)
929  }
930  }
931  }
932  }
933 
934  return legendre_map;
935  }

947  template <unsigned int lmax, unsigned int nopers = (lmax + 1) * (lmax + 1)>
948  std::pair<MultipoleMoment<double, lmax>, MultipoleMoment<double, lmax>>
949  form_symm_and_antisymm_moments(const MultipoleMoment<double, lmax> &M,
950  Ord2lmMap &ord_to_lm_map) {
951  assert(M.size() == nopers);
952  assert(ord_to_lm_map.size() == nopers);
953  MultipoleMoment<double, lmax> Mplus, Mminus;
954  for (auto op = 0u; op != nopers; ++op) {
955  auto m = ord_to_lm_map[op].second;
956  auto sign_of_m = (m > 0) ? 1 : ((m < 0) ? -1 : 0);
957  auto nega1_m = (m % 2 == 0) ? 1.0 : -1.0; // (-1)^m
958  switch (sign_of_m) {
959  case 1:
960  Mplus[op] = M[op];
961  Mminus[op] = -nega1_m * M[op - 2 * m];
962  break;
963  case 0:
964  Mplus[op] = M[op];
965  Mminus[op] = 0.0;
966  break;
967  case -1:
968  Mplus[op] = nega1_m * M[op - 2 * m];
969  Mminus[op] = M[op];
970  break;
971  default:
972  throw ProgrammingError("Invalid sign of m.", __FILE__, __LINE__);
973  }
974  }
975  return std::make_pair(Mplus, Mminus);
976  }

987  template <unsigned int lmax, unsigned int nopers = (lmax + 1) * (lmax + 1)>
988  MultipoleMoment<double, lmax> build_local_potential(
989  const MultipoleMoment<double, lmax> &O,
990  const MultipoleMoment<double, 2 * lmax> &M, Ord2lmMap &O_ord_to_lm_map,
991  Ord2lmMap &M_ord_to_lm_map) {
992  MultipoleMoment<double, lmax> result;
993 
994  // form O+, O-
995  auto O_pm = form_symm_and_antisymm_moments<lmax>(O, O_ord_to_lm_map);
996  auto &O_plus = O_pm.first;
997  auto &O_minus = O_pm.second;
998 
999  // form M+, M-
1000  auto M_pm = form_symm_and_antisymm_moments<2 * lmax>(M, M_ord_to_lm_map);
1001  auto &M_plus = M_pm.first;
1002  auto &M_minus = M_pm.second;
1003 
1004  // form multipole expansion of local potential created by the remote unit
1005  // cell
1006  for (auto op1 = 0; op1 != nopers; ++op1) {
1007  auto l1 = O_ord_to_lm_map[op1].first;
1008  auto m1 = O_ord_to_lm_map[op1].second;
1009  auto accu = 0.0;
1010  for (auto op2 = 0; op2 != nopers; ++op2) {
1011  auto l1pl2 = l1 + O_ord_to_lm_map[op2].first;
1012  auto m1pm2 = m1 + O_ord_to_lm_map[op2].second;
1013  auto ord_M = l1pl2 * l1pl2 + l1pl2 + m1pm2;
1014  if (m1 >= 0) {
1015  accu += M_plus[ord_M] * O_plus[op2];
1016  accu += M_minus[ord_M] * O_minus[op2];
1017  } else {
1018  accu += M_minus[ord_M] * O_plus[op2];
1019  accu -= M_plus[ord_M] * O_minus[op2];
1020  }
1021  }
1022  result[op1] = accu;
1023  }
1024 
1025  return result;
1026  }

1035  template <unsigned int lmax, unsigned int nopers = (lmax + 1) * (lmax + 1)>
1036  double compute_energy(const MultipoleMoment<double, lmax> &O,
1037  const MultipoleMoment<double, lmax> &L,
1038  Ord2lmMap &O_ord_to_lm_map) {

1046  result += 0.5 * sign * delta * O[op] * L[op];
1047  }

1049  return result;
