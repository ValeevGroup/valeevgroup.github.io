94  void check_valid() const {
95  const unsigned int rank = array_.trange().tiles().rank();
96  // Check the dimension of the lower block bound
97  if(TiledArray::detail::size(lower_bound_) != rank) {
98  if(World::get_default().rank() == 0) {
99  TA_USER_ERROR_MESSAGE( \
100  "The size lower bound of the block is not equal to rank of " \
101  "the array: " \
102  << "\n array rank = " << array_.trange().tiles().rank() \
103  << "\n lower bound = " << lower_bound_ );
104 
105  TA_EXCEPTION("The size lower bound of the block is not equal to " \
106  "rank of the array.");
107  }
108  }
109 
110  // Check the dimension of the upper block bound
111  if(TiledArray::detail::size(upper_bound_) != rank) {
112  if(World::get_default().rank() == 0) {
113  TA_USER_ERROR_MESSAGE( \
114  "The size upper bound of the block is not equal to rank of " \
115  "the array: " \
116  << "\n array rank = " << rank \
117  << "\n upper bound = " << upper_bound_ );
118 
119  TA_EXCEPTION("The size upper bound of the block is not equal to " \
120  "rank of the array.");
121  }
122  }
123 
124  const bool lower_bound_check =
125  std::equal(std::begin(lower_bound_), std::end(lower_bound_),
126  array_.trange().tiles().lobound_data(),
127  [] (std::size_t l, std::size_t r) { return l >= r; });
128  const bool upper_bound_check =
129  std::equal(std::begin(upper_bound_), std::end(upper_bound_),
130  array_.trange().tiles().upbound_data(),
131  [] (std::size_t l, std::size_t r) { return l <= r; });
132  if(! (lower_bound_check && upper_bound_check)) {
133  if(World::get_default().rank() == 0) {
134  TA_USER_ERROR_MESSAGE( \
135  "The block range is not a sub-block of the array range: " \
136  << "\n array range = " << array_.trange().tiles() \
137  << "\n block range = [ " << lower_bound_ << " , " << upper_bound_ << " )");
138  }
139 
140  TA_EXCEPTION("The block range is not a sub-block of the array range.");
141  }
142 
143  const bool lower_upper_bound_check =
144  std::equal(std::begin(lower_bound_), std::end(lower_bound_),
145  std::begin(upper_bound_),
146  [] (std::size_t l, std::size_t r) { return l < r; });
147  if(! lower_upper_bound_check) {
148  if(World::get_default().rank() == 0) {
149  TA_USER_ERROR_MESSAGE( \
150  "The block lower bound is not less than the upper bound: " \
151  << "\n lower bound = " << lower_bound_ \
152  << "\n upper bound = " << upper_bound_);
153  }
154 
155  TA_EXCEPTION("The block lower bound is not less than the upper bound.");
156  }
157  }
