80  static void lazy_deleter(const impl_type* const pimpl) {
81  try {
82  if(pimpl) {
83  if(madness::initialized()) {
84  World& world = pimpl->get_world();
85  const madness::uniqueidT id = pimpl->id();
86  cleanup_counter_++;
87  world.gop.lazy_sync(id, [pimpl]() {
88  delete pimpl;
89  Array_::cleanup_counter_--;
90  });
91  } else {
92  delete pimpl;
93  }
94  }
95  } catch(...) {
96  fprintf(stderr, "!! ERROR TiledArray: An error occurred in Array::lazy_deleter()\n");
97  // Abort since we cannot throw from a destructor.
98  SafeMPI::COMM_WORLD.Abort(1);
99  }
100  }

108  static std::shared_ptr<impl_type>
109  init(World& world, const trange_type& trange, const shape_type& shape,
110  std::shared_ptr<pmap_interface> pmap)
111  {
112  // User level validation of input
113 
114  // Validate the tiled range
115  TA_USER_ASSERT(trange.tiles().dim() == DIM,
116  "Array::Array() -- The dimension of tiled range is not equal to the array.");
117 
118  if(! pmap) {
119  // Construct a default process map
120  pmap = Policy::default_pmap(world, trange.tiles().volume());
121  } else {
122  // Validate the process map
123  TA_USER_ASSERT(pmap->size() == trange.tiles().volume(),
124  "Array::Array() -- The size of the process map is not equal to the number of tiles in the TiledRange object.");
125  TA_USER_ASSERT(pmap->rank() == typename pmap_interface::size_type(world.rank()),
126  "Array::Array() -- The rank of the process map is not equal to that of the world object.");
127  TA_USER_ASSERT(pmap->procs() == typename pmap_interface::size_type(world.size()),
128  "Array::Array() -- The number of processes in the process map is not equal to that of the world object.");
129  }
130 
131  // Validate the shape
132  TA_USER_ASSERT(! shape.empty(),
133  "Array::Array() -- The shape is not initialized.");
134  TA_USER_ASSERT(shape.validate(trange.tiles()),
135  "Array::Array() -- The range of the shape is not equal to the tiles range.");
136 
137  return std::shared_ptr<impl_type>(new impl_type(world, trange, shape, pmap), lazy_deleter);
138  }

329  template <typename Op>
330  void init_local_tiles(Op&& op, const bool wait = false) {
331  check_pimpl();
332  madness::Range<typename pmap_interface::const_iterator>
333  range(pimpl_->pmap()->begin(), pimpl_->pmap()->end(), 8);
334 
335  Array_ array(*this);
336 
337  Future<bool> result = pimpl_->get_world().taskq.for_each(range,
338  [=] (const typename pmap_interface::const_iterator& it) mutable {
339  const size_type index = *it;
340  if(! array.is_zero(index))
341  array.set(index, op(array.trange().make_tile_range(*it)));
342  return true;
343  });
344 
345  // Wait for all tiles to be set
346  if(wait)
347  result.get();
348  }

496  void make_replicated() {
497  check_pimpl();
498  if((! pimpl_->pmap()->is_replicated()) && (get_world().size() > 1)) {
499  // Construct a replicated array
500  std::shared_ptr<pmap_interface> pmap(new detail::ReplicatedPmap(get_world(), size()));
501  Array_ result = Array_(get_world(), trange(), get_shape(), pmap);
502 
503  // Create the replicator object that will do an all-to-all broadcast of
504  // the local tile data.
505  std::shared_ptr<detail::Replicator<Array_> > replicator(
506  new detail::Replicator<Array_>(*this, result));
507 
508  // Put the replicator pointer in the deferred cleanup object so it will
509  // be deleted at the end of the next fence.
510  TA_ASSERT(replicator.unique()); // Required for deferred_cleanup
511  madness::detail::deferred_cleanup(get_world(), replicator);
