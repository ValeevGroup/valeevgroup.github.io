70 template <typename Array>
71 Array compute_cs_ccsd_r1(const Array& t1, const Array& t2, const Array& tau,
72  cc::Integrals<Array>& ints, const Array& u = Array()) {
73  Array r1;
74  {
75  // intermediates for t1
76  // external index i and a
77  // vir index a b c d
78  // occ index i j k l
79  Array h_kc, h_ki, h_ac;
80 
81  // compute residual r1(n) (at convergence r1 = 0)
82  // external index i and a
83  r1("a,i") =
84  ints.Fia("i,a") - 2.0 * (ints.Fia("k,c") * t1("c,i")) * t1("a,k");
85 
86  Array g_ijab_bar;
87  g_ijab_bar("i,j,a,b") = 2.0 * ints.Gijab("i,j,a,b") - ints.Gijab("j,i,a,b");
88  {
89  h_ac = compute_cs_ccsd_F_vv(ints.Fab, g_ijab_bar, tau);
90  ints.FAB = h_ac;
91  r1("a,i") += h_ac("a,c") * t1("c,i");
92  }
93 
94  {
95  h_ki = compute_cs_ccsd_F_oo(ints.Fij, g_ijab_bar, tau);
96  ints.FIJ = h_ki;
97  r1("a,i") -= t1("a,k") * h_ki("k,i");
98  }
99 
100  {
101  h_kc = compute_cs_ccsd_F_ov(ints.Fia, g_ijab_bar, t1);
102 
103  r1("a,i") += h_kc("k,c") * (2.0 * t2("c,a,k,i") - t2("c,a,i,k") +
104  t1("c,i") * t1("a,k"));
105  }
106 
107  r1("a,i") +=
108  (2.0 * ints.Gijab("k,i,c,a") - ints.Giajb("k,a,i,c")) * t1("c,k");
109 
110  if (!u.is_initialized()) {
111  Array g_iabc_bar;
112  g_iabc_bar("k,a,c,d") =
113  2.0 * ints.Giabc("k,a,c,d") - ints.Giabc("k,a,d,c");
114  r1("a,i") += g_iabc_bar("k,a,c,d") * tau("c,d,k,i");
115  } else {
116  r1("a,i") +=
117  (2.0 * u("p,r,k,i") - u("p,r,i,k")) * ints.Ci("p,k") * ints.Ca("r,a");
118  }
119 
120  r1("a,i") -=
121  (2.0 * ints.Gijka("l,k,i,c") - ints.Gijka("k,l,i,c")) * tau("c,a,k,l");
122  }
123  return r1;
124 };

136 template <typename Array>
137 Array compute_cs_ccsd_r1_df(const Array& t1, const Array& t2, const Array& tau,
138  cc::Integrals<Array>& ints) {
139  Array r1;
140  {
141  // intermediates for t1
142  // external index i and a
143  // vir index a b c d
144  // occ index i j k l
145  Array h_kc, h_ki, h_ac;
146 
147  Array X_ai_tau;
148 
149  X_ai_tau("X,a,i") = 2.0 * ints.Xai("X,b,j") * tau("a,b,i,j") -
150  ints.Xai("X,b,j") * tau("a,b,j,i");
151 
152  // compute residual r1(n) (at convergence r1 = 0)
153  // external index i and a
154  r1("a,i") = ints.Fia("i,a") - 2.0 * ints.Fia("k,c") * t1("c,i") * t1("a,k");
155 
156  {
157  h_ac("a,c") = ints.Fab("a,c") - X_ai_tau("K,a,l") * ints.Xai("K,c,l");
158  ints.FAB = h_ac;
159  r1("a,i") += h_ac("a,c") * t1("c,i");
160  }
161 
162  {
163  h_ki("k,i") = ints.Fij("k,i") + X_ai_tau("K,c,i") * ints.Xai("K,c,k");
164  ints.FIJ = h_ki;
165  r1("a,i") -= t1("a,k") * h_ki("k,i");
166  }
167 
168  {
169  h_kc("k,c") =
170  ints.Fia("k,c") +
171  (2.0 * ints.Gijab("k,l,c,d") - ints.Gijab("k,l,d,c")) * t1("d,l");
172  r1("a,i") += h_kc("k,c") * (2.0 * t2("c,a,k,i") - t2("c,a,i,k") +
173  t1("c,i") * t1("a,k"));
174  }
175 
176  r1("a,i") +=
177  (2.0 * ints.Gijab("k,i,c,a") - ints.Giajb("k,a,i,c")) * t1("c,k");
178 
179  r1("a,i") += X_ai_tau("K,c,i") * ints.Xab("K,a,c");
180 
181  r1("a,i") -= X_ai_tau("K,a,l") * ints.Xij("K,l,i");
182  }
183  return r1;
184 }

198 template <typename Array>
199 Array compute_cs_ccsd_r2(const Array& t1, const Array& t2, const Array& tau,
200  const cc::Integrals<Array>& ints,
201  const Array& u = Array()) {
202  Array r2;
203 
204  // compute residual r2(n) (at convergence r2 = 0)
205  // permutation part
206  {
207  r2("a,b,i,j") =
208  (ints.Giabc("i,c,a,b") - ints.Giajb("k,b,i,c") * t1("a,k")) * t1("c,j");
209 
210  r2("a,b,i,j") -=
211  (ints.Gijka("j,i,k,a") + ints.Gijab("i,k,a,c") * t1("c,j")) * t1("b,k");
212  }
213 
214  {
215  // compute g intermediate
216  Array g_ki, g_ac;
217 
218  Array g_ijka_bar;
219  g_ijka_bar("i,j,k,a") = 2.0 * ints.Gijka("i,j,k,a") - ints.Gijka("j,i,k,a");
220  g_ki = cc::compute_cs_ccsd_F_OO(ints.FIJ, ints.Fia, g_ijka_bar, t1);
221 
222  Array g_iabc_bar;
223  g_iabc_bar("k,a,d,b") = 2.0 * ints.Giabc("k,a,d,b") - ints.Giabc("k,a,b,d");
224  g_ac = cc::compute_cs_ccsd_F_VV(ints.FAB, ints.Fia, g_iabc_bar, t1);
225 
226  r2("a,b,i,j") += g_ac("a,c") * t2("c,b,i,j") - g_ki("k,i") * t2("a,b,k,j");
227  }
228 
229  {
230  Array j_akic;
231  Array k_kaic;
232  // compute j and k intermediate
233  {
234  Array T;
235 
236  T("d,b,i,l") = 0.5 * t2("d,b,i,l") + t1("d,i") * t1("b,l");
237 
238  j_akic("a,k,i,c") =
239  ints.Gijab("i,k,a,c") - ints.Gijka("l,k,i,c") * t1("a,l") +
240  ints.Giabc("k,a,c,d") * t1("d,i") -
241  ints.Gijab("k,l,c,d") * T("d,a,i,l") +
242  0.5 * (2.0 * ints.Gijab("k,l,c,d") - ints.Gijab("k,l,d,c")) *
243  t2("a,d,i,l");
244 
245  k_kaic("k,a,i,c") = ints.Giajb("k,a,i,c") -
246  ints.Gijka("k,l,i,c") * t1("a,l") +
247  ints.Giabc("k,a,d,c") * t1("d,i") -
248  ints.Gijab("k,l,d,c") * T("d,a,i,l");
249  }
250 
251  r2("a,b,i,j") += 0.5 * (2.0 * j_akic("a,k,i,c") - k_kaic("k,a,i,c")) *
252  (2.0 * t2("c,b,k,j") - t2("b,c,k,j"));
253 
254  Array tmp;
255  tmp("a,b,i,j") = k_kaic("k,a,i,c") * t2("c,b,j,k");
256  r2("a,b,i,j") -= 0.5 * tmp("a,b,i,j") + tmp("b,a,i,j");
257  }
258 
259  // perform the permutation
260  r2("a,b,i,j") = r2("a,b,i,j") + r2("b,a,j,i");
261 
262  r2("a,b,i,j") += ints.Gijab("i,j,a,b");
263 
264  {
265  // compute a intermediate
266  Array a_klij =
267  cc::compute_cs_ccsd_W_oooo(ints.Gijkl, ints.Gijka, ints.Gijab, tau, t1);
268 
269  r2("a,b,i,j") += a_klij("k,l,i,j") * tau("a,b,k,l");
270  }
271 
272  {
273  Array b_abij;
274  if (!u.is_initialized()) {
275  b_abij("a,b,i,j") = tau("c,d,i,j") * ints.Gabcd("a,b,c,d");
276  Array tmp;
277  tmp("k,a,i,j") = ints.Giabc("k,a,c,d") * tau("c,d,i,j");
278  b_abij("a,b,i,j") -= tmp("k,a,j,i") * t1("b,k");
279 
280  b_abij("a,b,i,j") -= tmp("k,b,i,j") * t1("a,k");
281  } else {
282  b_abij("a,b,i,j") =
283  (u("p,r,i,j") * ints.Ca("r,b") -
284  ints.Ci("r,k") * t1("b,k") * u("p,r,i,j")) *
285  ints.Ca("p,a") -
286  u("p,r,i,j") * ints.Ci("p,k") * ints.Ca("r,b") * t1("a,k");
287  }
288 
289  r2("a,b,i,j") += b_abij("a,b,i,j");
290  }
291 
292  return r2;
293 }

376  cc::compute_cs_ccsd_W_oooo(ints.Gijkl, ints.Gijka, ints.Gijab, tau, t1);
377 
378  r2("a,b,i,j") += a_klij("k,l,i,j") * tau("a,b,k,l");

382  // compute b intermediate

529 template <typename Array>
530 Array compute_cs_mp1_r2_df(const Array& t2, const cc::Integrals<Array>& ints) {
531  // compute residual r2(n) (at convergence r2 = 0)
532 
533  Array r2;
534 
535  r2("a,b,i,j") += ints.Fab("a,c") * t2("c,b,i,j") - t2("a,b,i,k") * ints.Fij("k,j");
536  // 1<->2 symmetrize
537  r2("a,b,i,j") = r2("a,b,i,j") + r2("b,a,j,i");
538  r2("a,b,i,j") += ints.Gijab("i,j,a,b");
