201  void evaluate(Energy *energy) override {
202  auto target_precision = energy->target_precision(0);
203  // compute only if never computed, or requested with higher precision than
204  // before
205  if (!this->computed() || computed_precision_ > target_precision) {
206  // compute reference to higher precision than required of correlation
207  // energy
208  auto target_ref_precision = target_precision / 100.;
209  auto ref_energy =
210  std::make_shared<Energy>(ref_wfn_, target_ref_precision);
211  ::mpqc::evaluate(*ref_energy, ref_wfn_);
212 
213  auto &world = this->wfn_world()->world();
214  auto time0 = mpqc::fenced_now(world);
215 
216  this->init_sdref(ref_wfn_, target_ref_precision);
217 
218  f_pq_diagonal_ = make_diagonal_fpq(this->lcao_factory(),
219  this->ao_factory(), this->is_df());
220 
221  // set up the solver
222  if (solver_str_ == "jacobi_diis") {
223  const auto n_occ = this->trange1_engine()->get_occ();
224  const auto n_frozen = this->trange1_engine()->get_nfrozen();
225  const auto n_act_occ = n_occ - n_frozen;
226  const auto n_uocc = f_pq_diagonal_->rows() - n_occ;
227  solver_ = std::make_shared<cc::JacobiDIISSolver<TArray>>(
228  kv_, f_pq_diagonal_->segment(n_frozen, n_act_occ),
229  f_pq_diagonal_->segment(n_occ, n_uocc));
230  } else if (solver_str_ == "pno") {
231  kv_.assign("use_df", df_);
232  solver_ = std::make_shared<cc::PNOSolver<
233  TArray, typename LCAOFactory<Tile, Policy>::DirectTArray>>(
234  kv_, this->lcao_factory());
235  kv_.erase("use_df");
236  }
237  else
238  throw ProgrammingError("unknown solver string", __FILE__, __LINE__);
239 
240  // set the precision
241  target_precision_ = energy->target_precision(0);
242 
243  TArray t1;
244  TArray t2;
245 
246  if (method_ == "direct" || method_ == "direct_df") {
247  direct_ao_array_ = this->ao_factory().compute_direct(L"(Î¼ Î½| G|Îº Î»)");
248  }
249 
250  ccsd_corr_energy_ = compute_ccsd(t1, t2);
251 
252  T1_ = t1;
253  T2_ = t2;
254 
255  // delete current solver
256  solver_.reset();
257 
258  this->computed_ = true;
259  this->set_value(energy, ref_energy->energy() + ccsd_corr_energy_);
260 
261  auto time1 = mpqc::fenced_now(world);
262  auto duration0 = mpqc::duration_in_s(time0, time1);
263  ExEnv::out0() << "CCSD Time in CCSD: " << duration0 << " S" << std::endl;
264  }

268  double compute_ccsd(TArray &t1, TArray &t2) {
269  auto &world = this->wfn_world()->world();
270  bool accurate_time = this->lcao_factory().accurate_time();
271 
272  if (method_ == "standard") {
273  mpqc::utility::print_par(world, "Use Conventional CCSD Compute \n");
274  } else if (method_ == "df") {
275  mpqc::utility::print_par(world, "Use DF CCSD Compute \n");
276  } else if (method_ == "direct") {
277  mpqc::utility::print_par(world, "Use Direct CCSD Compute \n");
278  }
279 
280  auto tmp_time0 = mpqc::now(world, accurate_time);
281  cc::Integrals<TArray> ints;
282  // fock matrix
283  ints.Fia = this->get_fock_ia();
284  ints.Fij = this->get_fock_ij();
285  ints.Fab = this->get_fock_ab();
286  // get all two electron integrals
287  ints.Gijab = this->get_ijab();
288  ints.Gijkl = this->get_ijkl();
289  ints.Giajb = this->get_iajb();
290  ints.Gijka = this->get_ijka();
291 
292  if (method_ == "standard" || (method_ == "df" && !reduced_abcd_memory_)) {
293  if(!cp_ccsd_) {
294  ints.Gabcd = this->get_abcd();
295  }
296  ints.Giabc = this->get_iabc();
297  } else if (method_ == "direct") {
298  ints.Giabc = this->get_iabc();
299  }
300 
301  if (df_) {
302  ints.Xai = this->get_Xai();
303  ints.Xij = this->get_Xij();
304  ints.Xab = this->get_Xab();
305  }
306 
307 #ifdef MADNESS_LINALG_USE_LAPACKE
308  if(cp_ccsd_){
309  this->get_factors(ints.Xab_factors);
310  }
311 #endif // MADNESS_LINALG_USE_LAPACKE
312 
313  if (method_ == "direct" || method_ == "direct_df") {
314  ints.Ci = this->lcao_factory()
315  .orbital_registry()
316  .retrieve(OrbitalIndex(L"i"))
317  .coefs();
318  ints.Ca = this->lcao_factory()
319  .orbital_registry()
320  .retrieve(OrbitalIndex(L"a"))
321  .coefs();
322  }
323 
324  this->lcao_factory().registry().purge_formula(L"(i Î½| G |Îº Î» )");
325  this->lcao_factory().registry().purge_formula(L"(a Î½| G |Îº Î» )");
326 
327  auto tmp_time1 = mpqc::now(world, accurate_time);
328  auto tmp_time = mpqc::duration_in_s(tmp_time0, tmp_time1);
329  mpqc::utility::print_par(world, "Integral Prepare Time: ", tmp_time, "\n");
330 
331  TArray f_ai;
332  f_ai("a,i") = ints.Fia("i,a");
333  // initial guess = 0
334  t1 = TArray(f_ai.world(), f_ai.trange(), f_ai.shape(), f_ai.pmap());
335  t1.fill(0.0);
336  TArray g_abij;
337  g_abij("a,b,i,j") = ints.Gijab("i,j,a,b");
338  t2 = TArray(g_abij.world(), g_abij.trange(), g_abij.shape(), g_abij.pmap());
339  t2.fill(0.0);
340 
341  TArray tau;
342  tau("a,b,i,j") = t2("a,b,i,j") + t1("a,i") * t1("b,j");
343 
344  double E0;
345  double E1 = 0.0;
346  double dE;
347 
348  // optimize t1 and t2
349  std::size_t iter = 0ul;
350  double error = 1.0;
351  TArray r1(f_ai);
352  TArray r2(g_abij);
353 
354  if (world.rank() == 0) {
355  std::cout << "Start Iteration" << std::endl;
356  std::cout << "Max Iteration: " << max_iter_ << std::endl;
357  std::cout << "Target precision: " << target_precision_ << std::endl;
358  std::cout << "AccurateTime: " << accurate_time << std::endl;
359  std::cout << "Verbose: " << verbose_ << std::endl;
360  std::cout << "Reduced ABCD Memory Approach: "
361  << (reduced_abcd_memory_ ? "Yes" : "No") << std::endl;
362  }
363 
364  // CCSD solver loop
365  mpqc::time_point time0;
366  double duration_u;
367  while (iter < max_iter_) {
368  TArray::wait_for_lazy_cleanup(world);
369 
370  // zero out singles if want CCD
371  // TArray r1_new(r1.world(), r1.trange(), r1.shape()); r1_new.fill(0.0);
372  // r1("a,i") = r1_new("a,i");
373  error = solver_->error(r1, r2);
374 
375  // recompute energy
376  E0 = E1;
377  E1 = 2.0 * TA::dot(f_ai("a,i") + r1("a,i"), t1("a,i")) +
378  TA::dot(g_abij("a,b,i,j") + r2("a,b,i,j"),
379  2 * tau("a,b,i,j") - tau("b,a,i,j"));
380  dE = std::abs(E0 - E1);
381 
382  if (iter == 0 || !solver_->is_converged(target_precision_, error, dE)) {
383  tmp_time0 = mpqc::now(world, accurate_time);
384 
385  assert(solver_);
386  solver_->update(t1, t2, r1, r2, E1);
387 
388  // recompute tau
389  tau("a,b,i,j") = t2("a,b,i,j") + t1("a,i") * t1("b,j");
390  tmp_time1 = mpqc::now(world, accurate_time);
391  tmp_time = mpqc::duration_in_s(tmp_time0, tmp_time1);
392  if (verbose_) {
393  mpqc::utility::print_par(world, "Solver::update time: ", tmp_time,
394  "\n");
395  }
396 
397  // log the iteration
398  auto time1 = mpqc::fenced_now(world);
399  if (world.rank() == 0 && iter > 0) {
400  auto duration = mpqc::duration_in_s(time0, time1);
401  if(method_ == "direct" || method_ == "direct_df"){
402  detail::print_ccsd_direct(iter, dE, error, E1, duration_u, duration);
403  }
404  else{
405  detail::print_ccsd(iter, dE, error, E1, duration);
406  }
407  }
408  } else { // break out of the solver loop, if converged
409  // log the iteration
410  auto time1 = mpqc::fenced_now(world);
411  MPQC_ASSERT(iter > 0);
412  if (world.rank() == 0) {
413  auto duration = mpqc::duration_in_s(time0, time1);
414  detail::print_ccsd(iter, dE, error, E1, duration);
415  }
416  break;
417  }
418 
419  // start iteration timer
420  time0 = mpqc::fenced_now(world);
421 
422  TArray U;
423  auto tu0 = mpqc::now(world, accurate_time);
424  if (method_ == "direct" || method_ == "direct_df") {
425  U = compute_u2_u11(t2, t1);
426  }
427  auto tu1 = mpqc::now(world, accurate_time);
428  duration_u = mpqc::duration_in_s(tu0, tu1);
429 
430  auto t1_time0 = mpqc::now(world, accurate_time);
431 
432  if (method_ == "standard") {
433  r1 = cc::compute_cs_ccsd_r1(t1, t2, tau, ints);
434  } else if (method_ == "df") {
435  r1 = cc::compute_cs_ccsd_r1_df(t1, t2, tau, ints);
436  } else if (method_ == "direct" || method_ == "direct_df") {
437  r1 = cc::compute_cs_ccsd_r1(t1, t2, tau, ints, U);
438  }
439 
440  auto t1_time1 = mpqc::now(world, accurate_time);
441  auto t1_time = mpqc::duration_in_s(t1_time0, t1_time1);
442  if (verbose_) {
443  mpqc::utility::print_par(world, "t1 total time: ", t1_time, "\n");
444  }
445 
446  // intermediates for t2
447  // external index i j a b
448 
449  auto t2_time0 = mpqc::now(world, accurate_time);
450 
451  if (method_ == "standard") {
452  r2 = cc::compute_cs_ccsd_r2(t1, t2, tau, ints);
453  } else if (method_ == "df") {
454  r2 = cc::compute_cs_ccsd_r2_df(t1, t2, tau, ints);
455  } else if (method_ == "direct") {
456  r2 = cc::compute_cs_ccsd_r2(t1, t2, tau, ints, U);
457  } else {
458  r2 = cc::compute_cs_ccsd_r2_df(t1, t2, tau, ints, U);
459  }
460 
461  auto t2_time1 = mpqc::now(world, accurate_time);
462  auto t2_time = mpqc::duration_in_s(t2_time0, t2_time1);
463  if (verbose_) {
464  mpqc::utility::print_par(world, "t2 total time: ", t2_time, "\n");
465  }
466 
467  ++iter;
468  } // CCSD solver loop
469  if (iter >= max_iter_) {
470  utility::print_par(this->wfn_world()->world(),
471  "\n Warning!! Exceed Max Iteration! \n");
472  }
473  if (world.rank() == 0) {
474  std::cout << "CCSD Energy " << E1 << std::endl;
475  }
476  return E1;
477  }
