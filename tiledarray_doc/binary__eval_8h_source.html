127  virtual int internal_eval() {
128 
129  // Evaluate child tensors
130  left_.eval();
131  right_.eval();
132 
133  // Task function argument types
134  typedef typename madness::if_<std::is_const<typename op_type::first_argument_type>,
135  const typename left_type::value_type,
136  typename left_type::value_type>::type &
137  left_argument_type;
138  typedef typename madness::if_<std::is_const<typename op_type::second_argument_type>,
139  const typename right_type::value_type,
140  typename right_type::value_type>::type &
141  right_argument_type;
142  typedef ZeroTensor<typename left_type::value_type::value_type> zero_left_type;
143  typedef ZeroTensor<typename right_type::value_type::value_type> zero_right_type;
144 
145  size_type task_count = 0ul;
146 
147  // Construct local iterator
148  TA_ASSERT(left_.pmap() == right_.pmap());
149  std::shared_ptr<BinaryEvalImpl_> self = shared_from_this();
150  typename pmap_interface::const_iterator it = left_.pmap()->begin();
151  const typename pmap_interface::const_iterator end = left_.pmap()->end();
152 
153  if(left_.is_dense() && right_.is_dense() && TensorImpl_::is_dense()) {
154  // Evaluate tiles where both arguments and the result are dense
155  for(; it != end; ++it) {
156  // Get tile indices
157  const size_type source_index = *it;
158  const size_type target_index = DistEvalImpl_::perm_index_to_target(source_index);
159 
160  // Schedule tile evaluation task
161  TensorImpl_::get_world().taskq.add(self,
162  & BinaryEvalImpl_::template eval_tile<left_argument_type, right_argument_type>,
163  target_index, left_.get(source_index), right_.get(source_index));
164 
165  ++task_count;
166  }
167  } else {
168  // Evaluate tiles where the result or one of the arguments is sparse
169  for(; it != end; ++it) {
170  // Get tile indices
171  const size_type index = *it;
172  const size_type target_index = DistEvalImpl_::perm_index_to_target(index);
173 
174  if(! TensorImpl_::is_zero(target_index)) {
175  // Schedule tile evaluation task
176  if(left_.is_zero(index)) {
177  TensorImpl_::get_world().taskq.add(self,
178  & BinaryEvalImpl_::template eval_tile<const zero_left_type, right_argument_type>,
179  target_index, zero_left_type(), right_.get(index));
180  } else if(right_.is_zero(index)) {
181  TensorImpl_::get_world().taskq.add(self,
182  & BinaryEvalImpl_::template eval_tile<left_argument_type, const zero_right_type>,
183  target_index, left_.get(index), zero_right_type());
184  } else {
185  TensorImpl_::get_world().taskq.add(self,
186  & BinaryEvalImpl_::template eval_tile<left_argument_type, right_argument_type>,
187  target_index, left_.get(index), right_.get(index));
188  }
189 
190  ++task_count;
191  } else {
192  // Cleanup unused tiles
193  if(! left_.is_zero(index))
194  left_.get(index);
195  if(! right_.is_zero(index))
196  right_.get(index);
197  }
198  }
199  }
200 
201  // Wait for child tensors to be evaluated, and process tasks while waiting.
202  left_.wait();
203  right_.wait();
204 
205  return task_count;
206  }
