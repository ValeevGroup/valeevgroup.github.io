72  void compute(const numeric_type &e,
73  TA::DistArray<Tile, Policy> &guess) const {
74  const auto &eps_v = this->eps_v;
75  const auto &eps_o = this->eps_o;
76 
77  auto task = [&eps_v, &eps_o, e](Tile &result_tile) {
78  const auto &range = result_tile.range();
79  float norm = 0.0;
80  for (const auto &i : range) {
81  const auto result = result_tile[i] / (e + eps_o[i[0]] - eps_v[i[1]]);
82  result_tile[i] = result;
83  norm += result * result;
84  }
85  return std::sqrt(norm);
86  };
87 
88  TA::foreach_inplace(guess, task);
89  guess.world().gop.fence();
90  }
91  };

188 template <typename Tile, typename Policy>
189 void CIS<Tile, Policy>::evaluate(ExcitationEnergy *ex_energy) {
190  if (!this->computed()) {
191  auto &world = this->wfn_world()->world();
192  auto target_precision = ex_energy->target_precision(0);
193 
194  const auto &orbital_registry = this->ao_factory().orbital_registry();
195  // if required orbitals not initialized, then initialize
196  if (!orbital_registry.have(OrbitalIndex(L"i")) ||
197  !orbital_registry.have(OrbitalIndex(L"a"))) {
198  auto target_ref_precision = target_precision / 100.0;
199 
200  this->init_sdref(ref_wfn_, target_ref_precision);
201  }
202 
203  auto time0 = mpqc::fenced_now(world);
204 
205  ExEnv::out0() << indent << "\nCIS Excitation Energy \n";
206  auto n_roots = ex_energy->n_roots();
207  auto n_guess = ex_energy->n_guess();
208 
209  std::vector<numeric_type> result;
210 
211  if (ex_energy->singlets()) {
212  if (method_ == "standard") {
213  result = compute_cis(n_roots, n_guess, target_precision);
214  } else if (method_ == "df") {
215  result = compute_cis_df(n_roots, n_guess, target_precision);
216  }
217  }
218 
219  // TODO separate singlets and triplets energy
220  if (ex_energy->triplets()) {
221  decltype(result) triplet_result;
222  if (method_ == "standard") {
223  triplet_result = compute_cis(n_roots, n_guess, target_precision, true);
224  } else if (method_ == "df") {
225  triplet_result =
226  compute_cis_df(n_roots, n_guess, target_precision, true);
227  }
228  result.insert(result.end(), triplet_result.begin(), triplet_result.end());
229  }
230 
231  this->computed_ = true;
232  eigen_value_ = result;
233  this->set_value(ex_energy, result);
234 
235  auto time1 = mpqc::fenced_now(world);
236  ExEnv::out0() << "CIS Total Time: " << mpqc::duration_in_s(time0, time1)
237  << " S\n";
238  }

241 template <typename Tile, typename Policy>
242 std::vector<typename CIS<Tile, Policy>::numeric_type>
243 CIS<Tile, Policy>::compute_cis(std::size_t n_roots, std::size_t n_guess,
244  double converge, bool triplets) {
245  ExEnv::out0() << "\n";
246  ExEnv::out0() << indent
247  << "CIS standard: " << (triplets ? "Triplets" : "Singlets")
248  << "\n";
249  ExEnv::out0() << "\n";
250 
251  auto &world = this->wfn_world()->world();
252  auto time0 = mpqc::fenced_now(world);
253 
254  auto &factory = this->lcao_factory();
255 
256  // compute required integrals
257  auto F_ab = factory.compute(L"<a|F|b>");
258  auto F_ij = factory.compute(L"<i|F|j>");
259  auto I_ab = factory.compute(L"<a|I|b>");
260  auto I_ij = factory.compute(L"<i|I|j>");
261  auto G_iajb = factory.compute(L"<i a|G|j b>");
262 
263  // initialize diagonal
264  if (eps_o_.size() == 0) {
265  eps_o_ = math::array_to_eigen(F_ij).diagonal();
266  }
267  if (eps_v_.size() == 0) {
268  eps_v_ = math::array_to_eigen(F_ab).diagonal();
269  }
270 
271  // get guess vector
272  auto guess = init_guess_vector(n_guess);
273 
274  // compute H
275  TA::DistArray<Tile, Policy> H;
276 
277  // singlets
278  if (!triplets) {
279  auto G_ijab = factory.compute(L"<i j|G|a b>");
280  H("i,j,a,b") = I_ij("i,j") * F_ab("a,b") - F_ij("i,j") * I_ab("a,b") +
281  2.0 * G_ijab("i,j,a,b") - G_iajb("i,a,j,b");
282  }
283  // triplets
284  else {
285  H("i,j,a,b") = I_ij("i,j") * F_ab("a,b") - F_ij("i,j") * I_ab("a,b") -
286  G_iajb("i,a,j,b");
287  }
288 
289  auto time1 = mpqc::fenced_now(world);
290  auto time = mpqc::duration_in_s(time0, time1);
291 
292  ExEnv::out0() << indent << "Computed H matrix. Time: " << time << " S\n";
293 
294  // davidson object
295  DavidsonDiag<TA::DistArray<Tile, Policy>> dvd(n_roots, true, 2, 10,
296  10 * converge);
297 
298  auto pred = std::make_unique<Preconditioner>(eps_o_, eps_v_);
299 
300  auto oper = [&H](std::vector<TArray> &guess) {
301  const auto n_v = guess.size();
302  std::vector<TA::DistArray<Tile, Policy>> HB(n_v);
303  // product of H with guess vector
304  for (std::size_t i = 0; i < n_v; i++) {
305  // std::cout << guess[i] << std::endl;
306  HB[i]("i,a") = H("i,j,a,b") * guess[i]("j,b");
307  }
308  return HB;
309  };
310  // solve the lowest n_roots eigenvalues
311  auto eig = dvd.solve(guess, oper, pred.get(), converge, max_iter_);
312 
313  ExEnv::out0() << "\n";
314  util::print_excitation_energy(eig, triplets);
315  // get the latest eigen vector
316  auto &eigen_vector = dvd.eigen_vector();
317 
318  for (std::size_t i = 0; i < n_roots; i++) {
319  auto dominants = array_abs_max_n_index(eigen_vector[i], 5);
320 
321  ExEnv::out0() << "Dominant determinants of excited wave function " << i + 1
322  << "\n";
323  util::print_cis_dominant_elements(dominants);
324  ExEnv::out0() << "\n";
325  }
326 
327  eigen_vector_.insert(eigen_vector_.end(), eigen_vector.begin(),
328  eigen_vector.end());
329 
330  return std::vector<numeric_type>(eig.data(), eig.data() + eig.size());
331 }

333 template <typename Tile, typename Policy>
334 std::vector<typename CIS<Tile, Policy>::numeric_type>
335 CIS<Tile, Policy>::compute_cis_df(std::size_t n_roots, std::size_t n_guess,
336  double converge, bool triplets) {
337  ExEnv::out0() << "\n";
338  ExEnv::out0() << indent << "CIS Density-fitting: "
339  << (triplets ? "Triplets" : "Singlets") << "\n";
340  ExEnv::out0() << "\n";
341 
342  auto &factory = this->lcao_factory();
343 
344  // compute required integrals
345  auto F_ab = factory.compute(L"<a|F|b>[df]");
346  auto F_ij = factory.compute(L"<i|F|j>[df]");
347  auto I_ab = factory.compute(L"<a|I|b>");
348  auto I_ij = factory.compute(L"<i|I|j>");
349  auto X_ab = factory.compute(L"(Îš|G|a b)[inv_sqr]");
350  auto X_ij = factory.compute(L"(Îš|G|i j)[inv_sqr]");
351  auto X_ia = factory.compute(L"(Îš|G|i a)[inv_sqr]");
352 
353  // initialize diagonal
354  if (eps_o_.size() == 0) {
355  eps_o_ = math::array_to_eigen(F_ij).diagonal();
356  }
357  if (eps_v_.size() == 0) {
358  eps_v_ = math::array_to_eigen(F_ab).diagonal();
359  }
360 
361  // get guess vector
362  auto guess = init_guess_vector(n_guess);
363 
364  // davidson object
365  DavidsonDiag<TA::DistArray<Tile, Policy>> dvd(n_roots, true, 2, 10,
366  10 * converge);
367 
368  auto pred = std::make_unique<Preconditioner>(eps_o_, eps_v_);
369 
370  auto oper = [&F_ab, &F_ij, &I_ab, &I_ij, &X_ab, &X_ij, &X_ia,
371  triplets](std::vector<TArray> &guess) {
372  const auto n_v = guess.size();
373  std::vector<TA::DistArray<Tile, Policy>> HB(n_v);
374  for (std::size_t i = 0; i < n_v; i++) {
375  // std::cout << guess[i] << std::endl;
376  const auto &vec = guess[i];
377  // singlets
378  if (!triplets) {
379  HB[i]("j,b") = vec("i,a") * I_ij("i,j") * F_ab("a,b") -
380  F_ij("i,j") * vec("i,a") * I_ab("a,b") +
381  2.0 * vec("i,a") * X_ia("x,i,a") * X_ia("x,j,b") -
382  X_ab("x,a,b") * vec("i,a") * X_ij("x,i,j");
383  }
384  // triplets
385  else {
386  HB[i]("j,b") = vec("i,a") * I_ij("i,j") * F_ab("a,b") -
387  F_ij("i,j") * vec("i,a") * I_ab("a,b") +
388  -X_ab("x,a,b") * vec("i,a") * X_ij("x,i,j");
389  }
390  }
391  return HB;
392  };
393  // solve the lowest n_roots eigenvalues
394  auto eig = dvd.solve(guess, oper, pred.get(), converge, max_iter_);
395 
396  ExEnv::out0() << "\n";
397  util::print_excitation_energy(eig, triplets);
398  // get the latest eigen vector
399  auto &eigen_vector = dvd.eigen_vector();
400 
401  for (std::size_t i = 0; i < n_roots; i++) {
402  auto dominants = array_abs_max_n_index(eigen_vector[i], 5);
403 
404  ExEnv::out0() << "Dominant determinants of excited wave function " << i + 1
405  << "\n";
406  util::print_cis_dominant_elements(dominants);
407  ExEnv::out0() << "\n";
408  }
409 
410  eigen_vector_.insert(eigen_vector_.end(), eigen_vector.begin(),
411  eigen_vector.end());
412 
413  return std::vector<numeric_type>(eig.data(), eig.data() + eig.size());
414 }

416 template <typename Tile, typename Policy>
417 std::vector<typename CIS<Tile, Policy>::TArray>
418 CIS<Tile, Policy>::init_guess_vector(std::size_t n_roots) {
419  std::vector<TA::DistArray<Tile, Policy>> guess_vector(n_roots);
420 
421  auto &factory = this->lcao_factory();
422  auto& world = factory.world();
423 
424  std::size_t n_m = factory.orbital_registry().retrieve("m").rank();
425  const auto &orbital_i = factory.orbital_registry().retrieve("i");
426  const auto &orbital_a = factory.orbital_registry().retrieve("a");
427  std::size_t n_i = orbital_i.rank();
428  std::size_t n_a = orbital_a.rank();
429  std::size_t n_frozen = n_m - n_i;
430 
431  auto canonical_orbs_provider = std::dynamic_pointer_cast<
432  typename CanonicalOrbitalSpace<TArray>::Provider>(ref_wfn_);
433  bool has_canonical_orbs =
434  canonical_orbs_provider && canonical_orbs_provider->can_evaluate();
435 
436  // these two will be computed if is not canonical
437  TArray local_to_canonical_occ;
438  TArray local_to_canonical_unocc;
439 
440  EigenVector<numeric_type> energy_occ;
441  EigenVector<numeric_type> energy_unocc;
442  if (has_canonical_orbs) {
443  energy_occ = eps_o_;
444  energy_unocc = eps_v_;
445  } else {
446  auto f_pq =
447  df_ ? factory.compute(L"<p|F|q>[df]") : factory.compute(L"<p|F|q>");
448 
449  auto f_eig = math::array_to_eigen(f_pq);
450  Eigen::VectorXd evals;
451  RowMatrixXd C;
452  if (world.rank() == 0) {
453  Eigen::SelfAdjointEigenSolver<decltype(f_eig)> es(f_eig);
454  evals = es.eigenvalues();
455  C = es.eigenvectors();
456  }
457  world.gop.broadcast_serializable(evals, 0);
458  world.gop.broadcast_serializable(C, 0);
459 
460  energy_occ = evals.segment(n_frozen, n_i);
461  energy_unocc = evals.segment(n_m, n_a);
462 
463  decltype(C) C_i = C.block(n_frozen, n_frozen, n_i, n_i);
464  decltype(C) C_a = C.block(n_m, n_m, n_a, n_a);
465 
466  auto &tr_i = orbital_i.trange();
467  auto &tr_a = orbital_a.trange();
468  auto &world = f_pq.world();
469 
470  local_to_canonical_occ =
471  math::eigen_to_array<Tile, Policy>(world, C_i, tr_i, tr_i);
472  local_to_canonical_unocc =
473  math::eigen_to_array<Tile, Policy>(world, C_a, tr_a, tr_a);
474  }
475  // use f_ia for shape
476  auto f_ia =
477  df_ ? factory.compute(L"<i|F|a>[df]") : factory.compute(L"<i|F|a>");
478  auto range = f_ia.trange();
479 
480  // find the minimum n_roots of F_ii - F_aa
481  std::vector<detail::IndexSort<numeric_type>> index(n_i * n_a);
482  {
483  // TODO might want to parallel this section
484  for (std::size_t i = 0; i < n_i; ++i) {
485  for (std::size_t a = 0; a < n_a; ++a) {
486  index[i * n_a + a] = std::move(detail::IndexSort<numeric_type>(
487  i, a, energy_unocc[a] - energy_occ[i]));
488  }
489  }
490 
491  std::sort(index.begin(), index.end());
492 
493  // for (std::size_t i = 0; i < n_roots; i++) {
494  // std::cout << "index: " << index[i].index.first << " "
495  // << index[i].index.second << " " << index[i].value << "\n";
496  // }
497  }
498 
499  for (std::size_t i = 0; i < n_roots; i++) {
500  TA::DistArray<Tile, Policy> guess(f_ia.world(), range, f_ia.shape());
501 
502  guess.fill(numeric_type(0.0));
503 
504  std::size_t idx_i = index[i].index.first;
505  std::size_t idx_a = index[i].index.second;
506  // std::size_t idx_i = n_i - 1;
507  // std::size_t idx_a = i;
508  // std::cout << "element index" << std::endl;
509  // std::cout << idx_i << std::endl;
510  // std::cout << idx_a << std::endl;
511 
512  // fill in with 1.0
513  std::vector<std::size_t> element_idx{{idx_i, idx_a}};
514 
515  auto tile_idx = range.element_to_tile(element_idx);
516 
517  if (guess.is_local(tile_idx)) {
518  // get tile
519  auto tile = guess.find(tile_idx).get();
520  // set value
521  tile[element_idx] = numeric_type(1.0);
522  }
523 
524  guess.truncate();
525  // convert to local basis
526  if (!has_canonical_orbs) {
527  guess("i,a") = guess("i1, a1") * local_to_canonical_occ("i,i1") *
528  local_to_canonical_unocc("a,a1");
529  }
530 
531  guess_vector[i] = guess;
532  }
533 
534  return guess_vector;
535 };
