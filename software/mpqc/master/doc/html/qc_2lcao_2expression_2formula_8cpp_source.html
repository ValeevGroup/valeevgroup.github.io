22 Formula::Formula(std::wstring string) {
23  if (string.empty()) {
24  throw std::runtime_error(utility::to_string(string) + " Empty Formula! \n");
25  }
26 
27  // detect the brackets <> or ()
28  auto left_mark = [](const wchar_t letter) {
29  return letter == L'<' || letter == L'(';
30  };
31 
32  // find formula between < > or ( )
33  auto bra_symbol = std::find_if(string.cbegin(), string.cend(), left_mark);
34 
35  auto ket_symbol = string.cbegin();
36 
37  if (*bra_symbol == L'<') {
38  auto pos = string.find_last_of(L'>');
39  ket_symbol += pos;
40  } else {
41  auto pos = string.find_last_of(L')');
42  ket_symbol += pos;
43  }
44 
45  TA_ASSERT(bra_symbol < ket_symbol);
46 
47  if (bra_symbol == string.cend() || ket_symbol == string.cend()) {
48  throw std::runtime_error(
49  utility::to_string(string) +
50  " Formula should start with < or ( and end with > or ). \n");
51  }
52 
53  TA_ASSERT((*bra_symbol == L'<' && *ket_symbol == L'>') ||
54  (*bra_symbol == L'(' && *ket_symbol == L')'));
55 
56  if (*bra_symbol == L'<') {
57  notation_ = Notation::Physical;
58  } else {
59  notation_ = Notation::Chemical;
60  }
61 
62  std::wstring formula(bra_symbol + 1, ket_symbol);
63 
64  // find option between [ and ]
65  auto option_left = std::find(string.cbegin(), string.cend(), L'[');
66  auto option_right = std::find(string.cbegin(), string.cend(), L']');
67 
68  std::wstring opt;
69 
70  if ((option_left != string.cend()) && (option_right != string.cend())) {
71  TA_ASSERT(option_left < option_right);
72  opt = std::wstring(option_left + 1, option_right);
73  }
74 
75  // parse options (separators = {",", " "})
76  if (!opt.empty()) {
77  std::vector<std::wstring> split_option;
78  boost::split(split_option, opt, boost::is_any_of(L", "),
79  boost::token_compress_on);
80 
81  std::vector<Formula::Option> result;
82  for (const auto& op : split_option) {
83  // check if this is one of Options
84  auto opt_iter = std::find_if(
85  begin(option_to_string), end(option_to_string),
86  [=](const std::pair<Formula::Option, std::wstring> item) {
87  return item.second == op;
88  });
89  const auto has_opt = (opt_iter != option_to_string.end());
90 
91  // or math::PetiteList::Symmetry
92  auto op_str = utility::to_string(op);
93  auto symm_iter = std::find_if(
94  begin(math::PetiteList::symmetry_to_string),
95  end(math::PetiteList::symmetry_to_string),
96  [=](const std::pair<math::PetiteList::Symmetry, std::string> item) {
97  return item.second == op_str;
98  });
99  const auto has_symm =
100  (symm_iter != math::PetiteList::symmetry_to_string.end());
101 
102  if (!has_opt && !has_symm)
103  throw ProgrammingError(
104  (utility::to_string(op) + " Invalid Option! \n").c_str(), __FILE__,
105  __LINE__);
106  if (has_opt) result.push_back(opt_iter->first);
107  if (has_symm) symm_ = symm_iter->first;
108  }
109  std::sort(result.begin(), result.end());
110  options_ = result;
111  }
112 
113  // split the formula by |
114  std::vector<std::wstring> split_formula;
115  boost::split(split_formula, formula, boost::is_any_of(L"|"),
116  boost::token_compress_on);
117 
118  if (split_formula.size() == 2) {
119  std::wstring left_formula = std::move(split_formula[0]);
120  std::wstring right_formula = std::move(split_formula[1]);
121 
122  oper_ = Operator(L" ");
123  bra_indices_ = check_orbital_index(left_formula);
124  ket_indices_ = check_orbital_index(right_formula);
125 
126  } else if (split_formula.size() == 3) {
127  std::wstring left_formula = std::move(split_formula[0]);
128  std::wstring operation = std::move(split_formula[1]);
129  std::wstring right_formula = std::move(split_formula[2]);
130 
131  oper_ = Operator(operation);
132  bra_indices_ = check_orbital_index(left_formula);
133  ket_indices_ = check_orbital_index(right_formula);
134 
135  } else {
136  throw std::runtime_error(utility::to_string(string) +
137  " Formula in wrong length. \n");
138  }
139 
140  // error detecting
141 
142  // one body operation
143  if (oper_.is_onebody() && (bra_indices_.size() != 1) &&
144  (ket_indices_.size() != 1)) {
145  throw std::runtime_error(utility::to_string(string) +
146  "One body Operator with Wrong Index Size! \n");
147  }
148 
149  // more than three index
150  if ((bra_indices_.size() >= 3) || (ket_indices_.size() >= 3)) {
151  throw std::runtime_error(utility::to_string(string) +
152  " Wrong Number of Index! \n");
153  }
154 }

164  boost::token_compress_on);
165 
166  if (split_index.empty()) {

180 bool Formula::operator<(const Formula& other) const {
181  if (oper() != other.oper()) {
182  return oper() < other.oper();
183  } else if (options_ != other.options_) {
184  return options_ < other.options_;
185  } else if (symm_ != other.symm_) {
186  return symm_ < other.symm_;
187  } else if ((this->rank() != 2) && (notation_ != other.notation())) {
188  return notation_ < other.notation();
189  } else if (bra_indices() != other.bra_indices()) {
190  return bra_indices() < other.bra_indices();
191  } else {
192  return ket_indices() < other.ket_indices();
193  }
194 }

209 std::wstring Formula::string() const {
210  std::wstring left, right, result;
211  for (const auto& index : bra_indices_) {
212  left += index.name() + L" ";
213  }
214 
215  for (const auto& index : ket_indices_) {
216  right += L" " + index.name();
217  }
218 
219  // add operation
220  auto oper_str = oper_.oper_string();
221  if (!oper_str.empty()) {
222  result = left + L"|" + oper_.oper_string() + L"|" + right;
223  } else {
224  result = left + L"|" + right;
225  }
226 
227  if (notation_ == Notation::Chemical) {
228  result = L"( " + result + L" )";
229  } else {
230  result = L"< " + result + L" >";
231  }
232 
233  // append options
234  auto option_string = [=]() {
235  std::wstring result;
236  const auto has_options = !options_.empty();
237  const auto has_symm = symm_ != math::PetiteList::Symmetry::e;
238  if (!has_options && !has_symm) {
239  return result;
240  }
241  result = L"[";
242  for (const auto& option : options_) {
243  result += option_to_string.find(option)->second + L",";
244  }
245  // include symmetry, if not trivial
246  if (has_symm) result += utility::to_wstring(to_string(symm_)) + L",";
247  result.back() = L']';
248  return result;
249  };
250 
251  result = result + option_string();
252 
253  return result;
254 }
