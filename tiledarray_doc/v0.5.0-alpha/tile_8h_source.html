210  template <typename Archive,
211  typename std::enable_if<madness::archive::is_output_archive<Archive>::value>::type* = nullptr>
212  void serialize(Archive &ar) const {
213  // Serialize data for empty tile check
214  bool empty = !static_cast<bool>(pimpl_);
215  ar & empty;
216  if (!empty) {
217  // Serialize tile data
218  ar & *pimpl_;
219  }
220  }
221 
222  template <typename Archive,
223  typename std::enable_if<madness::archive::is_input_archive<Archive>::value>::type* = nullptr>
224  void serialize(Archive &ar) {
225  // Check for empty tile
226  bool empty = false;
227  ar & empty;
228 
229  if (!empty) {
230  // Deserialize tile data
231  tensor_type tensor;
232  ar & tensor;
233 
234  // construct a new pimpl
235  pimpl_ = std::make_shared<T>(std::move(tensor));
236  } else {
237  // Set pimpl to an empty tile
238  pimpl_.reset();
239  }
240  }
