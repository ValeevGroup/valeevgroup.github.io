90 TA::Tensor<float> SchwarzScreen::norm_estimate(
91  madness::World &world, std::vector<gaussian::Basis> const &bs_array,
92  TA::Pmap const &pmap, bool replicate) const {
93  const auto ndims = bs_array.size();
94  auto trange = gaussian::detail::create_trange(bs_array);
95  auto norms = TA::Tensor<float>(trange.tiles_range(), 0.0);
96 
97  if (ndims == 3) {
98  auto const &Ta = Qbra_->Qtile();
99  auto const &Tbc = Qket_->Qtile();
100  auto ord = 0ul;
101  for (auto a = 0l; a < Ta.size(); ++a) {
102  const float a_val = Ta(a);
103  for (auto b = 0l; b < Tbc.rows(); ++b) {
104  for (auto c = 0l; c < Tbc.cols(); ++c, ++ord) {
105  if (pmap.is_local(ord)){
106  norms[ord] = std::sqrt(a_val * Tbc(b, c));
107  }
108  }
109  }
110  }
111  } else if (ndims == 4) {
112  auto const &Tab = Qbra_->Qtile();
113  auto const &Tcd = Qket_->Qtile();
114  auto ord = 0ul;
115  for (auto a = 0l; a < Tab.rows(); ++a) {
116  for (auto b = 0l; b < Tab.cols(); ++b) {
117  const float ab = Tab(a, b);
118  for (auto c = 0l; c < Tcd.rows(); ++c) {
119  for (auto d = 0l; d < Tcd.cols(); ++d, ++ord) {
120  if (pmap.is_local(ord)) {
121  norms[ord] = std::sqrt(ab * Tcd(c, d));
122  }
123  }
124  }
125  }
126  }
127  } else {
128  norms = Screener::norm_estimate(world, bs_array, pmap);
129  }
130  world.gop.fence();
131 
132  // If we want to replicate and the size of the tensor is larger than max_int
133  // then we will have to do it using multiple sums. This is necessary because
134  // MPI_ISend can only send an integer (int) number of things in a single
135  // message.
136  if (replicate) { // construct the sum
137  // First get the size in a 64 bit int, if that overflows then it probably
138  // wasn't going to fit on 1 node anyways (2017, maybe one day I'll be wrong)
139  int64_t size = norms.size();
140 
141  const int64_t int_max = std::numeric_limits<int>::max();
142  if (size < int_max) { // If size fits into an int then life is easy
143  world.gop.sum(norms.data(), size);
144  } else {
145  // Blah, testing on NewRiver gave failures when trying to write in chunks
146  // of both int_max and int_max/2. For now I'll be conservative and just
147  // write in small chunks. Writing in chunks of int_max/10 is slow, but
148  // worked on NR.
149  const int64_t write_size = int_max / 10;
150  auto i = 0;
151  while (size > write_size) {
152  const auto next_ptr = norms.data() + i * write_size;
153  world.gop.sum(next_ptr, write_size);
154  size -= write_size;
155  ++i;
156  }
157 
158  // get the remaining elements
159  world.gop.sum(norms.data() + i * write_size, size);
160  }
161  }
162  world.gop.fence();
163 
164  return norms;
165 }
