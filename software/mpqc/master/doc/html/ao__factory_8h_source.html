35 template <typename Tile, typename Policy>
36 std::shared_ptr<AOFactory<Tile, Policy>> construct_ao_factory(
37  const KeyVal& kv) {
38  std::shared_ptr<AOFactory<Tile, Policy>> ao_factory;
39  if (kv.exists_class("wfn_world:ao_factory")) {
40  ao_factory = kv.class_ptr<AOFactory<Tile, Policy>>("wfn_world:ao_factory");
41  } else {
42  ao_factory = std::make_shared<AOFactory<Tile, Policy>>(kv);
43  std::shared_ptr<DescribedClass> ao_factory_base = ao_factory;
44  KeyVal& kv_nonconst = const_cast<KeyVal&>(kv);
45  kv_nonconst.keyval("wfn_world").assign("ao_factory", ao_factory_base);
46  }
47  return ao_factory;
48 };
