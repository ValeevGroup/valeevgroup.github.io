126 void
127 FormIO::init_ostream(std::ostream &o)
128 {
129  if (!xalloc_inited_) {
130  xalloc_inited_ = 1;
131  nindent_ = std::ios::xalloc();
132  indent_size_ = std::ios::xalloc();
133  skip_indent_ = std::ios::xalloc();
134  verbose_ = std::ios::xalloc();
135  initialized_ = std::ios::xalloc();
136  }
137 
138  if (o.iword(initialized_)) return;
139 
140  o.iword(skip_indent_) = 0;
141  o.iword(indent_size_) = 0;
142  o.iword(nindent_) = 2;
143  o.iword(verbose_) = 0;
144  o.iword(initialized_) = 1;
145 }

156 std::ios&
157 FormIO::indent(std::ios&o)
158 {
159  if (!ready_) init();
160  long &skip = o.iword(skip_indent_);
161  if (skip) {
162  skip--;
163  return o;
164  }
165  if (debug_ && parallel_) {
166  char nn[24];
167  sprintf(nn,"node %5d:",me_);
168  for (size_t i=0; i < strlen(nn); i++) o.rdbuf()->sputc(nn[i]);
169  }
170  long n = o.iword(nindent_);
171  for (int i=0; i<n; i++) o.rdbuf()->sputc(' ');
172  return o;
173 }

175 std::ios&
176 FormIO::incindent(std::ios&o)
177 {
178  if (!ready_) init();
179  long &n = o.iword(nindent_);
180  long size = o.iword(indent_size_);

183  n += size;
184  return o;
185 }

187 std::ios&
188 FormIO::decindent(std::ios&o)
189 {
190  if (!ready_) init();
191  long &n = o.iword(nindent_);
192  long size = o.iword(indent_size_);
193  if (size == 0) size = 2;

306 template <>
307 mpqcprintf<char>::mpqcprintf(const char* fmt, ...) {
308  va_list args;
309 
310  va_start(args, fmt);
311 
312  str_[0] = '\0';
313 
314  // hopefully this won't overflow
315  if (fmt && fmt[0] != '\0') {
316  if (std::vsprintf(str_, fmt, args) > 1023) {
317  ExEnv::errn() << indent << "mpqcprintf overflow\n";
318  abort();
319  }
320  }
321 
322  va_end(args);
323 }
