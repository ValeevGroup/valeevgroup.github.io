56  template <typename Perm, typename Arg, typename Result>
57  inline void permute_array(const Perm& perm, const Arg& arg, Result& result) {
58  TA_ASSERT(size(result) == size(arg));
59  const unsigned int n = size(arg);
60  for(unsigned int i = 0u; i < n; ++i) {
61  const typename Perm::index_type pi = perm[i];
62  TA_ASSERT(i < size(arg));
63  TA_ASSERT(pi < size(result));
64  result[pi] = arg[i];
65  }
66  }

246  std::vector<std::vector<index_type> > cycles() const {
247 
248  std::vector<std::vector<index_type>> result;
249 
250  std::vector<bool> placed_in_cycle(p_.size(), false);
251 
252  // 1. for each i compute its orbit
253  // 2. if the orbit is longer than 1, sort and add to the list of cycles
254  for(index_type i=0; i!= p_.size(); ++i) {
255  if (not placed_in_cycle[i]) {
256  std::vector<index_type> cycle(1,i);
257  placed_in_cycle[i] = true;
258 
259  index_type next_i = p_[i];
260  while (next_i != i) {
261  cycle.push_back(next_i);
262  placed_in_cycle[next_i] = true;
263  next_i = p_[next_i];
264  }
265 
266  if (cycle.size() != 1) {
267  std::sort(cycle.begin(), cycle.end());
268  result.emplace_back(cycle);
269  }
270 
271  } // this i already in a cycle
272  } // loop over i
273 
274  return result;
275  }

335  Permutation pow(int n) const {
336  // Initialize the algorithm inputs
337  int power;
338  Permutation value;
339  if(n < 0) {
340  value = inv();
341  power = -n;
342  } else {
343  value = *this;
344  power = n;
345  }
346 
347  Permutation result = identity(p_.size());
348 
349  // Compute the power of value with the exponentiation by squaring.
350  while(power) {
351  if(power & 1)
352  result = result.mult(value);
353  value = value.mult(value);
354  power >>= 1;
355  }
356 
357  return result;
358  }
