182 template <typename T, typename P>
183 std::array<pno::Array<T, P>, 2> PaoPnoRMP2<T, P>::localized_orbitals() {
184  auto& lcao_factory = this->lcao_factory();
185  auto& ao_factory = this->ao_factory();
186  auto& world = lcao_factory.world();
187 
188  pno::Array<T, P> Ci =
189  lcao_factory.orbital_registry().retrieve(OrbitalIndex(L"i")).coefs();
190 
191  pno::Array<T, P> Ci_loc;
192  auto U = mpqc::scf::FosterBoysLocalization{}(Ci, r_xyz_);
193  Ci_loc("mu, i") = Ci("mu, k") * U("k, i");
194  Ci_loc.truncate();
195 
196  auto nclusters = Ci_loc.trange().tiles_range().extent()[1];
197  if (cluster_coeffs_) {
198  scf::clustered_coeffs(r_xyz_, Ci_loc, nclusters);
199  }
200 
201  // Truncate blocks of Ci_loc that have real (not scaled) norms < tcutc
202  if (tcutc_ > 0) {
203  TA::foreach_inplace(Ci_loc, [this](T& t) {
204  const auto norm = t.norm();
205  if (norm >= tcutc_) {
206  return norm;
207  } else {
208  return 0.0;
209  }
210  });
211  }
212 
213 
214  world.gop.fence();
215 
216  //
217  // Project Fao into our new occupied subspace
218  //
219  // Assume that we used [DF]
220  auto Fao = ao_factory.compute(L"(μ | F | ν)[df]");
221 
222  pno::Array<T, P> Fij;
223  Fij("i,j") = Ci_loc("mu, i") * Fao("mu, nu") * Ci_loc("nu, j");
224  Fij.truncate();
225  world.gop.fence();
226 
227  return {{Ci_loc, Fij}};
228 }

230 template <typename T, typename P>
231 std::array<pno::Array<T, P>, 3> PaoPnoRMP2<T, P>::pao_projector() {
232  auto& lcao_factory = this->lcao_factory();
233  auto& ao_factory = this->ao_factory();
234  auto& world = lcao_factory.world();
235 
236  pno::Array<T, P> Cv =
237  lcao_factory.orbital_registry().retrieve(OrbitalIndex(L"a")).coefs();
238 
239  // Assume DF reference
240  auto Fao = ao_factory.compute(L"(μ | F | ν)[df]");
241  auto Sao = ao_factory.compute(L"(μ | ν)");
242 
243  pno::Array<T, P> Pv, Fab, Sab;
244  Pv("a, nu") = Cv("mu, a") * Sao("mu, nu");
245  Pv("mu, nu") = Cv("mu, a") * Pv("a,nu");
246  Pv.truncate();
247 
248  Fab("a, b") = Pv("mu, a") * Fao("mu, nu") * Pv("nu, b");
249  Sab("a, b") = Pv("mu, a") * Sao("mu, nu") * Pv("nu, b");
250  Fab.truncate();
251  Sab.truncate();
252 
253  world.gop.fence();
254 
255  return {{Pv, Fab, Sab}};
256 }

260  auto target_precision = result->target_precision(0);
261 
262  if (!this->computed() || computed_precision_ > target_precision) {

266  auto ref_energy = std::make_shared<Energy>(ref_wfn_, target_ref_precision);
267  ::mpqc::evaluate(*ref_energy, ref_wfn_);
268 
269  this->init_sdref(ref_wfn_, target_ref_precision);
270 
271  double energy = 0.0;
272  std::string cluster_lmos = (cluster_coeffs_) ? "true" : "false";
273  ExEnv::out0() << "Initialized PaoPnoRMP2 with:\n";
274  ExEnv::out0() << indent << "Cluster LMOs: " << cluster_lmos << "\n";
275  ExEnv::out0() << indent << "TCUTC : " << tcutc_ << "\n";

278  ExEnv::out0() << indent << "TCUTSpao : " << tcutSpao_ << "\n";
279  ExEnv::out0() << indent << "TCUTFIJ : " << tcutFIJ_ << "\n";
280  std::string print_maps = (print_maps_) ? "true" : "false";
281  ExEnv::out0() << indent << "Print Maps: " << print_maps << "\n";
282  ExEnv::out0() << indent << indent << "TCUTIX : " << tcutIX_ << "\n";
283  ExEnv::out0() << indent << indent << "TCUTIA : " << tcutIA_ << "\n";
284  ExEnv::out0() << indent << indent << "TCUTIJ : " << tcutIJ_ << "\n";

311  auto& world = lcao_factory.world();

315  energy = this->compute_mp2(target_precision);
316  auto fast_mp21 = fenced_now(world);
317  auto fast_mp2_time = duration_in_s(fast_mp20, fast_mp21);
318  ExEnv::out0() << "Total time for fast PNO MP2: " << fast_mp2_time
319  << std::endl;
320 
321  mp2_corr_energy_ = energy;
322

325  computed_precision_ = target_precision;

329 template <typename T, typename P>
330 double PaoPnoRMP2<T, P>::compute_mp2(double target_precision) {
331  auto& ao_factory = gaussian::to_ao_factory(this->ao_factory());
332  auto& world = ao_factory.world();
333  const auto nproc = world.size(); // Number of nodes
334  const auto me = world.rank(); // Which node am I on?
335 
336  /*
337  * To start a little notation:
338  *
339  * lower case letters (r s t u ...) are for AO element indices
340  * lower case letters (i j k l ...) are for occupied element indices
341  * lower case letters (a b c d...) are for pao element indices
342  * lower case letters (w x y z) are for auxiliary basis element indices
343  *
344  * lower case letters (R S T U ...) are for AO block indices
345  * lower case letters (I J K L ...) are for occupied block indices
346  * lower case letters (A B C D ...) are for pao block indices
347  * lower case letters (W X Y Z) are for auxiliary basis block indices
348  *
349  */
350 
351  //
352  // Get the matrices that we will need from the AO factory
353  //
354  auto M = ao_factory.compute(L"(Κ | G | Λ)");
355  auto Sao = ao_factory.compute(L"(μ | ν)");
356 
357  //
358  // Compute Fock matrices and overlap
359  //
360 
361  // Computes Ci and Fij
362  auto ij_mats = localized_orbitals();
363  pno::Array<T, P>& Ci = ij_mats[0];
364  pno::Array<T, P>& Fij = ij_mats[1];
365 
366  // Computes P(pao), Fpao and Spao
367  auto ab_mats = pao_projector();
368  pno::Array<T, P>& Pv = ab_mats[0];
369  pno::Array<T, P>& Fpao = ab_mats[1];
370  pno::Array<T, P>& Spao = ab_mats[2];
371 
372  //
373  // Grab all of the sizes that we are going to be working with.
374  //
375  auto const& Ctrange = Ci.trange();
376  const auto nocc_blocks = Ctrange.tiles_range().extent_data()[1];
377 
378  /* Build initial sparse maps
379  *
380  * (I -> A) For the PAOs
381  * (I -> R) For the AO -> LMO transform
382  * (R -> S) For the μν ao products in (X| μ ν)
383  * (I -> X) For the local density fitting domains
384  * (I -> J) We will initialize this to the identity and then build it up with
385  * the OSV estimator
386  *
387  * WARNING
388  * Because of the estimator function we will assume that IX is blocked
389  * the same as IA
390  */
391 
392  auto map_time0 = mpqc::fenced_now(world);
393 
394  // These guys are initialized with population analysis
395  pno::SparseMap IX, IA;
396  std::tie(IX, IA) = pno::iX_and_ipao(Ci, Sao, tcutIX_, tcutIA_);
397 
398  if (print_maps_) {
399  if (world.rank() == 0) {
400  std::string world_size =
401  (world.size() == 1) ? ""
402  : "_" + std::to_string(world.size()) + "_nodes";
403  std::string file_name_IX = "IX" + world_size + "_" + unique_id_ + ".csv";
404  std::string file_name_IA = "IA" + world_size + "_" + unique_id_ + ".csv";
405 
406  write_csv(IX, file_name_IX);
407  write_csv(IA, file_name_IA);
408  }
409  }
410 
411  auto map_time1 = mpqc::fenced_now(world);
412  auto map_time = mpqc::duration_in_s(map_time0, map_time1);
413  ExEnv::out0() << indent << "Time for Initial maps: " << map_time << std::endl;
414 
415  // We will start with IJ as the identity and later fill it out using OSVs
416  pno::SparseMap IJ = pno::SparseMap::Identity(nocc_blocks, nocc_blocks);
417 
418  bool diagonalIJ = true;
419  auto Z = pno::Zia(ao_factory, Ci, Pv, IX, IA, IJ, fast_E_, fast_Zir_,
420  fast_Zia_, tcutRS_, tcutScreen_, diagonalIJ);
421 
422  auto z_storage = ::mpqc::detail::array_storage(Z);
423  ExEnv::out0() << indent << "Z(for osvs) Storage " << z_storage[1] << ", "
424  << 100.0 - 100.0 * double(z_storage[1]) / double(z_storage[0])
425  << "% sparsity" << std::endl;
426 
427  /*
428  * Compute the diagonal PNOs (OSVs)
429  *
430  * We are going to just distribute this over J blocks later to avoid
431  * communication and because there should roughly be an okay number of I
432  * blocks for each J. So we will compute all JJ PNOs here
433  *
434  */
435  auto osv_time0 = mpqc::fenced_now(world);
436  Fij.make_replicated();
437  Eigen::VectorXd Fii = array_ops::array_to_eigen(Fij).diagonal();
438  const auto J_tr1 = Z.trange().data()[1];
439  TA::TiledRange osv_trange({J_tr1});
440  pno::Array<pno::PnoPair, TA::DensePolicy> osvs(world, osv_trange);
441 
442  bool use_osvs_as_intermediate_basis = false;
443  pno::PnoBuilder osv_builder(Z, Fii, tcutSpao_, tcutosv_, average_densities_,
444  use_osvs_as_intermediate_basis, osvs);
445  pno::MatrixGetter<T, P> osv_getter(Spao, Fpao, M, Z);
446 
447  pno::IdxMap<3, madness::Future<T>> local_Ztiles;
448 
449  M.make_replicated();
450  Fpao.make_replicated();
451  Spao.make_replicated();
452 
453  for (auto J = me; J < nocc_blocks; J += nproc) {
454  pno::SparseMap JJA = IA.row(J);
455  pno::SparseMap JJX = IX.row(J);
456  auto ZMFS = osv_getter.get_ZMFS(JJX, JJA, J, J);
457  // double dummy_time = 0;
458  // int64_t dummy_elements_written = 0;
459  // bool serial_MZ_for_osvs = false;
460 
461  // auto ZMFS_old = pno::get_pao_mats_GFS(
462  // Z, M, Fpao, Spao, IA, IX, J, J, local_Ztiles, JJA, tcutSpao_,
463  // dummy_time, dummy_time, &dummy_time, &mut_, dummy_elements_written,
464  // serial_MZ_for_osvs);
465 
466  auto task = [&osv_builder, &osvs, J, JJA = std::move(JJA),
467  ZMFS = std::move(ZMFS)]() {
468  osvs.set(J, osv_builder(std::move(ZMFS), std::move(JJA), J, J));
469  };
470 
471  world.taskq.add(task);
472  }
473  auto osv_time1 = mpqc::fenced_now(world);
474  auto osv_time = mpqc::duration_in_s(osv_time0, osv_time1);
475  ExEnv::out0() << indent << "Time for OSVs: " << osv_time << std::endl;
476 
477  // USE OSVS TO COMPUTE THE IJ MAPS, this might actually be slow
478  auto ij_time0 = mpqc::fenced_now(world);
479  IJ = this->IJ_map_generator(Ci, Pv, osvs, Fii);
480  if (print_maps_) {
481  if (world.rank() == 0) {
482  std::string world_size =
483  (world.size() == 1) ? ""
484  : "_" + std::to_string(world.size()) + "_nodes";
485  std::string file_name_IJ = "IJ" + world_size + "_" + unique_id_ + ".csv";
486  write_csv(IJ, file_name_IJ);
487  }
488  }
489  auto nsig_ij_pairs = IJ.count();
490  auto ij_time1 = mpqc::fenced_now(world);
491  auto ij_time = mpqc::duration_in_s(ij_time0, ij_time1);
492  ExEnv::out0() << indent << "Time for IJmap: " << ij_time << std::endl;
493  ExEnv::out0() << indent << "Number of significant IJ pairs: " << nsig_ij_pairs
494  << ", " << 100.0 * double(nsig_ij_pairs) / IJ.size() << "%"
495  << std::endl;
496 
497  // Recompute Z to take into account all of our unions given our new IJ
498  // maps, unless we used exact E, Zir, and Zia in that case we don't need to
499  // recompute Z
500  if (fast_E_ || fast_Zir_ || fast_Zia_) {
501  ExEnv::out0()
502  << "Pair-extending 3-center integrals\n";
503  diagonalIJ = false;
504  Z = pno::Zia(ao_factory, Ci, Pv, IX, IA, IJ, fast_E_, fast_Zir_, fast_Zia_,
505  tcutRS_, tcutScreen_, diagonalIJ);
506  z_storage = ::mpqc::detail::array_storage(Z);
507  ExEnv::out0() << indent << "Z(for pnos) Storage " << z_storage[1] << ", "
508  << 100.0 - 100.0 * double(z_storage[1]) / double(z_storage[0])
509  << "% sparsity" << std::endl;
510  }
511 
512  /*
513  * Compute the all PNOs
514  *
515  * We are going to just distribute this over J blocks later to avoid
516  * communication and because there should roughly be an okay number of I per J
517  *
518  */
519  auto pno_time0 = mpqc::fenced_now(world);

692 void PaoPnoRMP2<T, P>::sort_orbitals(Vector3d const& O, pno::Array<T, P>& Ci) {

694 
695  std::vector<std::pair<int, Vector3d>> tagged_centers;

698  tagged_centers.emplace_back(std::make_pair(i++, std::move(c)));

701  std::sort(tagged_centers.begin(), tagged_centers.end(),
702  [&O](std::pair<int, Vector3d> const& apair,
703  std::pair<int, Vector3d> const& bpair) {
704  auto const& a = apair.second;
705  auto const& b = bpair.second;
706 
707  return (a - O).squaredNorm() < (b - O).squaredNorm();

709

713  for (auto const& tc : tagged_centers) {
714  J(i++) = tc.first;
715  }

717  RowMatrixXd Perm = Eigen::PermutationWrapper<Eigen::VectorXi>(J);
718  const auto& I_tr1 = Ci.trange().data()[1];
719  auto TPerm = array_ops::eigen_to_array<T, P>(Ci.world(), Perm, I_tr1, I_tr1);
720  Ci("m, i") = Ci("m,j") * TPerm("j,i");
721  Ci.world().gop.fence();

724 template <typename T, typename P>
725 pno::SparseMap PaoPnoRMP2<T, P>::IJ_map_generator(
726  pno::Array<T, P> const& Ci, pno::Array<T, P> const& Pv,
727  pno::Array<pno::PnoPair, TA::DensePolicy> const& osvs,
728  Eigen::VectorXd const& Fii) {
729  auto& world = Ci.world();
730 
731  pno::Array<T, P> rix, riy, riz;
732  rix("i,rho") = Ci("mu, i") * r_xyz_[0]("mu, nu") * Pv("nu, rho");
733  riy("i,rho") = Ci("mu, i") * r_xyz_[1]("mu, nu") * Pv("nu, rho");
734  riz("i,rho") = Ci("mu, i") * r_xyz_[2]("mu, nu") * Pv("nu, rho");
735 
736  const auto nocc_blocks = Ci.trange().tiles_range().extent_data()[1];
737  pno::SparseMap IJ(nocc_blocks, nocc_blocks);
738  IJ.setIdentity(); // Always need pair II
739 
740  auto orb_centers = this->orbital_centers(Ci);
741 
742  std::pair<double,std::mutex> energy_of_discarded_pairs;
743  energy_of_discarded_pairs.first = 0.0;
744 
745  std::atomic<std::int64_t> counter(0);
746  int64_t ntasks = 0;
747  for (auto J = 0; J < nocc_blocks; ++J) {
748  if (!osvs.is_local(J)) {
749  continue;
750  }
751 
752  auto const& osv_J = osvs.find(J).get(); // Should be local
753  auto const& epsil_J = osv_J.epsil_ab;
754  auto const& U_J = osv_J.U;
755  auto const& J_domains = osv_J.pao_domains;
756  const auto J_start = osv_J.i_start;
757 
758  RowMatrixXd xJ = pno::orbital_domain_row(rix, J_domains, J) * U_J;
759  RowMatrixXd yJ = pno::orbital_domain_row(riy, J_domains, J) * U_J;
760  RowMatrixXd zJ = pno::orbital_domain_row(riz, J_domains, J) * U_J;
761 
762  for (auto I = J + 1; I < nocc_blocks; ++I) {
763  // Might not be local but task might be a bit tricky
764  auto osv_I = osvs.find(I).get();
765  auto const& epsil_I = osv_I.epsil_ab;
766  auto const& U_I = osv_I.U;
767  auto const& I_domains = osv_I.pao_domains;
768  const auto I_start = osv_I.i_start;
769 
770  RowMatrixXd xI = pno::orbital_domain_row(rix, I_domains, I) * U_I;
771  RowMatrixXd yI = pno::orbital_domain_row(riy, I_domains, I) * U_I;
772  RowMatrixXd zI = pno::orbital_domain_row(riz, I_domains, I) * U_I;
773 
774  const auto pair_thresh = tcutIJ_;
775  auto pair_energy_est = [&Fii, &orb_centers, &IJ, epsil_I, epsil_J,
776  pair_thresh, I, J, &counter, &energy_of_discarded_pairs](
777  int i_start, int j_start,
778  RowMatrixXd const& xJ, RowMatrixXd const& yJ,
779  RowMatrixXd const& zJ, RowMatrixXd const& xI,
780  RowMatrixXd const& yI, RowMatrixXd const& zI) {
781  const auto ni = xI.rows();
782  const auto na = xI.cols();
783 
784  const auto nj = xJ.rows();
785  const auto nb = xJ.cols();
786 
787  auto energy = 0.0;
788 
789  for (auto i = 0; i < ni; ++i) {
790  const auto global_i = i_start + i;
791  auto const& center_i = orb_centers[global_i];
792  const auto fii = Fii(global_i);
793 
794  for (auto j = 0; j < nj; ++j) {
795  const auto global_j = j_start + j;
796  const auto fjj = Fii(global_j);
797  auto const& center_j = orb_centers[global_j];
798  double Rij = (center_i - center_j).norm();
799 
800  auto pair_energy = 0.0;
801  for (auto a = 0; a < na; ++a) {
802  const Vector3d vec_i({xI(i, a), yI(i, a), zI(i, a)});
803  const auto ea = epsil_I(a);
804 
805  for (auto b = 0; b < nb; ++b) {
806  const Vector3d vec_j({xJ(j, b), yJ(j, b), zJ(j, b)});
807  const auto eb = epsil_J(b);
808 
809  double dotp = vec_i.dot(vec_j);
810  pair_energy += (dotp * dotp) / (ea + eb - fii - fjj);
811  }
812  }
813  pair_energy *= -8.0 / std::pow(Rij, 6);
814  energy += pair_energy; // sum all OSV pairs for the given IJ pair
815  }
816  }
817 
818  if (std::abs(energy) > pair_thresh) { // set IJ map element
819  IJ(I, J) = 1;
820  IJ(J, I) = 1;
821  }
822  else {
823  energy_of_discarded_pairs.second.lock(); // CRITICAL SECTION!!!
824  energy_of_discarded_pairs.first += energy;
825  energy_of_discarded_pairs.second.unlock(); // END OF CRITICAL SECTION
826  }
827  ++counter;
828  };
829 
830  world.taskq.add(pair_energy_est, I_start, J_start, xJ, yJ, zJ, xI, yI,
831  zI);
832  ++ntasks;
833  }
834  }
835 
836  world.await([&counter, ntasks] { return counter == ntasks; });
837 
838  // Sum up data for all nodes
839  world.gop.sum(IJ.data(), IJ.size());
840 
841  // Min data to 1
842  for (auto I = 0; I < IJ.rows(); ++I) {
843  for (auto J = 0; J < IJ.cols(); ++J) {
844  IJ(I, J) = std::min(IJ(I, J), 1);
845  }
846  }
847 
848  return IJ;
849 } // namespace lcao
