301  void reduce(std::shared_ptr<result_type>& result) {
302  while(result) {
303  lock_.lock(); // <<< Begin critical section
304  if(ready_object_) {
305  // Get the ready argument
306  ReduceObject* ready_object = const_cast<ReduceObject*>(ready_object_);
307  ready_object_ = nullptr;
308  lock_.unlock(); // <<< End critical section
309 
310  // Reduce the argument that was held by ready_object_
311  op_(*result, ready_object->arg());
312 
313  // cleanup the argument
314  ReduceObject::destroy(ready_object);
315  this->dec();
316  } else if(ready_result_) {
317  // Get the ready result
318  std::shared_ptr<result_type> ready_result = ready_result_;
319  ready_result_.reset();
320  lock_.unlock(); // <<< End critical section
321 
322  // Reduce the result that was held by ready_result_
323  op_(*result, *ready_result);
324 
325  // cleanup the result
326  ready_result.reset();
327  } else {
328  // Nothing is ready, so place result in the ready state.
329  ready_result_ = result;
330  result.reset();
331  lock_.unlock(); // <<< End critical section
332  }
333  }
334  }

340  void reduce_result_object(std::shared_ptr<result_type> result, const ReduceObject* object) {
341  // Reduce the argument
