97  template<typename Tile, typename Policy>
98  void CCSD_PNO<Tile, Policy>::decom_t2(TA::DistArray<Tile, Policy> &t2_mp2,
99  const DecomType decom_method) {
100 
101  double threshold = tcut_;
102 
103  // decompose and reconstruct T2
104  auto decom = [&](Tile &in_tile) {
105 
106  // copy in_tile which is ab matrix of T^ij
107  Tile tab_tile = in_tile.clone();
108 
109  // compute ij pair index
110  const auto orb_i = in_tile.range().upbound()[2];
111  const auto orb_j = in_tile.range().upbound()[3];
112 
113  // get the dimensions of input tile
114  const auto extent = in_tile.range().extent();
115  const auto a = extent[0];
116  const auto b = extent[1];
117 
118  int rank = 0;
119 
120  switch(decom_method) {
121  // use eigen decomposition
122  case eigen_decom: {
123  auto Tab = TA::eigen_map(tab_tile, a, b);
124  // compute virtual or PNO density
125  TA::EigenMatrixXd Dab = Tab * (4.0 * Tab - 2.0 * Tab.transpose()).transpose()
126  + Tab.transpose() * (4.0 * Tab - 2.0 * Tab.transpose());
127  // when i=j, Dab = Tab*Tab' + Tab'*Tab
128  if (orb_i == orb_j)
129  Dab = Dab/2.0;
130 
131  // compute eigenvalues and eigenvectors of Dab
132  Eigen::SelfAdjointEigenSolver<TA::EigenMatrixXd> es(Dab);
133  TA::EigenVectorXd S_es = es.eigenvalues();
134  TA::EigenMatrixXd C_es = es.eigenvectors();
135 
136  // truncate eigenvectors with corresponding eigenvalues smaller than threshold
137  int pos_trunc = a - 1;
138  for(; pos_trunc >= 0; --pos_trunc) {
139  if(std::abs(S_es(pos_trunc)) < threshold)
140  break;
141  }
142 
143  rank = a - pos_trunc - 1;
144  if (rank > 0) {
145  auto tab_pno = TA::eigen_map(in_tile, a, b);
146  tab_pno.noalias() = C_es.rightCols(rank) * C_es.rightCols(rank).transpose()
147  * Tab
148  * C_es.rightCols(rank) * C_es.rightCols(rank).transpose();
149  } else {
150  std::fill(in_tile.begin(), in_tile.end(), 0.0);
151  }
152 
153  break;
154  }
155  // use singular value decompostion
156  case svd: {
157  const auto x = std::min(a,b);
158 
159  // allocate memory for SVD
160  auto u_ax = std::make_unique<double[]>(a * x);
161  auto v_xb = std::make_unique<double[]>(b * x);
162  auto s = std::make_unique<double[]>(x);
163 
164  // SVD of tile: (t_ab)^T
165  math::svd(in_tile.data(), s.get(), u_ax.get(), v_xb.get(), a, b, 'A');
166 
167  for(; rank < x; ++rank) {
168  // check singular value for truncation threshold
169  if(std::abs(s.get()[rank]) < threshold)
170  break;
171 
172  // fold s into u_ab:
173  // u_{a} = u_{a} * s_{a}
174  madness::cblas::scal(a, s.get()[rank], u_ax.get() + (a * rank), 1);
175  }
176 
177  // compute output tile: (t_ab)^T = u_a{c} * v_{c}b
178  madness::cblas::gemm(madness::cblas::NoTrans, madness::cblas::NoTrans,
179  a, b, rank, 1.0, u_ax.get(), a, v_xb.get(), b, 0.0, in_tile.data(), a);
180 
181  break;
182  }
183  default:
184  throw std::runtime_error("Invalid decomposition type for decomposing MP1 T2!");
185  break;
186  } // end of switch
187 
188  // compute the norm of difference between input and output tile
189  std::stringstream ss;
190  ss << "("<< orb_i << "," << orb_j << ") pair rank: " << rank
191  << " norm of (t_new - t_old): " << (TA::subt(in_tile, tab_tile)).norm()
192  << std::endl;
193 
194  ExEnv::out0() << indent << ss.str();
195 
196  return in_tile.norm();
197  };
198 
199  ExEnv::out0() << std::endl << "Using MP1 T2 amplitudes for PNO decomposition" << std::endl;
200  ExEnv::out0() << "Tcut is " << tcut_ << std::endl;
201  TA::foreach_inplace(t2_mp2, decom);
202  }

514  r2("a,b,i,j") -= t2("a,b,i,j");
515

517  t2("a,b,i,j") = t2("a,b,i,j") + r2("a,b,i,j");
518  t1.truncate();
519  t2.truncate();
520 
521  auto t2_time1 = mpqc::now(world, accurate_time);
522  auto t2_time = mpqc::duration_in_s(t2_time0, t2_time1);
523  if (this->verbose_) {
524  mpqc::utility::print_par(world, "t2 total time: ", t2_time, "\n");
525  }
526  // recompute energy
527  E0 = E1;
528  E1 = 2.0 * TA::dot(f_ai("a,i"), t1("a,i")) +
529  TA::dot((2.0 * g_abij("a,b,i,j") - g_abij("b,a,i,j")),
530  tau("a,b,i,j"));
531  dE = std::abs(E0 - E1);
532 
533  double error_r1 = r1("a,i").norm().get();
534  double error_r2 = r2("a,b,i,j").norm().get();
535 
536  if (dE >= this->target_precision_ || error >= this->target_precision_) {
537  tmp_time0 = mpqc::now(world, accurate_time);
538  ::mpqc::cc::TPack<TArray> t(t1, t2);
539  ::mpqc::cc::TPack<TArray> r(r1, r2);
540  error = r.norm() / (size(t1) + size(t2)); // error = residual norm per element
541  diis.extrapolate(t, r);
542 
543  // update t1 and t2
544  t1("a,i") = t.t1("a,i");
545  t2("a,b,i,j") = t.t2("a,b,i,j");
546 
547  if (this->verbose_) {
548  mpqc::detail::print_size_info(r2, "R2");
549  mpqc::detail::print_size_info(t2, "T2");
550  }
551 
552  tau("a,b,i,j") = t2("a,b,i,j") + t1("a,i") * t1("b,j");
553  tmp_time1 = mpqc::now(world, accurate_time);
554  tmp_time = mpqc::duration_in_s(tmp_time0, tmp_time1);
555  if (this->verbose_) {
556  mpqc::utility::print_par(world, "diis time: ", tmp_time, "\n");
557  }
558 
559  auto time1 = mpqc::fenced_now(world);
560  auto duration = mpqc::duration_in_s(time0, time1);
561 
562  if (world.rank() == 0) {
563  detail::print_ccsd(iter, dE, error, error_r1, error_r2, E1, duration);
564  }
565 
566  iter += 1ul;
567  } else {
568  auto time1 = mpqc::fenced_now(world);
569  auto duration = mpqc::duration_in_s(time0, time1);
570 
571  if (world.rank() == 0) {
572  detail::print_ccsd(iter, dE, error, error_r1, error_r2, E1, duration);
573  }
574 
575  break;
576  }
577  }
578  if (iter >= this->max_iter_) {
579  utility::print_par(this->wfn_world()->world(),
580  "\n Warning!! Exceed Max Iteration! \n");
581  }
582  if (world.rank() == 0) {
583  std::cout << "CCSD Energy " << E1 << std::endl;
584  }
585  return E1;

592

599 
600  // compute converting matrices for reblocking T2 amplitudes
601  // in each block: n_i=n_j=1, n_a=n_b=nvir
602  TArray occ_convert, vir_convert;

604 
605  // obtain MP1 T2 with new blocking structure
606  TArray t2_mp2 = t2_mp2_orig.clone();
607  t2_mp2("a,b,i,j") = t2_mp2("c,d,k,l")
608  * vir_convert("c,a") * vir_convert("d,b")
609  * occ_convert("k,i") * occ_convert("l,j");
610 
611  ExEnv::out0() << "Decomposing T2 amplitudes" << std::endl;
612  decom_t2(t2_mp2);
613 
614  // transform MP1 T2 amplitudes back into its original blocking structure
615  t2_mp2("a,b,i,j") = t2_mp2("c,d,k,l")
616  * vir_convert("a,c") * vir_convert("b,d")
617  * occ_convert("i,k") * occ_convert("j,l");
618 
619  // compute the difference between original and decomposed MP1 T2
620  ExEnv::out0() << "Norm of (t2 - t2 (decomposed)): "
621  << (t2_mp2_orig("a,b,i,j") - t2_mp2("a,b,i,j")).norm().get()
622  << std::endl << std::endl;
623 
624  // compute CCSD with decomposed MP1 T2 as initial guess
625  TA::DistArray<Tile, Policy> t1, t2;
626  t2 = t2_mp2.clone();
627 
628  ExEnv::out0() << std::endl
629  << "Computing CCSD with decomposed MP1 T2 as initial guess" << std::endl;
630  return compute_ccsdpno_df(t1,t2);
