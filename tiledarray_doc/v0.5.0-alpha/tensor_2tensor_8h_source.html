451  temp->data_ = temp->allocate(n);

1257  value_type trace() const {
1258  TA_ASSERT(pimpl_);
1259 
1260  // Get pointers to the range data
1261  const size_type n = pimpl_->range_.rank();
1262  const size_type* restrict const lower = pimpl_->range_.lobound_data();
1263  const size_type* restrict const upper = pimpl_->range_.upbound_data();
1264  const size_type* restrict const stride = pimpl_->range_.stride_data();
1265 
1266  // Search for the largest lower bound and the smallest upper bound
1267  size_type lower_max = 0ul, upper_min =
1268  std::numeric_limits<size_type>::max();
1269  for(size_type i = 0ul; i < n; ++i) {
1270  const size_type lower_i = lower[i];
1271  const size_type upper_i = upper[i];
1272 
1273  lower_max = std::max(lower_max, lower_i);
1274  upper_min = std::min(upper_min, upper_i);
1275  }
1276 
1277  value_type result = 0;
1278 
1279  if(lower_max < upper_min) {
1280  // Compute the first and last ordinal index
1281  size_type first = 0ul, last = 0ul, trace_stride = 0ul;
1282  for(size_type i = 0ul; i < n; ++i) {
1283  const size_type lower_i = lower[i];
1284  const size_type stride_i = stride[i];
1285 
1286  first += (lower_max - lower_i) * stride_i;
1287  last += (upper_min - lower_i) * stride_i;
1288  trace_stride += stride_i;
1289  }
1290 
1291  // Compute the trace
1292  const value_type* restrict const data = pimpl_->data_;
1293  for(; first < last; first += trace_stride)
1294  result += data[first];
1295  }
1296 
1297  return result;
1298  }
