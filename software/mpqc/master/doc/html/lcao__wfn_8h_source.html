97  virtual void init_sdref(std::shared_ptr<lcao::Wavefunction> ref_wfn,
98  double target_ref_precision) {
99  using TArray = TA::DistArray<Tile, Policy>;
100 
101  std::size_t n_frozen_core = 0;
102  if (frozen_core_) {
103  auto n_core_electrons = this->atoms()->core_electrons();
104  ExEnv::out0() << indent << "Frozen Core: " << n_core_electrons
105  << " electrons" << std::endl;
106  n_frozen_core = n_core_electrons / 2;
107  }
108  const auto nelectrons = this->atoms()->total_atomic_number() - charge_;
109  if (nelectrons % 2 != 0)
110  throw ProgrammingError(
111  "LCAOWavefunction::init_sdref: closed-shell determinant requires an "
112  "even number of electrons",
113  __FILE__, __LINE__);
114  const auto ndocc = nelectrons / 2;
115 
116  // dispatch request for either Canonical or Populated orbitals
117  // order of preference:
118  // - canonical orbs, since these have most information
119  // - populated orbs, since don't have to compute anything
120  // - compute canonical orbs ourselves, since this costs
121  auto canonical_orbs_provider = std::dynamic_pointer_cast<
122  typename CanonicalOrbitalSpace<TArray>::Provider>(ref_wfn);
123  auto has_canonical_orbs =
124  canonical_orbs_provider && canonical_orbs_provider->can_evaluate();
125  auto populated_orbs_provider = std::dynamic_pointer_cast<
126  typename PopulatedOrbitalSpace<TArray>::Provider>(ref_wfn);
127  auto has_populated_orbs =
128  populated_orbs_provider && populated_orbs_provider->can_evaluate();
129 
130  if (has_canonical_orbs || !has_populated_orbs) { // use canonical orbs
131  auto orbs = std::make_shared<CanonicalOrbitalSpace<TArray>>();
132  if (has_canonical_orbs)
133  evaluate(*orbs, canonical_orbs_provider, target_ref_precision);
134  else // last resort: compute canonical orbitals
135  orbs = make_closed_shell_canonical_orbitals(ao_factory_, ndocc,
136  unocc_block_);
137 
138  make_closed_shell_canonical_sdref_subspaces(
139  lcao_factory_,
140  std::const_pointer_cast<const CanonicalOrbitalSpace<TArray>>(orbs),
141  ndocc, n_frozen_core, occ_block_, unocc_block_);
142  }
143  // else ask for populated spaces
144  else {
145  make_closed_shell_sdref_subspaces(
146  ao_factory_, populated_orbs_provider, target_ref_precision, ndocc,
147  n_frozen_core, occ_block_, unocc_block_);
148  }
149 
150  if(export_orbital_){
151 
152  auto orbital_registry = this->lcao_factory().orbital_registry();
153  // prepare to Molden
154  const auto libint2_atoms = ::mpqc::to_libint_atom(this->wfn_world()->atoms()->atoms());
155  auto C_p_eig =
156  math::array_to_eigen(orbital_registry.retrieve("p").coefs());
157  const auto libint2_shells =
158  this->lcao_factory().basis_registry()->retrieve(L"Î¼")->flattened_shells();
159 
160 
161  // do writing on node 0
162  if(this->wfn_world()->world().rank() == 0){
163  // write out orbitals
164 
165  std::size_t n_mo = C_p_eig.cols();
166 
167  std::size_t n_occ = orbital_registry.retrieve("m").rank();
168 
169  auto occs = Eigen::VectorXd::Constant(n_occ, 2.0);
170  auto unoccs = Eigen::VectorXd::Constant(n_mo - n_occ, 0.0);
171 
172  Eigen::VectorXd op(n_mo);
173  op << occs, unoccs;
174 
175  // set orbital energy to zeroes
176  auto energies = Eigen::VectorXd::Constant(C_p_eig.cols(), 0.0);
177  // set orbital irrep labels to A
178  std::vector<std::string> labels(n_mo, "A");
179 
180  libint2::molden::Export xport(libint2_atoms, libint2_shells, C_p_eig,
181  op, energies, labels);
182 
183  std::string filename = FormIO::fileext_to_fullpathname(".orbital.molden");
184  xport.write(filename);
185 
186  }
187 
188  }
