38 void MolecularFormula::compute(const mpqc::Molecule& mol) {
39  std::map<int, size_t> count; // maps atomic number -> number of atoms
40 
41  auto atoms = mol.atoms();
42 
43  for (const auto& atom : atoms) {
44  auto Z = atom.atomic_number();
45  if (count.find(Z) == count.end()) count[Z] = 0;
46  ++count[Z];
47  }
48 
49  formula_.reserve(count.size());
50 
51  std::ostringstream sstr;
52  // pattern: CxHy... where "..." are the rest of the elements in the order of
53  // increasing atomic number
54  if (count.find(6) != count.end()) {
55  sstr << "C";
56  if (count[6] > 1) sstr << count[6];
57  formula_.emplace_back(std::make_pair(6, count[6]));
58  count.erase(6);
59  }
60  if (count.find(1) != count.end()) {
61  sstr << "H";
62  if (count[1] > 1) sstr << count[1];
63  formula_.emplace_back(std::make_pair(1, count[1]));
64  count.erase(1);
65  }
66  for (const auto& it : count) {
67  auto Z = it.first;
68  auto c = it.second;
69  formula_.emplace_back(std::make_pair(Z, c));
70  assert(Z != 0);
71  sstr << libint2::chemistry::get_element_info().at(Z - 1).symbol;
72  if (c > 1) sstr << c;
73  }
74  formula_str_ = sstr.str();
75 }
