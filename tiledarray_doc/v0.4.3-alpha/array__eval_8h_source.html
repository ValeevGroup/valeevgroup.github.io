169  virtual Future<value_type> get_tile(size_type i) const {
170 
171  // Get the array index that corresponds to the target index
172  const size_type array_index = DistEvalImpl_::perm_index_to_source(i);
173 
174  // Get the tile from array_, which may be located on a remote node.
175  Future<typename array_type::value_type> tile =
176  array_.find(array_index);
177 
178  const bool consumable_tile = ! array_.is_local(array_index);
179  // Insert the tile into this evaluator for subsequent processing
180  if(tile.probe()) {
181  // Skip the task since the tile is ready
182  Future<value_type> result;
183  result.set(make_tile(tile, consumable_tile));
184  const_cast<ArrayEvalImpl_*>(this)->notify();
185  return result;
186  } else {
187  // Spawn a task to set the tile the input tile is ready.
188  Future<value_type> result =
189  TensorImpl_::get_world().taskq.add(shared_from_this(),
190  & ArrayEvalImpl_::make_tile, tile, consumable_tile,
191  madness::TaskAttributes::hipri());
192 
193  result.register_callback(const_cast<ArrayEvalImpl_*>(this));
194  return result;
195  }
196  }

225  virtual int internal_eval() {
226  // Counter for the number of tasks submitted by this object
227  int task_count = 0;
228 
229  // Get a count of the number of local tiles.
230  if(TensorImpl_::shape().is_dense()) {
231  task_count = TensorImpl_::pmap()->local_size();
232  } else {
233  // Create iterator to tiles that are local for this evaluator.
234  typename array_type::pmap_interface::const_iterator it =
235  TensorImpl_::pmap()->begin();
236  const typename array_type::pmap_interface::const_iterator end =
237  TensorImpl_::pmap()->end();
238 
239  for(; it != end; ++it) {
240  if(! TensorImpl_::is_zero(*it))
241  ++task_count;
242  }
243  }
244 
245  return task_count;
246  }
