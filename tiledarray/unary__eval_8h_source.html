114  virtual int internal_eval() {
115  // Convert pimpl to this object type so it can be used in tasks
116  std::shared_ptr<UnaryEvalImpl_> self =
117  std::enable_shared_from_this<UnaryEvalImpl_>::shared_from_this();
118 
119  // Evaluate argument
120  arg_.eval();
121 
122  // Counter for the number of tasks submitted by this object
123  size_type task_count = 0ul;
124 
125  // Make sure all local tiles are present.
126  const typename pmap_interface::const_iterator end = arg_.pmap()->end();
127  typename pmap_interface::const_iterator it = arg_.pmap()->begin();
128  for(; it != end; ++it) {
129  // Get argument tile index
130  const size_type index = *it;
131 
132  if(! arg_.is_zero(index)) {
133  // Get target tile index
134  const size_type target_index = DistEvalImpl_::perm_index_to_target(index);
135 
136  // Schedule tile evaluation task
137  TensorImpl_::get_world().taskq.add(self, & UnaryEvalImpl_::eval_tile,
138  target_index, arg_.get(index));
139 
140  ++task_count;
141  }
142  }
143 
144  // Wait for local tiles of argument to be evaluated
145  arg_.wait();
146 
147  return task_count;
148  }
