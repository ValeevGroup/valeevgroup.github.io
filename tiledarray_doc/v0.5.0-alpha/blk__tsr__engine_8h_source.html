157  trange_type make_trange() const {
158  const unsigned int rank = array_.trange().tiles().rank();
159 
160  std::vector<TiledRange1> trange_data;
161  trange_data.reserve(rank);
162  std::vector<std::size_t> trange1_data;
163 
164  // Get temporary data pointers
165  const auto* restrict const trange = array_.trange().data().data();
166  const auto* restrict const lower = lower_bound_.data();
167  const auto* restrict const upper = upper_bound_.data();
168 
169  for(unsigned int d = 0u; d < rank; ++d) {
170  // Copy the tiling for the block
171  const auto lower_d = lower[d];
172  const auto upper_d = upper[d];
173 
174  // Copy and shift the tiling for the block
175  auto i = lower_d;
176  const auto base_d = trange[d].tile(i).first;
177  trange1_data.emplace_back(0ul);
178  for(; i < upper_d; ++i)
179  trange1_data.emplace_back(trange[d].tile(i).second - base_d);
180 
181  // Add the trange1 to the tiled range data
182  trange_data.emplace_back(trange1_data.begin(), trange1_data.end());
183  trange1_data.resize(0ul);
184  }
185 
186  return TiledRange(trange_data.begin(), trange_data.end());
187  }

193  trange_type make_trange(const Permutation& perm) const {
194  const unsigned int rank = array_.trange().tiles().rank();
195 
196  std::vector<TiledRange1> trange_data;
197  trange_data.reserve(rank);
198  std::vector<std::size_t> trange1_data;
199 
200  // Get temporary data pointers
201  const auto* restrict const trange = array_.trange().data().data();
202  const auto* restrict const lower = lower_bound_.data();
203  const auto* restrict const upper = upper_bound_.data();
204 
205  // Construct the inverse permutation
206  const Permutation inv_perm = -perm;
207  for(unsigned int d = 0u; d < rank; ++d) {
208  const auto inv_perm_d = inv_perm[d];
209 
210  // Copy the tiling for the block
211  const auto lower_i = lower[inv_perm_d];
212  const auto upper_i = upper[inv_perm_d];
213 
214  // Copy, shift, and permute the tiling of the block
215  auto i = lower_i;
216  const auto base_d = trange[inv_perm_d].tile(i).first;
217  trange1_data.emplace_back(0ul);
218  for(; i < upper_i; ++i)
219  trange1_data.emplace_back(trange[inv_perm_d].tile(i).second - base_d);
220 
221  // Add the trange1 to the tiled range data
222  trange_data.emplace_back(trange1_data.begin(), trange1_data.end());
223  trange1_data.resize(0ul);
224  }
225 
226  return TiledRange(trange_data.begin(), trange_data.end());
227  }

335  op_type make_tile_op() const {
336  const unsigned int rank = trange_.tiles().rank();
337 
338  // Construct and allocate memory for the shift range
339  std::vector<long> range_shift;
340  range_shift.reserve(rank);
341 
342  // Get temporary data pointers
343  const auto* restrict const trange = array_.trange().data().data();
344  const auto* restrict const lower = lower_bound_.data();
345 
346  // Initialize the range shift vector
347  for(unsigned int d = 0u; d < rank; ++d) {
348  const auto lower_d = lower[d];
349  const auto base_d = trange[d].tile(lower_d).first;
350  range_shift.emplace_back(-base_d);
351  }
352 
353  return op_type(op_base_type(range_shift));
