13 template <typename Tile, typename Policy>
14 std::shared_ptr<::mpqc::utility::TRange1Engine>
15 closed_shell_dual_basis_mo_build_steele(LCAOFactory<Tile, Policy> &lcao_factory,
16  Eigen::VectorXd &ens, std::size_t nocc,
17  const Molecule &mols, bool frozen_core,
18  std::size_t occ_blocksize,
19  std::size_t vir_blocksize) {
20  auto &ao_factory = lcao_factory.ao_factory();
21  auto &world = ao_factory.world();
22  auto &orbital_registry = lcao_factory.orbital_space();
23  using TArray = TA::DistArray<Tile, Policy>;
24 
25  auto mo_time0 = mpqc::fenced_now(world);
26  utility::print_par(
27  world,
28  "\nBuilding ClosedShell Dual Basis MO Orbital (Steele Approach) \n");
29 
30  std::size_t n_frozen_core = 0;
31  if (frozen_core) {
32  n_frozen_core = mols.core_electrons();
33  utility::print_par(world, "Frozen Core: ", n_frozen_core, " electrons. \n");
34  n_frozen_core = n_frozen_core / 2;
35  }
36 
37  // solving occupied orbitals on node 0, then broadcasting
38  TArray F;
39  if (ao_factory.registry().have(Formula(L"<μ|F|ν>"))) {
40  F = ao_factory.registry().retrieve(Formula(L"<μ|F|ν>"));
41  } else {
42  F = ao_factory.registry().retrieve(Formula(L"<μ|F|ν>[df]"));
43  }
44 
45  auto S = ao_factory.compute(L"<κ|λ>");
46 
47  RowMatrixXd F_eig = math::array_to_eigen(F);
48  RowMatrixXd S_eig = math::array_to_eigen(S);
49 
50  // solve mo coefficients
51  RowMatrixXd C_occ;
52  if (world.rank() == 0) {
53  Eigen::GeneralizedSelfAdjointEigenSolver<RowMatrixXd> es(F_eig, S_eig);
54  C_occ = es.eigenvectors().leftCols(nocc);
55  }
56  world.gop.broadcast_serializable(C_occ, 0);
57 
58  // finished solving occupied orbitals
59 
60  // get all the sizes
61  utility::print_par(world, "OccBlockSize: ", occ_blocksize, "\n");
62  utility::print_par(world, "VirBlockSize: ", vir_blocksize, "\n");
63 
64  std::size_t n_obs = S.trange().elements_range().extent()[0];
65  auto tre = std::make_shared<TRange1Engine>(nocc, n_obs, occ_blocksize,
66  vir_blocksize, n_frozen_core);
67  auto tr_obs = S.trange().data().back();
68  auto tr_occ = tre->compute_range(nocc, occ_blocksize);
69  auto tr_corr_occ = tre->get_active_occ_tr1();
70  mpqc::detail::parallel_print_range_info(world, tr_occ, "Occ");
71 
72  // convert to TA
73  auto C_occ_ta =
74  math::eigen_to_array<Tile, Policy>(world, C_occ, tr_obs, tr_occ);
75 
76  // project to large basis set
77 
78  TArray S_vbs_inv = ao_factory.compute(L"<Α|Β>[inv]");
79  TArray S_vbs_obs = ao_factory.compute(L"<Α|μ>");
80  TArray S_vbs_vbs = ao_factory.compute(L"<Α|Β>");
81  auto n_vbs = S_vbs_inv.trange().elements_range().extent()[0];
82  auto tr_vbs = S_vbs_inv.trange().data().back();
83 
84  TArray t;
85  t("A,mu") = S_vbs_inv("A,B") * S_vbs_obs("B,mu");
86 
87  C_occ_ta("A,i") = t("A,mu") * C_occ_ta("mu,i");
88 
89  // insert to registry
90  using OrbitalSpaceTArray = OrbitalSpace<TA::DistArray<Tile, Policy>>;
91  auto occ_space =
92  OrbitalSpaceTArray(OrbitalIndex(L"m"), OrbitalIndex(L"Α"), C_occ_ta);
93  orbital_registry.add(occ_space);
94 
95  // solving the new orbitals in large basis set
96  // find fock matrix
97  TArray F_vbs;
98  // if use density fitting
99  if (ao_factory.registry().have(Formula(L"<μ|F|ν>[df]"))) {
100  F_vbs = ao_factory.compute(Formula(L"<Α|F|Β>[df]"));
101  } else {
102  F_vbs = ao_factory.compute(Formula(L"<Α|F|Β>"));
103  }
104  tre = std::make_shared<TRange1Engine>(nocc, n_vbs, occ_blocksize,
105  vir_blocksize, n_frozen_core);
106 
107  RowMatrixXd C_vir;
108  RowMatrixXd C_corr_occ;
109  {
110  RowMatrixXd F_vbs_eigen = math::array_to_eigen(F_vbs);
111  RowMatrixXd S_vbs_eigen = math::array_to_eigen(S_vbs_vbs);
112  if (world.rank() == 0) {
113  Eigen::GeneralizedSelfAdjointEigenSolver<RowMatrixXd> es(F_vbs_eigen,
114  S_vbs_eigen);
115 
116  assert(es.info() == Eigen::ComputationInfo::Success);
117  ens = es.eigenvalues();
118  auto env = es.eigenvectors();
119 
120  C_occ = env.leftCols(nocc);
121  C_corr_occ = C_occ.rightCols(nocc - n_frozen_core);
122  C_vir = env.rightCols(n_vbs - nocc);
123  }
124  world.gop.broadcast_serializable(ens, 0);
125  world.gop.broadcast_serializable(C_occ, 0);
126  world.gop.broadcast_serializable(C_corr_occ, 0);
127  world.gop.broadcast_serializable(C_vir, 0);
128  }
129 
130  // get all the trange1s
131  auto tr_vir = tre->get_vir_tr1();
132  mpqc::detail::parallel_print_range_info(world, tr_vir, "Vir");
133 
134  C_occ_ta =
135  math::eigen_to_array<Tile, Policy>(world, C_occ, tr_vbs, tr_occ);
136 
137  auto C_corr_occ_ta = math::eigen_to_array<Tile, Policy>(
138  world, C_corr_occ, tr_vbs, tr_corr_occ);
139 
140  auto C_vir_ta =
141  math::eigen_to_array<Tile, Policy>(world, C_vir, tr_vbs, tr_vir);
142 
143  // insert to registry
144  occ_space =
145  OrbitalSpaceTArray(OrbitalIndex(L"m"), OrbitalIndex(L"Α"), C_occ_ta);
146  orbital_registry.remove(OrbitalIndex(L"m"));
147  orbital_registry.add(occ_space);
148 
149  auto corr_occ_space =
150  OrbitalSpaceTArray(OrbitalIndex(L"i"), OrbitalIndex(L"Α"), C_corr_occ_ta);
151  orbital_registry.add(corr_occ_space);
152 
153  auto vir_space =
154  OrbitalSpaceTArray(OrbitalIndex(L"a"), OrbitalIndex(L"Α"), C_vir_ta);
155  orbital_registry.add(vir_space);
156 
157  auto mo_time1 = mpqc::fenced_now(world);
158  auto mo_time = mpqc::duration_in_s(mo_time0, mo_time1);
159  utility::print_par(world, "ClosedShell Dual Basis MO Build Time: ", mo_time,
160  " S\n");
161 
162  return tre;
163 }

178  if (!this->compute()) {
