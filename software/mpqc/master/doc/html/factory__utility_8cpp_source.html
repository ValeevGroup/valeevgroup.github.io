13 libint2::Operator to_libint2_operator(Operator::Type mpqc_oper) {
14  TA_USER_ASSERT((Operator::Type::__first_1body_operator <= mpqc_oper &&
15  mpqc_oper <= Operator::Type::__last_1body_operator) ||
16  (Operator::Type::__first_2body_operator <= mpqc_oper &&
17  mpqc_oper <= Operator::Type::__last_2body_operator),
18  "invalid Operator::Type");
19 
20  libint2::Operator result;
21  switch (mpqc_oper) {
22  case Operator::Type::Overlap: {
23  result = libint2::Operator::overlap;
24  } break;
25  case Operator::Type::Kinetic: {
26  result = libint2::Operator::kinetic;
27  } break;
28  case Operator::Type::Nuclear: {
29  result = libint2::Operator::nuclear;
30  } break;
31  case Operator::Type::SphericalMultipole: {
32  result = libint2::Operator::sphemultipole;
33  } break;
34  case Operator::Type::Coulomb: {
35  result = libint2::Operator::coulomb;
36  } break;
37  case Operator::Type::CADF: {
38  result = libint2::Operator::coulomb;
39  } break;
40  case Operator::Type::cGTG: {
41  result = libint2::Operator::cgtg;
42  } break;
43  case Operator::Type::cGTG2: {
44  result = libint2::Operator::cgtg;
45  } break;
46  case Operator::Type::cGTGCoulomb: {
47  result = libint2::Operator::cgtg_x_coulomb;

60 libint2::any to_libint2_operator_params(
61  Operator::Type mpqc_oper, const Molecule &molecule,
62  const std::vector<std::pair<double, double>> &gtg_params) {
63  TA_USER_ASSERT((Operator::Type::__first_1body_operator <= mpqc_oper &&
64  mpqc_oper <= Operator::Type::__last_1body_operator) ||
65  (Operator::Type::__first_2body_operator <= mpqc_oper &&
66  mpqc_oper <= Operator::Type::__last_2body_operator),
67  "invalid Operator::Type");
68 
69  libint2::any result;
70  switch (mpqc_oper) {
71  case Operator::Type::Nuclear: {
72  result = make_q(molecule);
73  } break;
74  case Operator::Type::cGTG:
75  case Operator::Type::cGTGCoulomb:
76  case Operator::Type::DelcGTG2: {
77  TA_USER_ASSERT(not gtg_params.empty(),
78  "Gaussian-type geminal not initialized");
79  result = gtg_params;
80  } break;
81  case Operator::Type::cGTG2: {
82  TA_USER_ASSERT(not gtg_params.empty(),
83  "Gaussian-type geminal not initialized");
84  const auto &cgtg_params = gtg_params;
85  const auto ng = cgtg_params.size();
86  typename std::decay<decltype(cgtg_params)>::type cgtg2_params;
87  cgtg2_params.reserve(ng * (ng + 1) / 2);
88  for (auto b = 0ul; b < ng; ++b) {
89  for (decltype(b) k = 0; k <= b; ++k) {
90  const auto gexp = cgtg_params[b].first + cgtg_params[k].first;
91  const auto gcoeff = cgtg_params[b].second * cgtg_params[k].second *
92  (b == k ? 1 : 2); // if a != b include ab and ba
93  cgtg2_params.push_back(std::make_pair(gexp, gcoeff));
94  }
95  }
96  result = cgtg2_params;
97  } break;
98  default:; // nothing to do
99  }
100  return result;
101 }

242 std::array<Formula, 3> get_fock_formula(const Formula &formula) {
243  std::array<Formula, 3> result;
244  Formula h(formula);
245  h.set_operator(Operator(L"H"));
246  // H doesn't need DF
247  if (formula.has_option(Formula::Option::DensityFitting)) {
248  h.remove_option(Formula::Option::DensityFitting);
249  }
250  decltype(h) j(formula);
251  j.set_operator_type(Operator::Type::J);
252  decltype(h) k(formula);
253  if (formula.oper().type() == Operator::Type::Fock) {
254  k.set_operator_type(Operator::Type::K);
255  } else if (formula.oper().type() == Operator::Type::FockAlpha) {
256  k.set_operator_type(Operator::Type::KAlpha);
257  } else if (formula.oper().type() == Operator::Type::FockBeta) {
258  k.set_operator_type(Operator::Type::KBeta);
259  }
260 
261  result[0] = h;
262  result[1] = j;
263  result[2] = k;
264  return result;

267 std::vector<std::vector<size_t>> parallel_compute_shellpair_list(
268  madness::World &world, const Basis &basis1, const Basis &basis2,
269  double threshold, double engine_precision) {
270  using ::mpqc::lcao::gaussian::make_engine_pool;
271  // initialize engine
272  auto engine_pool = make_engine_pool(
273  libint2::Operator::overlap, utility::make_array_of_refs(basis1, basis2),
274  libint2::BraKet::x_x);
275 
276  const auto &shv1 = basis1.flattened_shells();
277  const auto &shv2 = basis2.flattened_shells();
278  const auto nsh1 = shv1.size();
279  const auto nsh2 = shv2.size();
280 
281  std::vector<std::vector<size_t>> result;
282  result.reserve(nsh1);
283 
284  auto compute = [&](size_t input_s1) {
285  auto engine = engine_pool->local();
286  engine.set_precision(engine_precision);
287  const auto &buf = engine.results();
288 
289  auto n1 = shv1[input_s1].size();
290  for (auto s2 = 0ul; s2 != nsh2; ++s2) {
291  auto on_same_center = (shv1[input_s1].O == shv2[s2].O);
292  bool significant = on_same_center;
293  if (!on_same_center) {
294  auto n2 = shv2[s2].size();
295  engine.compute1(shv1[input_s1], shv2[s2]);
296  Eigen::Map<const RowMatrixXd> buf_mat(buf[0], n1, n2);
297  auto norm = buf_mat.norm();
298  significant = (norm >= threshold);
299  }
300 
301  if (significant) {
302  result[input_s1].emplace_back(s2);
303  }
304  }
305  };
306 
307  for (auto s1 = 0ul; s1 != nsh1; ++s1) {
308  result.emplace_back(std::vector<size_t>());
309  world.taskq.add(compute, s1);
310  }
311  world.gop.fence();
312 
313  engine_pool.reset();
314 
315  // resort shell list in increasing order
316  for (auto s1 = 0ul; s1 != nsh1; ++s1) {
317  auto &list = result[s1];
318  std::sort(list.begin(), list.end());
319  }
320 
321  return result;
322 }
