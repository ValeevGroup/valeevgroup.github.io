181  template <typename Archive,
182  typename std::enable_if<madness::archive::is_output_archive<Archive>::value>::type* = nullptr>
183  void serialize(Archive &ar) const {
184  // Serialize data for empty tile check
185  bool empty = !static_cast<bool>(pimpl_);
186  ar & empty;
187  if (!empty) {
188  // Serialize tile data
189  ar & *pimpl_;
190  }
191  }
192 
193  template <typename Archive,
194  typename std::enable_if<madness::archive::is_input_archive<Archive>::value>::type* = nullptr>
195  void serialize(Archive &ar) {
196  // Check for empty tile
197  bool empty = false;
198  ar & empty;
199 
200  if (!empty) {
201  // Deserialize tile data
202  tensor_type tensor;
203  ar & tensor;
204 
205  // construct a new pimpl
206  pimpl_ = std::make_shared<T>(std::move(tensor));
207  } else {
208  // Set pimpl to an empty tile
209  pimpl_.reset();
210  }
211  }
