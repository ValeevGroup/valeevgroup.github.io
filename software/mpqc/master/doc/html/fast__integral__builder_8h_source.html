24 // Function to compute the direct tiles of (X | μ ν)
25 template <typename Tile, typename Policy>
26 auto compute_fast_E(gaussian::AOFactory<Tile, Policy>& aof,
27  std::shared_ptr<gaussian::SchwarzScreen> const& screener,
28  SparseMap const& IX, SparseMap const& IR,
29  SparseMap const& RSao) {
30  auto& world = aof.world();
31 
32  gaussian::Basis obs = *aof.basis_registry()->retrieve(OrbitalIndex(L"λ"));
33 
34  gaussian::Basis dfbs = *aof.basis_registry()->retrieve(OrbitalIndex(L"Κ"));
35 
36  auto three_center_pool = gaussian::make_engine_pool(
37  libint2::Operator::coulomb, utility::make_array_of_refs(dfbs, obs, obs),
38  libint2::BraKet::xs_xx);
39 
40  const auto bs_array = gaussian::BasisVector({dfbs, obs, obs});
41  auto trange3 = gaussian::detail::create_trange(bs_array);
42  auto pmap3 =
43  TA::SparsePolicy::default_pmap(world, trange3.tiles_range().volume());
44 
45  const auto nX = IX.cols();
46  const auto nR = RSao.rows();
47  // If XR or XS is important for any orbital we must keep it.
48  SparseMap XR = IX.transpose() * IR;
49  SparseMap RS;
50  { // Form RS using RSao and the IR map
51 
52  // if RS is important for any orbital then we must keep it
53  SparseMap RSI = IR.transpose() * IR;
54 
55  // RS will be the array product of RSI and RSao since if either is zero then
56  // RS should be 0
57  RS = RSI.array() * RSao.array();
58 
59  for (auto R = 0; R < nR; ++R) {
60  for (auto S = 0; S < nR; ++S) {
61  const auto val = RS(R, S);
62  RS(R, S) = std::min(val, 1);
63  }
64  }
65  }
66 
67  const bool replicate = true;
68  auto norm_ests = screener->norm_estimate(world, bs_array, *pmap3, replicate);
69 
70  using NormPair = std::pair<std::array<int, 3>, float>;
71 
72  std::vector<NormPair> E_shape_data;
73  for (auto X = 0; X < nX; ++X) {
74  for (auto R = 0; R < nR; ++R) {
75  if (XR(X, R) == 0) {
76  continue;
77  }
78 
79  for (auto S = 0; S < nR; ++S) {
80  // XR is the same as XS
81  if (XR(X, S) == 0 || RS(R, S) == 0) {
82  continue;
83  }
84 
85  std::array<int, 3> IRS{{X, R, S}};
86  const float norm_est = norm_ests(X, R, S);
87  auto IRS_pair = std::make_pair(IRS, norm_est);
88  E_shape_data.emplace_back(std::move(IRS_pair));
89  }
90  }
91  }
92 
93  return gaussian::direct_sparse_integrals(world, three_center_pool, bs_array,
94  E_shape_data, screener);
95 }

122 // Function to compute RS given the screener for (X | μ ν)
123 inline SparseMap make_map_RS(gaussian::SchwarzScreen const& SS, double TCUTRS) {
124  auto const& RS_est = SS.Qket().Qtile();
125  const auto nR = RS_est.rows();
126  const auto nS = RS_est.cols();
127 
128  SparseMap RS(nR, nS);
129  RS.setZero();
130 
131  for (auto R = 0; R < nR; ++R) {
132  for (auto S = 0; S < nS; ++S) {
133  if (RS_est(R, S) >= TCUTRS) {
134  RS(R, S) = 1;
135  }
136  }
137  }
138 
139  return RS;

143 inline SparseMap unionize_for_IJ(SparseMap const& Iany, SparseMap const& IJ) {
144  SparseMap Iunion = IJ * Iany;
145 
146  const auto nI = Iany.rows();
147  const auto nother = Iany.cols();

150  for (auto A = 0; A < nother; ++A) { // Return mapped indices to 1

152  Iunion(I,A) = std::min(val, 1);
153  }
154  }

160 template <typename T>
161 SparseMap map_from_shape(TA::Tensor<T> mat_shape) {
162  auto const& range = mat_shape.range();
163  const auto ext = range.extent_data();
164 
165  assert(range.rank() == 2); // ensure matrix
166  const auto nA = ext[0];
167  const auto nB = ext[1];
168 
169  SparseMap AB(nA, nB);
170  AB.setZero();
171 
172  for (auto A = 0; A < nA; ++A) {
173  for (auto B = 0; B < nB; ++B) {
174  if (mat_shape(A, B) > 0.0) {
175  AB(A, B) = 1;
176  }
177  }
178  }
179 
180  return AB;
181 }

183 // Compute the (i -> AO) map using the (i -> AO) map based on LMOs, then use the
184 // IA map along with the PAO projector to figure out which AOs are needed for
185 // each PAO
186 //
187 // IRlmo should come from the LMOs
188 // IA is the orbital PAO domains
189 // P is the PAO projector
190 template <typename Tile, typename Policy>
191 SparseMap compute_IR(TA::DistArray<Tile, Policy> const& P, SparseMap const& IA,
192  SparseMap const& IRlmo) {
193  const auto nI = IA.rows();
194  const auto nA = IA.cols();
195  const auto nR = IRlmo.cols();
196  SparseMap IR(nI, nR);
197  IR.setZero();
198 
199  TA::Tensor<float> Pshape = P.shape().data();
200 
201  // First things first we need to figure out which AOs are important for each
202  // PAO in I
203  for (auto I = 0; I < nI; ++I) { // Loop over I
204  for (auto A = 0; A < nA; ++A) {
205  if (IA(I, A) == 0) { // Only include this PAO if it is in the IA domain
206  continue;
207  }
208 
209  for (auto R = 0; R < nR; ++R) {
210  // Check if the PAO projector couples AO block R with the PAO block A
211  if (Pshape(R, A) > 0) {
212  // R is connected to I via the PAO projector then copy that element of
213  // IRlmo
214  IR(I, R) = IRlmo(I, R);
215  }
216  }
217  }
218  }
219 
220  return IR;
221 }

226 template <typename DirectArray, typename Tile, typename Policy>
227 TA::DistArray<Tile, Policy> EToZir(DirectArray const& E,
228  TA::DistArray<Tile, Policy> const& Ci,
229  SparseMap const& IX, SparseMap const& IR) {
230  auto const& Etrange = E.array().trange();
231  const auto Eext = Etrange.tiles_range().extent_data();
232  const auto X_tr1 = Etrange.data()[0];
233  const auto R_tr1 = Etrange.data()[2];
234 
235  auto const& Ctrange = Ci.trange();
236  const auto Cext = Ctrange.tiles_range().extent_data();
237  const auto I_tr1 = Ctrange.data()[1];
238 
239  TA::TiledRange Zir_trange({X_tr1, I_tr1, R_tr1});
240 
241  const auto nX = Eext[0];
242  const auto nI = Cext[1];
243  const auto nR = Eext[2];
244 
245  TA::Tensor<float> shape_data(TA::Range({nX, nI, nR}), 0.0);
246  for (auto I = 0; I < nI; ++I) {
247  for (auto X = 0; X < nX; ++X) {
248  if (IX(I, X) == 0) {
249  continue;
250  }
251 
252  for (auto R = 0; R < nR; ++R) {
253  if (IR(I, R) == 0) {
254  continue;
255  }
256 
257  shape_data(X, I, R) = std::numeric_limits<float>::max();
258  }
259  }
260  }
261 
262  TA::SparseShape<float> Zir_shape(shape_data, Zir_trange);
263 
264  TA::DistArray<Tile, Policy> Zir;
265  Zir("X, i, r") = (E("X,r,s") * Ci("s, i")).set_shape(Zir_shape);
266  Zir.truncate();
267  return Zir;
268 }

287  auto Zia_trange = TA::TiledRange({X_tr1, I_tr1, A_tr1});
288 
289  // The shape for the output Zia
290  const auto nX = Ztext[0];
291  const auto nI = Ztext[1];
292  const auto nA = Ptext[1];
293  TA::Tensor<float> shape_data(TA::Range({nX, nI, nA}), 0.0);
294 
295  for (auto X = 0; X < nX; ++X) {
296  for (auto I = 0; I < nI; ++I) {
297  if (IX(I, X) == 0) {
298  continue;
299  }
300 
301  for (auto A = 0; A < nA; ++A) {
302  if (IA(I, A) == 0) {
303  continue;
304  }
305 
306  shape_data(X, I, A) = std::numeric_limits<float>::max();
307  }
308  }
309  }
310 
311  // No need for Madness to replicate since sparse maps are already replicated.
312  TA::SparseShape<float> Zia_shape(shape_data, Zia_trange);
313 
314  // Use the new sparse shape to screen the construction of Zia
315  Zir("X,i,a") = (Zir("X,i,mu") * P("mu, a")).set_shape(Zia_shape);
316  Zir.truncate();
317  world.gop.fence();

353 template <typename Tile, typename Policy>
354 TA::DistArray<Tile, Policy> Zia(gaussian::AOFactory<Tile, Policy>& aof,
355  TA::DistArray<Tile, Policy> const& Ci,
356  TA::DistArray<Tile, Policy> const& P,
357  SparseMap const& IX, SparseMap const& IA,
358  SparseMap const& IJ, bool fastE, bool fastZir,
359  bool fastZia, double tcutRS, double tcutScreen,
360  bool diagonalIJ = false) {
361  auto& world = Ci.world();
362  TA::DistArray<Tile, Policy> Zia;
363 
364  // Create output shape for Zia based on our sparse map inputs
365  auto IX_unions = IX;
366  auto IA_unions = IA;
367 
368  if (!diagonalIJ) {
369  IX_unions = detail::unionize_for_IJ(IX, IJ);
370  IA_unions = detail::unionize_for_IJ(IA, IJ);
371  }
372 
373  SparseMap IR;
374  if (fastE || fastZir) { // We only need IR for E or Zir
375  SparseMap IRlmo = detail::map_from_shape(Ci.shape().data());
376  // Transpose since Ci is (R x I)
377  IRlmo.transposeInPlace();
378  IR = detail::compute_IR(P, IA_unions, IRlmo);
379  }
380 
381  //
382  // Compute direct (X | μ ν) if fast we limit the allowed direct tiles
383  //
384  auto etime0 = mpqc::fenced_now(world);
385  using DirectArray = gaussian::DirectArray<
386  Tile, Policy, gaussian::DirectIntegralBuilder<Tile, libint2::Engine>>;
387  DirectArray E;
388  if (!fastE) {
389  E = aof.compute_direct(L"(Λ | G | μ ν)");
390  } else {
391  // Get the maps that we didn't pass into our function
392  // Get a screener to use for the AO integrals
393  auto screener = detail::get_schwarz_screener(aof, tcutScreen);
394  SparseMap RS = detail::make_map_RS(*screener, tcutRS);
395  E = detail::compute_fast_E(aof, screener, IX, IR, RS);
396  }
397  auto etime1 = mpqc::fenced_now(world);
398  auto etime = mpqc::duration_in_s(etime0, etime1);
399 
400  //
401  // Compute (X | i μ)
402  //
403  auto Zirtime0 = mpqc::fenced_now(world);
404  if (!fastZir) {
405  Zia("X,i,r") = E("X,r,s") * Ci("s,i");
406  Zia.truncate();
407  } else {
408  Zia = EToZir(E, Ci, IX_unions, IR);
409  }
410  auto Zirtime1 = mpqc::fenced_now(world);
411  auto Zirtime = mpqc::duration_in_s(Zirtime0, Zirtime1);
412 
413  //
414  // Compute (X | i μ)
415  //
416  auto Ziatime0 = mpqc::fenced_now(world);
417  if (!fastZia) {
418  Zia("X,i,a") = Zia("X,i,r") * P("r,a");
419  Zia.truncate();
420  world.gop.fence();
421  } else {
422  ZirToZia(Zia, P, IX_unions, IA_unions);
423  }
424  auto Ziatime1 = mpqc::fenced_now(world);
425  auto Ziatime = mpqc::duration_in_s(Ziatime0, Ziatime1);
426 
427  if (diagonalIJ) {
428  ExEnv::out0() << "\n" << indent << "Computed Zia using diagonal IJ:\n";
429  } else {
430  ExEnv::out0() << "\n" << indent << "Computed Zia using IJ map:\n";
431  }
432 
433  std::string emeth = (fastE) ? "fast" : "canon";
434  ExEnv::out0() << indent << "E time(" << emeth << "): " << etime << "\n";
435 
436  std::string rmeth = (fastZir) ? "fast" : "canon";
437  ExEnv::out0() << indent << "Zir time(" << rmeth << "): " << Zirtime << "\n";
438 
439  std::string ameth = (fastZia) ? "fast" : "canon";
440  ExEnv::out0() << indent << "Zia time(" << ameth << "): " << Ziatime << "\n";
441  ExEnv::out0() << indent << "Total time: " << etime + Ziatime + Zirtime
442  << std::endl;
443 
444  return Zia;
445 }
