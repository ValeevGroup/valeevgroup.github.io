98  DistEvalImpl(World& world, const trange_type& trange,
99  const shape_type& shape, const std::shared_ptr<pmap_interface>& pmap,

167  void wait() const {
168  const int task_count = task_count_;
169  if(task_count > 0) {
170  try {
171  TensorImpl_::get_world().await([this,task_count] ()
172  { return set_counter_ == task_count; });
173  } catch(...) {
174  std::stringstream ss;
175  ss << "!! ERROR TiledArray: Aborting due to exception.\n"
176  << "!! ERROR TiledArray: rank=" << TensorImpl_::get_world().rank()
177  << " id=" << id_ << " " << set_counter_ << " of " << task_count << " tiles set\n";
178  std::cout << ss.str().c_str();
179  throw;
180  }
181  }
