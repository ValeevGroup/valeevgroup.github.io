99 template <typename T>
100 inline void print_excitation_energy(const EigenVector<T> &eig, bool triplets) {
101  const auto &unit_factory = UnitFactory::get_default();
102  const auto Hartree_to_eV = unit_factory->make_unit("eV").from_atomic_units();
103  const auto Hartree_to_wavenumber =
104  unit_factory->make_unit("energy_wavenumber[cm]").from_atomic_units();
105 
106  ExEnv::out0() << "Excitation Energy: ( "
107  << (triplets ? "Triplets" : "Singlets") << " )\n";
108 
109  ExEnv::out0() << mpqc::printf("%5s \t %10s \t %10s \t %10s \n", "state", "au",
110  "eV", "cm^-1");
111 
112  const std::size_t size = eig.size();
113 
114  for (std::size_t i = 1; i <= size; i++) {
115  T e = eig[i - 1];
116  ExEnv::out0() << mpqc::printf("%5i \t %10.8f \t %10.5f \t %10.2f \n", i, e,
117  e * Hartree_to_eV, e * Hartree_to_wavenumber);
118  }
119  ExEnv::out0() << "\n";
120 }
