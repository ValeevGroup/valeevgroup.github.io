212 std::string Formula::to_ta_expression(Transformer transform_op) const {
213  std::string ta_expression;
214  std::size_t rank = this->rank();
215  std::size_t count = 0;
216 
217  // add left index
218  for (const auto &index : bra_indices_) {
219  std::string index_expression = transform_op(index.to_ta_expression());
220  ta_expression.append(index_expression.begin(), index_expression.end());
221  ++count;
222  ta_expression.append(", ");
223  }
224 
225  // add right index
226  for (const auto &index : ket_indices_) {
227  std::string index_expression = transform_op(index.to_ta_expression());
228  ta_expression.append(index_expression.begin(), index_expression.end());
229  ++count;
230  if (count < rank) {
231  ta_expression.append(", ");
232  }
233  }
234
