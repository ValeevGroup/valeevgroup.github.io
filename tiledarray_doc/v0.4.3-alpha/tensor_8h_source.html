112  size_type * restrict const fused_weight,
113  const size_type * restrict const size, const Permutation& perm)
114  {
115  const unsigned int ndim1 = perm.dim() - 1u;
116 
117  int i = ndim1;
118  fused_size[3] = size[i--];
119  while((i >= 0) && (perm[i + 1u] == (perm[i] + 1u)))
120  fused_size[3] *= size[i--];
121  fused_weight[3] = 1u;
122 
123 
124 
125  if((i >= 0) && (perm[i] != ndim1)) {
126  fused_size[2] = size[i--];
127  while((i >= 0) && (perm[i] != ndim1))
128  fused_size[2] *= size[i--];
129 
130  fused_weight[2] = fused_size[3];
131 
132  fused_size[1] = size[i--];
133  while((i >= 0) && (perm[i + 1] == (perm[i] + 1u)))
134  fused_size[1] *= size[i--];
135 
136  fused_weight[1] = fused_size[2] * fused_weight[2];
137  } else {
138  fused_size[2] = 1ul;
139  fused_weight[2] = 0ul;
140 
141  fused_size[1] = size[i--];

143  fused_size[1] *= size[i--];

145  fused_weight[1] = fused_size[3];

212  template <typename U, typename AU>
213  Tensor(const Tensor<U, AU>& other, const Permutation& perm) :
214  pimpl_(new Impl(perm ^ other.range()))
215  {
216  // Check inputs.
217  TA_ASSERT(! other.empty());
218  TA_ASSERT(perm);
219  TA_ASSERT(perm.dim() == other.range().dim());
220 
221  detail::PermIndex perm_index_op(other.range(), perm);
222 
223  // Cache constants
224  const unsigned int ndim = other.range().dim();
225  const unsigned int ndim1 = ndim - 1u;
226  const size_type volume = other.range().volume();
227 
228  if(perm[ndim1] == ndim1) {
229  // This is the simple case where the last dimension is not permuted.
230  // Therefore, it can be shuffled in chunks.
231 
232  // Determine which dimensions can be permuted with the least significant
233  // dimension.
234  size_type block_size = other.range().size()[ndim1];
235  for(int i = -1 + ndim1 ; i >= 0; --i) {
236  if(int(perm[i]) != i)
237  break;
238  block_size *= other.range().size()[i];
239  }
240 
241  // Permute the data
242  for(size_type index = 0ul; index < volume; index += block_size) {
243  const size_type perm_index = perm_index_op(index);
244 
245  // Copy the block
246  math::uninitialized_copy_vector(block_size, other.data() + index,
247  pimpl_->data_ + perm_index);
248  }
249 
250  } else {
251  // This is the more complicated case. Here we permute in terms of matrix
252  // transposes. The data layout of the input and output matrices are
253  // chosen such that they both contain stride one dimensions.
254 
255  size_type other_fused_size[4];
256  size_type other_fused_weight[4];
257  fuse_dimensions(other_fused_size, other_fused_weight,
258  other.range().size().data(), perm);
259 
260  // Compute the fused stride for the result matrix transpose.
261  size_type result_outer_stride = 1ul;
262  for(unsigned int i = perm[ndim1] + 1u; i < ndim; ++i)
263  result_outer_stride *= pimpl_->range_.size()[i];
264 
265  // Copy data from the input to the output matrix via a series of matrix
266  // transposes.
267  for(size_type i = 0ul; i < other_fused_size[0]; ++i) {
268  size_type index = i * other_fused_weight[0];
269  for(size_type j = 0ul; j < other_fused_size[2]; ++j, index += other_fused_weight[2]) {
270  // Compute the ordinal index of the input and output matrices.
271  size_type perm_index = perm_index_op(index);
272 
273  // Copy a transposed matrix from the input tensor to the this tensor.
274  auto copy_op = [] (param_value_type<Tensor<U, AU> > a) ->
275  param_value_type<Tensor<U, AU> > { return a; };
276  math::uninitialized_transpose(copy_op,
277  other_fused_size[1], other_fused_size[3],
278  result_outer_stride, pimpl_->data_ + perm_index,
279  other_fused_weight[1], other.data() + index);
280  }
281  }
282  }
283  }
