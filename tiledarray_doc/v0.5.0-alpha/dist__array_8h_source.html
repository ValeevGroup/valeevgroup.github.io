80  static void lazy_deleter(const impl_type* const pimpl) {
81  if(pimpl) {
82  if(madness::initialized()) {
83  World& world = pimpl->get_world();
84  const madness::uniqueidT id = pimpl->id();
85  cleanup_counter_++;
86 
87  try {
88  world.gop.lazy_sync(id, [pimpl]() {
89  delete pimpl;
90  DistArray_::cleanup_counter_--;
91  });
92  }
93  catch(madness::MadnessException& e) {
94  fprintf(stderr, "!! ERROR TiledArray: madness::MadnessException thrown in Array::lazy_deleter().\n"
95  "%s\n"
96  "!! ERROR TiledArray: The exception has been absorbed.\n"
97  "!! ERROR TiledArray: rank=%i\n", e.what(), world.rank());
98 
99  cleanup_counter_--;
100  delete pimpl;
101  }
102  catch(std::exception& e) {
103  fprintf(stderr, "!! ERROR TiledArray: std::exception thrown in Array::lazy_deleter().\n"
104  "%s\n"
105  "!! ERROR TiledArray: The exception has been absorbed.\n"
106  "!! ERROR TiledArray: rank=%i\n", e.what(), world.rank());
107 
108  cleanup_counter_--;
109  delete pimpl;
110  }
111  catch(...) {
112  fprintf(stderr, "!! ERROR TiledArray: An unknown exception was thrown in Array::lazy_deleter().\n"
113  "!! ERROR TiledArray: The exception has been absorbed.\n"
114  "!! ERROR TiledArray: rank=%i\n", world.rank());
115 
116  cleanup_counter_--;
117  delete pimpl;
118  }
119  } else {
120  delete pimpl;
121  }
122  }
123  }

131  static std::shared_ptr<impl_type>
132  init(World& world, const trange_type& trange, const shape_type& shape,
133  std::shared_ptr<pmap_interface> pmap)
134  {
135  // User level validation of input
136 
137  if(! pmap) {
138  // Construct a default process map
139  pmap = Policy::default_pmap(world, trange.tiles().volume());
140  } else {
141  // Validate the process map
142  TA_USER_ASSERT(pmap->size() == trange.tiles().volume(),
143  "Array::Array() -- The size of the process map is not equal to the number of tiles in the TiledRange object.");
144  TA_USER_ASSERT(pmap->rank() == typename pmap_interface::size_type(world.rank()),
145  "Array::Array() -- The rank of the process map is not equal to that of the world object.");
146  TA_USER_ASSERT(pmap->procs() == typename pmap_interface::size_type(world.size()),
147  "Array::Array() -- The number of processes in the process map is not equal to that of the world object.");
148  }
149 
150  // Validate the shape
151  TA_USER_ASSERT(! shape.empty(),
152  "Array::Array() -- The shape is not initialized.");
153  TA_USER_ASSERT(shape.validate(trange.tiles()),
154  "Array::Array() -- The range of the shape is not equal to the tiles range.");
155 
156  return std::shared_ptr<impl_type>(new impl_type(world, trange, shape, pmap), lazy_deleter);
157  }

396  template <typename Op>
397  void init_tiles(Op&& op) {
398  check_pimpl();
399 
400  auto it = pimpl_->pmap()->begin();
401  const auto end = pimpl_->pmap()->end();
402  for(; it != end; ++it) {
403  const auto index = *it;
404  if(! pimpl_->is_zero(index)) {
405  Future<value_type> tile = pimpl_->get_world().taskq.add(
406  [] (DistArray_& array, const size_type index, const Op& op) -> value_type
407  { return op(array.trange().make_tile_range(index)); },
408  *this, index, op);
409  set(index, tile);
410  }
411  }
412  }

563  void make_replicated() {
564  check_pimpl();
565  if((! pimpl_->pmap()->is_replicated()) && (get_world().size() > 1)) {
566  // Construct a replicated array
567  std::shared_ptr<pmap_interface> pmap(new detail::ReplicatedPmap(get_world(), size()));
568  DistArray_ result = DistArray_(get_world(), trange(), get_shape(), pmap);
569 
570  // Create the replicator object that will do an all-to-all broadcast of
571  // the local tile data.
572  std::shared_ptr<detail::Replicator<DistArray_> > replicator(
573  new detail::Replicator<DistArray_>(*this, result));
574 
575  // Put the replicator pointer in the deferred cleanup object so it will
576  // be deleted at the end of the next fence.
577  TA_ASSERT(replicator.unique()); // Required for deferred_cleanup
578  madness::detail::deferred_cleanup(get_world(), replicator);
579 
580  DistArray_::operator=(result);
581  }
582  }
