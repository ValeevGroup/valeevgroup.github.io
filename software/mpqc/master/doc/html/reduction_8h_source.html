46 template <typename Tile>
47 class MaxNReduction {
48  public:
49  // typedefs
50  using element_type = typename Tile::numeric_type;
51  using result_type = std::vector<element_type>;
52  using argument_type = Tile;
53 
54  // constructor
55  MaxNReduction(std::size_t n)
56  : n_(n), default_(std::numeric_limits<element_type>::min()) {}
57  ~MaxNReduction() = default;
58 
59  // Reduction functions
60 
61  // Make an empty result object
62  result_type operator()() const { return result_type(n_, default_); }
63 
64  // Post process the result
65  const result_type& operator()(const result_type& result) const {
66  return result;
67  }
68 
69  // Reduce two result objects
70  void operator()(result_type& result, const result_type& arg) const {
71  result.insert(result.end(), arg.begin(), arg.end());
72  if (result.size() > n_) {
73  std::partial_sort(result.begin(), result.begin() + n_, result.end(),
74  std::greater<element_type>());
75  result.erase(result.begin() + n_, result.end());
76  }
77  }
78 
79  // Reduce an argument
80  void operator()(result_type& result, const argument_type& arg) const {
81  // make sure n does not exceed number of elements in arg
82  std::size_t n = std::min(n_, arg.size());
83  result_type arg_result(n);
84  std::partial_sort_copy(arg.begin(), arg.end(), arg_result.begin(),
85  arg_result.end(), std::greater<element_type>());
86  operator()(result, arg_result);
87  }
88 
89  private:
90  std::size_t n_; // number of max number to return
91  element_type default_; // default initialized value
92 
93 }; // class MaxNReduction
