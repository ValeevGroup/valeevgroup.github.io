176 std::size_t qr_rank(double const *data, std::size_t rows, std::size_t cols,
177  double threshold) {
178  const auto full_rank = std::min(cols, rows);
179  auto out_rank = full_rank;
180 
181  auto M = Eigen::Map<const Eigen::MatrixXd>(data, rows, cols);
182 
183  auto squared_sum = 0.0;
184  auto thresh2 = threshold * threshold;
185  for (int i = (full_rank - 1); i >= 0; --i) { // rows of R
186  for (int j = (cols - 1); j >= i; --j) { // cols of R
187  squared_sum += M(i, j) * M(i, j);
188  }
189 
190  if (squared_sum >= thresh2) {
191  return out_rank;
192  }
193 
194  --out_rank; // Decriment rank and go to next row.
195  }
196 
197  return std::max(out_rank, 1ul);
198 }

532 void recompress(DecomposedTensor<double> &t) {
533  assert(t.ndecomp() >= 2);
534 
535  // Given W = S * T;
536  // we want S = Ls * Rs and T = Lt * Rt
537  // W = Ls * Rs * Lt * Rt
538  TA::Tensor<double> Ls, Rs, Lt, Rt;
539  ta_tensor_lq(t.tensor(0), Ls, Rs); // LQ Left to end up with QR since Row
540  // Major
541  ta_tensor_qr(t.tensor(1), Lt, Rt); // QR right to end up with LQ since Row
542  // Major
543 
544  // Form an M matrix where M = Rs * Lt
545  constexpr auto NoT = madness::cblas::CBLAS_TRANSPOSE::NoTrans;
546  const auto gh = TA::math::GemmHelper(NoT, NoT, 2, 2, 2);
547  auto M = Rs.gemm(Lt, 1.0, gh);
548 
549  // want to always do the full decomp so make input
550  // max rank larger than rank of M.
551  TA::Tensor<double> Lm, Rm;
552 
553  // Form M = Lm * Rm
554  ta_tensor_col_pivoted_qr(M, Lm, Rm, t.cut());
555 
556  // Form new Left side Snew = Ls * Lm
557  auto newL = Ls.gemm(Lm, 1.0, gh);
558 
559  // Form new Right side Tnew = Rm * Rt
560  auto ord = t.orders();
561  const auto gh_r = TA::math::GemmHelper(NoT, NoT, ord[1], ord[0], ord[1]);
562  auto newR = Rm.gemm(Rt, 1.0, gh_r);
563 
564  // W = Snew * Rnew
565  t = DecomposedTensor<double>(t.cut(), std::move(newL), std::move(newR));
566 }
