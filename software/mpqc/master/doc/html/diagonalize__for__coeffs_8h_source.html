23 inline TA::TiledRange1 tr_occupied(int guess, int occ) {
24  auto nblocks = (guess < occ) ? guess : occ;
25  auto block_size = std::max(occ / nblocks, 1);
26  std::vector<std::size_t> blocks;
27  blocks.reserve(nblocks + 1);
28  blocks.push_back(0);
29  for (auto i = block_size; i < occ; i += block_size) {
30  blocks.push_back(i);
31  }
32  blocks.push_back(occ);
33  return TA::TiledRange1(blocks.begin(), blocks.end());
34 }

36 inline Array2 Coeffs_from_fock(Array2 const &F, Array2 const &S, TA::TiledRange1 tr_i,
37  unsigned int occ, unsigned int nocc_clusters,
38  bool use_chol_vectors = false) {
39  auto F_eig = math::array_to_eigen(F);
40  auto S_eig = math::array_to_eigen(S);
41 
42  Eigen::GeneralizedSelfAdjointEigenSolver<decltype(S_eig)> es(F_eig, S_eig);
43  decltype(S_eig) C = es.eigenvectors().leftCols(occ);
44 
45  if (use_chol_vectors) {
46  decltype(S_eig) D = C * C.transpose();
47  unsigned int rank = math::piv_cholesky(D);
48  if (rank == occ) {

50  } else {
51  std::cout << "Cholesky Rank was " << rank << std::endl;
52  throw std::runtime_error("The rank of the cholesky vectors was not "
53  "equal to the number of occupied "

55  }
56  }
57 
58  auto tr_ao = S.trange().data()[0];
59
