157 Eigen::RowVectorXi sub_basis_map(const Basis &basis, const Basis &sub_basis) {
158  auto shells = basis.flattened_shells();
159  auto sub_shells = sub_basis.flattened_shells();
160 
161  std::size_t n_functions = basis.nfunctions();
162  std::size_t n_sub_functions = sub_basis.nfunctions();
163  TA_ASSERT(n_functions >= n_sub_functions);
164 
165  std::size_t n_shells = basis.nshells();
166  std::size_t n_sub_shells = sub_basis.nshells();
167 
168  Eigen::RowVectorXi result = Eigen::RowVectorXi::Zero(n_functions);
169 
170  std::size_t sub_shell_lowbound = 0;
171  std::size_t sub_shell_highbound = 0;
172  for (std::size_t i = 0; i < n_sub_shells; i++) {
173  auto &sub_shell = sub_shells[i];
174  sub_shell_lowbound = sub_shell_highbound;
175  sub_shell_highbound += sub_shell.size();
176 
177  std::size_t shell_lowbound = 0;
178  std::size_t shell_highbound = 0;
179 
180  bool find = false;
181 
182  // locate the position in shells
183  for (std::size_t j = 0; j < n_shells; j++) {
184  auto &shell = shells[j];
185  shell_lowbound = shell_highbound;
186  shell_highbound += shell.size();
187 
188  if (sub_shell == shell) {
189  find = true;
190 
191  // fill in the value
192  const std::size_t gap = sub_shell_lowbound - shell_lowbound;
193  for (std::size_t k = shell_lowbound; k < shell_highbound; k++) {
194  result[k] = gap + k + 1;
195  }
196 
197  break;
198  }
199  }
200 
201  if (find == false) {
202  throw std::runtime_error("\n Not a Sub Basis! \n");
203  }
204  }
205 
206  return result;
207 }

260 void AtomicBasis::compute_shell_to_atom() {
261  const auto ncluster = shells_.size();
262  shell_to_atom_.resize(ncluster);
263 
264  auto atoms = molecule_->atoms();
265  const auto natoms = atoms.size();
266  size_t current = 0;
267  for (size_t cluster = 0; cluster != ncluster; ++cluster) {
268  for (const auto &shell : shells_[cluster]) {
269  while (!detail::equal(shell.O,atoms[current].center()) && current != natoms) {
270  ++current;
271  }
272  if (current == natoms)
273  throw ProgrammingError("invalid shells_ in AtomicBasis", __FILE__, __LINE__);
274  shell_to_atom_[cluster].push_back(current);
275  }
276  }
277  assert(current == natoms-1);
278 }
