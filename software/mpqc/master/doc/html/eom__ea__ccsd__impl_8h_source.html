13 template <typename Tile, typename Policy>
14 void EOM_EA_CCSD<Tile, Policy>::evaluate(ExcitationEnergy* ex_energy) {
15  auto target_precision = ex_energy->target_precision(0);
16  if (vector_threshold_ == 0.0) {
17  vector_threshold_ = 10 * target_precision;
18  }
19  if (!this->computed()) {
20  auto& world = this->wfn_world()->world();
21 
22  auto ccsd_energy =
23  std::make_shared<Energy>(this->shared_from_this(), target_precision);
24  // do CCSD energy
25  CCSD<Tile, Policy>::evaluate(ccsd_energy.get());
26 
27  auto time0 = mpqc::fenced_now(world);
28 
29  ExEnv::out0() << indent << "\nEOM-EA-CCSD Electron Affinity\n";
30  auto n_roots = ex_energy->n_roots();
31  ExEnv::out0() << indent << "Number of roots: " << n_roots << "\n";
32  ExEnv::out0() << indent << "Max number of vector per root: " << max_vector_
33  << "\n";
34  ExEnv::out0() << indent
35  << "Threshold for norm of new vector: " << vector_threshold_
36  << "\n";
37 
38  // initialize guest vector
39  ExEnv::out0() << indent << "\nInitialize Guess Vector \n";
40  auto C = init_guess_vector(n_roots);
41 
42  // initialize intermediates
43  ExEnv::out0() << indent << "\nInitialize Intermediates \n";
44  auto imds = cc::compute_eom_intermediates(this->lcao_factory(),
45  this->ao_factory(), this->t2(),
46  this->t1(), this->is_df(), "ea");
47 
48  auto max_iter = this->max_iter_;
49  auto result =
50  ea_eom_ccsd_davidson_solver(C, imds, max_iter, target_precision);
51 
52  this->computed_ = true;
53  ExcitationEnergy::Provider::set_value(
54  ex_energy, std::vector<numeric_type>(result.data(),
55  result.data() + result.size()));
56 
57  auto time1 = mpqc::fenced_now(world);
58  ExEnv::out0() << "EOM-EA-CCSD Total Time: "
59  << mpqc::duration_in_s(time0, time1) << " S\n";
60  }
61 }

63 template <typename Tile, typename Policy>
64 std::vector<typename EOM_EA_CCSD<Tile, Policy>::GuessVector>
65 EOM_EA_CCSD<Tile, Policy>::init_guess_vector(std::size_t n_roots) {
66  auto& world = this->wfn_world()->world();
67 
68  std::vector<typename EOM_EA_CCSD<Tile, Policy>::GuessVector> guess_vector(
69  n_roots);
70 
71  // occ range
72  auto range_o = this->lcao_factory()
73  .orbital_registry()
74  .retrieve(OrbitalIndex(L"i"))
75  .trange();
76 
77  // unocc range
78  auto range_v = this->lcao_factory()
79  .orbital_registry()
80  .retrieve(OrbitalIndex(L"a"))
81  .trange();
82 
83  TA::TiledRange trange_ca = {range_v};
84  TA::TiledRange trange_cabi = {range_v, range_v, range_o};
85 
86 #if TA_DEFAULT_POLICY == 1
87  // making fake shape of 1
88  TA::SparseShape<float> shape_v(TA::Tensor<float>(trange_ca.tiles_range(), 1),
89  trange_ca);
90  TA::SparseShape<float> shape_vvo(
91  TA::Tensor<float>(trange_cabi.tiles_range(), 0.0), trange_cabi);
92 #endif
93 
94  for (std::size_t i = 0; i < n_roots; ++i) {
95  // Ca
96  guess_vector[i] = ::mpqc::cc::TPack<TArray>(2);
97 
98 #if TA_DEFAULT_POLICY == 0
99  guess_vector[i][0] = TA::DistArray<Tile, Policy>(world, trange_ca);
100 #elif TA_DEFAULT_POLICY == 1
101  guess_vector[i][0] = TA::DistArray<Tile, Policy>(world, trange_ca, shape_v);
102 #endif
103 
104  guess_vector[i][0].fill(0.0);
105  // make unit vector
106  std::vector<std::size_t> element_idx = {i};
107 
108  auto tile_idx = trange_ca.element_to_tile(element_idx);
109 
110  if (guess_vector[i][0].is_local(tile_idx)) {
111  auto tile = guess_vector[i][0].find(tile_idx).get();
112  tile[element_idx] = numeric_type(1.0);
113  }
114  guess_vector[i][0].truncate();
115 
116  // Cabi
117 #if TA_DEFAULT_POLICY == 0
118  guess_vector[i][1] = TA::DistArray<Tile, Policy>(world, trange_cabi);
119 #elif TA_DEFAULT_POLICY == 1
120  guess_vector[i][1] =
121  TA::DistArray<Tile, Policy>(world, trange_cabi, shape_vvo);
122 #endif
123  guess_vector[i][1].fill(0.0);
124  guess_vector[i][1].truncate();
125  }
126  return guess_vector;
127 }

129 template <typename Tile, typename Policy>
130 EigenVector<typename Tile::numeric_type>
131 EOM_EA_CCSD<Tile, Policy>::ea_eom_ccsd_davidson_solver(
132  std::vector<typename EOM_EA_CCSD<Tile, Policy>::GuessVector>& C,
133  const cc::Intermediates<TA::DistArray<Tile, Policy>>& imds,
134  std::size_t max_iter, double convergence) {
135  std::size_t n_roots = C.size();
136 
138  std::shared_ptr<DavidsonDiagPred<GuessVector>> pred;
139  {
140  EigenVector<numeric_type> eps_o =
141  math::array_to_eigen(imds.FIJ).diagonal();
142  EigenVector<numeric_type> eps_v =
143  math::array_to_eigen(imds.FAB).diagonal();
144 
145  pred = std::make_shared<cc::EAPred<TArray>>(eps_o, eps_v);
146  }
147 
149  auto op = [this, &imds](const std::vector<GuessVector>& vec) {
150  std::size_t dim = vec.size();
151  // ExEnv::out0() << "vector dimension: " << dim << std::endl;
152 
153  // compute product of H with guess vector
154  std::vector<GuessVector> HC(dim);
155  for (std::size_t i = 0; i < dim; ++i) {
156  if (vec[i][0].is_initialized() && vec[i][1].is_initialized()) {
157  HC[i] = ::mpqc::cc::TPack<TArray>(2);
158  HC[i][0] = compute_HS1(vec[i][0], vec[i][1], imds);
159  HC[i][1] = compute_HS2(vec[i][0], vec[i][1], imds);
160 
161  } else {
162  throw ProgrammingError("Guess Vector not initialized", __FILE__,
163  __LINE__);
164  }
165  }
166 
167  return HC;
168 
169  };
170 
172  DavidsonDiag<GuessVector> dvd(n_roots, false, 2, max_vector_,
173  vector_threshold_);
174 
175  EigenVector<numeric_type> eig = EigenVector<numeric_type>::Zero(n_roots);
176 
177  eig = dvd.solve(C, op, pred.get(), convergence, max_iter);
178 
179  ExEnv::out0() << "\n";
180  util::print_excitation_energy(eig, false);
181 
182  return eig;
183 }

185 template <typename Tile, typename Policy>
186 TA::DistArray<Tile, Policy> EOM_EA_CCSD<Tile, Policy>::compute_HS1(
187  const TArray& Ca, const TArray& Cabi,
188  const cc::Intermediates<TA::DistArray<Tile, Policy>>& imds) {
189  TArray HS1;
190 
191  {
192  HS1("a") = imds.FAB("a,c") * Ca("c");
193 
194  HS1("a") +=
195  imds.FIA("k,c") * (2.0 * Cabi("a,c,k") - Cabi("c,a,k")) +
196  (2.0 * imds.Waibc("a,k,b,c") - imds.Waibc("a,k,c,b")) * Cabi("b,c,k");
197  }
198  return HS1;
199 }

201 template <typename Tile, typename Policy>
202 TA::DistArray<Tile, Policy> EOM_EA_CCSD<Tile, Policy>::compute_HS2(
203  const TArray& Ca, const TArray& Cabi,
204  const cc::Intermediates<TA::DistArray<Tile, Policy>>& imds) {
205  TArray HS2;
206  {
207  HS2("a,b,i") = imds.Wabci("a,b,c,i") * Ca("c");
208 
209  HS2("a,b,i") +=
210  imds.FAB("a,c") * Cabi("c,b,i") + imds.FAB("b,c") * Cabi("a,c,i") -
211  imds.FIJ("k,i") * Cabi("a,b,k") +
212  (2 * imds.Wiabj("k,b,c,i") - imds.Wiajb("k,b,i,c")) * Cabi("a,c,k") -
213  imds.Wiajb("k,a,i,c") * Cabi("c,b,k") -
214  imds.Wiabj("k,b,c,i") * Cabi("c,a,k") +
215  imds.Wabcd("a,b,c,d") * Cabi("c,d,i") -
216  (2 * imds.Wijab("k,l,c,d") - imds.Wijab("k,l,d,c")) * Cabi("c,d,l") *
217  this->t2()("a,b,k,i");
218  }
219  return HS2;
220 }
