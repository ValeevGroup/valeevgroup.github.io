35 void OrbitalIndex::init(const std::wstring &string) {
36  const wchar_t *letter = string.c_str();
37 
38  auto length = wcslen(letter);
39 
40  // single letter case, for example i, a , p
41 
42  const auto first = letter[0];
43  if (length == 1) {
44  index_ = wchar_to_index(first);
45  }
46  // letter with ' or letter with number, number is ignored
47  // i1, a2, or A', a', P'
48  else if (length == 2) {
49  if (letter[1] == L'\'') {
50  index_ = wchar_with_prime_to_index(first);
51  } else if (isdigit(letter[1])) {
52  index_ = wchar_to_index(first);
53  } else {
54  std::string error_message =
55  "Wrong Key Index " + utility::to_string(string) + " ! \n";
56  throw std::runtime_error(error_message);
57  }
58  }
59  // letter with number and prime
60  // A'1, a'1, P'12, i12
61  else if (wcslen(letter) >= 3) {
62  // with prime
63  if (letter[1] == L'\'') {
64  // check number
65  for (auto i = 2ul; i < length; i++) {
66  TA_ASSERT(isdigit(letter[i]));
67  }
68 
69  index_ = wchar_with_prime_to_index(first);
70  }
71  // without prime
72  else if (isdigit(letter[1])) {
73  // check number
74  for (auto i = 1ul; i < length; i++) {
75  TA_ASSERT(isdigit(letter[i]));
76  }
77 
78  index_ = wchar_to_index(first);
79  } else {
80  std::string error_message =
81  "Wrong Key Index " + utility::to_string(string) + " ! \n";
82  throw std::runtime_error(error_message);
83  }
84  }
85 }

118 OrbitalIndex::Type OrbitalIndex::wchar_to_index(const wchar_t first) {
119  if (first >= occ_wchar[0] && first <= occ_wchar[1]) {
120  return Type::occ;
121  } else if (first >= corr_occ_wchar[0] && first <= corr_occ_wchar[1]) {
122  return Type::corr_occ;
123  } else if (first >= active_wchar[0] && first <= active_wchar[1]) {
124  return Type::active;
125  } else if (first >= virt_wchar[0] && first <= virt_wchar[1]) {
126  return Type::virt;
127  } else if (first >= any_wchar[0] && first <= any_wchar[1]) {
128  return Type::any;
129  } else if (first >= obs_wchar[0] && first <= obs_wchar[3]) {
130  return Type::obs;
131  } else if (first >= vbs_wchar[0] && first <= vbs_wchar[3]) {
132  return Type::vbs;
133  } else if (first >= abs_wchar[0] && first <= abs_wchar[3]) {
134  return Type::abs;
135  } else if (first >= dfbs_wchar[0] && first <= dfbs_wchar[3]) {
136  return Type::dfbs;
137  } else if (first >= ribs_wchar[0] && first <= ribs_wchar[3]) {
138  return Type::ribs;
139  } else if (first == ubs_wchar[0]) {
140  return Type::ubs;
141  } else {
142  std::string error_message = "Wrong Key Index " +
143  utility::to_string(std::wstring(1, first)) +
144  " ! \n";
145  throw std::runtime_error(error_message);
146  return Type();
147  }
148 }

204 std::string OrbitalIndex::to_ta_expression() const {
205  std::string ta_expression;
206  int length;
207  const wchar_t *pt;
208  pt = name_.c_str();
209  char buffer[MB_CUR_MAX];
210 
211  while (*pt) {
212  length = std::wctomb(buffer, *pt);
213  TA_USER_ASSERT(
214  length != -1,
215  "OrbitalIndex::to_ta_expression encountered invalid character");
216  // multiple byte, convert to english name
217  if (length > 1) {
218  auto pos = greek_to_english_name.find(*pt);
219  if (pos == greek_to_english_name.end()) {
220  throw std::runtime_error(
221  "Couldn't Find The English Name of Greek Letter");
222  }
223  std::string convert_name = pos->second;
224  ta_expression.append(convert_name.begin(), convert_name.end());
225  }
226  // single byte, do nothing
227  else if (length == 1) {
228  ta_expression.push_back(*buffer);
229  }
230  ++pt;
231  }
232  return ta_expression;
233 }
