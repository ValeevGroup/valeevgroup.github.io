105 template <typename Tile, typename Policy>
106 TA::DistArray<Tile, typename std::enable_if<std::is_same<Policy, TA::SparsePolicy>::value, TA::SparsePolicy>::type >
107 eigen_to_array(
108  madness::World &world, Matrix<typename Tile::numeric_type> const &M,
109  TA::TiledRange1 tr0, TA::TiledRange1 tr1, double cut = 1e-7) {
110 
111  // make sure dimensions of M match the dimensions of tiled ranges
112  assert(std::size_t(M.rows()) == tr0.extent() && "eigen_to_array(): row dimensions do not match");
113  assert(std::size_t(M.cols()) == tr1.extent() && "eigen_to_array(): col dimensions do not match");
114 
115  TA::TiledRange trange{tr0, tr1};
116  TA::Tensor<float> norms(
117  trange.tiles_range(),
118  trange.elements_range().volume() / trange.tiles_range().volume());
119 
120  TA::SparseShape<float> shape(world, norms, trange);
121  TA::DistArray<Tile, TA::SparsePolicy> array(world, trange, shape);
122 
123  auto const &pmap = array.pmap();
124  const auto end = pmap->end();
125  for (auto it = pmap->begin(); it != end; ++it) {
126  if (!array.is_zero(*it)) {
127  auto range = trange.make_tile_range(*it);
128  madness::Future<Tile> tile =
129  world.taskq.add(mat_to_tile<Tile>, range, &M, cut);
130  array.set(*it, tile);
131  }
132  }
133 
134  world.gop.fence(); // Be careful because M is ref
135  array.truncate(); // Be lazy and fix shape like this.
136  return array;
137 }
