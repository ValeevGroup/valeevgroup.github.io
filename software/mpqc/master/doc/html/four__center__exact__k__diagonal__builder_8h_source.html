366  const auto& trange1 = trange_D_.dim(0);

614  shellpair_list_t compute_shellpair_list(
615  const ShellVec& shv1,
616  const ShellVec& _shv2 = std::vector<Shell>({Shell()}),
617  double threshold = 1e-12) {
618  const ShellVec& shv2 =
619  ((_shv2.size() == 1 && _shv2[0] == Shell()) ? shv1 : _shv2);
620  const auto nsh1 = shv1.size();
621  const auto nsh2 = shv2.size();
622  const auto shv1_equiv_shv2 = (&shv1 == &shv2);
623 
624  // determine max # of primitives in a shell cluster
625  auto max_nprim = [](const ShellVec& shv) {
626  size_t n = 0;
627  for (auto shell : shv) n = std::max(shell.nprim(), n);
628  return n;
629  };
630  const auto max_nprim_1 = max_nprim(shv1);
631  const auto max_nprim_2 = max_nprim(shv2);
632 
633  // determine max angular momentum of a shell cluster
634  auto max_l = [](const ShellVec& shv) {
635  int l = 0;
636  for (auto shell : shv)
637  for (auto c : shell.contr) l = std::max(c.l, l);
638  return l;
639  };
640  const auto max_l_1 = max_l(shv1);
641  const auto max_l_2 = max_l(shv2);
642 
643  // initialize libint2 engine
644  auto engine = libint2::Engine(libint2::Operator::overlap,
645  std::max(max_nprim_1, max_nprim_2),
646  std::max(max_l_1, max_l_2), 0);
647  const auto& buf = engine.results();
648  shellpair_list_t result;
649 
650  // compute non-negligible shell-pair list
651  for (auto s1 = 0ul, s12 = 0ul; s1 != nsh1; ++s1) {
652  result.insert(std::make_pair(s1, std::vector<size_t>()));
653  auto n1 = shv1[s1].size();
654 
655  auto s2_max = shv1_equiv_shv2 ? s1 : nsh2 - 1;
656  for (auto s2 = 0ul; s2 <= s2_max; ++s2, ++s12) {
657  auto on_same_center = (shv1[s1].O == shv2[s2].O);
658  bool significant = on_same_center;
659  if (!on_same_center) {
660  auto n2 = shv2[s2].size();
661  engine.compute(shv1[s1], shv2[s2]);
662  Eigen::Map<const RowMatrixXd> buf_mat(buf[0], n1, n2);
663  auto norm = buf_mat.norm();
664  significant = (norm >= threshold);
665  }
666 
667  if (significant) result[s1].emplace_back(s2);
668  }
669  }
670 
671  // resort shell list in increasing order
672  for (auto s1 = 0ul; s1 != nsh1; ++s1) {
673  auto& list = result[s1];
674  std::sort(list.begin(), list.end());
675  }
676 
677  return result;
678  }
