171 void CCSD_F12<Tile>::compute_f12() {
172  auto lazy_two_electron_int = this->get_direct_ao_integral();
173  Matrix Eij_F12;
174  if (method_ == "standard") {
175  Eij_F12 = compute_ccsd_f12(lazy_two_electron_int);
176  } else if (method_ == "df" || method_ == "direct_df") {
177  Eij_F12 = compute_ccsd_f12_df(lazy_two_electron_int, approximation_);
178  }
179 
180  f12_energy_ = Eij_F12.sum();

183 template <typename Tile>
184 typename CCSD_F12<Tile>::Matrix CCSD_F12<Tile>::compute_ccsd_f12_df(
185  const DirectArray& darray, const char approach) {
186  auto& lcao_factory = this->lcao_factory();
187  auto& ao_factory = this->ao_factory();
188  auto& world = lcao_factory.world();
189  Matrix Eij_F12;
190 
191  utility::print_par(world, "\n Computing CCSD_F12 ", approach, " Approach \n");
192 
193  auto n_active_occ = this->trange1_engine()->get_active_occ();
194 
195  // create shape
196  auto occ_tr1 = this->trange1_engine()->get_active_occ_tr1();
197  TiledArray::TiledRange occ4_trange({occ_tr1, occ_tr1, occ_tr1, occ_tr1});
198  auto ijij_ijji_shape = f12::make_ijij_ijji_shape(occ4_trange);
199 
200  // compute B term
201  {
202  TArray B_ijij_ijji;
203 
204  if (approach == 'C') {
205  B_ijij_ijji = f12::compute_B_ijij_ijji_C_df(lcao_factory, ao_factory,
206  ijij_ijji_shape);
207  } else if (approach == 'D') {
208  B_ijij_ijji = f12::compute_B_ijij_ijji_D_df(lcao_factory, ao_factory,
209  ijij_ijji_shape);
210  }
211 
212  Matrix eij = B_ijij_ijji("i1,j1,i2,j2")
213  .reduce(f12::F12PairEnergyReductor<Tile>(
214  f12::CC_ijij_bar, f12::CC_ijji_bar, n_active_occ));
215  utility::print_par(world, "E_B: ", eij.sum(), "\n");
216  Eij_F12 = eij;
217  }
218 
219  ao_factory.purge_operator(L"R");
220 
221  // compute X term
222  TArray X_ijij_ijji =
223  f12::compute_X_ijij_ijji_df(lcao_factory, ao_factory, ijij_ijji_shape);
224 
225  lcao_factory.purge_operator(L"R2");
226 
227  auto Fij = lcao_factory.compute(L"<i|F|j>[df]");
228  auto Fij_eigen = math::array_to_eigen(Fij);
229  f12::convert_X_ijkl(X_ijij_ijji, Fij_eigen);
230  {
231  Matrix eij = X_ijij_ijji("i1,j1,i2,j2")
232  .reduce(f12::F12PairEnergyReductor<Tile>(
233  f12::CC_ijij_bar, f12::CC_ijji_bar, n_active_occ));
234  eij *= -1;
235  utility::print_par(world, "E_X: ", eij.sum(), "\n");
236  Eij_F12 += eij;
237  }
238 
239  // compute V_ijij_ijji
240  TArray V_ijij_ijji =
241  f12::compute_V_ijij_ijji_df(lcao_factory, ao_factory, ijij_ijji_shape);
242 
243  // VT2 contribution
244  if (darray.array().is_initialized()) {
245  TArray tmp = f12::compute_VT2_ijij_ijji_df_direct(
246  lcao_factory, ao_factory, this->t2(), ijij_ijji_shape, darray, vt_cabs_);
247  V_ijij_ijji("i1,j1,i2,j2") += tmp("i1,j1,i2,j2");
248  } else {
249  TArray tmp = f12::compute_VT2_ijij_ijji_df(lcao_factory, this->t2(),
250  ijij_ijji_shape, vt_cabs_);
251  V_ijij_ijji("i1,j1,i2,j2") += tmp("i1,j1,i2,j2");
252  }
253 
254  // VT1 contribution
255  {
256  TArray tmp = f12::compute_VT1_ijij_ijji_df(
257  lcao_factory, ao_factory, this->t1(), ijij_ijji_shape, vt_cabs_);
258  V_ijij_ijji("i1,j1,i2,j2") += tmp("i1,j1,i2,j2");
259  }
260 
261  // V contribution to energy
262  Matrix e_ij =
263  V_ijij_ijji("i1,j1,i2,j2")
264  .reduce(f12::F12PairEnergyReductor<Tile>(
265  2 * f12::C_ijij_bar, 2 * f12::C_ijji_bar, n_active_occ));
266  Eij_F12 += e_ij;
267  utility::print_par(world, "E_V: ", e_ij.sum(), "\n");
268 
269  return Eij_F12;
270 }
