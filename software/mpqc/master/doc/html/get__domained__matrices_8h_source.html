24 inline void write_local_mat_tile(RowMatrixXd* Mptr, double const* tptr,
25  std::pair<int, int> mpair, std::pair<int, int> npair,
26  std::atomic<int>* counter) {
27  RowMatrixXd& M = *Mptr;
28  const auto m_start = mpair.first;
29  const auto m_size = mpair.second;
30  const auto m_end = m_start + m_size;
31  const auto n_start = npair.first;
32  const auto n_size = npair.second;
33  for (auto m = m_start; m < m_end; ++m) {
34  double* data = &M(m, n_start);
35  std::copy(tptr, tptr + n_size, data);
36  tptr += n_size;
37  }
38  ++(*counter);
39 }

126  void fill_matrix_map(TA::DistArray<T, P> const& Mat,
127  IdxMap<2, TA::Tensor<double>*>& Tmap,
128  bool lower_triangle = false) {
129  const auto ext = Mat.trange().tiles_range().extent_data();
130  const auto nK = ext[0];
131  const auto nL = ext[1];
132 
133  for (auto K = 0; K < nK; ++K) {
134  if (lower_triangle) {
135  for (auto L = 0; L <= K; ++L) {
136  if (!Mat.is_zero({K, L})) {
137  Tmap[{{K, L}}] = &(Mat.find({K, L}).get());
138  }
139  }
140  } else {
141  for (auto L = 0; L < nL; ++L) {
142  if (!Mat.is_zero({K, L})) {
143  Tmap[{{K, L}}] = &(Mat.find({K, L}).get());
144  }
145  }
146  }
147  }
148  }

150  void fill_Z_map() {
151  const auto ext = Z_.trange().tiles_range().extent_data();
152  const auto nX = ext[0];
153  const auto nI = ext[1];
154  const auto nA = ext[2];
155 
156  for (auto X = 0; X < nX; ++X) {
157  for (auto I = 0; I < nI; ++I) {
158  for (auto A = 0; A < nA; ++A) {
159  if (Z_.is_local({X, I, A}) && !Z_.is_zero({X, I, A})) {
160  localZtiles_[{{X, I, A}}] = &(Z_.find({X, I, A}).get());
161  }
162  }
163  }
164  }
165  }

168  MatrixGetter(TA::DistArray<T, P>& S, TA::DistArray<T, P>& F,
169  TA::DistArray<T, P>& M, TA::DistArray<T, P>& Z)
170  : world_(Z.world()), Z_(Z) {
171  // Ensure matrices are replicated
172  S.make_replicated();
173  F.make_replicated();
174  M.make_replicated();
175 
176  // Get the TiledRange1 objects
177  auto const& Ztrange = Z.trange();
178  auto const& X_tr1 = Ztrange.data()[0];
179  auto const& I_tr1 = Ztrange.data()[1];
180  auto const& A_tr1 = Ztrange.data()[2];
181 
182  const auto X_tiles = Ztrange.tiles_range().extent_data()[0];
183  const auto I_tiles = Ztrange.tiles_range().extent_data()[1];
184  const auto A_tiles = Ztrange.tiles_range().extent_data()[2];
185 
186  aux_dim_sizes_.reserve(X_tiles);
187  for (auto X = 0; X < X_tiles; ++X) {
188  const auto size_pair = *(X_tr1.begin() + X);
189  const auto size = size_pair.second - size_pair.first;
190  aux_dim_sizes_.emplace_back(size);
191  }
192 
193  occ_dim_sizes_.reserve(I_tiles);
194  for (auto I = 0; I < I_tiles; ++I) {
195  const auto size_pair = *(I_tr1.begin() + I);
196  const auto size = size_pair.second - size_pair.first;
197  occ_dim_sizes_.emplace_back(size);
198  }
199 
200  pao_dim_sizes_.reserve(A_tiles);
201  for (auto A = 0; A < A_tiles; ++A) {
202  const auto size_pair = *(A_tr1.begin() + A);
203  const auto size = size_pair.second - size_pair.first;
204  pao_dim_sizes_.emplace_back(size);
205  }
206 
207  auto lower_triangle = true;
208  fill_matrix_map(S, Stiles_, lower_triangle);
209  fill_matrix_map(M, Mtiles_, lower_triangle);
210 
211  lower_triangle = false;
212  fill_matrix_map(F, Ftiles_, lower_triangle);
213 
214  fill_Z_map();
215  }

217  // Only write the lower triangle of M and S, but full F
218  std::array<RowMatrixXd, 4> get_ZMFS(SparseMap const& IJX,
219  SparseMap const& IJA, int I, int J) {
220  TileToEl aux_t2e;
221  const auto nX = aux_dim_sizes_.size();
222  auto nX_els = 0;
223  for (auto X = 0; X < nX; ++X) {
224  if (IJX(X) > 0) {
225  aux_t2e.emplace_back(std::make_pair(X, nX_els));
226  nX_els += aux_dim_sizes_[X];
227  }
228  }
229 
230  const auto I_size = occ_dim_sizes_[I];
231  const auto J_size = occ_dim_sizes_[J];
232  const auto IJsize = (I == J) ? I_size : I_size + J_size;
233 
234  TileToEl pao_t2e;
235  const auto nA = pao_dim_sizes_.size();
236  auto nA_els = 0;
237  for (auto A = 0; A < nA; ++A) {
238  if (IJA(A) > 0) {
239  pao_t2e.emplace_back(std::make_pair(A, nA_els));
240  nA_els += pao_dim_sizes_[A];
241  }
242  }
243 
244  // Zero M so we don't have to submit tasks for the upper triangle
245  RowMatrixXd M(nX_els, nX_els);
246  M.setZero();
247 
248  // Don't zero Z
249  RowMatrixXd Z(nX_els, IJsize * nA_els);
250 
251  // Zero S, but not F
252  RowMatrixXd F(nA_els, nA_els);
253  RowMatrixXd S(nA_els, nA_els);
254  S.setZero();
255 
256  auto Send = Stiles_.end();
257  auto Fend = Ftiles_.end();
258  auto Mend = Mtiles_.end();
259  auto Z_dist_end = nonlocalZtiles_.end();
260 
261  std::atomic<int> counter(0);
262  int ntasks = 0;
263  const auto I_start = 0; // In the matrix I always starts at zero
264  const auto J_start = I_size; // J comes after I
265  const auto Ipair = std::make_pair(I_start, I_size);
266  const auto Jpair = std::make_pair(J_start, J_size);
267 
268  for (auto const& Xt2e : aux_t2e) {
269  const auto X = Xt2e.first;
270  const auto Xpair = std::make_pair(Xt2e.second, aux_dim_sizes_[X]);
271 
272  // Loop for Second M tile
273  for (auto const& Yt2e : aux_t2e) {
274  const auto Y = Yt2e.first;
275  if (Y > X) break;
276  const auto Ypair = std::make_pair(Yt2e.second, aux_dim_sizes_[Y]);
277 
278  auto Mtile = Mtiles_.find({{X, Y}});
279  if (Mtile != Mend) {
280  const auto tile_ptr = Mtile->second;
281  world_.taskq.add(tasks::write_local_mat_tile, &M, tile_ptr->data(),
282  Xpair, Ypair, &counter);
283  ++ntasks;
284  }
285  }
286 
287  for (auto const& At2e : pao_t2e) {
288  const auto A = At2e.first;
289  const auto Apair = std::make_pair(At2e.second, pao_dim_sizes_[A]);
290 
291  // Write Z(X,I,A) tile
292  if (!Z_.is_zero({X, I, A})) { // We have data to write
293  if (Z_.is_local({X, I, A})) { // Write Z from out stored ptr
294  const auto tile_ptr = localZtiles_.at({{X, I, A}});
295  world_.taskq.add(tasks::write_local_Z_tile, &Z, tile_ptr->data(),
296  Xpair, Ipair, Apair, nA_els, &counter);
297  ++ntasks;
298  } else { // Have to do some caching work
299  auto ZXIA_iter = nonlocalZtiles_.find({{X, I, A}});
300  // First check if there is a future to this tile
301  if (ZXIA_iter == nonlocalZtiles_.end()) { // No future
302  auto ZXIA_future = Z_.find({X, I, A});
303  std::array<int, 3> XIA{{X, I, A}};
304 
305  // Grab the iterator after we insert
306  ZXIA_iter =
307  nonlocalZtiles_
308  .insert(std::make_pair(XIA, std::move(ZXIA_future)))
309  .first;
310  }
311  // get a reference to the future
312  auto& ZXIA_fut = ZXIA_iter->second;
313 
314  // Check if the future is ready if so call fast function
315  if (ZXIA_fut.probe()) {
316  world_.taskq.add(tasks::write_local_Z_tile, &Z,
317  ZXIA_fut.get().data(), Xpair, Ipair, Apair,
318  nA_els, &counter);
319  ++ntasks;
320  } else { // Submit task with the future calls malloc on tile copy
321  world_.taskq.add(tasks::write_remote_Z_tile, &Z, ZXIA_fut, Xpair,
322  Ipair, Apair, nA_els, &counter);
323  ++ntasks;
324  }
325  }
326  } else { // Ztile was zero
327  world_.taskq.add(tasks::write_zero_Z_tile, &Z, Xpair, Ipair, Apair,
328  nA_els, &counter);
329  ++ntasks;
330  }
331  if (I != J) {
332  if (!Z_.is_zero({X, J, A})) {
333  if (Z_.is_local({X, J, A})) { // Write Z from out stored ptr
334  const auto tile_ptr = localZtiles_.at({{X, J, A}});
335  world_.taskq.add(tasks::write_local_Z_tile, &Z, tile_ptr->data(),
336  Xpair, Jpair, Apair, nA_els, &counter);
337  ++ntasks;
338  } else { // Have to do some caching work
339  auto ZXJA_iter = nonlocalZtiles_.find({{X, J, A}});
340  // First check if there is a future to this tile
341  if (ZXJA_iter == nonlocalZtiles_.end()) { // No future
342  auto ZXJA_future = Z_.find({X, J, A});
343  std::array<int, 3> XJA{{X, J, A}};
344 
345  // Grab the iterator after we insert
346  ZXJA_iter =
347  nonlocalZtiles_
348  .insert(std::make_pair(XJA, std::move(ZXJA_future)))
349  .first;
350  }
351  // get a reference to the future
352  auto& ZXJA_fut = ZXJA_iter->second;
353 
354  // Check if the future is ready if so call fast function
355  if (ZXJA_fut.probe()) {
356  world_.taskq.add(tasks::write_local_Z_tile, &Z,
357  ZXJA_fut.get().data(), Xpair, Jpair, Apair,
358  nA_els, &counter);
359  ++ntasks;
360  } else { // Submit task with the future calls malloc on tile copy
361  world_.taskq.add(tasks::write_remote_Z_tile, &Z, ZXJA_fut,
362  Xpair, Jpair, Apair, nA_els, &counter);
363  ++ntasks;
364  }
365  }
366  } else { // Ztile was zero
367  world_.taskq.add(tasks::write_zero_Z_tile, &Z, Xpair, Jpair, Apair,
368  nA_els, &counter);
369  ++ntasks;
370  }
371  }
372 
373  for (auto const& Bt2e : pao_t2e) {
374  const auto B = Bt2e.first;
375  const auto Bpair = std::make_pair(Bt2e.second, pao_dim_sizes_[B]);
376 
377  auto Ftile = Ftiles_.find({{A, B}});
378  if (Ftile != Fend) {
379  const auto tile_ptr = Ftile->second;
380  world_.taskq.add(tasks::write_local_mat_tile, &F, tile_ptr->data(),
381  Apair, Bpair, &counter);
382  ++ntasks;
383  } else {
384  world_.taskq.add(tasks::write_zero_mat_tile, &F, Apair, Bpair,
385  &counter);
386  ++ntasks;
387  }
388 
389  if (B > A) continue;
390 
391  auto Stile = Stiles_.find({{A, B}});
392  if (Stile != Send) {
393  const auto tile_ptr = Stile->second;
394  world_.taskq.add(tasks::write_local_mat_tile, &S, tile_ptr->data(),
395  Apair, Bpair, &counter);
396  ++ntasks;
397  }
398  }
399  }
400  }
401 
402  world_.await([&counter, ntasks] { return counter == ntasks; });
403 
404  return std::array<RowMatrixXd, 4>{
405  {std::move(Z), std::move(M), std::move(F), std::move(S)}};
406  }
407 };
