129  template <typename Shape, typename ProcMap>
130  madness::Group make_group(const Shape& shape, size_type index,
131  const size_type end, const size_type stride, const size_type max_group_size,
132  const size_type k, const size_type key_offset, const ProcMap& proc_map) const
133  {
134  // Generate the list of processes in rank_row
135  std::vector<ProcessID> proc_list(max_group_size, -1);
136 
137  // Flag the root processes of the broadcast, which may not be included
138  // by shape.
139  size_type p = k % max_group_size;
140  proc_list[p] = proc_map(p);
141  size_type count = 1ul;
142 
143  // Flag all process that have non-zero tiles
144  for(p = 0ul; (index < end) && (count < max_group_size); index += stride,
145  p = (p + 1u) % max_group_size)
146  {
147  if((proc_list[p] != -1) || (shape.is_zero(index))) continue;
148 
149  proc_list[p] = proc_map(p);
150  ++count;
151  }
152 
153  // Remove processes from the list that will not be in the group
154  for(size_type x = 0ul, p = 0ul; x < count; ++p) {
155  if(proc_list[p] == -1) continue;
156  proc_list[x++] = proc_list[p];
157  }
158 
159  // Truncate invalid process id's
160  proc_list.resize(count);
161 
162  return madness::Group(TensorImpl_::get_world(), proc_list,
163  madness::DistributedID(DistEvalImpl_::id(), k + key_offset));
164  }

382  void bcast_col_range_task(size_type k, const size_type end) const {
383  // Compute the first local row of right
384  const size_type Pcols = proc_grid_.proc_cols();
385  k += (Pcols - ((k + Pcols - proc_grid_.rank_col()) % Pcols)) % Pcols;
386 
387  for(; k < end; k += Pcols) {
388 
389  // Compute local iteration limits for column k of left_.
390  size_type index = left_start_local_ + k;
391 
392  // Search column k of left for non-zero tiles
393  for(; index < left_end_; index += left_stride_local_) {
394  if(left_.shape().is_zero(index)) continue;
395 
396  // Construct broadcast group
397  const madness::Group row_group = make_row_group(k);
398  const ProcessID group_root = get_row_group_root(k, row_group);
399 
400  if(row_group.size() > 1) {
401  // Broadcast column k of left_.
402  for(; index < left_end_; index += left_stride_local_) {
403  if(left_.shape().is_zero(index)) continue;
404 
405  // Broadcast the tile
406  const madness::DistributedID key(DistEvalImpl_::id(), index);
407  auto tile = get_tile(left_, index);
408  TensorImpl_::get_world().gop.bcast(key, tile, group_root, row_group);
409  }
410  } else {
411  // Discard column k of left_.
412  for(; index < left_end_; index += left_stride_local_) {
413  if(left_.shape().is_zero(index)) continue;
414  left_.discard(index);
415  }
416  }
417 
418  break;
419  }
420  }
421  }

423  void bcast_row_range_task(size_type k, const size_type end) const {
424  // Compute the first local row of right
425  const size_type Prows = proc_grid_.proc_rows();
426  k += (Prows - ((k + Prows - proc_grid_.rank_row()) % Prows)) % Prows;
427 
428  for(; k < end; k += Prows) {
429 
430  // Compute local iteration limits for row k of right_.
431  size_type index = k * proc_grid_.cols();
432  const size_type row_end = index + proc_grid_.cols();
433  index += proc_grid_.rank_col();
434 
435  // Search for and broadcast non-zero row
436  for(; index < row_end; index += right_stride_local_) {
437  if(right_.shape().is_zero(index)) continue;
438 
439  // Construct broadcast group
440  const madness::Group col_group = make_col_group(k);
441  const ProcessID group_root = get_col_group_root(k, col_group);
442 
443  if(col_group.size() > 1) {
444  // Broadcast row k of right_.
445  for(; index < row_end; index += right_stride_local_) {
446  if(right_.shape().is_zero(index)) continue;
447 
448  // Broadcast the tile
449  const madness::DistributedID key(DistEvalImpl_::id(), index + left_.size());
450  auto tile = get_tile(right_, index);
451  TensorImpl_::get_world().gop.bcast(key, tile, group_root, col_group);
452  }
453  } else {
454  // Broadcast row k of right_.
455  for(; index < row_end; index += right_stride_local_) {
456  if(right_.shape().is_zero(index)) continue;
457  right_.discard(index);
458  }
459  }
460 
461  break;
462  }
463  }

477  size_type iterate_row(size_type k) const {
478  // Iterate over k's until a non-zero tile is found or the end of the
479  // matrix is reached.
480  size_type end = k * proc_grid_.cols();
481  for(; k < k_; ++k) {
482  // Search for non-zero tiles in row k of right
483  size_type i = end + proc_grid_.rank_col();
484  end += proc_grid_.cols();
485  for(; i < end; i += right_stride_local_)
486  if(! right_.shape().is_zero(i))
487  return k;
488  }
489 
490  return k;
491  }

501  size_type iterate_col(size_type k) const {
502  // Iterate over k's until a non-zero tile is found or the end of the

524  size_type iterate_sparse(const size_type k) const {
525  // Initial step for k_col and k_row.
526  size_type k_col = iterate_col(k);
527  size_type k_row = iterate_row(k_col);
528 
529  // Search for a row and column that both have non-zero tiles
530  while(k_col != k_row) {
531  if(k_col < k_row) {
532  k_col = iterate_col(k_row);
533  } else {
534  k_row = iterate_row(k_col);
535  }
536  }
537 
538  if(k < k_row) {
539  // Spawn a task to broadcast any local columns of left that were skipped
540  TensorImpl_::get_world().taskq.add(shared_from_this(),
541  & Summa_::bcast_col_range_task, k, k_row,
542  madness::TaskAttributes::hipri());
543 
544  // Spawn a task to broadcast any local rows of right that were skipped
545  TensorImpl_::get_world().taskq.add(shared_from_this(),
546  & Summa_::bcast_row_range_task, k, k_col,
547  madness::TaskAttributes::hipri());
548  }
549 
550  return k_col;
551  }
