34  template<typename T, unsigned int DIM, typename Tile>
35  elem::DistMatrix<T> array_to_elem(const Array<T,DIM, Tile> &array,
36  const elem::Grid &grid){
37  // Check that the Array is 2-d
38  TA_USER_ASSERT(DIM == 2u,
39  "TiledArray::array_to_elem(): The array dimension must be 2.");
40 
41  // Construct the elemental matrix
42  std::vector<std::size_t> sizes = array.trange().elements().size();
43  elem::DistMatrix<T> mat(sizes[0], sizes[1], grid);
44  elem::Zero(mat);
45 
46  //Create the Axpy interface to fill elemental matrix
47  elem::AxpyInterface<T> interface;
48  // Attach matrix to interface
49  interface.Attach(elem::LOCAL_TO_GLOBAL, mat);
50 
51  // Get array iterators
52  typename Array<T,DIM,Tile>::iterator it = array.begin();
53  typename Array<T,DIM,Tile>::iterator end = array.end();
54 
55  for(; it != end; ++it){
56  // Get tile matrix location info
57  const typename Array<T,DIM,Tile>::value_type tile = *it;
58  const std::size_t t0start = tile.range().start()[0];
59  const std::size_t t1start = tile.range().start()[1];
60  const std::size_t t0size = tile.range().size()[0];
61  const std::size_t t1size = tile.range().size()[1];
62 
63  // Create Eigen RowMajor Map of tile
64  const Eigen::Map<
65  const Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>,
66  Eigen::AutoAlign> eig_row_map = eigen_map(tile, t0size, t1size);
67 
68  // Create ColMajor EigenMatrix from RowMajor Map
69  Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> CMatrix = eig_row_map;
70 
71  // Make Elemental local matrix and attach the data.
72  elem::Matrix<T> ElemBlock;
73  ElemBlock.Attach(CMatrix.rows(), CMatrix.cols(), CMatrix.data(), CMatrix.rows());
74 
75  // Attach elem local matrix to elem global matrix
76  interface.Axpy(1.0, ElemBlock, t0start, t1start);
77  }
78  interface.Detach(); // Does communication using elemental
79 
80  return mat;
81  }

83  template<typename T, unsigned int DIM, typename Tile>
84  void elem_to_array(Array<T,DIM,Tile> &array, elem::DistMatrix<T> &mat){
85  TA_USER_ASSERT(DIM==2u, "TiledArray::elem_to_array(): requires the array to have dimension 2");
86  TA_USER_ASSERT((array.trange().elements().extent()[0]==mat.Height()) &&
87  (array.trange().elements().extent()[1] == mat.Width()),
88  "TiledArray::elem_to_array(): requires the shape of the elem matrix and the array to be the same.");
89 
90  // Make interface and attach mat
91  elem::AxpyInterface<T> interface;
92  interface.Attach(elem::GLOBAL_TO_LOCAL, mat);
93 
94  // Get iterators to array
95  typename Array<T,DIM, Tile>::iterator it = array.begin();
96  typename Array<T,DIM, Tile>::iterator end = array.end();
97 
98  // Loop over tiles and improperly assign the data to them in column major
99  // format.
100  for(;it != end; ++it){
101  // Get tile matrix location info
102  typename Array<T,DIM,Tile>::value_type tile = *it;
103  std::size_t t0start = tile.range().start()[0];
104  std::size_t t1start = tile.range().start()[1];
105  std::size_t t0size = tile.range().size()[0];
106  std::size_t t1size = tile.range().size()[1];
107 
108  // Make Elemental local matrix and attach the data.
109  elem::Matrix<T> ElemBlock;
110  ElemBlock.Attach(t0size, t1size, tile.data(), t0size);
111 
112  // need to zero tile so Axpy doesn't add to it.
113  std::fill(ElemBlock.Buffer(), ElemBlock.Buffer()+t0size*t1size, 0.0);
114 
115  // Attach elem local matrix to elem global matrix
116  interface.Axpy(1.0, ElemBlock, t0start, t1start);
117  }
118  interface.Detach(); // Does communication using elemental
119 
120  // now we have to go back and fix it so the tiles are row major ordered.
121  it = array.begin();
122  for(;it != end; ++it){
123  // Get tile and size
124  typename Array<T,DIM,Tile>::value_type tile = *it;
125  std::size_t t0size = tile.range().size()[0];
126  std::size_t t1size = tile.range().size()[1];
127 
128  // copy to row major matrix
129  Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor> row_mat =
130  Eigen::Map<const Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>,
131  Eigen::AutoAlign > (tile.data(),t0size,t1size);
132 
133  // Finally copy the data back into the tile in the correct format.
134  std::copy(row_mat.data(), row_mat.data()+row_mat.size(), tile.data());
135  }
136  }
