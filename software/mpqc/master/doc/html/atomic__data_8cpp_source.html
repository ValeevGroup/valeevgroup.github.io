23 boost::optional<double> AtomicData::isotope_mass(size_t atomic_number,
24  int64_t mass_number) const {
25  auto range = isotope_data_.equal_range(atomic_number);
26  if (range.second == range.first)
27  throw Uncomputable("did not find isotopes for this atomic number", __FILE__,
28  __LINE__);
29 
30  auto the_iter =
31  (mass_number == -1)
32  ? most_abundant_(range.first, range.second)
33  : std::find_if(range.first, range.second,
34  [=](const std::pair<size_t, Isotope>& a) {
35  assert(a.first == atomic_number);
36  return a.second.mass_number == mass_number;
37  });
38 
39  typedef boost::optional<double> result_t;
40  return (the_iter == range.second) ? result_t()
41  : result_t(the_iter->second.mass);
42 }

58 std::multimap<size_t, AtomicData::Isotope>::const_iterator
59 AtomicData::most_abundant_(
60  const std::multimap<size_t, AtomicData::Isotope>::const_iterator& first,
61  const std::multimap<size_t, AtomicData::Isotope>::const_iterator& second) {
62  return std::max_element(
63  first, second, [](const std::pair<size_t, Isotope>& a,
64  const std::pair<size_t, Isotope>& b) {
65  const auto& a_ab = a.second.abundance;
66  const auto& b_ab = b.second.abundance;
67  if (std::isnan(
68  a_ab)) { // neither has abundance or only b has it? b wins
69  return true;
70  } else {
71  if (std::isnan(b_ab)) // only a has abundance? a wins
72  return false;
73  else
74  return a_ab < b_ab;
75  }
76  });
77 }
