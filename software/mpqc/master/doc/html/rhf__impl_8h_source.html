32 template <typename Tile, typename Policy>
33 RHF<Tile, Policy>::RHF(const KeyVal& kv)
34  : AOWavefunction<Tile, Policy>(kv), kv_(kv) {
35  auto mol = *this->wfn_world()->atoms();
36 
37  // get the molecular charge
38  const auto charge = kv.value<int>("charge", 0);
39  if (mol.total_atomic_number() <= charge)
40  throw InputError(
41  "net charge cannot be greater than the total nuclear charge", __FILE__,
42  __LINE__, "charge");
43  nelectrons_ = mol.total_atomic_number() - charge;
44  if (nelectrons_ % 2 != 0)
45  throw InputError("RHF requires an even number of electrons", __FILE__,
46  __LINE__, "charge");
47 
48  max_iter_ = kv.value<int>("max_iter", 30);
49 
50  density_builder_str_ =
51  kv.value<std::string>("density_builder", "eigen_solve");
52 
53  if (kv.exists("localizer") || (kv.exists("localize") && kv.value<bool>("localize"))) {
54  const bool localizer_given = static_cast<bool>(kv.count("localizer"));
55  if (localizer_given) {
56  localizer_ = kv.class_ptr<OrbitalLocalizer<Tile, Policy>>("localizer", false, true);
57  }
58  if (!localizer_) {
59  localizer_ = std::make_shared<scf::FosterBoysLocalizer<Tile, Policy>>();
60  }
61  if (localizer_)
62  localize_core_ = kv.value<bool>("localize_core", true);
63  } else {
64  const auto localize = kv.value<bool>("localize", false);
65  const auto localization_method =
66  kv.value<std::string>("", "boys-foster", "localization_method");
67  if (localize) {
68  if (localization_method == "boys-foster" || localization_method == "boys-foster(valence)")
69  localizer_ = std::make_shared<scf::FosterBoysLocalizer<Tile,Policy>>(KeyVal{});
70  else if (localization_method == "rrqr" || localization_method == "rrqr(valence)")
71  localizer_ = std::make_shared<scf::RRQRLocalizer<Tile,Policy>>(KeyVal{});
72  if (localization_method.find("valence") != std::string::npos)
73  localize_core_ = false;
74  }
75  }
76 
77  clustered_coeffs_ = kv.value<bool>("clustered_coeffs", false);
78  if (clustered_coeffs_ && localize_core_) {
79  throw InputError("RHF: clustered_coeffs=true is incompatible with localize_core=true", __FILE__,
80  __LINE__, "clustered_coeffs");
81  }
82  t_cut_c_ = kv.value<double>("t_cut_c", 0.0);
83 }

85 template <typename Tile, typename Policy>
86 void RHF<Tile, Policy>::init(const KeyVal& kv) {
87  auto& ao_factory = this->ao_factory();
88  auto& world = this->wfn_world()->world();
89  const auto& mol = *this->wfn_world()->atoms();
90 
91  // Overlap ints
92  S_ = ao_factory.compute(L"<κ|λ>");
93  // H core int
94  H_ = ao_factory.compute(L"<κ|H|λ>");
95 
96  // fock builder
97  init_fock_builder();
98 
99  // emultipole integral TODO better interface to compute this
100  const auto& basis =
101  *this->wfn_world()->basis_registry()->retrieve(OrbitalIndex(L"λ"));
102  const auto bs_array = utility::make_array(basis, basis);
103  auto multi_pool = gaussian::make_engine_pool(
104  libint2::Operator::emultipole1, utility::make_array_of_refs(basis));
105  auto r_xyz =
106  gaussian::xyz_integrals<Tile, Policy>(world, multi_pool, bs_array);
107  if (localizer_)
108  localizer_->initialize(S_, r_xyz);
109 
110  const auto nocc = nelectrons_ / 2;
111  const auto ncore = mol.core_electrons() / 2;
112 
113  // density builder
114  std::size_t n_cluster = mol.nclusters();
115  if (density_builder_str_ == "purification") {
116  auto density_builder = scf::PurificationDensityBuilder<Tile, Policy>(
117  S_, r_xyz, nocc, ncore, n_cluster, t_cut_c_, localizer_,
118  localize_core_,clustered_coeffs_);
119  d_builder_ =
120  std::make_unique<decltype(density_builder)>(std::move(density_builder));
121  } else if (density_builder_str_ == "eigen_solve") {
122  std::string decompo_type =
123  kv.value<std::string>("decompo_type", "conditioned");
124  double s_tolerance = kv.value<double>("s_tolerance", 1e8);
125  auto density_builder = scf::ESolveDensityBuilder<Tile, Policy>(
126  S_, r_xyz, nocc, ncore, n_cluster, t_cut_c_, decompo_type, s_tolerance,
127  localizer_, localize_core_, clustered_coeffs_);
128  d_builder_ =
129  std::make_unique<decltype(density_builder)>(std::move(density_builder));
130  } else {
131  throw std::runtime_error("Unknown DensityBuilder name! \n");
132  }
133 
134  if (!F_.is_initialized()) {
135  // soad
136  F_ = gaussian::fock_from_soad(world, mol, basis, H_);
137  }
138 
139  F_diis_ = F_;
140  compute_density();
141 }

177 template <typename Tile, typename Policy>
178 void RHF<Tile, Policy>::solve(int64_t max_iters, double thresh) {
179  // for now use same precision for energy and orbital gradient
180  const auto target_energy_precision = thresh;
181  const auto target_orbgrad_precision = thresh;
182 
183  auto& world = F_.world();
184 
185  if (world.rank() == 0) {
186  std::cout << "Starting SCF:\n"
187  << "\tThreshold: " << thresh << "\n"
188  << "\tMaximum number of iterations: " << max_iters << "\n";
189  }
190 
191  madness::print_meminfo(world.rank(), "RHF:begin");
192 
193  TiledArray::DIIS<array_type> diis;
194 
195  auto iter = 0;
196  auto error = std::numeric_limits<double>::max();
197  auto rms_error = std::numeric_limits<double>::max();
198  auto old_energy = compute_energy();
199  const double volume = F_.trange().elements_range().volume();
200 
201  while (iter < max_iters &&
202  (target_energy_precision < error ||
203  target_orbgrad_precision < (rms_error / volume))) {
204  auto s0 = mpqc::fenced_now(world);
205 
206  madness::print_meminfo(world.rank(), "RHF:before_fock");
207  build_F();
208  auto b1 = mpqc::fenced_now(world);
209  madness::print_meminfo(world.rank(), "RHF:after_fock");
210 
211  auto current_energy = compute_energy();
212  error = std::abs(old_energy - current_energy);
213  old_energy = current_energy;
214 
215  array_type Grad;
216  Grad("i,j") =
217  F_("i,k") * D_("k,l") * S_("l,j") - S_("i,k") * D_("k,l") * F_("l,j");
218  madness::print_meminfo(world.rank(), "RHF:orbgrad");
219 
220  rms_error = Grad("i,j").norm().get();
221 
222  F_diis_ = F_;
223  diis.extrapolate(F_diis_, Grad);
224  madness::print_meminfo(world.rank(), "RHF:diis");
225 
226  auto d0 = mpqc::fenced_now(world);
227  compute_density();
228  auto s1 = mpqc::fenced_now(world);
229  madness::print_meminfo(world.rank(), "RHF:density");
230 
231  rhf_times_.push_back(mpqc::duration_in_s(s0, s1));
232  d_times_.push_back(mpqc::duration_in_s(d0, s1));
233  build_times_.push_back(mpqc::duration_in_s(s0, b1));
234 
235  if (world.rank() == 0) {
236  std::cout << "iteration: " << iter << "\n"
237  << "\tEnergy: " << old_energy << "\n"
238  << "\tabs(Energy Change): " << error << "\n"
239  << "\t(Gradient Norm)/n^2: " << (rms_error / volume) << "\n"
240  << "\tScf Time: " << rhf_times_.back() << "\n"
241  << "\t\tDensity Time: " << d_times_.back() << "\n"
242  << "\t\tFock Build Time: " << build_times_.back() << "\n";
243  }
244  f_builder_->print_iter("\t\t");
245  d_builder_->print_iter("\t\t");
246  ++iter;
247  }
248 
249  if (iter == max_iters)
250  throw MaxIterExceeded("RHF SCF did not converge", __FILE__, __LINE__,
251  max_iters);
252 
253  // commit the converged state
254  this->computed_ = true;
255  energy_ = old_energy;
256  // store fock matix in registry
257  auto& registry = this->ao_factory().registry();
258  f_builder_->register_fock(F_, registry);
259  // update precision
260  computed_precision_ = thresh;
261 }

306 template <typename Tile, typename Policy>
307 void RHF<Tile, Policy>::evaluate(CanonicalOrbitalSpace<array_type>* result,
308  double target_energy_precision,
309  std::size_t target_blocksize) {
310  if (!can_evaluate(result))
311  throw ProgrammingError(
312  "RHF: canonical orbitals requested, but can't compute them", __FILE__,
313  __LINE__);
314 
315  do_evaluate(target_energy_precision);
316 
317  // Eigen-based density builder provides full set of canonical orbitals, other
318  // build them
319  auto d_eigen_builder_ =
320  dynamic_cast<scf::ESolveDensityBuilder<Tile, Policy>*>(
321  &*d_builder_.get());
322  assert(d_eigen_builder_ != nullptr && !d_eigen_builder_->localize());
323 
324  auto eps = d_eigen_builder_->orbital_energies();
325  auto C = d_eigen_builder_->C();
326 
327  std::vector<double> eps_vec(eps.rows());
328  std::copy(eps.data(), eps.data() + eps.rows(), eps_vec.begin());
329  *result = CanonicalOrbitalSpace<array_type>(OrbitalIndex(L"p"),
330  OrbitalIndex(L"κ"), C, eps_vec);
331 }
