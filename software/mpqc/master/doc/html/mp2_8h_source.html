22 template <typename Tile>
23 struct Mp2Energy {
24  using result_type = double;
25  using argument_type = Tile;
26 
27  std::shared_ptr<const Eigen::VectorXd> vec_;
28  std::size_t n_occ_;
29  std::size_t n_frozen_;
30 
31  Mp2Energy(std::shared_ptr<const Eigen::VectorXd> vec, std::size_t n_occ,
32  std::size_t n_frozen)
33  : vec_(vec), n_occ_(n_occ), n_frozen_(n_frozen) {}
34 
35  Mp2Energy(Mp2Energy const &) = default;
36 
37  result_type operator()() const { return 0.0; }
38 
39  result_type operator()(result_type const &t) const { return t; }
40 
41  void operator()(result_type &me, result_type const &other) const {
42  me += other;
43  }
44 
45  void operator()(result_type &me, argument_type const &tile) const {
46  auto const &range = tile.range();
47  auto const &vec = *vec_;
48  auto const st = range.lobound_data();
49  auto const fn = range.upbound_data();
50  auto tile_idx = 0;
51 
52  auto sti = st[0];
53  auto fni = fn[0];
54  auto stj = st[1];
55  auto fnj = fn[1];
56  auto sta = st[2];
57  auto fna = fn[2];
58  auto stb = st[3];
59  auto fnb = fn[3];
60 
61  for (auto i = sti; i < fni; ++i) {
62  const auto e_i = vec[i + n_frozen_];
63  for (auto j = stj; j < fnj; ++j) {
64  const auto e_ij = e_i + vec[j + n_frozen_];
65  for (auto a = sta; a < fna; ++a) {
66  const auto e_ija = e_ij - vec[a + n_occ_];
67  for (auto b = stb; b < fnb; ++b, ++tile_idx) {
68  const auto e_iajb = e_ija - vec[b + n_occ_];
69  me += 1.0 / (e_iajb)*tile.data()[tile_idx];
70  }
71  }
72  }
73  }
74  }
75 };
