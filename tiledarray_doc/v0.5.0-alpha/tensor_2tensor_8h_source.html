989  template <typename U, typename AU>
990  Tensor_ gemm(const Tensor<U, AU>& other, const numeric_type factor,
991  const math::GemmHelper& gemm_helper) const
992  {
993  // Check that this tensor is not empty and has the correct rank
994  TA_ASSERT(pimpl_);
995  TA_ASSERT(pimpl_->range_.rank() == gemm_helper.left_rank());
996 
997  // Check that the arguments are not empty and have the correct ranks
998  TA_ASSERT(!other.empty());
999  TA_ASSERT(other.range().rank() == gemm_helper.right_rank());
1000 
1001  // Construct the result Tensor
1002  Tensor_ result(gemm_helper.make_result_range<range_type>(pimpl_->range_, other.range()));

1107  const size_type* restrict const upper = pimpl_->range_.upbound_data();

1109

1111  size_type lower_max = 0ul, upper_min =
1112  std::numeric_limits<size_type>::max();
1113  for(size_type i = 0ul; i < n; ++i) {
1114  const size_type lower_i = lower[i];
1115  const size_type upper_i = upper[i];
1116 
1117  lower_max = std::max(lower_max, lower_i);
1118  upper_min = std::min(upper_min, upper_i);
1119  }
1120 
1121  value_type result = 0;
1122 
1123  if(lower_max < upper_min) {
1124  // Compute the first and last ordinal index
1125  size_type first = 0ul, last = 0ul, trace_stride = 0ul;
1126  for(size_type i = 0ul; i < n; ++i) {
1127  const size_type lower_i = lower[i];
1128  const size_type stride_i = stride[i];
1129 
1130  first += (lower_max - lower_i) * stride_i;
1131  last += (upper_min - lower_i) * stride_i;
1132  trace_stride += stride_i;
1133  }
1134 
1135  // Compute the trace
1136  const value_type* restrict const data = pimpl_->data_;
1137  for(; first < last; first += trace_stride)
1138  result += data[first];
1139  }
1140

1142  }
