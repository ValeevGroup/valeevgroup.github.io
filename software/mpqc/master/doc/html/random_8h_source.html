18 RowMatrix<T> random_unitary(int size, int seed = 42) {
19  static_assert(std::is_floating_point<T>::value, "random_unitary is only implemented for real types so far");
20 
21  auto engine = std::mt19937{}; engine.seed(seed);
22  auto dist = std::uniform_real_distribution<T>(0, 1);
23 
24  RowMatrix<T> rmat(size, size); // random real
25  for(int r=0; r!=size; ++r)
26  for(int c=0; c!=size; ++c)
27  rmat(r, c) = dist(engine);
28 
29  Eigen::HouseholderQR<RowMatrix<T>> qr(rmat);
30  RowMatrix<T> Q = qr.householderQ();
31  // fix phase of Q: see p. 9 of http://home.lu.lv/~sd20008/papers/essays/Random%20unitary%20[paper].pdf
32  RowMatrix<T> R = qr.matrixQR().template triangularView<Eigen::Upper>();
33  for(int c=0; c!=size; ++c) {
34  if (R(c,c) < 0) {
35  Q.col(c) *= -1;
36  }
37  }
38  return Q;
39 }
