77  auto canonicalize_and_form_G(std::array<RowMatrixXd, 4> ZMFS, int I, int J,
78  int i_size, int j_size) const {
79  RowMatrixXd& Z = ZMFS[0];
80  RowMatrixXd& M = ZMFS[1];
81  RowMatrixXd& F = ZMFS[2];
82  RowMatrixXd& S = ZMFS[3];
83 
84  // We will need to first make an OSV basis for this pair
85  RowMatrixXd UIJ;
86  if (use_osv_basis_ && I != J) {
87  // Calling get should be okay since we replicated osvs_
88  PnoPair& osv_I = osvs_.find(I).get();
89  PnoPair& osv_J = osvs_.find(J).get();
90 
91  RowMatrixXd UI = osv_I.U;
92  auto& domains_I = osv_I.pao_domains;
93 
94  RowMatrixXd UJ = osv_J.U;
95  auto& domains_J = osv_J.pao_domains;
96 
97  // We need PAO domains for the rows and the cols will be the OSV space
98  const auto nrows = S.rows();
99  const auto ncols_I = UI.cols();
100  const auto ncols_J = UJ.cols();
101  const auto ncols_IJ = ncols_I + ncols_J;
102 
103  UIJ.resize(nrows, ncols_IJ);
104  UIJ.setZero();
105 
106  SparseMap pair_domain = domains_I + domains_J;
107 
108  // Let's do this in a dumb way just loop over all domains first
109  auto IJ_row = 0; // Keep track of rows of UIJ
110  auto I_row = 0; // Keep track of rows of UI
111  auto J_row = 0; // Keep track of rows of UJ
112 
113  for (auto d = 0; d < pao_domain_sizes_.size(); ++d) {
114  const auto drows = pao_domain_sizes_[d];
115 
116  if (domains_I(d) > 0) { // Write UI to UIJ
117  auto UIJ_block = UIJ.block(IJ_row, 0, drows, ncols_I);
118  UIJ_block = UI.block(I_row, 0, drows, ncols_I);
119  I_row += drows;
120  }
121 
122  if (domains_J(d) > 0) { // Write UJ to UIJ
123  auto UIJ_block = UIJ.block(IJ_row, ncols_I, drows, ncols_J);
124  UIJ_block = UJ.block(J_row, 0, drows, ncols_J);
125  J_row += drows;
126  }
127 
128  // If this domain was in IJ
129  if (pair_domain(d) > 0) {
130  IJ_row += drows;
131  }
132  }
133 
134  // Rotate F and S
135  F = AT_B_C(UIJ, F, UIJ);
136 
137  // Write the upper triangle
138  for (auto r = 0; r < S.rows(); ++r) {
139  for (auto c = r + 1; c < S.cols(); ++c) {
140  S(r, c) = S(c, r);
141  }
142  }
143  S = AT_B_C(UIJ, S, UIJ);
144 
145  }
146 
147  RowMatrixXd Usc;
148  Eigen::VectorXd sc_evals;
149  std::tie(Usc, sc_evals) = general_semi_canonicalize(F, S, tcutSPAO_);
150 
151  if (use_osv_basis_ && I != J) {
152  // Fix the transformation basis to be in the full space since we shrank
153  // it for the semi-canonicalization
154  Usc = (UIJ * Usc).eval();
155  Smut my_mut(mut_);
156  num_osvs_ij_ += Usc.cols();
157  }
158 
159  const auto naux = Z.rows();
160  const auto nij = (I != J) ? i_size + j_size : i_size;
161  const auto na = Usc.rows();
162  const auto nsc = Usc.cols();
163 
164  // Do the Usc rotation in a single GEMM
165  Z.resize(naux * nij, na);
166  Z = A_x_B(Z, Usc);
167  Z.resize(naux, nij * nsc);
168 
169  RowMatrixXd Giajb;
170  if (I == J) {
171  RowMatrixXd C = llt_solve(M, Z);
172  Giajb = AT_x_B(C, Z);
173  } else {
174  RowMatrixXd Zia = Z.leftCols(i_size * nsc);
175  RowMatrixXd Zjb = Z.rightCols(j_size * nsc);
176  RowMatrixXd Cia = llt_solve(M, Zia);
177  Giajb = AT_x_B(Cia, Zjb);
178  }
179 
180  return std::make_tuple(std::move(Giajb), std::move(Usc),
181  std::move(sc_evals));
182  }

217  const auto nI = ij_size.first;
218  const auto nJ = ij_size.second;

220  auto tsc0 = now();

222  RowMatrixXd Usc;

225  canonicalize_and_form_G(std::move(ZMFS), I, J, nI, nJ);
226  auto tsc1 = now();

304  void rotate_G(int ni, int nj, RowMatrixXd& G, RowMatrixXd const& U) const {
305  // In simple case just use math function
306  if (ni == 1 && nj == 1) {
307  G = AT_B_C(U, G, U);
308  return;
309  }
310 
311  auto nsc = U.cols();
312  const auto na = U.rows();
313 
314  // G(i a j, b) * U(b, bp)
315  G.resize(ni * na * nj, na);
316  // G = (G * U).eval();
317  G = A_x_B(G, U);
318  G.resize(ni * na, nj * nsc);
319 
320  // G(j bsc i, a) * U(a, asc)
321  G.transposeInPlace();
322  G.resize(nj * nsc * ni, na);
323 
324  G = A_x_B(G, U);
325  G.resize(nj * nsc, ni * nsc);
326  G.transposeInPlace();
327  }

329  PnoPair make_pnos(RowMatrixXd G, RowMatrixXd const& Usc,
330  Eigen::VectorXd const& sc_evals, SparseMap const& IJA,
331  int I, int I_size, int I_start, int J, int J_size,
332  int J_start) const {
333  auto const& ni = I_size;
334  auto const& nj = J_size;
335 
336  pno::PnoPair pij;
337 
338  pij.i_size = I_size;
339  pij.i_start = I_start;
340 
341  pij.j_size = J_size;
342  pij.j_start = J_start;
343 
344  pij.pao_domains = IJA;
345 
346  auto& Ta = pij.T;
347  Ta = G;
348 
349  const auto nsc = Usc.cols();
350 
351  for (auto i = 0; i < ni; ++i) {
352  const auto ei = Fii_(i + I_start);
353 
354  for (auto a = 0; a < nsc; ++a) {
355  const auto eia = ei - sc_evals(a);
356  const auto ia = i * nsc + a;
357 
358  for (auto j = 0; j < nj; ++j) {
359  const auto eiaj = eia + Fii_(j + J_start);
360 
361  for (auto b = 0; b < nsc; ++b) {
362  const auto eiajb = eiaj - sc_evals(b);
363  const auto jb = j * nsc + b;
364 
365  Ta(ia, jb) /= eiajb;
366  }
367  }
368  }
369  }
370 
371  double pair_energy = 0.0;
372 
373  for (auto i = 0; i < ni; ++i) {
374  const auto i_global = i + I_start;
375  for (auto j = 0; j < nj; ++j) {
376  const auto j_global = j + J_start;
377 
378  if (i_global < j_global) {
379  continue;
380  }
381 
382  else {
383  auto scale = (i_global == j_global) ? 1.0 : 2.0;
384  RowMatrixXd G_ij = G.block(i * nsc + 0, j * nsc + 0, nsc, nsc);
385  RowMatrixXd T_ij = Ta.block(i * nsc + 0, j * nsc + 0, nsc, nsc);
386  RowMatrixXd Tbar = 2 * T_ij - T_ij.transpose();
387  pair_energy += scale * (G_ij.array() * Tbar.array()).sum();
388  }
389  }
390  }
391 
392 
393  {
394  Smut smut(mut_);
395  sc_energy_ += pair_energy;
396  }
397 
398 
399  RowMatrixXd Dab(nsc, nsc);
400  Dab.setZero();
401 
402  // Accumulate over Dab
403  for (auto i = 0; i < ni; ++i) {
404  for (auto j = 0; j < nj; ++j) {
405  // Get the ij specific ab block of T, leave zeros for
406  // readability
407  auto scale = (i + I_start == j + J_start) ? 0.5 : 1.0;
408  RowMatrixXd Tmap = Ta.block(i * nsc + 0, j * nsc + 0, nsc, nsc);
409  RowMatrixXd Tt = scale * (4 * Tmap - 2 * Tmap.transpose());
410 
411  // Dab += Tt.transpose() * Tmap + Tt * Tmap.transpose();
412  Dab += AT_x_B(Tt, Tmap) + A_x_BT(Tt, Tmap);
413  }
414  }
415 
416  if (average_densities_) {
417  Dab /= double(ni + nj);
418  }
419 
420  auto e_pair = pno::SymmEVD(Dab);
421  auto const& evals = e_pair.second;
422  const auto nevals = evals.size();
423  auto rank = 0;
424  for (auto e = 0; e < nevals; ++e) {
425  if (evals(e) >= tcutPNO_) {
426  ++rank;
427  }
428  }
429 
430  if(rank == 0){
431  pij.is_zero = true;
432  return pij;
433  }
434 
435  auto& U = pij.U;
436  U = e_pair.first.rightCols(rank);
437  const auto npnos = rank;
438  pij.a_size = npnos;
439  pij.b_size = npnos;
440 
441  RowMatrixXd Fpno = U.transpose() * sc_evals.asDiagonal() * U;
442  e_pair = pno::SymmEVD(Fpno);
443 
444  // Rotate the rotation matrix
445  U = A_x_B(U, e_pair.first);
446 
447  // Store the sc evals
448  pij.epsil_ab = e_pair.second;
449 
450  // Rotate G to pno space
451  rotate_G(ni, nj, G, U);
452 
453  pij.K = std::move(G);
454 
455  Ta.resize(ni * npnos, nj * npnos);
456  Ta.setZero();
457 
458  // Make pij R but just leave it as zero
459  pij.R = Ta;
460 
461  // Finally put U into the redundant pao space
462  U = A_x_B(Usc, U);
463 
464  return pij;
465  }
